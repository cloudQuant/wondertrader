121 else if(m_nFeeAlg == 1)

126 case 0: ret = m_dOpenFee * amount; break;
127 case 1: ret = m_dCloseFee * amount; break;
128 case 2: ret = m_dCloseTFee * amount; break;
129 default: ret = 0.0; break;
130 }
131 }
