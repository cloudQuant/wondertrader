78void WtUftDtMgr::handle_push_quote(const char* stdCode, WTSTickData* newTick)
79{
80 if (newTick == NULL)
81 return;
82
83 if (_rt_tick_map == NULL)
84 _rt_tick_map = DataCacheMap::create();
85
86 _rt_tick_map->add(stdCode, newTick, true);
87
88 if(_ticks_cache != NULL)
89 {
90 WTSHisTickData* tData = (WTSHisTickData*)_ticks_cache->get(stdCode);
91 if (tData == NULL)
92 return;
93
94 // 如果只需要有效数据且当前成交量为0，则跳过
95 if (tData->isValidOnly() && newTick->volume() == 0)
96 return;
97
98 // 将新的Tick数据追加到历史数据中
99 tData->appendTick(newTick->getTickStruct());
100 }
101}
