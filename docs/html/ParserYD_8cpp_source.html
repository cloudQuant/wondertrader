124void ParserYD::notifyMarketData(const YDMarketData *pDepthMarketData)
125{
126 if (m_pBaseDataMgr == NULL)
127 return;
128
129 const YDInstrument* instInfo = pDepthMarketData->m_pInstrument;
130 const YDExchange* exchgInfo = instInfo->m_pExchange;
131 uint32_t actDate = pDepthMarketData->TradingDay;
132 uint32_t actTime = pDepthMarketData->TimeStamp;
133 uint32_t actHour = actTime / 10000000;
134
135 WTSContractInfo* contract = m_pBaseDataMgr->getContract(instInfo->InstrumentID, exchgInfo->ExchangeID);
136 if (contract == NULL)
137 return;
138
139 WTSTickData* tick = WTSTickData::create(instInfo->InstrumentID);
140 WTSTickStruct& quote = tick->getTickStruct();
141 wt_strcpy(quote.exchg, contract->getExchg());
142 tick->setContractInfo(contract);
143
144 quote.action_date = actDate;
145 quote.action_time = actTime;
146
147 quote.price = pDepthMarketData->LastPrice;
148 quote.total_volume = pDepthMarketData->Volume;
149 quote.trading_date = m_uTradingDate;
150 quote.total_turnover = pDepthMarketData->Turnover;
151
152 quote.open_interest = pDepthMarketData->OpenInterest;
153
154 quote.upper_limit = pDepthMarketData->UpperLimitPrice;
155 quote.lower_limit = pDepthMarketData->LowerLimitPrice;
156
157 quote.pre_close = pDepthMarketData->PreClosePrice;
158 quote.pre_settle = pDepthMarketData->PreSettlementPrice;
159 quote.pre_interest = pDepthMarketData->PreOpenInterest;
160
161 //委卖价格
162 quote.ask_prices[0] = pDepthMarketData->AskPrice;
163 //委买价格
164 quote.bid_prices[0] = pDepthMarketData->BidPrice;
165 //委卖量
166 quote.ask_qty[0] = pDepthMarketData->AskVolume;
167 //委买量
168 quote.bid_qty[0] = pDepthMarketData->BidVolume;
169
170 if (m_sink)
171 m_sink->handleQuote(tick, 1);
172
173 tick->release();
174}

235void ParserYD::DoSubscribe()
236{
237 CodeSet codeFilter = m_filterSubs;
238 if(codeFilter.empty())
239 {//如果订阅礼包只空的,则取出全部合约列表
240 return;
241 }
242
243 int nCount = 0;
244 for(auto& code : codeFilter)
245 {
246 std::size_t pos = code.find('.');
247 const YDInstrument* instInfo = NULL;
248 if (pos != std::string::npos)
249 {
250 instInfo = m_pUserAPI->getInstrumentByID((char*)code.c_str() + pos + 1);
251 }
252 else
253 {
254 instInfo = m_pUserAPI->getInstrumentByID((char*)code.c_str());
255 }
256
257 if(instInfo == NULL)
258 continue;
259
260 bool bRet = m_pUserAPI->subscribe(instInfo);
261 write_log(m_sink, LL_ERROR, "[ParserYD] Subscribe md of {} {}", code.c_str(), bRet?"succeed":"failed");
262 }
263}
