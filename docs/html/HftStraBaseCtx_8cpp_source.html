1267void HftStraBaseCtx::do_set_position(const char* stdCode, double qty, double price /* = 0.0 */, const char* userTag /*= ""*/)
1268{
1269 // 获取合约的持仓信息
1270 PosInfo& pInfo = _pos_map[stdCode];
1271 // 处理价格，如果传入价格为0，则使用当前市场价格
1272 double curPx = price;
1273 if (decimal::eq(price, 0.0))
1274 curPx = _price_map[stdCode];
1275 // 生成当前时间戳
1276 uint64_t curTm = (uint64_t)_engine->get_date() * 1000000000 + (uint64_t)_engine->get_raw_time() * 100000 + _engine->get_secs();
1277 // 获取当前交易日
1278 uint32_t curTDate = _engine->get_trading_date();
1279
1280 //手数相等则不用操作了
1281 if (decimal::eq(pInfo._volume, qty))
1282 return;
1283
1284 // 记录持仓变化日志
1285 log_info("Target position updated: {} -> {}", pInfo._volume, qty);
1286
1287 // 获取合约信息
1288 WTSCommodityInfo* commInfo = _engine->get_commodity_info(stdCode);
1289
1290 //成交价
1291 double trdPx = curPx;
1292
1293 // 计算持仓变化量
1294 double diff = qty - pInfo._volume;
1295 // 判断是买入还是卖出
1296 bool isBuy = decimal::gt(diff, 0.0);
1297 if (decimal::gt(pInfo._volume*diff, 0))//当前持仓和仓位变化方向一致, 增加一条明细, 增加数量即可
1298 {
1299 // 更新持仓数量
1300 pInfo._volume = qty;
1301
1302 // 如果设置了滑点，调整成交价格
1303 if (_slippage != 0)
1304 {
1305 trdPx += _slippage * commInfo->getPriceTick()*(isBuy ? 1 : -1);
1306 }
1307
1308 // 创建新的持仓明细
1309 DetailInfo dInfo;
1310 dInfo._long = decimal::gt(qty, 0);
1311 dInfo._price = trdPx;
1312 dInfo._volume = abs(diff);
1313 dInfo._opentime = curTm;
1314 dInfo._opentdate = curTDate;
1315 wt_strcpy(dInfo._usertag, userTag);
1316 pInfo._details.emplace_back(dInfo);
1317
1318 // 计算并记录手续费
1319 double fee = commInfo->calcFee(trdPx, abs(diff), 0);
1320 _fund_info._total_fees += fee;
1321
1322 // 记录成交日志
1323 log_trade(stdCode, dInfo._long, true, curTm, trdPx, abs(diff), fee, userTag);
1324 }
1325 else
1326 {//持仓方向和仓位变化方向不一致,需要平仓
1327 // 计算需要平仓的数量
1328 double left = abs(diff);
1329
1330 // 如果设置了滑点，调整成交价格
1331 if (_slippage != 0)
1332 trdPx += _slippage * commInfo->getPriceTick()*(isBuy ? 1 : -1);
1333
1334 // 更新持仓数量
1335 pInfo._volume = qty;
1336 // 如果持仓量变为0，清空浮动盈亏
1337 if (decimal::eq(pInfo._volume, 0))
1338 pInfo._dynprofit = 0;
1339 // 用于记录需要清理的持仓明细数量
1340 uint32_t count = 0;
1341 // 遍历所有持仓明细，进行平仓处理
1342 for (auto it = pInfo._details.begin(); it != pInfo._details.end(); it++)
1343 {
1344 DetailInfo& dInfo = *it;
1345 // 计算当前明细需要平仓的数量
1346 double maxQty = min(dInfo._volume, left);
1347 if (decimal::eq(maxQty, 0))
1348 continue;
1349
1350 // 按比例计算最大盈利和最大亏损
1351 double maxProf = dInfo._max_profit * maxQty / dInfo._volume;
1352 double maxLoss = dInfo._max_loss * maxQty / dInfo._volume;
1353
1354 // 更新当前明细的持仓量
1355 dInfo._volume -= maxQty;
1356 left -= maxQty;
1357
1358 // 如果明细持仓量变为0，标记需要清理
1359 if (decimal::eq(dInfo._volume, 0))
1360 count++;
1361
1362 // 计算平仓盈亏
1363 double profit = (trdPx - dInfo._price) * maxQty * commInfo->getVolScale();
1364 if (!dInfo._long)
1365 profit *= -1;
1366 // 更新平仓盈亏和浮动盈亏
1367 pInfo._closeprofit += profit;
1368 pInfo._dynprofit = pInfo._dynprofit*dInfo._volume / (dInfo._volume + maxQty);//浮盈也要做等比缩放
1369 _fund_info._total_profit += profit;
1370
1371 // 计算并记录手续费
1372 double fee = commInfo->calcFee(trdPx, maxQty, dInfo._opentdate == curTDate ? 2 : 1);
1373 _fund_info._total_fees += fee;
1374 //这里写成交记录
1375 log_trade(stdCode, dInfo._long, false, curTm, trdPx, maxQty, fee, userTag);
1376 //这里写平仓记录
1377 log_close(stdCode, dInfo._long, dInfo._opentime, dInfo._price, curTm, trdPx, maxQty, profit, maxProf, maxLoss, pInfo._closeprofit, dInfo._usertag, userTag);
1378
1379 // 如果已经平仓完所需数量，结束循环
1380 if (left == 0)
1381 break;
1382 }
1383
1384 //需要清理掉已经平仓完的明细
1385 while (count > 0)
1386 {
1387 auto it = pInfo._details.begin();
1388 pInfo._details.erase(it);
1389 count--;
1390 }
1391
1392 //最后,如果还有剩余的,则需要反手了
1393 if (left > 0)
1394 {
1395 // 计算反手数量，保持方向一致
1396 left = left * qty / abs(qty);
1397
1398 // 创建新的持仓明细
1399 DetailInfo dInfo;
1400 dInfo._long = decimal::gt(qty, 0);
1401 dInfo._price = trdPx;
1402 dInfo._volume = abs(left);
1403 dInfo._opentime = curTm;
1404 dInfo._opentdate = curTDate;
1405 wt_strcpy(dInfo._usertag, userTag);
1406 pInfo._details.emplace_back(dInfo);
1407
1408 //这里还需要写一笔成交记录
1409 double fee = commInfo->calcFee(trdPx, abs(left), 0);
1410 _fund_info._total_fees += fee;
1411 //_engine->mutate_fund(fee, FFT_Fee);
1412 log_trade(stdCode, dInfo._long, true, curTm, trdPx, abs(left), fee, userTag);
1413 }
1414 }
