88bool ParserShm::connect()
89{
90 _thrd_parser.reset(new StdThread([this]() {
91
92 write_log(_sink, LL_INFO, "[ParserShm] loading {} ...", _path);
93 while (!StdFile::exists(_path.c_str()))
94 {
95 write_log(_sink, LL_WARN, "[ParserShm] {} not exist yet, waiting for 2 seconds", _path);
96 std::this_thread::sleep_for(std::chrono::seconds(2));
97 continue;
98 }
99
100 _mapfile.reset(new BoostMappingFile);
101 _mapfile->map(_path.c_str());
102 _queue = (CastQueue*)_mapfile->addr();
103 uint32_t cast_pid = _queue->_pid;
104
105 if (_sink)
106 {
107 _sink->handleEvent(WPE_Connect, 0);
108 _sink->handleEvent(WPE_Login, 0);
109 }
110 write_log(_sink, LL_INFO, "[ParserShm] {} loaded, start to receiving", _path);
111
112 uint64_t lastIdx = UINT64_MAX;
113 while(!_stopped)
114 {
115 //如果pid不同，说明datakit重启了
116 if(cast_pid != _queue->_pid)
117 {
118 lastIdx = UINT64_MAX;
119 write_log(_sink, LL_WARN, "ShareMemory queue has been reset justnow");
120 cast_pid = _queue->_pid;
121 }
122
123 if (_queue->_readable == UINT64_MAX) //刚分配好，还没数据进来
124 {
125 lastIdx = NODATA_FLAG;
126 if(_check_span != 0)
127 std::this_thread::sleep_for(std::chrono::microseconds(_check_span));
128 continue;
129 }
130
131 if (lastIdx == UINT64_MAX) //有数据，第一次检查，则直接定位到最后一条数据
132 {
133 lastIdx = _queue->_readable;
134 if (_check_span != 0)
135 std::this_thread::sleep_for(std::chrono::microseconds(_check_span));
136 continue;
137 }
138 else if (lastIdx == NODATA_FLAG) //之前没数据的时候检查了一次，现在有数据了，从0开始读取
139 {
140 lastIdx = 0;
141 }
142 else if (lastIdx >= _queue->_readable) //没有新的数据进来
143 {
144 if (_check_span != 0)
145 std::this_thread::sleep_for(std::chrono::microseconds(_check_span));
146 continue;
147 }
148 else
149 {
150 lastIdx++; //普通情况，下标递增
151 }
152
153 DataItem& item = _queue->_items[lastIdx % _queue->_capacity];
154 switch (item._type)
155 {
156 case 0:
157 {
158 const char* fullCode = fmtutil::format("{}.{}", item._tick.exchg, item._tick.code);
159 auto it = _set_subs.find(fullCode);
160 if (it != _set_subs.end())
161 {
162 WTSTickData* newData = WTSTickData::create(item._tick);
163 if (_sink)
164 _sink->handleQuote(newData, 0);
165 newData->release();
166
167 static uint32_t recv_cnt = 0;
168 recv_cnt++;
169 if (recv_cnt % _gpsize == 0)
170 write_log(_sink, LL_DEBUG, "[ParserShm] {} ticks received in total", recv_cnt);
171 }
172 }
173 break;
174 case 1:
175 {
176 const char* fullCode = fmtutil::format("{}.{}", item._queue.exchg, item._queue.code);
177 auto it = _set_subs.find(fullCode);
178 if (it != _set_subs.end())
179 {
180 WTSOrdQueData* newData = WTSOrdQueData::create(item._queue);
181 if (_sink)
182 _sink->handleOrderQueue(newData);
183 newData->release();
184
185 static uint32_t recv_cnt = 0;
186 recv_cnt++;
187 if (recv_cnt % _gpsize == 0)
188 write_log(_sink, LL_DEBUG, "[ParserShm] {} queues received in total", recv_cnt);
189 }
190 }
191 break;
192 case 2:
193 {
194 const char* fullCode = fmtutil::format("{}.{}", item._order.exchg, item._order.code);
195 auto it = _set_subs.find(fullCode);
196 if (it != _set_subs.end())
197 {
198 WTSOrdDtlData* newData = WTSOrdDtlData::create(item._order);
199 if (_sink)
200 _sink->handleOrderDetail(newData);
201 newData->release();
202
203 static uint32_t recv_cnt = 0;
204 recv_cnt++;
205 if (recv_cnt % _gpsize == 0)
206 write_log(_sink, LL_DEBUG, "[ParserShm] {} orders received in total", recv_cnt);
207 }
208 }
209 break;
210 case 3:
211 {
212 const char* fullCode = fmtutil::format("{}.{}", item._trans.exchg, item._trans.code);
213 auto it = _set_subs.find(fullCode);
214 if (it != _set_subs.end())
215 {
216 WTSTransData* newData = WTSTransData::create(item._trans);
217 if (_sink)
218 _sink->handleTransaction(newData);
219 newData->release();
220
221 static uint32_t recv_cnt = 0;
222 recv_cnt++;
223 if (recv_cnt % _gpsize == 0)
224 write_log(_sink, LL_DEBUG, "[ParserShm] {} transactions received in total", recv_cnt);
225 }
226 }
227 break;
228 default:
229 break;
230 }
231 }
232 }));
233
234 return true;
235}
