24bool ShmCaster::init(WTSVariant* cfg)
25{
26 // 检查配置项指针是否有效
27 if (cfg == NULL)
28 return false;
29
30 // 检查广播器是否激活
31 if (!cfg->getBoolean("active"))
32 return false;
33
34 // 获取共享内存文件路径
35 _path = cfg->getCString("path");
36
37 // 每次启动都重置该队列，确保数据一致性
38 {
39 BoostFile bf;
40 bf.create_or_open_file(_path.c_str());
41 bf.truncate_file(sizeof(CastQueue));
42 bf.close_file();
43 }
44
45 // 创建内存映射文件并进行映射
46 _mapfile.reset(new BoostMappingFile);
47 _mapfile->map(_path.c_str());
48 _queue = (CastQueue*)_mapfile->addr();
49
50 // 在映射内存上调用CastQueue的构造函数，初始化队列
51 new(_mapfile->addr()) CastQueue();
52
53 // 设置进程标识，根据不同平台选择不同的获取方式
54#ifdef _MSC_VER
55 _queue->_pid = _getpid();
56#else
57 _queue->_pid = getpid();
58#endif
59
60 // 设置初始化标记并输出日志
61 _inited = true;
62 WTSLogger::info("ShmCaste initialized @ {}", _path.c_str());
63
64 return true;
65}

72void ShmCaster::broadcast(WTSTickData* curTick)
73{
74 // 参数有效性检查，确保广播器已初始化且数据非空
75 if (curTick == NULL || _queue == NULL || !_inited)
76 return;
77
78 /*
79 * 先移动写的下标，然后写入数据
80 * 写完了以后，再移动读的下标
81 */
82 // 原子操作获取并且递增写指针
83 uint64_t wIdx = _queue->_writable++;
84 // 计算实际的环形缓冲区下标
85 uint64_t realIdx = wIdx % _queue->_capacity;
86 // 设置数据类型为Tick数据
87 _queue->_items[realIdx]._type = 0;
88 // 将Tick数据复制到队列中的相应位置
89 memcpy(&_queue->_items[realIdx]._tick, &curTick->getTickStruct(), sizeof(WTSTickStruct));
90 // 更新可读指针，让其他进程知道有新数据可读
91 _queue->_readable = wIdx;
92}
