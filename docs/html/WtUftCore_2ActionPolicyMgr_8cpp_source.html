51bool ActionPolicyMgr::init(const char* filename)
52{
53 // 从配置文件加载数据
54 WTSVariant* cfg = WTSCfgLoader::load_from_file(filename);
55 if (cfg == NULL)
56 return false;
57
58 // 遍历配置文件中的所有规则组
59 auto keys = cfg->memberNames();
60 for (auto it = keys.begin(); it != keys.end(); it++)
61 {
62 // 获取规则组名称
63 const char* gpName = (*it).c_str();
64 WTSVariant* vGpItem = cfg->get(gpName);
65 // 为每个规则组创建一个全新的ActionRuleGroup
66 ActionRuleGroup& gp = _rules[gpName];
67
68 // 加载该规则组的所有交易动作规则
69 WTSVariant* vOrds = vGpItem->get("order");
70 if(vOrds != NULL && vOrds->isArray())
71 {
72 // 遍历处理每一个交易动作规则
73 for (uint32_t i = 0; i < vOrds->size(); i++)
74 {
75 WTSVariant* vObj = vOrds->get(i);
76 ActionRule aRule;
77
78 // 从配置中提取动作类型和限制值
79 const char* action = vObj->getCString("action");
80 uint32_t uLimit = vObj->getUInt32("limit");
81 uint32_t uLimitS = vObj->getUInt32("limit_s");
82 uint32_t uLimitL = vObj->getUInt32("limit_l");
83
84 // 将字符串动作类型转换为ActionType枚举值
85 if (wt_stricmp(action, "open") == 0)
86 aRule._atype = AT_Open;
87 else if (wt_stricmp(action, "close") == 0)
88 aRule._atype = AT_Close;
89 else if (wt_stricmp(action, "closetoday") == 0)
90 aRule._atype = AT_CloseToday;
91 else if (wt_stricmp(action, "closeyestoday") == 0)
92 aRule._atype = AT_CloseYestoday;
93 else
94 {
95 // 如果动作类型不识别，记录错误并跳过该条规则
96 WTSLogger::error("Loading action policy failed: unrecognized type {}", action);
97 continue;
98 }
99
100 // 设置规则的限制参数
101 aRule._limit = uLimit; // 总手数限制
102 aRule._limit_s = uLimitS; // 空头限制
103 aRule._limit_l = uLimitL; // 多头限制
104 aRule._pure = vObj->getBoolean("pure"); // 是否为纯净模式
105
106 // 将规则添加到规则组中
107 gp.emplace_back(aRule);
108 }
109 }
110
111 // 加载该规则组适用的品种过滤器
112 WTSVariant* filters = vGpItem->get("filters");
113 if(filters!=NULL && filters->isArray() && filters->size()>0)
114 {
115 // 遍历处理每一个品种代码
116 for (uint32_t i = 0; i < filters->size(); i++)
117 {
118 // 将品种代码与规则组名称建立映射关系
119 const char* commid = filters->get(i)->asCString();
120 _comm_rule_map[commid] = gpName;
121 }
122 }
123 }
124
125 // 释放配置对象内存
126 cfg->release();
127 return true;
128}

139const ActionRuleGroup& ActionPolicyMgr::getActionRules(const char* pid)
140{
141 // 默认使用"default"规则组
142 std::string gpName = "default";
143
144 {// 先找到品种对应的规则组名称
145 auto it = _comm_rule_map.find(pid);
146 if (it != _comm_rule_map.end())
147 gpName = it->second; // 如果找到对应的规则组，则使用该规则组
148 }
149
150 {
151 // 根据规则组名称查找规则组
152 auto it = _rules.find(gpName);
153 if (it == _rules.end())
154 {
155 // 如果指定的规则组不存在，则使用默认规则组
156 it = _rules.find("default");
157 WTSLogger::error("Action policy group {} not exists, changed to default group", gpName.c_str());
158 }
159
160 // 断言规则组必定存在，这是一个程序防御机制
161 assert(it != _rules.end());
162 return it->second; // 返回找到的规则组
163 }
164}
