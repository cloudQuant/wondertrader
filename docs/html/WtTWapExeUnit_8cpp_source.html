152 _total_times = cfg->getUInt32("total_times"); // 总执行次数
153 _price_mode = cfg->getUInt32("price_mode"); // 价格模式（0-最新价,1-最优价,2-对手价）

183void WtTWapExeUnit::on_order(uint32_t localid, const char* stdCode, bool isBuy, double leftover, double price, bool isCanceled)
184{
185 // 如果订单不在监控中，则忽略此回报
186 if (!_orders_mon.has_order(localid))
187 return;
188
189 // 如果订单已撤销或已完全成交，从订单监控器中移除
190 if (isCanceled || leftover == 0)
191 {
192 // 从订单监控器中删除该订单
193 _orders_mon.erase_order(localid);
194
195 // 减少撤单计数
196 if (_cancel_cnt > 0)
197 _cancel_cnt--;
198
199 // 记录日志
200 _ctx->writeLog(fmt::format("Order {} updated cancelcnt -> {}", localid, _cancel_cnt).c_str());
201 }
202 /***---begin---23.5.19---zhaoyk***/
203 // 如果订单完全成交且不是因为撤单，重置撤单次数
204 if (leftover == 0 && !isCanceled)
205 {
206 // 订单成交后重置撤单次数
207 _cancel_times = 0;
208 // 记录订单成交日志
209 _ctx->writeLog(fmtutil::format("Order {} has filled", localid));
210 }
211 /***---end---23.5.19---zhaoyk***/
212 // 如果所有订单已撤销（撤单计数为0），处理超时撤单情况
213 if (isCanceled && _cancel_cnt == 0)
214 {
215 // 获取当前实际仓位
216 double realPos = _ctx->getPosition(stdCode);
217
218 // 如果实际仓位与本轮目标仓位不一致，需要重新发单
219 if (!decimal::eq(realPos, _this_target))
220 {
221 /***---begin---23.5.22---zhaoyk***/
222 // 记录订单撤销后重发日志
223 _ctx->writeLog(fmtutil::format("Order {} of {} canceled, re_fire will be done", localid, stdCode));
224
225 // 增加撤单次数，用于下次发单时的价格偏移
226 _cancel_times++;
227
228 // 撤单后重新发单，确保发单数量不小于最小开仓数量
229 // fire_at_once(_this_target - realPos);
230 fire_at_once(max(_min_open_lots, _this_target - realPos));
231 /***---end---23.5.22---zhaoyk***/
232 }
233 }
234 /***---begin---23.5.22---zhaoyk***/
235 // 处理异常情况：订单已撤销但撤单计数不为0，这可能是一个错误状态
236 if (isCanceled && _cancel_cnt != 0)
237 {
238 // 记录错误日志并返回，等待后续检查
239 _ctx->writeLog(fmtutil::format("Order {} of {} hasn't canceled, error will be return ", localid, stdCode));
240 return;
241 }
242 /***---end---23.5.22---zhaoyk***/
243}

253void WtTWapExeUnit::on_channel_ready()
254{
255 // 标记交易通道为就绪状态
256 _channel_ready = true;
257
258 // 获取未完成的订单数量
259 double undone = _ctx->getUndoneQty(_code.c_str());
260 /***---begin---23.5.18---zhaoyk***/
261 // 情况1: 如果有未完成订单但本地没有订单记录，需要清除未被管理的订单
262 if (!decimal::eq(undone, 0) && !_orders_mon.has_order())
263 //if (undone != 0 && !_orders_mon.has_order())
264 /***---end---23.5.18---zhaoyk***/
265 {
266 // 这说明有未完成订单不在监控中，需要先撤销
267 _ctx->writeLog(fmt::format("{} unmanaged orders of {}, cancel all", undone, _code).c_str());
268
269 // 根据未完成订单的正负判断是买入还是卖出订单
270 bool isBuy = (undone > 0);
271
272 // 撤销对应方向的所有未完成订单
273 OrderIDs ids = _ctx->cancel(_code.c_str(), isBuy);
274
275 // 将撤销的订单添加到监控器中进行跟踪
276 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime());
277
278 // 更新撤单计数
279 _cancel_cnt += ids.size();
280
281 // 记录撤单日志
282 _ctx->writeLog(fmt::format("Unmanaged order updated cancelcnt to {}", _cancel_cnt).c_str());
283 }
284 /***---begin---23.5.18---zhaoyk***/
285 // 情况2: 如果没有未完成订单但本地有订单记录，需要清除本地错误订单
286 else if (decimal::eq(undone, 0) && _orders_mon.has_order())
287 {
288 /*
289 * By Wesey @ 2021.12.13
290 * 如果未完成单为0，但是OMS中是有订单的
291 * 说明OMS中是错单，需要清理掉，不然超时撤单就会出错
292 * 这种情况，一般是断线重连以后，之前下出去的订单，并没有真正发送到柜台
293 * 所以这里需要清理掉本地订单
294 */
295 // 记录日志并清除所有本地订单
296 _ctx->writeLog(fmtutil::format("Local orders of {} not confirmed in trading channel, clear all", _code.c_str()));
297 _orders_mon.clear_orders();
298 }
299 // 情况3: 其他异常情况处理
300 else // 参考Minimpactunit实现
301 {
302 // 记录日志，显示当前未完成订单和本地监控状态
303 _ctx->writeLog(fmtutil::format("Unrecognized condition while channle ready,{:.2f} live orders of {} exists,local orders {}exist", undone, _code.c_str(), _orders_mon.has_order() ? "" : "not"));
304 }
305 /***---end---23.5.18---zhaoyk***/
306
307 // 通道就绪后触发一次计算，重新调整仓位
308 do_calc();
309}

329void WtTWapExeUnit::on_tick(WTSTickData* newTick)
330{
331 // 如果行情数据为空或者合约代码不匹配，则直接返回
332 if (newTick == NULL || _code.compare(newTick->code()) != 0)
333 return;
334
335 // 标记是否为首次收到行情
336 bool isFirstTick = false;
337
338 // 如果已有缓存的行情数据，需要先释放内存
339 if (_last_tick)
340 {
341 _last_tick->release();
342 }
343 /***---begin---23.5.18---zhaoyk***/
344 else
345 {
346 // 标记为首次收到行情
347 isFirstTick = true;
348
349 // 首次行情需要检查是否在交易时间内
350 // 如果行情时间不在交易时间内（如集合竞价时段），则过滤掉该行情
351 if (_sess_info != NULL && !_sess_info->isInTradingTime(newTick->actiontime() / 100000))
352 return;
353 }
354 /***---end---23.5.18---zhaoyk***/
355
356 // 保存新的行情数据并增加引用计数
357 _last_tick = newTick;
358 _last_tick->retain();
359
360 /*
361 * 这里可以考虑一下
362 * 如果写的上一次丢出去的单子不够达到目标仓位
363 * 那么在新的行情数据进来的时候可以再次触发核心逻辑
364 */
365
366 // 如果是首次收到行情，需要检查目标仓位与实际仓位是否一致
367 if (isFirstTick)
368 {
369 // 获取目标仓位
370 double newVol = _target_pos;
371 const char* stdCode = _code.c_str();
372
373 // 获取未完成订单数量和实际仓位
374 double undone = _ctx->getUndoneQty(stdCode);
375 double realPos = _ctx->getPosition(stdCode);
376
377 // 如果目标仓位与当前仓位加未完成订单不一致，需要调整仓位
378 if (!decimal::eq(newVol, undone + realPos))
379 {
380 // 触发仓位计算和调整
381 do_calc();
382 }
383 }
384 // 非首次行情处理，主要检查订单超时和定时发单
385 else
386 {
387 // 获取当前时间
388 uint64_t now = TimeUtils::getLocalTimeNow();
389
390 // 标记是否有订单被撤销
391 bool hasCancel = false;
392
393 // 如果设置了订单超时时间且有未完成订单，检查是否有订单需要撤销
394 if (_ord_sticky != 0 && _orders_mon.has_order())
395 {
396 // 检查订单是否超时，并对超时订单执行撤单操作
397 _orders_mon.check_orders(_ord_sticky, now, [this, &hasCancel](uint32_t localid) {
398 // 尝试撤销超时订单
399 if (_ctx->cancel(localid))
400 {
401 // 增加撤单计数
402 _cancel_cnt++;
403
404 // 记录订单超时撤销日志
405 _ctx->writeLog(fmt::format("Order expired, cancelcnt updated to {}", _cancel_cnt).c_str());
406
407 // 标记有订单被撤销
408 hasCancel = true;
409 }
410 });
411
412 }
413
414 // 如果没有订单被撤销，并且距离上次发单时间超过了设定的发单间隔，触发新一轮发单
415 if (!hasCancel && (now - _last_fire_time >= _fire_span * 1000))
416 {
417 // 执行仓位计算和发单操作
418 do_calc();
419 }
420 }
421}

468void WtTWapExeUnit::fire_at_once(double qty)
469{
470 // 如果数量为0，则直接返回
471 if (decimal::eq(qty, 0))
472 return;
473
474 // 增加行情数据引用计数，防止在使用过程中被释放
475 _last_tick->retain();
476 WTSTickData* curTick = _last_tick;
477
478 // 获取合约代码和当前时间
479 const char* code = _code.c_str();
480 uint64_t now = TimeUtils::getLocalTimeNow();
481
482 // 根据数量正负确定交易方向，正数为买入，负数为卖出
483 bool isBuy = decimal::gt(qty, 0);
484
485 // 初始化目标价格
486 double targetPx = 0;
487
488 // 根据价格模式设置确定委托基准价格
489 // 0-最新价, 1-最优价(买入用买一价，卖出用卖一价), 2-对手价(买入用卖一价，卖出用买一价)
490 if (_price_mode == 0)
491 {
492 // 模式0: 使用最新成交价
493 targetPx = curTick->price();
494 }
495 else if (_price_mode == 1)
496 {
497 // 模式1: 使用最优价（买入用买一价，卖出用卖一价）
498 targetPx = isBuy ? curTick->bidprice(0) : curTick->askprice(0);
499 }
500 else if (_price_mode == 2)
501 {
502 // 模式2: 使用对手价（买入用卖一价，卖出用买一价）
503 targetPx = isBuy ? curTick->askprice(0) : curTick->bidprice(0);
504 }
505
506 /***---begin---23.5.22---zhaoyk***/
507 // 根据撤单次数调整目标价格
508 targetPx += _comm_info->getPriceTick() * _cancel_times* (isBuy ? 1 : -1);
509 /***---end---23.5.22---zhaoyk***/
510
511 // 检查涨跌停价
512 //检查涨跌停价
513 isCanCancel = true;
514 if (isBuy && !decimal::eq(_last_tick->upperlimit(), 0) && decimal::gt(targetPx, _last_tick->upperlimit()))
515 {
516 _ctx->writeLog(fmt::format("Buy price {} of {} modified to upper limit price", targetPx, _code.c_str(), _last_tick->upperlimit()).c_str());
517 targetPx = _last_tick->upperlimit();
518 isCanCancel = false;//如果价格被修正为涨跌停价，订单不可撤销
519 }
520 if (isBuy != 1 && !decimal::eq(_last_tick->lowerlimit(), 0) && decimal::lt(targetPx, _last_tick->lowerlimit()))
521 {
522 _ctx->writeLog(fmt::format("Sell price {} of {} modified to lower limit price", targetPx, _code.c_str(), _last_tick->lowerlimit()).c_str());
523 targetPx = _last_tick->lowerlimit();
524 isCanCancel = false; //如果价格被修正为涨跌停价，订单不可撤销
525 }
526
527 OrderIDs ids;
528 if (qty > 0)
529 ids = _ctx->buy(code, targetPx, abs(qty));
530 else
531 ids = _ctx->sell(code, targetPx, abs(qty));
532
533 _orders_mon.push_order(ids.data(), ids.size(), now, isCanCancel);
534
535 curTick->release();
536}

544void WtTWapExeUnit::do_calc()
545{
546 // 使用标记类防止重入，确保同一时间只有一个线程在执行计算
547 CalcFlag flag(&_in_calc);
548 if (flag)
549 return;
550
551 // 如果交易通道未就绪，则直接返回
552 if (!_channel_ready)
553 return;
554
555 // 这里加一个锁，主要原因是实盘过程中发现
556 // 在修改目标仓位的时候，会触发一次do_calc
557 // 而ontick也会触发一次do_calc，两次调用是从两个线程分别触发的，所以会出现同时触发的情况
558 // 如果不加锁，就会引起问题
559 // 这种情况在原来的SimpleExecUnit没有出现，因为SimpleExecUnit只在set_position的时候触发
560 StdUniqueLock lock(_mtx_calc);
561
562 // 获取合约代码
563 const char* code = _code.c_str();
564
565 // 获取未完成订单数量
566 double undone = _ctx->getUndoneQty(code);
567
568 // 获取真实目标仓位（处理清仓标记DBL_MAX的情况）
569 double newVol = get_real_target(_target_pos);
570
571 // 获取当前实际仓位
572 double realPos = _ctx->getPosition(code);
573
574 // 计算目标仓位与实际仓位的差值，即需要交易的数量
575 double diffQty = newVol - realPos;
576
577 // 检查是否有正在撤销的订单，如果有则不能进行下一轮计算
578 if (_cancel_cnt != 0)
579 {
580 // 记录日志并退出本轮执行
581 _ctx->writeLog(fmt::format("{}尚有未完成撤单指令,暂时退出本轮执行", _code).c_str());
582 return;
583 }
584
585 // 如果目标仓位与实际仓位相等，则无需发单，直接返回
586 if (decimal::eq(diffQty, 0))
587 return;
588
589 // 检查是否有与目标仓位变动方向相反的未完成订单
590 // 每一次发单要保障成交，如果有与目标相反的未完成单，需要先撤销
591 if (decimal::lt(diffQty * undone, 0))
592 {
593 // 确定未完成订单的方向（正数为买入，负数为卖出）
594 bool isBuy = decimal::gt(undone, 0);
595
596 // 撤销相应方向的所有订单
597 OrderIDs ids = _ctx->cancel(code, isBuy);
598
599 // 如果成功撤销了订单，需要更新订单监控状态
600 if (!ids.empty())
601 {
602 // 将撤销的订单添加到监控器中
603 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime());
604
605 // 增加撤单计数
606 _cancel_cnt += ids.size();
607
608 // 记录撤单日志
609 _ctx->writeLog(fmtutil::format("[{}@{}] live opposite order of {} canceled, cancelcnt -> {}", __FILE__, __LINE__, _code.c_str(), _cancel_cnt));
610 }
611
612 // 撤单后退出本轮执行，等待撤单回报
613 return;
614 }
615 // 因为采用逐笔发单策略，如果有同向的未完成订单，则暂不发新单
616 if (!decimal::eq(undone, 0))
617 {
618 // 记录日志并退出本轮执行
619 _ctx->writeLog(fmt::format("{}上一轮有挂单未完成,暂时退出本轮执行", _code).c_str());
620 return;
621 }
622
623 // 保存当前仓位
624 double curPos = realPos;
625
626 // 检查是否有最新的行情数据
627 if (_last_tick == NULL)
628 {
629 // 没有行情数据则无法计算价格，退出执行
630 _ctx->writeLog(fmt::format("{}没有最新tick数据,退出执行逻辑", _code).c_str());
631 return;
632 }
633 // 检查当前仓位是否已经达到目标仓位
634 if (decimal::eq(curPos, newVol))
635 {
636 // 如果当前仓位和目标仓位匹配，且不是清仓指令，则直接退出
637 if (!is_clear(_target_pos))
638 return;
639
640 // 如果是清仓指令，需要进一步检查多头仓位
641 // 获取多头仓位
642 double lPos = _ctx->getPosition(code, true, 1);
643
644 // 如果多头仓位已经为0，说明已经完全清仓，直接退出
645 if (decimal::eq(lPos, 0))
646 return;
647
648 // 如果还有多头仓位，则将目标仓位设置为卖出指令，强制触发清仓
649 newVol = -min(lPos, _order_lots);
650
651 // 记录清仓过程触发日志
652 _ctx->writeLog(fmtutil::format("Clearing process triggered, target position of {} has been set to {}", _code.c_str(), newVol));
653 }
654
655 // 检查行情是否有更新，如果相比上次没有更新的tick，则先不下单
656 // 这是为了防止开盘前集中下单导致交易通道被封
657 uint64_t curTickTime = (uint64_t)_last_tick->actiondate() * 1000000000 + _last_tick->actiontime();
658
659 // 如果当前行情时间小于等于上次行情时间，说明行情未更新
660 if (curTickTime <= _last_tick_time)
661 {
662 // 记录日志并等待下次执行
663 _ctx->writeLog(fmtutil::format("No tick of {} updated, {} <= {}, execute later", _code, curTickTime, _last_tick_time));
664 return;
665 }
666
667 // 更新最新行情时间
668 _last_tick_time = curTickTime;
669
670 // 设置单次发单手数
671 double this_qty = _order_lots;
672 /***---end---23.5.26---zhaoyk***/
673
674 // 计算剩余的发单次数
675 uint32_t leftTimes = _total_times - _fired_times;
676
677 /***---begin---23.5.22---zhaoyk***/
678 // 记录当前发单次数日志
679 //_ctx->writeLog(fmt::format("第 {} 次发单", _fired_times).c_str());
680 _ctx->writeLog(fmt::format("第 {} 次发单", _fired_times+1).c_str());
681 /***---end---23.5.22---zhaoyk***/
682
683 // 标记是否需要全部发单（即“打板”）
684 bool bNeedShowHand = false;
685
686 // 计算本次发单数量
687 // 如果剩余发单次数为0且还有未完成的仓位差异，则需要全部发单
688 // 否则将剩余数量平均分配到剩余发单次数中
689 /***---begin---23.5.22---zhaoyk***/
690 // 计算本次发单数量
691 double curQty = 0;
692
693 // 如果剩余次数为0，则需要全部发单（打板）
694 if (leftTimes == 0 && !decimal::eq(diffQty, 0))
695 {
696 // 标记需要打板
697 bNeedShowHand = true;
698
699 // 取最小开仓数量和差异数量的最大值
700 curQty = max(_min_open_lots, diffQty);
701 }
702 else {
703 // 如果还有剩余次数，则将差异数量平均分配
704 // 取最小开仓数量和平均分配数量的最大值，并保持原始符号
705 curQty = std::max(_min_open_lots, round(abs(diffQty) / leftTimes)) * abs(diffQty) / diffQty;
706 }
707 /***---end---23.5.22---zhaoyk***/
708 /***---end---23.5.22---zhaoyk***/
709
710 // 设定本轮目标仓位（当前实际仓位 + 本次发单数量）
711 _this_target = realPos + curQty;
712
713 WTSTickData* curTick = _last_tick;
714 uint64_t now = TimeUtils::getLocalTimeNow();
715 bool isBuy = decimal::gt(diffQty, 0);
716 double targetPx = 0;
717 //根据价格模式设置,确定委托基准价格: 0-最新价,1-最优价,2-对手价
718 if (_price_mode == 0)
719 {
720 targetPx = curTick->price();
721 }
722 else if (_price_mode == 1)
723 {
724 targetPx = isBuy ? curTick->bidprice(0) : curTick->askprice(0);
725 }
726 else // if(_price_mode == 2)
727 {
728 targetPx = isBuy ? curTick->askprice(0) : curTick->bidprice(0);
729 }
730
731 //如果需要全部发单,则价格偏移5跳,以保障执行
732 if (bNeedShowHand) // last showhand time
733 {
734 targetPx += _comm_info->getPriceTick() * 5 * (isBuy ? 1 : -1);
735 }
736 else if (_price_offset != 0) //如果设置了价格偏移,也要处理一下
737 {
738 targetPx += _comm_info->getPriceTick() * _price_offset * (isBuy ? 1 : -1);
739 }
740 // 如果最后价格为0，再做一个修正
741 if (decimal::eq(targetPx, 0.0))
742 targetPx = decimal::eq(_last_tick->price(), 0.0) ? _last_tick->preclose() : _last_tick->price();
743
744 //检查涨跌停价
745 isCanCancel = true;
746 if (isBuy && !decimal::eq(_last_tick->upperlimit(), 0) && decimal::gt(targetPx, _last_tick->upperlimit()))
747 {
748 _ctx->writeLog(fmt::format("Buy price {} of {} modified to upper limit price", targetPx, _code.c_str(), _last_tick->upperlimit()).c_str());
749 targetPx = _last_tick->upperlimit();
750 isCanCancel = false;//如果价格被修正为涨跌停价，订单不可撤销
751 }
752 if (isBuy != 1 && !decimal::eq(_last_tick->lowerlimit(), 0) && decimal::lt(targetPx, _last_tick->lowerlimit()))
753 {
754 _ctx->writeLog(fmt::format("Sell price {} of {} modified to lower limit price", targetPx, _code.c_str(), _last_tick->lowerlimit()).c_str());
755 targetPx = _last_tick->lowerlimit();
756 isCanCancel = false; //如果价格被修正为涨跌停价，订单不可撤销
757 }
758 //最后发出指令
759 OrderIDs ids;
760 if (curQty > 0)
761 ids = _ctx->buy(code, targetPx, abs(curQty));
762 else
763 ids = _ctx->sell(code, targetPx, abs(curQty));
764
765 _orders_mon.push_order(ids.data(), ids.size(), now, isCanCancel);
766 _last_fire_time = now;
767 _fired_times += 1;
768
769 curTick->release();
770}
