68public:
69 static T* allocate()
70 {
71 /*
72 * By Wesley @ 2022.06.14
73 * 有用户反馈，这里使用了thread_local，线程销毁的话，内存池也销毁了
74 * 该用户在Trader里复现了这个bug，如果Trader底层销毁了一个API对象实例
75 * 那么这里内存池就已经析构了，如果有在系统中存储（retain）Trader创建的对象（WTSOrderInfo等），则会出现访问越界的问题
76 * 这里如果去掉thread_local，改成纯静态的，可能多线程并发的场景下也会有一些问题
77 * 总之如果要彻底安全，那么可能需要加一把锁才行，但是这样会带来性能开销
78 * 所以注释一下，如果有问题的可以参考一下
79 */
80 thread_local static MyPool pool;
81 thread_local static SpinMutex mtx;
82
83 mtx.lock();
84 T* ret = pool.construct();
85 mtx.unlock();
86 ret->_pool = &pool;
87 ret->_mutex = &mtx;
88 return ret;
89 }
