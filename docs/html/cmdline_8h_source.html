215struct range_reader{
216 range_reader(const T &low, const T &high): low(low), high(high) {}
217 T operator()(const std::string &s) const {
218 T ret=default_reader<T>()(s);
219 if (!(ret>=low && ret<=high)) throw cmdline::cmdline_error("range_error");
220 return ret;
221 }
222private:
223 T low, high;
224};

436 bool parse(const std::string &arg){
437 std::vector<std::string> args;
438
439 std::string buf;
440 bool in_quote=false;
441 for (std::string::size_type i=0; i<arg.length(); i++){
442 if (arg[i]=='\"'){
443 in_quote=!in_quote;
444 continue;
445 }
446
447 if (arg[i]==' ' && !in_quote){
448 args.push_back(buf);
449 buf="";
450 continue;
451 }
452
453 if (arg[i]=='\\'){
454 i++;
455 if (i>=arg.length()){
456 errors.push_back("unexpected occurrence of '\\' at end of string");
457 return false;
458 }
459 }
460
461 buf+=arg[i];
462 }
463
464 if (in_quote){
465 errors.push_back("quote is not closed");
466 return false;
467 }
468
469 if (buf.length()>0)
470 args.push_back(buf);
471
472 for (size_t i=0; i<args.size(); i++)
473 std::cout<<"\""<<args[i]<<"\""<<std::endl;
474
475 return parse(args);
476 }

478 bool parse(const std::vector<std::string> &args){
479 int argc=static_cast<int>(args.size());
480 std::vector<const char*> argv(argc);
481
482 for (int i=0; i<argc; i++)
483 argv[i]=args[i].c_str();
484
485 return parse(argc, &argv[0]);
486 }
487
488 bool parse(int argc, const char * const argv[]){
489 errors.clear();
490 others.clear();
491
492 if (argc<1){
493 errors.push_back("argument number must be longer than 0");
494 return false;
495 }
496 if (prog_name=="")
497 prog_name=argv[0];
498
499 std::map<char, std::string> lookup;
500 for (std::map<std::string, option_base*>::iterator p=options.begin();
501 p!=options.end(); p++){
502 if (p->first.length()==0) continue;
503 char initial=p->second->short_name();
504 if (initial){
505 if (lookup.count(initial)>0){
506 lookup[initial]="";
507 errors.push_back(std::string("short option '")+initial+"' is ambiguous");
508 return false;
509 }
510 else lookup[initial]=p->first;
511 }
512 }
513
514 for (int i=1; i<argc; i++){
515 if (strncmp(argv[i], "--", 2)==0){
516 const char *p=strchr(argv[i]+2, '=');
517 if (p){
518 std::string name(argv[i]+2, p);
519 std::string val(p+1);
520 set_option(name, val);
521 }
522 else{
523 std::string name(argv[i]+2);
524 if (options.count(name)==0){
525 errors.push_back("undefined option: --"+name);
526 continue;
527 }
528 if (options[name]->has_value()){
529 if (i+1>=argc){
530 errors.push_back("option needs value: --"+name);
531 continue;
532 }
533 else{
534 i++;
535 set_option(name, argv[i]);
536 }
537 }
538 else{
539 set_option(name);
540 }
541 }
542 }
543 else if (strncmp(argv[i], "-", 1)==0){
544 if (!argv[i][1]) continue;
545 char last=argv[i][1];
546 for (int j=2; argv[i][j]; j++){
547 last=argv[i][j];
548 if (lookup.count(argv[i][j-1])==0){
549 errors.push_back(std::string("undefined short option: -")+argv[i][j-1]);
550 continue;
551 }
552 if (lookup[argv[i][j-1]]==""){
553 errors.push_back(std::string("ambiguous short option: -")+argv[i][j-1]);
554 continue;
555 }
556 set_option(lookup[argv[i][j-1]]);
557 }
558
559 if (lookup.count(last)==0){
560 errors.push_back(std::string("undefined short option: -")+last);
561 continue;
562 }
563 if (lookup[last]==""){
564 errors.push_back(std::string("ambiguous short option: -")+last);
565 continue;
566 }
567
568 if (i+1<argc && options[lookup[last]]->has_value()){
569 set_option(lookup[last], argv[i+1]);
570 i++;
571 }
572 else{
573 set_option(lookup[last]);
574 }
575 }
576 else{
577 others.push_back(argv[i]);
578 }
579 }
580
581 for (std::map<std::string, option_base*>::iterator p=options.begin();
582 p!=options.end(); p++)
583 if (!p->second->valid())
584 errors.push_back("need option: --"+std::string(p->first));
585
586 return errors.size()==0;
587 }

618 std::string usage() const {
619 std::ostringstream oss;
620 oss<<"usage: "<<prog_name<<" ";
621 for (size_t i=0; i<ordered.size(); i++){
622 if (ordered[i]->must())
623 oss<<ordered[i]->short_description()<<" ";
624 }
625
626 oss<<"[options] ... "<<ftr<<std::endl;
627 oss<<"options:"<<std::endl;
628
629 size_t max_width=0;
630 for (size_t i=0; i<ordered.size(); i++){
631 max_width=std::max(max_width, ordered[i]->name().length());
632 }
633 for (size_t i=0; i<ordered.size(); i++){
634 if (ordered[i]->short_name()){
635 oss<<" -"<<ordered[i]->short_name()<<", ";
636 }
637 else{
638 oss<<" ";
639 }
640
641 oss<<"--"<<ordered[i]->name();
642 for (size_t j=ordered[i]->name().length(); j<max_width+4; j++)
643 oss<<' ';
644 oss<<ordered[i]->description()<<std::endl;
645 }
646 return oss.str();
647 }
