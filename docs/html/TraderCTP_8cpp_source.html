456int TraderCTP::orderInsert(WTSEntrust* entrust)
457{
458 if (m_pUserAPI == NULL || m_wrapperState != WS_ALLREADY)
459 {
460 write_log(m_sink, LL_ERROR, "[TraderCTP] Trading channel not ready");
461 return -1;
462 }
463
464 CThostFtdcInputOrderField req;
465 memset(&req, 0, sizeof(req));
466 wt_strcpy(req.BrokerID, m_strBroker.c_str(), m_strBroker.size());
467 wt_strcpy(req.InvestorID, m_strUser.c_str(), m_strUser.size());
468
469 wt_strcpy(req.InstrumentID, entrust->getCode());
470 wt_strcpy(req.ExchangeID, entrust->getExchg());
471
472 if (strlen(entrust->getUserTag()) == 0)
473 {
475 fmt::format_to(req.OrderRef, "{}", m_orderRef.fetch_add(0));
476 }
477 else
478 {
479 uint32_t fid, sid, orderref;
480 extractEntrustID(entrust->getEntrustID(), fid, sid, orderref);
482 fmt::format_to(req.OrderRef, "{}", orderref);
483 }
484
485 if (strlen(entrust->getUserTag()) > 0)
486 {
487 m_eidCache.put(entrust->getEntrustID(), entrust->getUserTag(), 0, [this](const char* message) {
488 write_log(m_sink, LL_WARN, message);
489 });
490 }
491
493 req.OrderPriceType = wrapPriceType(entrust->getPriceType(), strcmp(entrust->getExchg(), "CFFEX") == 0);
495 req.Direction = wrapDirectionType(entrust->getDirection(), entrust->getOffsetType());
497 req.CombOffsetFlag[0] = wrapOffsetType(entrust->getOffsetType());
499 req.CombHedgeFlag[0] = THOST_FTDC_HF_Speculation;
501 req.LimitPrice = entrust->getPrice();
503 req.VolumeTotalOriginal = (int)entrust->getVolume();
504
505 if(entrust->getOrderFlag() == WOF_NOR)
506 {
507 req.TimeCondition = THOST_FTDC_TC_GFD;
508 req.VolumeCondition = THOST_FTDC_VC_AV;
509 }
510 else if (entrust->getOrderFlag() == WOF_FAK)
511 {
512 req.TimeCondition = THOST_FTDC_TC_IOC;
513 req.VolumeCondition = THOST_FTDC_VC_AV;
514 }
515 else if (entrust->getOrderFlag() == WOF_FOK)
516 {
517 req.TimeCondition = THOST_FTDC_TC_IOC;
518 req.VolumeCondition = THOST_FTDC_VC_CV;
519 }
520 //req.MinVolume = 1;
521
523 req.ContingentCondition = THOST_FTDC_CC_Immediately;
525 req.ForceCloseReason = THOST_FTDC_FCC_NotForceClose;
527 req.IsAutoSuspend = 0;
529 req.UserForceClose = 0;
530
531 int iResult = m_pUserAPI->ReqOrderInsert(&req, genRequestID());
532 if (iResult != 0)
533 {
534 write_log(m_sink, LL_ERROR, "[TraderCTP] Order inserting failed: {}", iResult);
535 }
536
537 return 0;
