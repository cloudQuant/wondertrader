98 static inline std::size_t find(const char* src, char symbol = '.', bool bReverse = false)
99 {
100 std::size_t len = strlen(src);
101 if (len != 0)
102 {
103 if (bReverse)
104 {
105 for (std::size_t idx = len - 1; idx >= 0; idx--)
106 {
107 if (src[idx] == symbol)
108 return idx;
109 }
110 }
111 else
112 {
113 for (std::size_t idx = 0; idx < len; idx++)
114 {
115 if (src[idx] == symbol)
116 return idx;
117 }
118 }
119 }
120
121
122 return std::string::npos;
123 }

130 static bool isStdChnFutOptCode(const char* code)
131 {
132 /* 定义正则表达式 */
133 //static cregex reg_stk = cregex::compile("^[A-Z]+.[A-z]+\\d{4}.(C|P).\\d+$"); //CFFEX.IO2007.C.4000

136 std::size_t i = 0;
137 for(; ; i++)
138 {
139 char ch = code[i];
140 if(ch == '\0')
141 break;
142
143 if(state == 0)
144 {
145 if (!('A' <= ch && ch <= 'Z'))
146 return false;
147
148 state += 1;
149 }
150 else if (state == 1)
151 {
152 if ('A' <= ch && ch <= 'Z')
153 continue;
154
155 if (ch == '.')
156 state += 1;
157 else
158 return false;
159 }
160 else if (state == 2)
161 {
162 if (!('A' <= ch && ch <= 'z'))
163 return false;
164
165 state += 1;
166 }
167 else if (state == 3)
168 {
169 if ('A' <= ch && ch <= 'z')
170 continue;
171
172 if ('0' <= ch && ch <= '9')
173 state += 1;
174 else
175 return false;
176 }
177 else if (state >= 4 && state <= 6)
178 {
179 if ('0' <= ch && ch <= '9')
180 state += 1;
181 else
182 return false;
183 }
184 else if (state == 7)
185 {
186 if (ch == '.')
187 state += 1;
188 else
189 return false;
190 }
191 else if (state == 8)
192 {
193 if (ch == 'C' || ch == 'P')
194 state += 1;
195 else
196 return false;

199 {
200 if (ch == '.')

369 {
370 //using namespace boost::xpressive;
371 //最后3-6位都是数字，才是分月合约
372 //static cregex reg_stk = cregex::compile("^.*[A-z|-]\\d{3,6}$"); //CFFEX.IO.2007
373 //return regex_match(code, reg_stk);
374 auto len = strlen(code);
375 char state = 0;
376 for (std::size_t i = 0; i < len; i++)
377 {
378 char ch = code[len - i - 1];
379 if (0 <= state && state < 3)
380 {
381 if (!('0' <= ch && ch <= '9'))
382 return false;
383
384 state += 1;
385 }
386 else if (3 == state || 4 == state)
387 {
388 if ('0' <= ch && ch <= '9')
389 state += 1;
390
391 else if ('A' <= ch && ch <= 'z')
392 {
393 state = 7;
394 break;
395 }
396 }
397 else if (4 == state)
398 {

400 state += 1;
401
402 else if (('A' <= ch && ch <= 'z') || ch == '-')
403 {
404 state = 7;
405 break;
406 }
407 }
408 else if (state < 6)
409 {
410 if ('0' <= ch && ch <= '9')
411 state += 1;
412 else
413 return false;
414 }
415 else if (state >= 6)
416 {
417 if (('A' <= ch && ch <= 'z') || ch == '-')
418 {
419 state = 7;
420 break;
421 }
422 else
423 {
424 return false;
425 }
426 }
427 }

443 if(bMatch)

451 //郑商所上期所期权代码格式ZC010P11600

456 {
457 if(!isdigit(code[idx]))
458 break;
459 }
460

462 s.append(".");
463 s.append(code, idx-3);
464 if(strcmp(exchg, "CZCE") == 0)
465 s.append("2");
466 s.append(&code[idx - 3], 3);
467 s.append(".");
468 s.append(&code[idx], 1);
469 s.append(".");
470 s.append(&code[idx + 1]);
471 return s;

646 static CodeInfo extractStdCode(const char* stdCode, IHotMgr *hotMgr)
647 {
648 //期权的代码规则和其他都不一样，所以单独判断
649 if(isStdChnFutOptCode(stdCode))
650 {
651 return extractStdChnFutOptCode(stdCode);
652 }
653 else
654 {
655 /*
656 * By Wesley @ 2021.12.25
657 * 1、先看是不是Q和H结尾的，如果是复权标记确认以后，最后一段长度-1，复制到code，如SSE.STK.600000Q
658 * 2、再看是不是分月合约，如果是，则将product字段拼接月份给code（郑商所特殊处理），如CFFEX.IF.2112
659 * 3、最后看看是不是HOT和2ND结尾的，如果是，则将product拷贝给code，如DCE.m.HOT
660 * 4、如果都不是，则原样复制第三段，如BINANCE.DC.BTCUSDT/SSE.STK.600000
661 */
662 thread_local static CodeInfo codeInfo;
663 codeInfo.clear();
664 auto idx = StrUtil::findFirst(stdCode, '.');
665 wt_strcpy(codeInfo._exchg, stdCode, idx);
666
667 auto idx2 = StrUtil::findFirst(stdCode + idx + 1, '.');
668 if (idx2 == std::string::npos)
669 {
670 wt_strcpy(codeInfo._product, stdCode + idx + 1);
671
672 //By Wesley @ 2021.12.29
673 //如果是两段的合约代码，如OKEX.BTC-USDT
674 //则品种代码和合约代码一致
675 wt_strcpy(codeInfo._code, stdCode + idx + 1);
676 }
677 else
678 {
679 wt_strcpy(codeInfo._product, stdCode + idx + 1, idx2);
680 const char* ext = stdCode + idx + idx2 + 2;
681 std::size_t extlen = strlen(ext);
682 char lastCh = ext[extlen - 1];
683 if (lastCh == SUFFIX_QFQ || lastCh == SUFFIX_HFQ)
684 {
685 codeInfo._exright = (lastCh == SUFFIX_QFQ) ? 1 : 2;
686
687 extlen--;
688 lastCh = ext[extlen - 1];
689 }
690
691 if (extlen == 4 && '0' <= lastCh && lastCh <= '9')
692 {
693 //如果最后一段是4位数字，说明是分月合约
694 //TODO: 这样的判断存在一个假设，最后一位是数字的一定是期货分月合约，以后可能会有问题，先注释一下
695 //那么code得加上品种id
696 //郑商所得单独处理一下，这个只能hardcode了
697 auto i = wt_strcpy(codeInfo._code, codeInfo._product);
698 if (memcmp(codeInfo._exchg, "CZCE", 4) == 0)
699 wt_strcpy(codeInfo._code + i, ext + 1, extlen-1);
700 else
701 wt_strcpy(codeInfo._code + i, ext, extlen);
702 }
703 else
704 {
705 const char* ruleTag = (hotMgr != NULL) ? hotMgr->getRuleTag(ext) :"";
706 if (strlen(ruleTag) == 0)
707 wt_strcpy(codeInfo._code, ext, extlen);
708 else
709 {
710 wt_strcpy(codeInfo._code, codeInfo._product);
711 wt_strcpy(codeInfo._ruletag, ruleTag);
712 }
713 }
714 }
715
716 return codeInfo;
717 }
