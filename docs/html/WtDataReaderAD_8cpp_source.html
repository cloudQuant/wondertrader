149WTSTickSlice* WtDataReaderAD::readTickSlice(const char* stdCode, uint32_t count, uint64_t etime /* = 0 */)
150{
151 CodeHelper::CodeInfo cInfo = CodeHelper::extractStdCode(stdCode, _hot_mgr);
152 WTSCommodityInfo* commInfo = _base_data_mgr->getCommodity(cInfo._exchg, cInfo._product);
153 std::string stdPID = StrUtil::printf("%s.%s", cInfo._exchg, cInfo._product);
154
155 uint32_t curDate, curTime, curSecs;
156 if (etime == 0)
157 {
158 curDate = _sink->get_date();
159 curTime = _sink->get_min_time();
160 curSecs = _sink->get_secs();
161
162 etime = (uint64_t)curDate * 1000000000 + curTime * 100000 + curSecs;
163 }
164 else
165 {
166 //20190807124533900
167 curDate = (uint32_t)(etime / 1000000000);
168 curTime = (uint32_t)(etime % 1000000000) / 100000;
169 curSecs = (uint32_t)(etime % 100000);
170 }
171
172 uint32_t endTDate = _base_data_mgr->calcTradingDate(stdPID.c_str(), curDate, curTime, false);
173
174 std::string curCode = cInfo._code;
175 if (commInfo->isFuture())
176 {
177 const char* ruleTag = cInfo._ruletag;
178 if (strlen(ruleTag) > 0)
179 curCode = _hot_mgr->getCustomRawCode(ruleTag, cInfo.stdCommID(), endTDate);
180 //else if (cInfo.isHot())
181 // curCode = _hot_mgr->getRawCode(cInfo._exchg, cInfo._product, endTDate);
182 //else if (cInfo.isSecond())
183 // curCode = _hot_mgr->getSecondRawCode(cInfo._exchg, cInfo._product, endTDate);
184 }
185
186 uint32_t reload_flag = 0; //重载标记，0-不需要重载，1-加载最新，2-全部重载
187 std::string key = StrUtil::printf("%s.%s", cInfo._exchg, curCode.c_str());
188 //先检查缓存
189 TicksList& tickList = _ticks_cache[key];
190 uint64_t last_access_time = 0;
191 do
192 {
193 if(tickList._ticks.capacity() < count)
194 {
195 //容量不够，全部重载
196 reload_flag = 2;
197 tickList._ticks.rset_capacity(count);
198 tickList._ticks.clear(); //清除原有数据
199 }
200
201 if(tickList._last_req_time < etime)
202 {
203 //请求时间不够，只拉取最新的
204 reload_flag = 1;
205 last_access_time = tickList._last_req_time;
206 break;
207 }
208
209 } while (false);
210
211 //这里要改成从lmdb读取
212 WtLMDBPtr db = get_t_db(cInfo._exchg, cInfo._code);
213 if (db == NULL)
214 return NULL;
215
216 if(reload_flag == 1)
217 {
218 //增量更新，从上次的时间戳到本次的时间戳，读取最新的即可
219 last_access_time += 1;
220 WtLMDBQuery query(*db);
221 LMDBHftKey lKey(cInfo._exchg, cInfo._code, (uint32_t)(last_access_time / 1000000000), (uint32_t)(last_access_time % 1000000000));
222 LMDBHftKey rKey(cInfo._exchg, cInfo._code, (uint32_t)(etime / 1000000000), (uint32_t)(etime % 1000000000));
223 int cnt = query.get_range(std::string((const char*)&lKey, sizeof(lKey)),
224 std::string((const char*)&rKey, sizeof(rKey)), [this, &tickList](const ValueArray& ayKeys, const ValueArray& ayVals) {
225 for(const std::string& item : ayVals)
226 {
227 WTSTickStruct* curTick = (WTSTickStruct*)item.data();
228 tickList._ticks.push_back(*curTick);
229 }
230 });
231 if(cnt > 0)
232 pipe_reader_log(_sink, LL_DEBUG, "{} ticks after {} of {} append to cache", cnt, last_access_time, stdCode);
233 }
234 else if(reload_flag == 2)
235 {
236 //全部更新，从结束时间往前读取即可
237 WtLMDBQuery query(*db);
238 LMDBHftKey rKey(cInfo._exchg, cInfo._code, (uint32_t)(etime / 1000000000), (uint32_t)(etime % 1000000000));
239 LMDBHftKey lKey(cInfo._exchg, cInfo._code, 0, 0);
240 int cnt = query.get_lowers(std::string((const char*)&lKey, sizeof(lKey)), std::string((const char*)&rKey, sizeof(rKey)),
241 count, [this, &tickList](const ValueArray& ayKeys, const ValueArray& ayVals) {
242 tickList._ticks.resize(ayVals.size());
243 for (std::size_t i = 0; i < ayVals.size(); i++)
244 {
245 const std::string& item = ayVals[i];
246 memcpy(&tickList._ticks[i], item.data(), item.size());
247 }
248 });
249
250 pipe_reader_log(_sink, LL_DEBUG, "{} ticks of {} loaded to cache for the first time", cnt, stdCode);
251 }
252
253 tickList._last_req_time = etime;
254
255 //全部读取完成以后，再生成切片
256 //这里要注意一下，因为读取tick是顺序的，但是返回是需要从后往前取指定条数的，所以应该先读取array_two
257 count = min((uint32_t)tickList._ticks.size(), count); //先对count做一个修正
258 auto ayTwo = tickList._ticks.array_two();
259 auto cnt_2 = ayTwo.second;
260 if(cnt_2 >= count)
261 {
262 //如果array_two条数足够，则直接返回数据块
263 return WTSTickSlice::create(stdCode, &tickList._ticks[ayTwo.second - count], count);
264 }
265 else
266 {
267 //如果array_two条数不够，需要再从array_one提取
268 auto ayOne = tickList._ticks.array_one();
269 auto diff = count - cnt_2;
270 auto ret = WTSTickSlice::create(stdCode, &tickList._ticks[ayOne.second - diff], diff);
271 if(cnt_2 > 0)
272 ret->appendBlock(ayTwo.first, cnt_2);
273 return ret;
274 }
275}

354 });
355

380

382 for (std::size_t idx = 0; idx < cnt; idx++)

386 LMDBBarKey* barKey = (LMDBBarKey*)key.data();
387 printf("%u\r\n", reverseEndian(barKey->_bartime));

418WTSBarStruct* WtDataReaderAD::get_rt_cache_bar(const char* exchg, const char* code, WTSKlinePeriod period)
419{
420 RTBarCacheWrapper* wrapper = NULL;
421 if (period == KP_DAY)
422 wrapper = &_d1_cache;
423 else if (period == KP_Minute1)
424 wrapper = &_m1_cache;
425 else if (period == KP_Minute5)
426 wrapper = &_m5_cache;
427
428 bool isDay = (period == KP_DAY);
429
430 if (wrapper != NULL)
431 {
432 if (wrapper->empty())
433 {
434 //如果缓存为空，则加载缓存
435 do
436 {
437
438 //缓存为空，先自动加载
439 std::string filename = _base_dir + wrapper->_filename;
440 if (!StdFile::exists(filename.c_str()))
441 break;
442
443 wrapper->_file_ptr.reset(new BoostMappingFile);
444 wrapper->_file_ptr->map(filename.c_str());
445 wrapper->_cache_block = (RTBarCache*)wrapper->_file_ptr->addr();
446
447 wrapper->_cache_block->_size = min(wrapper->_cache_block->_size, wrapper->_cache_block->_capacity);
448 wrapper->_last_size = wrapper->_cache_block->_size;
449
450 for (uint32_t i = 0; i < wrapper->_cache_block->_size; i++)
451 {
452 const BarCacheItem& item = wrapper->_cache_block->_items[i];
453 wrapper->_idx[StrUtil::printf("%s.%s", item._exchg, item._code)] = i;
454 }
455 } while (false);
456 }
457 else
458 {
459 //如果缓存不为空，检查一下有没有新的合约进来，如果有的话，就把索引更新一下
460 if (wrapper->_last_size != wrapper->_cache_block->_size)
461 {
462 for (uint32_t i = wrapper->_last_size; i < wrapper->_cache_block->_size; i++)
463 {
464 const BarCacheItem& item = wrapper->_cache_block->_items[i];
465 wrapper->_idx[StrUtil::printf("%s.%s", item._exchg, item._code)] = i;
466 }
467 }
468 }
469
470 //最后看缓存里有没有改合约对应的K线缓存
471 auto it = wrapper->_idx.find(StrUtil::printf("%s.%s", exchg, code));
472 if (it != wrapper->_idx.end())
473 {
474 return &wrapper->_cache_block->_items[it->second]._bar;
475 }
476 }
477
478 return NULL;
479}

499WTSKlineSlice* WtDataReaderAD::readKlineSlice(const char* stdCode, WTSKlinePeriod period, uint32_t count, uint64_t etime /* = 0 */)
500{
501 CodeHelper::CodeInfo cInfo = CodeHelper::extractStdCode(stdCode, _hot_mgr);
502 WTSCommodityInfo* commInfo = _base_data_mgr->getCommodity(cInfo._exchg, cInfo._product);
503 std::string stdPID = StrUtil::printf("%s.%s", cInfo._exchg, cInfo._product);
504
505 uint32_t curDate, curTime, curSecs;
506 if (etime == 0)
507 {
508 curDate = _sink->get_date();
509 curTime = _sink->get_min_time();
510 curSecs = _sink->get_secs();
511
512 etime = (uint64_t)curDate * 1000000000 + curTime * 100000 + curSecs;
513 }
514 else
515 {
516 //20190807124533900
517 curDate = (uint32_t)(etime / 1000000000);
518 curTime = (uint32_t)(etime % 1000000000) / 100000;
519 curSecs = (uint32_t)(etime % 100000);
520 }
521
522 uint32_t endTDate = _base_data_mgr->calcTradingDate(stdPID.c_str(), curDate, curTime, false);
523
524 std::string curCode = cInfo._code;
525 if (commInfo->isFuture())
526 {
527 const char* ruleTag = cInfo._ruletag;
528 if (strlen(ruleTag) > 0)
529 curCode = _hot_mgr->getCustomRawCode(ruleTag, cInfo.stdCommID(), endTDate);
530 //else if (cInfo.isHot())
531 // curCode = _hot_mgr->getRawCode(cInfo._exchg, cInfo._product, endTDate);
532 //else if (cInfo.isSecond())
533 // curCode = _hot_mgr->getSecondRawCode(cInfo._exchg, cInfo._product, endTDate);
534 }
535
536 std::string key = StrUtil::printf("%s#%u", stdCode, period);
537 BarsList& barsList = _bars_cache[key];
538 if (barsList._bars.capacity() < count)
539 {
540 //容量不够，全部重载
541 barsList._bars.rset_capacity(count);
542 barsList._bars.clear(); //清除原有数据
543 cacheBarsFromStorage(key, stdCode, period, count);
544 }
545
546 //这里只需要检查一下RTBarCache的K线时间戳和etime是否一致
547 //如果一致，说明最后一条K线还没完成，但是系统要读取，这个时候就用缓存中的最后一条bar追加到最后
548 //如果不一致，说明Writer那边已经处理完成了，直接用缓存好的K线即可
549 //OnMinuteEnd的时候也要做类似的检查
550 if (barsList._bars.empty())
551 return NULL;
552
553 //数据条数做一个修正
554 count = min((uint32_t)barsList._bars.size(), count);
555
556 bool isDay = (period == KP_DAY);
557 etime = isDay ? curDate : ((curDate - 19900000)*10000 + curTime);
558 if(barsList._last_req_time < etime)
559 {
560 //上次请求的时间，小于当前请求的时间，则要检查最后一条K线
561 WTSBarStruct& lastBar = barsList._bars.back();
562 uint32_t lastBarTime = isDay ? lastBar.date : (uint32_t)lastBar.time;
563 if(lastBarTime < etime)
564 {
565 //如果最后一条K线的时间小于当前时间，先从数LMDB更新最新的K线
566 update_cache_from_lmdb(barsList, cInfo._exchg, curCode.c_str(), period, lastBarTime);
567
568 lastBar = barsList._bars.back();
569 lastBarTime = isDay ? lastBar.date : (uint32_t)lastBar.time;
570 }
571
572 //从lmdb读完了以后，再检查
573 //如果时间戳仍然小于截止时间
574 //则从缓存中读取
575 if(lastBarTime < etime)
576 {
577 WTSBarStruct* rtBar = get_rt_cache_bar(cInfo._exchg, curCode.c_str(), period);
578 if(rtBar != NULL)
579 {
580 uint64_t cacheBarTime = isDay ? rtBar->date : rtBar->time;
581 if (cacheBarTime > etime)
582 {
583 //缓存的K线时间戳大于截止时间，说明检查的过程中Writer已经将数据转储到lmdb中了
584 //这个时候就再读一次lmdb
585 update_cache_from_lmdb(barsList, cInfo._exchg, curCode.c_str(), period, lastBarTime);
586 barsList._last_from_cache = false;
587 }
588 else
589 {
590 //如果缓存的K线时间没有超过etime，则将缓存中的最后一条K线追加到队列中
591 barsList._bars.push_back(*rtBar);
592 barsList._last_from_cache = true;
593 pipe_reader_log(_sink, LL_DEBUG,
594 "{} bars @ {} of {} updated from cache instead of lmdb in {}", PERIOD_NAME[period], etime, stdCode, __FUNCTION__);
595 }
596 }
597 }
598 }
599
600 //全部处理完了以后，再从K线缓存里读取
601 barsList._last_req_time = etime;
602
603 //这里要注意一下，因为读取数据是顺序的，但是返回是需要从后往前取指定条数的，所以应该先读取array_two
604 count = min((uint32_t)barsList._bars.size(), count); //先对count做一个修正
605 auto ayTwo = barsList._bars.array_two();
606 auto cnt_2 = ayTwo.second;
607 if (cnt_2 >= count)
608 {
609 //如果array_two条数足够，则直接返回数据块
610 return WTSKlineSlice::create(stdCode, period, 1, &barsList._bars[ayTwo.second - count], count);
611 }
612 else
613 {
614 //如果array_two条数不够，需要再从array_one提取
615 auto ayOne = barsList._bars.array_one();
616 auto diff = count - cnt_2;
617 auto ret = WTSKlineSlice::create(stdCode, period, 1, &barsList._bars[ayOne.second - diff], diff);
618 if (cnt_2 > 0)
619 ret->appendBlock(ayTwo.first, cnt_2);
620 return ret;
621 }
622
623
624 return NULL;
625}

637void WtDataReaderAD::onMinuteEnd(uint32_t uDate, uint32_t uTime, uint32_t endTDate /* = 0 */)
638{
639 //这里应该触发检查
640 uint64_t nowTime = (uint64_t)uDate * 10000 + uTime;
641 if (nowTime <= _last_time)
642 return;
643
644 uint64_t endBarTime = (uDate - 19900000) * 10000 + uTime;
645
646 for (auto it = _bars_cache.begin(); it != _bars_cache.end(); it++)
647 {
648 BarsList& barsList = (BarsList&)it->second;
649 CodeHelper::CodeInfo cInfo = CodeHelper::extractStdCode(barsList._code.c_str(), _hot_mgr);
650 if (barsList._period != KP_DAY)
651 {
652 uint32_t lastBarTime = (uint32_t)barsList._bars.back().time;
653 pipe_reader_log(_sink, LL_DEBUG,
654 "Updating {} bars of {} in section ({},{}]", PERIOD_NAME[barsList._period], barsList._code, lastBarTime, endBarTime);
655 update_cache_from_lmdb(barsList, barsList._exchg.c_str(), cInfo._code, barsList._period, lastBarTime);
656 if(lastBarTime < endBarTime)
657 {
658 WTSBarStruct* rtBar = get_rt_cache_bar(cInfo._exchg, cInfo._code, barsList._period);
659 if(rtBar->time > lastBarTime && rtBar->time <=endBarTime)
660 {
661 barsList._bars.push_back(*rtBar);
662 barsList._last_from_cache = true;
663 _sink->on_bar(barsList._code.c_str(), barsList._period, rtBar);
664 pipe_reader_log(_sink, LL_DEBUG,
665 "{} bars @ {} of {} updated from cache instead of lmdb in {}", PERIOD_NAME[barsList._period], endBarTime, barsList._code, __FUNCTION__);
666 }
667 }
668 }
669 else if(endTDate != 0)
670 {
671 uint32_t lastBarTime = barsList._bars.back().date;
672 endBarTime = uDate;
673 update_cache_from_lmdb(barsList, barsList._exchg.c_str(), cInfo._code, barsList._period, lastBarTime);
674 if (lastBarTime < endBarTime)
675 {
676 WTSBarStruct* rtBar = get_rt_cache_bar(cInfo._exchg, cInfo._code, barsList._period);
677 if (rtBar->date > lastBarTime && rtBar->date <= endBarTime)
678 {
679 barsList._bars.push_back(*rtBar);
680 barsList._last_from_cache = true;
681 _sink->on_bar(barsList._code.c_str(), barsList._period, rtBar);
682 pipe_reader_log(_sink, LL_DEBUG,
683 "{} bars @ of {} updated from cache instead of lmdb in {}", PERIOD_NAME[barsList._period], endBarTime, barsList._code, __FUNCTION__);
684 }
685 }
686 }
687 }
688
689 if (_sink)
690 _sink->on_all_bar_updated(uTime);
691
692 _last_time = nowTime;
693}
