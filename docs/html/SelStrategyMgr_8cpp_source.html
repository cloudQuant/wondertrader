29bool SelStrategyMgr::loadFactories(const char* path)
30{
31 if (!StdFile::exists(path))
32 {
33 WTSLogger::error("Directory {} of SEL strategy factory not exists", path);
34 return false;
35 }
36
37 uint32_t count = 0;
38 boost::filesystem::path myPath(path);
39 boost::filesystem::directory_iterator endIter;
40 for (boost::filesystem::directory_iterator iter(myPath); iter != endIter; iter++)
41 {
42 if (boost::filesystem::is_directory(iter->path()))
43 continue;
44
45#ifdef _WIN32
46 if (iter->path().extension() != ".dll")
47 continue;
48#else //_UNIX
49 if (iter->path().extension() != ".so")
50 continue;
51#endif
52
53 DllHandle hInst = DLLHelper::load_library(iter->path().string().c_str());
54 if (hInst == NULL)
55 continue;
56
57 FuncCreateSelStraFact creator = (FuncCreateSelStraFact)DLLHelper::get_symbol(hInst, "createMfStrategyFact");
58 if (creator == NULL)
59 {
60 DLLHelper::free_library(hInst);
61 continue;
62 }
63
64 ISelStrategyFact* fact = creator();
65 if (fact != NULL)
66 {
67 StraFactInfo& fInfo = _factories[fact->getName()];
68 fInfo._module_inst = hInst;
69 fInfo._module_path = iter->path().string();
70 fInfo._creator = creator;
71 fInfo._remover = (FuncDeleteSelStraFact)DLLHelper::get_symbol(hInst, "deleteMfStrategyFact");
72 fInfo._fact = fact;
73
74 WTSLogger::info("SEL strategy factory[{}] loaded", fact->getName());
75
76 count++;
77 }
78 else
79 {
80 DLLHelper::free_library(hInst);
81 continue;
82 }
83
84 }
85
86 WTSLogger::info("{} SEL strategy factories in directory[{}] loaded", count, path);
87
88 return true;
89}
