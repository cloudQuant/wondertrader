46bool HftStrategyMgr::loadFactories(const char* path)
47{
48 // 检查路径是否存在
49 if (!StdFile::exists(path))
50 {
51 WTSLogger::error("Directory {} of HFT strategy factory not exists", path);
52 return false;
53 }
54
55 uint32_t count = 0;
56 boost::filesystem::path myPath(path);
57 boost::filesystem::directory_iterator endIter;
58 // 遍历目录中的文件
59 for (boost::filesystem::directory_iterator iter(myPath); iter != endIter; iter++)
60 {
61 // 跳过子目录
62 if (boost::filesystem::is_directory(iter->path()))
63 continue;
64
65 // 根据不同操作系统加载不同扩展名的动态库
66#ifdef _WIN32
67 if (iter->path().extension() != ".dll")
68 continue;
69#else //_UNIX
70 if (iter->path().extension() != ".so")
71 continue;
72#endif
73
74 // 加载动态库
75 DllHandle hInst = DLLHelper::load_library(iter->path().string().c_str());
76 if (hInst == NULL)
77 continue;
78
79 // 获取创建策略工厂的函数
80 FuncCreateHftStraFact creator = (FuncCreateHftStraFact)DLLHelper::get_symbol(hInst, "createStrategyFact");
81 if (creator == NULL)
82 {
83 // 如果没有找到创建函数，释放动态库
84 DLLHelper::free_library(hInst);
85 continue;
86 }
87
88 // 创建策略工厂实例
89 IHftStrategyFact* pFact = creator();
90 if (pFact != NULL)
91 {
92 // 创建成功，将工厂信息保存到映射表中
93 StraFactInfo& fInfo = _factories[pFact->getName()];
94
95 fInfo._module_inst = hInst;
96 fInfo._module_path = iter->path().string();
97 fInfo._creator = creator;
98 fInfo._remover = (FuncDeleteHftStraFact)DLLHelper::get_symbol(hInst, "deleteStrategyFact");
99 fInfo._fact = pFact;
100 WTSLogger::info("HFT strategy factory[{}] loaded", pFact->getName());
101
102 count++;
103 }
104 else
105 {
106 // 创建失败，释放动态库
107 DLLHelper::free_library(hInst);
108 continue;
109 }
110 }
111
112 // 记录加载的策略工厂数量
113 WTSLogger::info("{} HFT strategy factories in directory[{}] loaded", count, path);
114
115 return true;
116}

126HftStrategyPtr HftStrategyMgr::createStrategy(const char* factname, const char* unitname, const char* id)
127{
128 // 在已加载的工厂中查找指定名称的工厂
129 auto it = _factories.find(factname);
130 if (it == _factories.end())
131 return HftStrategyPtr();
132
133 // 获取工厂信息，创建策略实例
134 StraFactInfo& fInfo = (StraFactInfo&)it->second;
135 // 使用工厂创建策略实例，并用HftStraWrapper封装
136 HftStrategyPtr ret(new HftStraWrapper(fInfo._fact->createStrategy(unitname, id), fInfo._fact));
137 // 将策略添加到策略映射表中
138 _strategies[id] = ret;
139 return ret;
140}
