56const char* WTSHotMgr::getRuleTag(const char* stdCode)
57{
58 if (m_mapCustRules == NULL)
59 return "";
60
61 auto len = strlen(stdCode);
62 if (stdCode[len - 1] == '+' || stdCode[len - 1] == '-')
63 len--;
64
65 auto idx = StrUtil::findLast(stdCode, '.');
66 if (idx == std::string::npos)
67 {
68 auto it = m_mapCustRules->find(std::string(stdCode, len));
69 if (it == m_mapCustRules->end())
70 return "";
71
72 return it->first.c_str();
73 }
74
75 const char* tail = stdCode + idx + 1;
76 auto it = m_mapCustRules->find(std::string(tail, len - idx - 1));
77 if (it == m_mapCustRules->end())
78 return "";
79
80 return it->first.c_str();
81}

94double WTSHotMgr::getRuleFactor(const char* ruleTag, const char* fullPid, uint32_t uDate /* = 0 */ )
95{
96 if (m_mapCustRules == NULL)
97 return 1.0;
98
99 WTSProductHotMap* prodMap = (WTSProductHotMap*)m_mapCustRules->get(ruleTag);
100 if (prodMap == NULL)
101 return 1.0;
102
103 WTSDateHotMap* dtMap = STATIC_CONVERT(prodMap->get(fullPid), WTSDateHotMap*);
104 if (dtMap == NULL)
105 return 1.0;
106
107 if(uDate == 0)
108 {
109 WTSSwitchItem* pItem = STATIC_CONVERT(dtMap->rbegin()->second, WTSSwitchItem*);
110 return pItem->get_factor();
111 }
112
113 auto it = dtMap->lower_bound(uDate);
114 if(it == dtMap->end())
115 {
116 //找不到，说明记录的日期都比传入的日期小，所以返回最后一条的复权因子
117 WTSSwitchItem* pItem = STATIC_CONVERT(dtMap->rbegin()->second, WTSSwitchItem*);
118 return pItem->get_factor();
119 }
120 else
121 {
122 //找到了，就要看切换日期是否等于传入日期
123 //如果相等，说明刚好切换，那么就直接返回复权因子
124 WTSSwitchItem* pItem = STATIC_CONVERT(it->second, WTSSwitchItem*);
125 if (pItem->switch_date() == uDate)
126 {
127 return pItem->get_factor();
128 }
129 else
130 {
131 //如果切换日期大于传入日期，则要看前一个阶段
132 //如果已经是第一个了，则直接返回1.0
133 if (it == dtMap->begin())
134 {
135 return 1.0;
136 }
137 else
138 {
139 //如果不是第一个，则回退一个，再返回即可
140 it--;
141 WTSSwitchItem* pItem = STATIC_CONVERT(it->second, WTSSwitchItem*);
142 return pItem->get_factor();
143 }
144 }
145 }

352bool WTSHotMgr::loadCustomRules(const char* tag, const char* filename)
353{
354 if (!StdFile::exists(filename))
355 {
356 return false;
357 }
358
359 WTSVariant* root = WTSCfgLoader::load_from_file(filename);
360 if (root == NULL)
361 return false;
362
363 if (m_mapCustRules == NULL)
364 m_mapCustRules = WTSCustomSwitchMap::create();
365
366 WTSProductHotMap* prodMap = (WTSProductHotMap*)m_mapCustRules->get(tag);
367 if(prodMap == NULL)
368 {
369 prodMap = WTSProductHotMap::create();
370 m_mapCustRules->add(tag, prodMap, false);
371 }
372
373 for (const std::string& exchg : root->memberNames())
374 {
375 WTSVariant* jExchg = root->get(exchg);
376
377 for (const std::string& pid : jExchg->memberNames())
378 {
379 WTSVariant* jProduct = jExchg->get(pid);
380 std::string fullPid = fmt::format("{}.{}", exchg, pid);
381
382 WTSDateHotMap* dateMap = WTSDateHotMap::create();
383 prodMap->add(fullPid.c_str(), dateMap, false);
384
385 std::string lastCode;
386 double factor = 1.0;
387 for (uint32_t i = 0; i < jProduct->size(); i++)
388 {
389 WTSVariant* jHotItem = jProduct->get(i);
390 WTSSwitchItem* pItem = WTSSwitchItem::create(
391 exchg.c_str(), pid.c_str(),
392 jHotItem->getCString("from"), jHotItem->getCString("to"),
393 jHotItem->getUInt32("date"));
394
395 //计算复权因子
396 double oldclose = jHotItem->getDouble("oldclose");
397 double newclose = jHotItem->getDouble("newclose");
398 factor *= (decimal::eq(oldclose, 0.0) ? 1.0 : (oldclose/ newclose));
399 pItem->set_factor(factor);
400 dateMap->add(pItem->switch_date(), pItem, false);
401 lastCode = jHotItem->getCString("to");
402 }
403
404 std::string fullCode = fmt::format("{}.{}", exchg.c_str(), lastCode.c_str());
405 m_mapCustCodes[tag].insert(fullCode);
406 }
407 }
408
409 root->release();
410 return true;
411}

532bool WTSHotMgr::isCustomHot(const char* tag, const char* fullCode, uint32_t dt /* = 0 */)
533{
534 if (m_mapCustRules == NULL)
535 return false;
536
537 const auto& curHotCodes = m_mapCustCodes[tag];
538 if (curHotCodes.empty())
539 return false;
540
541 if (dt == 0)
542 {
543 auto it = curHotCodes.find(fullCode);
544 if (it == curHotCodes.end())
545 return false;
546 else
547 return true;
548 }
549
550 auto idx = StrUtil::findFirst(fullCode, '.');
551 const char* rawCode = fullCode + idx + 1;
552 std::string fullPid(fullCode, idx);
553 fullPid += ".";
554 fullPid += CodeHelper::rawMonthCodeToRawCommID(rawCode);
555
556 WTSProductHotMap* prodMap = (WTSProductHotMap*)m_mapCustRules->get(tag);
557 if (prodMap == NULL)
558 return "";
559
560 WTSDateHotMap* dtMap = STATIC_CONVERT(prodMap->get(fullPid), WTSDateHotMap*);
561 if (dtMap == NULL)
562 return "";
563
564 WTSDateHotMap::ConstIterator cit = dtMap->lower_bound(dt);
565 if (cit != dtMap->end())
566 {
567 WTSSwitchItem* pItem = STATIC_CONVERT(cit->second, WTSSwitchItem*);
568 //因为登记的换月日期是开始生效的交易日，如果是下午盘后确定主力的话
569 //那么dt就会是第二天，所以，dt必须大于等于切换日期
570 if (pItem->switch_date() > dt)
571 cit--;
572
573 pItem = STATIC_CONVERT(cit->second, WTSSwitchItem*);
574 if (strcmp(pItem->to(), rawCode) == 0)
575 return true;
576 }
577 else if (dtMap->size() > 0)
578 {
579 WTSSwitchItem* pItem = (WTSSwitchItem*)dtMap->last();
580 if (strcmp(pItem->to(), rawCode) == 0)
581 return true;
582 }
583
584 return false;
585}

600bool WTSHotMgr::splitCustomSections(const char* tag, const char* fullPid, uint32_t sDt, uint32_t eDt, HotSections& sections)
601{
602 if (m_mapCustRules == NULL)
603 return false;
604
605 WTSProductHotMap* prodMap = (WTSProductHotMap*)m_mapCustRules->get(tag);
606 if (prodMap == NULL)
607 return false;
608
609 WTSDateHotMap* dtMap = STATIC_CONVERT(prodMap->get(fullPid), WTSDateHotMap*);
610 if (dtMap == NULL)
611 return false;
612
613 uint32_t leftDate = sDt;
614 uint32_t lastDate = 0;
615 const char* curHot = "";
616 auto cit = dtMap->begin();
617 double prevFactor = 1.0;
618 for (; cit != dtMap->end(); cit++)
619 {
620 uint32_t curDate = cit->first;
621 WTSSwitchItem* hotItem = (WTSSwitchItem*)cit->second;
622
623 if (curDate > eDt)
624 {
625 sections.emplace_back(HotSection(hotItem->from(), leftDate, eDt, prevFactor));
626 }
627 else if (leftDate < curDate)
628 {
629 //如果开始日期小于当前切换的日期,则添加一段
630 if (strlen(hotItem->from()) > 0)//这里from为空,主要是第一条规则,如果真的遇到这种情况,也没有太好的办法,只能不要这一段数据了,一般情况下是够的
631 {
632 sections.emplace_back(HotSection(hotItem->from(), leftDate, TimeUtils::getNextDate(curDate, -1), prevFactor));
633 }
634
635 leftDate = curDate;
636 }
637
638 lastDate = curDate;
639 prevFactor = hotItem->get_factor();
640 curHot = hotItem->to();
641 }
642
643 if (leftDate >= lastDate && lastDate != 0)
644 {
645 sections.emplace_back(HotSection(curHot, leftDate, eDt, prevFactor));
646 }
647
648 return true;
649}
