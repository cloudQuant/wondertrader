159 {

191void WtMinImpactExeUnit::on_channel_ready()
192{
193 double undone = _ctx->getUndoneQty(_code.c_str());//获取未完成数量
194
195 if(!decimal::eq(undone, 0) && !_orders_mon.has_order())
196 {
197 /*
198 * 如果未完成单不为0，而OMS没有订单
199 * 这说明有未完成单不在监控之中,全部撤销掉
200 * 因为这些订单没有本地订单号，无法直接进行管理
201 * 这种情况，就是刚启动的时候，上次的未完成单或者外部的挂单
202 */
203 _ctx->writeLog(fmtutil::format("Unmanaged live orders with qty {} of {} found, cancel all", undone, _code.c_str()));
204
205 bool isBuy = (undone > 0);
206 OrderIDs ids = _ctx->cancel(_code.c_str(), isBuy);//根据本地订单号撤单
207 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime()); //push into orderpair
208 _cancel_cnt += ids.size();
209
210 _ctx->writeLog(fmtutil::format("[{}@{}]cancelcnt -> {}", __FILE__, __LINE__, _cancel_cnt));
211 }
212 else if (decimal::eq(undone, 0) && _orders_mon.has_order())
213 {
214 /*
215 * By Wesey @ 2021.12.13
216 * 如果未完成单为0，但是OMS中是有订单的
217 * 说明OMS中是错单，需要清理掉，不然超时撤单就会出错
218 * 这种情况，一般是断线重连以后，之前下出去的订单，并没有真正发送到柜台
219 * 所以这里需要清理掉本地订单
220 */
221 _ctx->writeLog(fmtutil::format("Local orders of {} not confirmed in trading channel, clear all", _code.c_str()));
222 _orders_mon.clear_orders();
223 }
224 else
225 {
226 _ctx->writeLog(fmtutil::format("Unrecognized condition while channle ready, {:.2f} live orders of {} exists, local orders {}exist",
227 undone, _code.c_str(), _orders_mon.has_order() ? "" : "not "));
228 }
229
230
231 do_calc();
232}

252{

341void WtMinImpactExeUnit::do_calc()
342{
343 CalcFlag flag(&_in_calc);
344 if (flag)
345 return;
346
347 if (_cancel_cnt != 0)
348 return;
349
350 //这里加一个锁，主要原因是实盘过程中发现
351 //在修改目标仓位的时候，会触发一次do_calc
352 //而ontick也会触发一次do_calc，两次调用是从两个线程分别触发的，所以会出现同时触发的情况
353 //如果不加锁，就会引起问题
354 //这种情况在原来的SimpleExecUnit没有出现，因为SimpleExecUnit只在set_position的时候触发
355 StdUniqueLock lock(_mtx_calc );
356
357 double newVol = get_real_target(_target_pos);//真实价格目标仓位 newvol
358 const char* stdCode = _code.c_str();
359
360 double undone = _ctx->getUndoneQty(stdCode); //undone 未完成订单量
361 double realPos = _ctx->getPosition(stdCode); //realpos 获取仓位
362 double diffPos = newVol - realPos; //diffpos = 真实目标仓位-获取仓位
363
364 //有未完成订单，与实际仓位变动方向相反
365 //则需要撤销现有订单
366 if (decimal::lt(diffPos * undone, 0)) //diff*undone 剩余需要完成*未完成订单 <0 true -> cancel
367 {
368 bool isBuy = decimal::gt(undone, 0); //未完成单>0 isbuy = 1 买方
369 OrderIDs ids = _ctx->cancel(stdCode, isBuy);
370 if(!ids.empty())
371 {
372 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime());
373 _cancel_cnt += ids.size();
374 _ctx->writeLog(fmtutil::format("[{}@{}] live opposite order of {} canceled, cancelcnt -> {}", __FILE__, __LINE__, _code.c_str(), _cancel_cnt));//相反的订单已取消
375 }
376 return;
377 }
378
379 //因为是逐笔发单，所以如果有不需要撤销的未完成单，则暂不发单
380 //****此处逐笔指 拆单的回合中上一笔订单量未完结，则不发单， 逐笔发单指每个回合成交的单
381 if (!decimal::eq(undone, 0))
382 return;
383
384 double curPos = realPos; //realPos获取仓位 curPos 为获取现在的仓位
385
386 if (_last_tick == NULL)
387 {
388 _ctx->writeLog(fmtutil::format("No lastest tick data of {}, execute later", _code.c_str()));
389 return;
390 }
391
392 //检查下单时间间隔
393 uint64_t now = TimeUtils::makeTime(_last_tick->actiondate(), _last_tick->actiontime());
394 if (now - _last_place_time < _entrust_span) //当前时间-上ticktime <发单时间间隔
395 return;
396
397 if (decimal::eq(curPos, newVol))
398 {
399 //当前仓位和最新仓位匹配时，如果不是全部清仓的需求，则直接退出计算了
400 if (!is_clear(_target_pos))
401 return;
402
403 //如果是清仓的需求，还要再进行对比
404 //如果多头为0，说明已经全部清理掉了，则直接退出
405 double lPos = _ctx->getPosition(stdCode, true , 1); //获取（多头）持仓 可用持仓 多头 //返回值 轧平后的仓位: 多仓>0, 空仓<0
406 if (decimal::eq(lPos, 0))
407 return;
408
409 //如果还有多头仓位，则将目标仓位设置为非0，强制触发
410 newVol = -min(lPos, _order_lots);// -min(获取（多头）持仓 单次发单手数)
411 _ctx->writeLog(fmtutil::format("Clearing process triggered, target position of {} has been set to {}", _code.c_str(), newVol));
412 }
413
414 bool bForceClose = is_clear(_target_pos); //target==del_max return 1
415
416 bool isBuy = decimal::gt(newVol, curPos);//如何判断买卖方向 真实目标仓位-获取持仓 > 0 多头
417
418 //如果相比上次没有更新的tick进来，则先不下单，防止开盘前集中下单导致通道被封
419 uint64_t curTickTime = (uint64_t)_last_tick->actiondate() * 1000000000 + _last_tick->actiontime();
420 if (curTickTime <= _last_tick_time)
421 {
422 _ctx->writeLog(fmtutil::format("No tick of {} updated, {} <= {}, execute later", _code, curTickTime, _last_tick_time));
423 return;
424 }
425
426 _last_tick_time = curTickTime;
427
428 double this_qty = _order_lots; //单次发单手数
429 if (_by_rate)//是否按照对手的挂单数的比例下单，如果是true，则rate字段生效，如果是false则lots字段生效
430 {
431 this_qty = isBuy ? _last_tick->askqty(0) : _last_tick->bidqty(0); //isbuy ture askqty 买价量
432 this_qty = round(this_qty*_qty_rate);
433 if (decimal::lt(this_qty, 1)) //this_qty <0 时，return this_qty =1
434 this_qty = 1;
435 }
436
437 //By Wesley @ 2022.09.13
438 //这里要对下单数量做一个修正
439 this_qty = min(this_qty, abs(newVol - curPos));//真实-获取仓位
440
441 //是否开仓，如果持仓大于等于0且买入，或者持仓小于等于0且卖出，就是开仓
442 bool isOpen = (isBuy && decimal::ge(curPos, 0)) || (!isBuy && decimal::le(curPos, 0));
443
444 //如果平仓的话
445 //对单次下单做一个修正，保证平仓和开仓不会同时下单
446 if (!isOpen)
447 {
448 this_qty = min(this_qty, abs(curPos)); //curPos 现在的仓位
449 }
450
451 /*
452 * By Wesley @ 2022.12.15
453 * 增加一个对最小下单数量的修正逻辑
454 */
455 if (isOpen && decimal::lt(this_qty, _min_open_lots))//if 开仓&&this_qty<min_open_lots（最小开仓数量）
456 {
457 this_qty = _min_open_lots;
458 _ctx->writeLog(fmtutil::format("Lots of {} changed from {} to {} due to minimum open lots", _code, this_qty, _min_open_lots));
459 }
460
461 double buyPx, sellPx;
462 if (_price_mode == 2)//价格类型,0-最新价,-1-最优价,1-对手价,2-自动,默认为0
463 {
464 double mp = (_last_tick->bidqty(0) - _last_tick->askqty(0))*1.0 / (_last_tick->bidqty(0) + _last_tick->askqty(0));
465 bool isUp = (mp > 0);
466 if (isUp)
467 {
468 buyPx = _last_tick->askprice(0);
469 sellPx = _last_tick->askprice(0);
470 }
471 else
472 {
473 buyPx = _last_tick->bidprice(0);
474 sellPx = _last_tick->bidprice(0);
475 }
476
477 /*
478 * By Wesley @ 2022.03.07
479 * 如果最后价格为0，再做一个修正 价格为0，可能当日没有交易，所以取上一个交易日的收盘价
480 */
481 if (decimal::eq(buyPx, 0.0)) //buypx==0，return ture
482 buyPx = decimal::eq(_last_tick->price(), 0.0) ? _last_tick->preclose() : _last_tick->price(); //如果上一tick ==0 取lasttick 收盘价
483
484 if (decimal::eq(sellPx, 0.0))
485 sellPx = decimal::eq(_last_tick->price(), 0.0) ? _last_tick->preclose() : _last_tick->price();
486
487 buyPx += _comm_info->getPriceTick() * _cancel_times; // // * 撤单次数 扩大追单步幅
488 sellPx -= _comm_info->getPriceTick() * _cancel_times;
489 }
490 else
491 {
492 if (_price_mode == -1)//0 最新价 -1 最优价 1 对手价
493 {
494 buyPx = _last_tick->bidprice(0);
495 sellPx = _last_tick->askprice(0);
496 }
497 else if (_price_mode == 0)
498 {
499 buyPx = _last_tick->price();
500 sellPx = _last_tick->price();
501 }
502 else if (_price_mode == 1)
503 {
504 buyPx = _last_tick->askprice(0);
505 sellPx = _last_tick->bidprice(0);
506 }
507
508 /*
509 * By Wesley @ 2022.03.07
510 * 如果最后价格为0，再做一个修正
511 */
512 if (decimal::eq(buyPx, 0.0))
513 buyPx = decimal::eq(_last_tick->price(), 0.0)? _last_tick->preclose(): _last_tick->price();
514
515 if (decimal::eq(sellPx, 0.0))
516 sellPx = decimal::eq(_last_tick->price(), 0.0) ? _last_tick->preclose() : _last_tick->price();
517
518 buyPx += _comm_info->getPriceTick() * _price_offset;//价格偏移跳数
519 sellPx -= _comm_info->getPriceTick() * _price_offset;
520 }
521
522
523 //检查涨跌停价
524 bool isCanCancel = true;
525 if (!decimal::eq(_last_tick->upperlimit(), 0) && decimal::gt(buyPx, _last_tick->upperlimit())) //upperlimit涨停价
526 {
527 _ctx->writeLog(fmtutil::format("Buy price {} of {} modified to upper limit price", buyPx, _code.c_str(), _last_tick->upperlimit()));//买入价，，，改为上限价
528 buyPx = _last_tick->upperlimit();
529 isCanCancel = false; //如果价格被修正为涨跌停价，订单不可撤销
530 }
531
532 if (!decimal::eq(_last_tick->lowerlimit(), 0) && decimal::lt(sellPx, _last_tick->lowerlimit()))
533 {
534 _ctx->writeLog(fmtutil::format("Sell price {} of {} modified to lower limit price", sellPx, _code.c_str(), _last_tick->lowerlimit()));
535 sellPx = _last_tick->lowerlimit();
536 isCanCancel = false; //如果价格被修正为涨跌停价，订单不可撤销
537 }
538
539 if (isBuy)
540 {
541 OrderIDs ids = _ctx->buy(stdCode, buyPx, this_qty, bForceClose);
542 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime(), isCanCancel);
543 }
544 else
545 {
546 OrderIDs ids = _ctx->sell(stdCode, sellPx, this_qty, bForceClose);
547 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime(), isCanCancel);
548 }
549
550 _last_place_time = now;
551}
