83void WtDtRunner::initialize(const char* cfgFile, bool isFile /* = true */, const char* modDir /* = "" */, const char* logCfg /* = "logcfg.yaml" */,
84 FuncOnTickCallback cbTick /* = NULL */, FuncOnBarCallback cbBar /* = NULL */)
85{
86 if(_is_inited)
87 {
88 WTSLogger::error("WtDtServo has already been initialized");
89 return;
90 }
91
92 _cb_tick = cbTick;
93 _cb_bar = cbBar;
94
95 WTSLogger::init(logCfg);
96 WtHelper::set_module_dir(modDir);
97
98 WTSVariant* config = isFile ? WTSCfgLoader::load_from_file(cfgFile) : WTSCfgLoader::load_from_content(cfgFile, false);
99 if(config == NULL)
100 {
101 WTSLogger::error("Loading config failed");
102 WTSLogger::log_raw(LL_INFO, cfgFile);
103 return;
104 }
105
106 if(!config->getBoolean("disable_dump"))
107 {
108#ifdef _MSC_VER
109 CMiniDumper::Enable(getModuleName(), true, WtHelper::get_cwd());
110#endif
111 }
112 //基础数据文件
113 WTSVariant* cfgBF = config->get("basefiles");
114 if (cfgBF->get("session"))
115 {
116 _bd_mgr.loadSessions(cfgBF->getCString("session"));
117 WTSLogger::info("Trading sessions loaded");
118 }
119
120 WTSVariant* cfgItem = cfgBF->get("commodity");
121 if (cfgItem)
122 {
123 if (cfgItem->type() == WTSVariant::VT_String)
124 {
125 _bd_mgr.loadCommodities(cfgItem->asCString());
126 }
127 else if (cfgItem->type() == WTSVariant::VT_Array)
128 {
129 for (uint32_t i = 0; i < cfgItem->size(); i++)
130 {
131 _bd_mgr.loadCommodities(cfgItem->get(i)->asCString());
132 }
133 }
134 }
135
136 cfgItem = cfgBF->get("contract");
137 if (cfgItem)
138 {
139 if (cfgItem->type() == WTSVariant::VT_String)
140 {
141 _bd_mgr.loadContracts(cfgItem->asCString());
142 }
143 else if (cfgItem->type() == WTSVariant::VT_Array)
144 {
145 for (uint32_t i = 0; i < cfgItem->size(); i++)
146 {
147 _bd_mgr.loadContracts(cfgItem->get(i)->asCString());
148 }
149 }
150 }
151
152 if (cfgBF->get("holiday"))
153 {
154 _bd_mgr.loadHolidays(cfgBF->getCString("holiday"));
155 WTSLogger::info("Holidays loaded");
156 }
157
158 if (cfgBF->get("hot"))
159 {
160 _hot_mgr.loadHots(cfgBF->getCString("hot"));
161 WTSLogger::info("Hot rules loaded");
162 }
163
164 if (cfgBF->get("second"))
165 {
166 _hot_mgr.loadSeconds(cfgBF->getCString("second"));
167 WTSLogger::info("Second rules loaded");
168 }
169
170 WTSArray* ayContracts = _bd_mgr.getContracts();
171 for (auto it = ayContracts->begin(); it != ayContracts->end(); it++)
172 {
173 WTSContractInfo* cInfo = (WTSContractInfo*)(*it);
174 bool isHot = _hot_mgr.isHot(cInfo->getExchg(), cInfo->getCode());
175 bool isSecond = _hot_mgr.isSecond(cInfo->getExchg(), cInfo->getCode());
176
177 std::string hotCode = cInfo->getFullPid();
178 if (isHot)
179 hotCode += ".HOT";
180 else if (isSecond)
181 hotCode += ".2ND";
182 else
183 hotCode = "";
184
185 cInfo->setHotFlag(isHot ? 1 : (isSecond ? 2 : 0), hotCode.c_str());
186 }
187 ayContracts->release();
188
189 initDataMgr(config->get("data"));
190
191 WTSVariant* cfgParser = config->get("parsers");
192 if (cfgParser)
193 {
194 if (cfgParser->type() == WTSVariant::VT_String)
195 {
196 const char* filename = cfgParser->asCString();
197 if (StdFile::exists(filename))
198 {
199 WTSLogger::info("Reading parser config from {}...", filename);
200 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
201 if (var)
202 {
203 initParsers(var->get("parsers"));
204 var->release();
205 }
206 else
207 {
208 WTSLogger::error("Loading parser config {} failed", filename);
209 }
210 }
211 else
212 {
213 WTSLogger::error("Parser configuration {} not exists", filename);
214 }
215 }
216 else if (cfgParser->type() == WTSVariant::VT_Array)
217 {
218 initParsers(cfgParser);
219 }
220 }
221 else
222 WTSLogger::log_raw(LL_WARN, "No parsers config, skipped loading parsers");
223
224 config->release();
225
226 start();
227
228 _is_inited = true;
229}

655void WtDtRunner::trigger_tick(const char* stdCode, WTSTickData* curTick)
656{
657 if (_cb_tick != NULL)
658 {
659 StdUniqueLock lock(_mtx_subs);
660 auto sit = _tick_sub_map.find(stdCode);
661 if (sit != _tick_sub_map.end())
662 {
663 SubFlags flags = sit->second;
664 for (uint32_t flag : flags)
665 {
666 if (flag == 0)
667 {
668 _cb_tick(stdCode, &curTick->getTickStruct());
669 }
670 else
671 {
672 std::string wCode = fmtutil::format("{}{}", stdCode, (flag == 1) ? SUFFIX_QFQ : SUFFIX_HFQ);
673 if (flag == 1)
674 {
675 _cb_tick(wCode.c_str(), &curTick->getTickStruct());
676 }
677 else //(flag == 2)
678 {
679 WTSTickData* newTick = WTSTickData::create(curTick->getTickStruct());
680 WTSTickStruct& newTS = newTick->getTickStruct();
681 newTick->setContractInfo(curTick->getContractInfo());
682
683 //这里做一个复权因子的处理
684 double factor = _data_mgr.get_exright_factor(stdCode, curTick->getContractInfo()->getCommInfo());
685 newTS.open *= factor;
686 newTS.high *= factor;
687 newTS.low *= factor;
688 newTS.price *= factor;
689
690 newTS.settle_price *= factor;
691
692 newTS.pre_close *= factor;
693 newTS.pre_settle *= factor;
694
695 _cb_tick(wCode.c_str(), &newTS);
696 newTick->release();
697 }
698 }
699 }
700
701 }
702 }
703
704 {
705 StdUniqueLock lock(_mtx_innersubs);
706 auto sit = _tick_innersub_map.find(stdCode);
707 if (sit == _tick_innersub_map.end())
708 return;
709
710 SubFlags flags = sit->second;
711 for (uint32_t flag : flags)
712 {
713 if (flag == 0)
714 {
715 _data_mgr.update_bars(stdCode, curTick);
716 }
717 else
718 {
719 std::string wCode = fmtutil::format("{}{}", stdCode, (flag == 1) ? SUFFIX_QFQ : SUFFIX_HFQ);
720 curTick->setCode(wCode.c_str());
721 if (flag == 1)
722 {
723 _data_mgr.update_bars(wCode.c_str(), curTick);
724 }
725 else //(flag == 2)
726 {
727 WTSTickData* newTick = WTSTickData::create(curTick->getTickStruct());
728 WTSTickStruct& newTS = newTick->getTickStruct();
729 newTick->setContractInfo(curTick->getContractInfo());
730
731 //这里做一个复权因子的处理
732 double factor = _data_mgr.get_exright_factor(stdCode, curTick->getContractInfo()->getCommInfo());
733 newTS.open *= factor;

764void WtDtRunner::sub_tick(const char* codes, bool bReplace, bool bInner /* = false */)
765{
766 if(bInner)
767 {
768 StdUniqueLock lock(_mtx_innersubs);
769 if (bReplace)
770 _tick_innersub_map.clear();
771
772 const char* stdCode = codes;
773 std::size_t length = strlen(stdCode);
774 uint32_t flag = 0;
775 if (stdCode[length - 1] == SUFFIX_QFQ || stdCode[length - 1] == SUFFIX_HFQ)
776 {
777 length--;
778
779 flag = (stdCode[length] == SUFFIX_QFQ) ? 1 : 2;
780 }
781
782 SubFlags& flags = _tick_innersub_map[std::string(stdCode, length)];
783 flags.insert(flag);
784 WTSLogger::info("Tick dada of {} subscribed with flag {} for inner use", stdCode, flag);
785 }
786 else
787 {
788 StdUniqueLock lock(_mtx_subs);
789 if (bReplace)
790 _tick_sub_map.clear();
791
792 StringVector ayCodes = StrUtil::split(codes, ",");
793 for (const std::string& code : ayCodes)
794 {
795 //如果是主力合约代码, 如SHFE.ag.HOT, 那么要转换成原合约代码, SHFE.ag.1912
796 //因为执行器只识别原合约代码
797 const char* stdCode = code.c_str();
798 std::size_t length = strlen(stdCode);
799 uint32_t flag = 0;
800 if (stdCode[length - 1] == SUFFIX_QFQ || stdCode[length - 1] == SUFFIX_HFQ)
801 {
802 length--;
803
804 flag = (stdCode[length] == SUFFIX_QFQ) ? 1 : 2;
805 }
806
807 SubFlags& flags = _tick_sub_map[std::string(stdCode, length)];
808 flags.insert(flag);
809 WTSLogger::info("Tick dada of {} subscribed with flag {}", stdCode, flag);
810 }
811 }
