21bool Dumper::config(const char* cfgfile, bool isFile, const char* modDir)
22{
23 WtHelper::set_module_dir(modDir);
24
25 WTSVariant* root = NULL;
26 if (isFile)
27 root = WTSCfgLoader::load_from_file(cfgfile);
28 else
29 root = WTSCfgLoader::load_from_content(cfgfile, false);
30
31 WTSVariant* cfg = root->get("config");
32 if(cfg)
33 {
34 if(cfg->has("refresh_span"))
35 {
36 _refresh_span = cfg->getUInt32("refresh_span");
37 }
38
39 }
40
41 //基础数据文件
42 WTSVariant* cfgBF = root->get("basefiles");
43 if (cfgBF->get("session"))
44 {
45 g_bdMgr.loadSessions(cfgBF->getCString("session"));
46 WTSLogger::info("Trading sessions loaded");
47 }
48
49 WTSVariant* cfgItem = cfgBF->get("commodity");
50 if (cfgItem)
51 {
52 if (cfgItem->type() == WTSVariant::VT_String)
53 {
54 g_bdMgr.loadCommodities(cfgItem->asCString());
55 }
56 else if (cfgItem->type() == WTSVariant::VT_Array)
57 {
58 for (uint32_t i = 0; i < cfgItem->size(); i++)
59 {
60 g_bdMgr.loadCommodities(cfgItem->get(i)->asCString());
61 }
62 }
63 }
64
65 cfgItem = cfgBF->get("contract");
66 if (cfgItem)
67 {
68 if (cfgItem->type() == WTSVariant::VT_String)
69 {
70 g_bdMgr.loadContracts(cfgItem->asCString());
71 }
72 else if (cfgItem->type() == WTSVariant::VT_Array)
73 {
74 for (uint32_t i = 0; i < cfgItem->size(); i++)
75 {
76 g_bdMgr.loadContracts(cfgItem->get(i)->asCString());
77 }
78 }
79 }
80
81 cfg = root->get("traders");
82 for (uint32_t idx = 0; idx < cfg->size(); idx++)
83 {
84 WTSVariant* cfgItem = cfg->get(idx);
85 if (!cfgItem->getBoolean("active"))
86 continue;
87
88 const char* channelid = cfgItem->getCString("channelid");
89
90 TraderAdapterPtr adapter(new TraderAdapter(&g_adapterMgr));
91 adapter->init(channelid, cfgItem, &g_bdMgr);
92 g_adapterMgr.addAdapter(channelid, adapter);
93 }
94
95 root->release();
96
97 WTSLogger::info("交易数据落地模块初始化完成，主动刷新间隔:{}s", _refresh_span);
98
99 return true;
