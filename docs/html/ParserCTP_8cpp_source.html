90inline uint32_t strToTime(const char* strTime)
91{
92 static char str[10] = { 0 };
93 const char *pos = strTime;
94 int idx = 0;
95 auto len = strlen(strTime);
96 for(std::size_t i = 0; i < len; i++)
97 {
98 if(strTime[i] != ':')
99 {
100 str[idx] = strTime[i];
101 idx++;
102 }
103 }
104 str[idx] = '\0';
105
106 return strtoul(str, NULL, 10);
107}

189#ifdef _WIN32

369void ParserCTP::OnRtnDepthMarketData( CThostFtdcDepthMarketDataField *pDepthMarketData )
370{
371 if(m_pBaseDataMgr == NULL)
372 {
373 return;
374 }
375
376 WTSContractInfo* contract = m_pBaseDataMgr->getContract(pDepthMarketData->InstrumentID, pDepthMarketData->ExchangeID);
377 if (contract == NULL)
378 return;
379
380 uint32_t actDate, actTime, actHour;
381
382 if(m_bLocaltime)
383 {
384 TimeUtils::getDateTime(actDate, actTime);
385 actHour = actTime / 10000000;
386 }
387 else
388 {
389 actDate = strtoul(pDepthMarketData->ActionDay, NULL, 10);
390 actTime = strToTime(pDepthMarketData->UpdateTime) * 1000 + pDepthMarketData->UpdateMillisec;
391 actHour = actTime / 10000000;
392
393 if (actDate == m_uTradingDate && actHour >= 20) {
394 //这样的时间是有问题,因为夜盘时发生日期不可能等于交易日
395 //这就需要手动设置一下
396 uint32_t curDate, curTime;
397 TimeUtils::getDateTime(curDate, curTime);
398 uint32_t curHour = curTime / 10000000;
399
400 //早上启动以后,会收到昨晚12点以前收盘的行情,这个时候可能会有发生日期=交易日的情况出现
401 //这笔数据直接丢掉
402 if (curHour >= 3 && curHour < 9)
403 return;
404
405 actDate = curDate;
406
407 if (actHour == 23 && curHour == 0) {
408 //行情时间慢于系统时间
409 actDate = TimeUtils::getNextDate(curDate, -1);
410 } else if (actHour == 0 && curHour == 23) {
411 //系统时间慢于行情时间
412 actDate = TimeUtils::getNextDate(curDate, 1);
413 }
414 }
415 }
416
417 WTSCommodityInfo* pCommInfo = contract->getCommInfo();
418
419 WTSTickData* tick = WTSTickData::create(pDepthMarketData->InstrumentID);
420 tick->setContractInfo(contract);
421
422 WTSTickStruct& quote = tick->getTickStruct();
423 strcpy(quote.exchg, pCommInfo->getExchg());
424
425 quote.action_date = actDate;
426 quote.action_time = actTime;
427
428 quote.price = checkValid(pDepthMarketData->LastPrice);
429 quote.open = checkValid(pDepthMarketData->OpenPrice);
430 quote.high = checkValid(pDepthMarketData->HighestPrice);
431 quote.low = checkValid(pDepthMarketData->LowestPrice);
432 quote.total_volume = pDepthMarketData->Volume;
433 quote.trading_date = m_uTradingDate;
434 if(pDepthMarketData->SettlementPrice != DBL_MAX)
435 quote.settle_price = checkValid(pDepthMarketData->SettlementPrice);
436 if(strcmp(quote.exchg, "CZCE") == 0)
437 {
438 quote.total_turnover = pDepthMarketData->Turnover*pCommInfo->getVolScale();
439 }
440 else
441 {
442 if(pDepthMarketData->Turnover != DBL_MAX)
443 quote.total_turnover = pDepthMarketData->Turnover;
444 }
445
446 quote.open_interest = pDepthMarketData->OpenInterest;
447
448 quote.upper_limit = checkValid(pDepthMarketData->UpperLimitPrice);
449 quote.lower_limit = checkValid(pDepthMarketData->LowerLimitPrice);
450
451 quote.pre_close = checkValid(pDepthMarketData->PreClosePrice);
452 quote.pre_settle = checkValid(pDepthMarketData->PreSettlementPrice);
453 quote.pre_interest = pDepthMarketData->PreOpenInterest;
454
455 //委卖价格
456 quote.ask_prices[0] = checkValid(pDepthMarketData->AskPrice1);
457 quote.ask_prices[1] = checkValid(pDepthMarketData->AskPrice2);
458 quote.ask_prices[2] = checkValid(pDepthMarketData->AskPrice3);
459 quote.ask_prices[3] = checkValid(pDepthMarketData->AskPrice4);
460 quote.ask_prices[4] = checkValid(pDepthMarketData->AskPrice5);
461
462 //委买价格
463 quote.bid_prices[0] = checkValid(pDepthMarketData->BidPrice1);
464 quote.bid_prices[1] = checkValid(pDepthMarketData->BidPrice2);
465 quote.bid_prices[2] = checkValid(pDepthMarketData->BidPrice3);
466 quote.bid_prices[3] = checkValid(pDepthMarketData->BidPrice4);
467 quote.bid_prices[4] = checkValid(pDepthMarketData->BidPrice5);
468
469 //委卖量
470 quote.ask_qty[0] = pDepthMarketData->AskVolume1;
471 quote.ask_qty[1] = pDepthMarketData->AskVolume2;
472 quote.ask_qty[2] = pDepthMarketData->AskVolume3;
473 quote.ask_qty[3] = pDepthMarketData->AskVolume4;
474 quote.ask_qty[4] = pDepthMarketData->AskVolume5;
475
476 //委买量
477 quote.bid_qty[0] = pDepthMarketData->BidVolume1;
478 quote.bid_qty[1] = pDepthMarketData->BidVolume2;
479 quote.bid_qty[2] = pDepthMarketData->BidVolume3;
480 quote.bid_qty[3] = pDepthMarketData->BidVolume4;
481 quote.bid_qty[4] = pDepthMarketData->BidVolume5;
482
483 if(m_sink)
484 m_sink->handleQuote(tick, 1);
485
486 tick->release();
487}

560void ParserCTP::DoSubscribeMD()
561{
562 CodeSet codeFilter = m_filterSubs;
563 if(codeFilter.empty())
564 {//如果订阅礼包只空的,则取出全部合约列表
565 return;
566 }
567
568 char ** subscribe = new char*[codeFilter.size()];
569 int nCount = 0;
570 for(auto& code : codeFilter)
571 {
572 std::size_t pos = code.find('.');
573 if (pos != std::string::npos)
574 subscribe[nCount++] = (char*)code.c_str() + pos + 1;
575 else
576 subscribe[nCount++] = (char*)code.c_str();
577 }
578
579 if(m_pUserAPI && nCount > 0)
580 {
581 int iResult = m_pUserAPI->SubscribeMarketData(subscribe, nCount);
582 if(iResult != 0)
583 {
584 if(m_sink)
585 write_log(m_sink, LL_ERROR, "[ParserCTP] Sending md subscribe request failed: {}", iResult);
586 }
587 else
588 {
589 if(m_sink)
590 write_log(m_sink, LL_INFO, "[ParserCTP] Market data of {} contracts subscribed totally", nCount);
591 }
592 }
593 codeFilter.clear();
594 delete[] subscribe;
595}
