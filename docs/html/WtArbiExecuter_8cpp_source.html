74bool WtArbiExecuter::init(WTSVariant* params)
75{
76 if (params == NULL)
77 return false;
78
79 _config = params;
80 _config->retain();
81
82 _scale = params->getDouble("scale");
83 _strict_sync = params->getBoolean("strict_sync");
84
85 uint32_t poolsize = params->getUInt32("poolsize");
86 if(poolsize > 0)
87 {
88 _pool.reset(new boost::threadpool::pool(poolsize));
89 }
90
91 /*
92 * By Wesley @ 2021.12.14
93 * 从配置文件中读取自动清理的策略
94 * active: 是否启用
95 * includes: 包含列表，格式如CFFEX.IF
96 * excludes: 排除列表，格式如CFFEX.IF
97 */
98 WTSVariant* cfgClear = params->get("clear");
99 if(cfgClear)
100 {
101 _auto_clear = cfgClear->getBoolean("active");
102 WTSVariant* cfgItem = cfgClear->get("includes");
103 if(cfgItem)
104 {
105 if (cfgItem->type() == WTSVariant::VT_String)
106 _clear_includes.insert(cfgItem->asCString());
107 else if (cfgItem->type() == WTSVariant::VT_Array)
108 {
109 for(uint32_t i = 0; i < cfgItem->size(); i++)
110 _clear_includes.insert(cfgItem->get(i)->asCString());
111 }
112 }
113
114 cfgItem = cfgClear->get("excludes");
115 if (cfgItem)
116 {
117 if (cfgItem->type() == WTSVariant::VT_String)
118 _clear_excludes.insert(cfgItem->asCString());
119 else if (cfgItem->type() == WTSVariant::VT_Array)
120 {
121 for (uint32_t i = 0; i < cfgItem->size(); i++)
122 _clear_excludes.insert(cfgItem->get(i)->asCString());
123 }
124 }
125 }
126
127 WTSVariant* cfgGroups = params->get("groups");
128 if (cfgGroups)
129 {
130 auto names = cfgGroups->memberNames();
131 for(const std::string& gpname : names)
132 {
133 CodeGroupPtr& gpInfo = _groups[gpname];
134 if (gpInfo == NULL)
135 {
136 gpInfo.reset(new CodeGroup);
137 wt_strcpy(gpInfo->_name, gpname.c_str(), gpname.size());
138 }
139
140 WTSVariant* cfgGrp = cfgGroups->get(gpname.c_str());
141 auto codes = cfgGrp->memberNames();
142 for(const std::string& code : codes)
143 {
144 gpInfo->_items[code] = cfgGrp->getDouble(code.c_str());
145 _code_to_groups[code] = gpInfo;
146 }
147 }
148 }
149
150 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Local executer inited, scale: {}, auto_clear: {}, strict_sync: {}, thread poolsize: {}, code_groups: {}",
151 _scale, _auto_clear, _strict_sync, poolsize, _groups.size());
152
153 return true;
154}

163ExecuteUnitPtr WtArbiExecuter::getUnit(const char* stdCode, bool bAutoCreate /* = true */)
164{
165 CodeHelper::CodeInfo codeInfo = CodeHelper::extractStdCode(stdCode, NULL);
166 std::string commID = codeInfo.stdCommID();
167
168 WTSVariant* policy = _config->get("policy");
169 std::string des = commID;
170 if (!policy->has(commID.c_str()))
171 des = "default";
172
173 SpinLock lock(_mtx_units);
174
175 auto it = _unit_map.find(stdCode);
176 if(it != _unit_map.end())
177 {
178 return it->second;
179 }
180
181 if (bAutoCreate)
182 {
183 WTSVariant* cfg = policy->get(des.c_str());
184
185 const char* name = cfg->getCString("name");
186 ExecuteUnitPtr unit = _factory->createExeUnit(name);
187 if (unit != NULL)
188 {
189 _unit_map[stdCode] = unit;
190 unit->self()->init(this, stdCode, cfg);
191
192 //如果通道已经就绪，则直接通知执行单元
193 if (_channel_ready)
194 unit->self()->on_channel_ready();
195 }
196 return unit;
197 }
198 else
199 {
200 return ExecuteUnitPtr();
201 }
202}

404void WtArbiExecuter::set_position(const wt_hashmap<std::string, double>& targets)
405{
406 /*
407 * 先要把目标头寸进行组合匹配
408 */
409 auto real_targets = targets;
410 for(auto& v : _groups)
411 {
412 const CodeGroupPtr& gpInfo = v.second;
413 bool bHit = false;
414 double gpQty = DBL_MAX;
415 for(auto& vi : gpInfo->_items)
416 {
417 double unit = vi.second;
418 auto it = real_targets.find(vi.first);
419 if (it == real_targets.end())
420 {
421 bHit = false;
422 break;
423 }
424 else
425 {
426 bHit = true;
427 //计算最小的组合单位数量
428 gpQty = std::min(gpQty, decimal::mod(it->second, unit));
429 }
430 }
431
432 if(bHit && decimal::gt(gpQty, 0))
433 {
434 real_targets[gpInfo->_name] = gpQty;
435 for (auto& vi : gpInfo->_items)
436 {
437 double unit = vi.second;
438 real_targets[vi.first] -= gpQty * unit;
439 }
440 }
441 }
442
443
444 for (auto it = targets.begin(); it != targets.end(); it++)
445 {
446 const char* stdCode = it->first.c_str();
447 double newVol = it->second;
448 ExecuteUnitPtr unit = getUnit(stdCode);
449 if (unit == NULL)
450 continue;
451
452 double oldVol = _target_pos[stdCode];
453 _target_pos[stdCode] = newVol;
454 // 账户的理论持仓要经过修正
455 double traderTarget = round(newVol * _scale);
456
457 if(!decimal::eq(oldVol, newVol))
458 {
459 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Target position of {} changed: {} -> {} : {} with scale{}", stdCode, oldVol, newVol, traderTarget, _scale);
460 }
461
462 if (_trader && !_trader->checkOrderLimits(stdCode))
463 {
464 WTSLogger::log_dyn("executer", _name.c_str(), LL_WARN, "{} is disabled due to entrust limit control ", stdCode);
465 continue;
466 }
467
468 if (_pool)
469 {
470 std::string code = stdCode;
471 _pool->schedule([unit, code, traderTarget](){
472 unit->self()->set_position(code.c_str(), traderTarget);
473 });
474 }
475 else
476 {
477 unit->self()->set_position(stdCode, traderTarget);
478 }
479 }
480
481 //在原来的目标头寸中，但是不在新的目标头寸中，则需要自动设置为0
482 for (auto it = _target_pos.begin(); it != _target_pos.end(); it++)
483 {
484 const char* code = it->first.c_str();
485 double& pos = (double&)it->second;
486 auto tit = targets.find(code);
487 if(tit != targets.end())
488 continue;
489
490 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "{} is not in target, set to 0 automatically", code);
491
492 ExecuteUnitPtr unit = getUnit(code);
493 if (unit == NULL)
494 continue;
495
496 //unit->self()->set_position(code, 0);
497 if (_pool)
498 {
499 _pool->schedule([unit, code](){
500 unit->self()->set_position(code, 0);
501 });
502 }
503 else
504 {
505 unit->self()->set_position(code, 0);
506 }
507
508 pos = 0;
509 }
510
511 //如果开启了严格同步，则需要检查通道持仓
512 //如果通道持仓不在管理中，则直接平掉
513 if(_strict_sync)
514 {
515 for(const std::string& stdCode : _channel_holds)
516 {
517 auto it = _target_pos.find(stdCode.c_str());
518 if(it != _target_pos.end())
519 continue;
520
521 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "{} is not in management, set to 0 due to strict sync mode", stdCode.c_str());
522
523 ExecuteUnitPtr unit = getUnit(stdCode.c_str());
524 if (unit == NULL)
525 continue;
526
527 if (_pool)
528 {
529 std::string code = stdCode.c_str();
530 _pool->schedule([unit, code]() {
531 unit->self()->set_position(code.c_str(), 0);
532 });
533 }
534 else
535 {
536 unit->self()->set_position(stdCode.c_str(), 0);
537 }
538 }
539 }

821void WtArbiExecuter::on_position(const char* stdCode, bool isLong, double prevol, double preavail, double newvol, double newavail, uint32_t tradingday)
822{
823 _channel_holds.insert(stdCode);
824
825 /*
826 * By Wesley @ 2021.12.14
827 * 先检查自动清理过期主力合约的标记是否为true
828 * 如果不为true，则直接退出该逻辑
829 */
830 if (!_auto_clear)
831 return;
832
833 //如果不是分月期货合约，直接退出
834 if (!CodeHelper::isStdMonthlyFutCode(stdCode))
835 return;
836
837 IHotMgr* hotMgr = _stub->get_hot_mon();
838 CodeHelper::CodeInfo cInfo = CodeHelper::extractStdCode(stdCode, NULL);
839 //获取上一期的主力合约
840 std::string prevCode = hotMgr->getPrevRawCode(cInfo._exchg, cInfo._product, tradingday);
841
842 //如果当前合约不是上一期的主力合约，则直接退出
843 if (prevCode != cInfo._code)
844 return;
845
846 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Prev hot contract of {}.{} on {} is {}", cInfo._exchg, cInfo._product, tradingday, prevCode);
847
848 thread_local static char fullPid[64] = { 0 };
849 fmtutil::format_to(fullPid, "{}.{}", cInfo._exchg, cInfo._product);
850
851 //先检查排除列表
852 //如果在排除列表中，则直接退出
853 auto it = _clear_excludes.find(fullPid);
854 if(it != _clear_excludes.end())
855 {
856 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Position of {}, as prev hot contract, won't be cleared for it's in exclude list", stdCode);
857 return;
858 }
859
860 //如果包含列表不为空，再检查是否在包含列表中
861 //如果为空，则全部清理，不再进入该逻辑
862 if(!_clear_includes.empty())
863 {
864 it = _clear_includes.find(fullPid);
865 if (it == _clear_includes.end())
866 {
867 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Position of {}, as prev hot contract, won't be cleared for it's not in include list", stdCode);
868 return;
869 }
870 }
871
872 //最后再进行自动清理
873 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Position of {}, as prev hot contract, will be cleared", stdCode);
874 ExecuteUnitPtr unit = getUnit(stdCode);
875 if (unit)
876 {
877 if (_pool)
878 {
879 std::string code = stdCode;
880 _pool->schedule([unit, code](){
881 unit->self()->clear_all_position(code.c_str());
882 });
883 }
884 else
885 {
886 unit->self()->clear_all_position(stdCode);
887 }
888 }
