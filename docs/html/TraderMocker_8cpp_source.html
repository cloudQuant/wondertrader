52std::vector<uint32_t> splitVolume(uint32_t vol, uint32_t minQty = 1, uint32_t maxQty = 100)
53{
54 uint32_t length = maxQty - minQty + 1;
55 std::vector<uint32_t> ret;
56 if (vol <= minQty)
57 {
58 ret.emplace_back(vol);
59 }
60 else
61 {
62 uint32_t left = vol;
63 srand((uint32_t)time(NULL));
64 while (left > 0)
65 {
66 uint32_t curVol = minQty + (uint32_t)rand() % length;
67
68 if (curVol >= left)
69 curVol = left;
70
71 if (curVol == 0)
72 continue;
73
74 ret.emplace_back(curVol);
75 left -= curVol;
76 }
77 }
78
79 return ret;
80}

138int TraderMocker::orderInsert(WTSEntrust* entrust)
139{
140 if (entrust == NULL)
141 {
142 return 0;
143 }
144
145 entrust->retain();
146 _io_service.post([this, entrust](){
147 StdUniqueLock lock(_mutex_api);
148
149 WTSContractInfo* ct = entrust->getContractInfo();
150 if(ct == NULL)
151 ct = _bd_mgr->getContract(entrust->getCode(), entrust->getExchg());
152
153 /*
154 * 1、开仓无需检查
155 * 2、平仓要先检查可平
156 * 3、检查通过了,平仓还要冻结持仓
157 * 4、还要考虑国际期货的问题
158 */
159
160 bool bPass = false;
161 std::string msg;
162 do
163 {
164 if (ct == NULL)
165 {
166 bPass = false;
167 msg = "品种不存在";
168 break;
169 }
170 WTSCommodityInfo* commInfo = ct->getCommInfo();
171
172 //检查价格类型的合法性
173 if (entrust->getPriceType() == WPT_ANYPRICE && commInfo->getPriceMode() == PM_Limit)
174 {
175 bPass = false;
176 msg = "价格类型不合法";
177 break;
178 }
179
180 //检查数量的合法性
181 if ((commInfo->getCategoty() == CC_Stock) && (entrust->getOffsetType() == WOT_OPEN) && !decimal::eq(decimal::mod(entrust->getVolume(), 100), 0))
182 {
183 bPass = false;
184 msg = "股票买入数量必须为100的整数倍";
185 break;
186 }
187
188 //检查方向的合法性
189 if(!commInfo->canShort() && entrust->getDirection() == WDT_SHORT)
190 {
191 bPass = false;
192 msg = "股票不能做空";
193 break;
194 }
195
196 //检查价格的合法性
197 if(!decimal::eq(entrust->getPrice(), 0))
198 {
199 double pricetick = commInfo->getPriceTick();
200 double v = entrust->getPrice() / pricetick;
201
202 if (!decimal::eq(decimal::mod(entrust->getPrice(), pricetick), 0)) //整除的检查方式,先小数相除得到商,然后商取整以后,再跟原来的商相减,如果等于0,则是整除,否则是
203 {
204 bPass = false;
205 msg = "委托价格不合法";
206 break;
207 }
208 }
209
210
211 //开仓直接通过,不检查资金
212 if (entrust->getOffsetType() == WOT_OPEN)
213 {
214 bPass = true;
215 break;
216 }
217
218 //如果不需要开平,则直接通过,主要针对国际期货
219 if (commInfo->getCoverMode() == CM_None)
220 {
221 bPass = true;
222 break;
223 }
224
225 //如果区分平昨平今,而委托的是平昨,则直接拒绝,因为mocker为了简化处理,不考虑昨仓
226 if (commInfo->getCoverMode() == CM_CoverToday && (entrust->getOffsetType() == WOT_CLOSE || entrust->getOffsetType() == WOT_CLOSEYESTERDAY))
227 {
228 bPass = false;
229 msg = "没有足够的可平仓位";
230 break;
231 }
232
233 //如果没有持仓或者持仓不够,也要
234 auto it = _positions.find(ct->getFullCode());
235 if(it == _positions.end())
236 {
237 bPass = false;
238 msg = "没有足够的可平仓位";
239 break;
240 }
241
242 PosItem& pItem = (PosItem&)it->second;
243 bool isLong = entrust->getDirection() == WDT_LONG;
244
245 double validQty = isLong ? (pItem._long._volume - pItem._long._frozen) : (pItem._short._volume - pItem._short._frozen);
246 if(decimal::lt(validQty, entrust->getVolume()))
247 {
248 bPass = false;
249 msg = "没有足够的可平仓位";
250 break;
251 }
252
253 //冻结持仓
254 if(isLong)
255 {
256 pItem._long._frozen += entrust->getVolume();
257 }
258 else
259 {
260 pItem._short._frozen += entrust->getVolume();
261 }
262
263 bPass = true;
264 msg = "下单成功";
265
266 } while (false);
267
268 if(bPass)
269 {
270 WTSOrderInfo* ordInfo = WTSOrderInfo::create();
271 ordInfo->setContractInfo(ct);
272 ordInfo->setCode(entrust->getCode());
273 ordInfo->setExchange(entrust->getExchg());
274 ordInfo->setDirection(entrust->getDirection());
275 ordInfo->setOffsetType(entrust->getOffsetType());
276 ordInfo->setUserTag(entrust->getUserTag());
277 ordInfo->setPrice(entrust->getPrice());
278 thread_local static char str[64];
279 fmtutil::format_to(str, "mo.{}.{}", _mocker_id, makeOrderID());
280 ordInfo->setOrderID(str);
281 ordInfo->setStateMsg(msg.c_str());
282 ordInfo->setOrderState(WOS_NotTraded_Queuing);
283 ordInfo->setOrderTime(TimeUtils::getLocalTimeNow());
284 ordInfo->setVolume(entrust->getVolume());
285 ordInfo->setVolLeft(entrust->getVolume());
286 ordInfo->setPriceType(entrust->getPriceType());
287 ordInfo->setOrderFlag(entrust->getOrderFlag());
288
289 if (_listener != NULL)
290 {
291 _listener->onRspEntrust(entrust, NULL);
292 _listener->onPushOrder(ordInfo);
293 }
294
295 _codes.insert(ct->getFullCode());
296
297 if(_listener)
298 {
299 write_log(_listener,LL_INFO, "共有{}个品种有待撮合订单", _codes.size());
300 }
301
302 if (_orders == NULL)
303 _orders = WTSArray::create();
304 _orders->append(ordInfo, false);
305
306 if (_awaits == NULL)
307 _awaits = OrderCache::create();
308
309 _awaits->add(ordInfo->getOrderID(), ordInfo, true);
310
311 save_positions();
312 }
313 else
314 {
315 WTSError* err = WTSError::create(WEC_ORDERINSERT, msg.c_str());
316 if (_listener != NULL)
317 {
318 _listener->onRspEntrust(entrust, err);
319 }
320 err->release();
321 }
322 entrust->release();
323 });
324
325 return 0;
326}

328int32_t TraderMocker::match_once()
329{
330 if (_terminated || _orders == NULL || _orders->size() == 0 || _ticks == NULL)
331 return 0;
332
333 int32_t count = 0;
334 for (const std::string& fullcode : _codes)
335 {
336 if (_terminated)
337 return count;
338
339 std::string code, exchg;
340 auto pos = fullcode.find(".");
341 exchg = fullcode.substr(0, pos);
342 code = fullcode.substr(pos + 1);
343
344 WTSContractInfo* ct = _bd_mgr->getContract(code.c_str(), exchg.c_str());
345 if (ct == NULL)
346 continue;
347
348 WTSCommodityInfo* commInfo = ct->getCommInfo();
349
350 WTSTickData* curTick = (WTSTickData*)_ticks->grab(fullcode);
351 if (curTick && strcmp(curTick->code(), ct->getCode())==0)
352 {
353 StdUniqueLock lock(_mtx_awaits);
354 uint64_t tickTime = (uint64_t)curTick->actiondate() * 1000000000 + curTick->actiontime();
355 if (decimal::gt(curTick->price(), 0) /*&& tickTime >= _last_match_time*/)
356 {
357 //开始处理订单
358 //处理记录
359 std::vector<std::string> to_erase;
360
361 for (auto it = _awaits->begin(); it != _awaits->end(); it++)
362 {
363 WTSOrderInfo* ordInfo = (WTSOrderInfo*)it->second;
364 if (ordInfo->getVolLeft() == 0 || strcmp(ordInfo->getCode(), curTick->code()) != 0)
365 continue;
366
367 bool isBuy = (ordInfo->getDirection() == WDT_LONG && ordInfo->getOffsetType() == WOT_OPEN) || (ordInfo->getDirection() != WDT_LONG && ordInfo->getOffsetType() != WOT_OPEN);
368
369 double uPrice, uVolume;
370 if (isBuy)
371 {
372 uPrice = curTick->askprice(0);
373 uVolume = curTick->askqty(0);
374 }
375 else
376 {
377 uPrice = curTick->bidprice(0);
378 uVolume = curTick->bidqty(0);
379 }
380
381 if (decimal::eq(uVolume, 0))
382 continue;
383
384 if (_use_newpx)
385 {
386 uPrice = curTick->price();
387 }
388
389 if (decimal::eq(uPrice, 0))
390 continue;
391
392 double target = ordInfo->getPrice();
393 //买入的时候,委托价格小于最新价则不成交,卖出的时候,委托价大于最新价则不成交
394 if (ordInfo->getPriceType() == WPT_LIMITPRICE && ((isBuy && decimal::lt(target, uPrice)) || (!isBuy && decimal::gt(target, uPrice))))
395 continue;
396
397 count++;
398
399 double maxVolume = min(uVolume, ordInfo->getVolLeft());
400 std::vector<uint32_t> ayVol = splitVolume((uint32_t)maxVolume, (uint32_t)_min_qty, (uint32_t)_max_qty);
401 for (uint32_t curVol : ayVol)
402 {
403
404 WTSTradeInfo* trade = WTSTradeInfo::create(curTick->code(), curTick->exchg());
405 trade->setDirection(ordInfo->getDirection());
406 trade->setOffsetType(ordInfo->getOffsetType());
407 trade->setContractInfo(ct);
408
409 trade->setPrice(uPrice);
410 trade->setVolume(curVol);
411
412 trade->setRefOrder(ordInfo->getOrderID());
413
414 char str[64];
415 fmtutil::format_to(str, "mt.{}.{}", _mocker_id, makeTradeID());
416 trade->setTradeID(str);
417
418 trade->setTradeTime(TimeUtils::getLocalTimeNow());
419 trade->setUserTag(ordInfo->getUserTag());
420
421 //更新订单数据
422 ordInfo->setVolLeft(ordInfo->getVolLeft() - curVol);
423 ordInfo->setVolTraded(ordInfo->getVolTraded() - curVol);
424 if (decimal::eq(ordInfo->getVolLeft(), 0))
425 {
426 ordInfo->setOrderState(WOS_AllTraded);
427 ordInfo->setStateMsg("AllTrd");
428 to_erase.emplace_back(ordInfo->getOrderID());
429 }
430 else
431 {
432 ordInfo->setOrderState(WOS_PartTraded_Queuing);
433 ordInfo->setStateMsg("PartTrd");
434 }
435
436 PosItem& pItem = _positions[ct->getFullCode()];
437 //第一次的话要给代码和交易所赋值
438 if(strlen(pItem._code) == 0)
439 {
440 strcpy(pItem._code, ct->getCode());
441 strcpy(pItem._exchg, ct->getExchg());
442 }
443
444 if(commInfo->getCoverMode() == CM_None)
445 {
446
447 }
448 else
449 {
450 if (ordInfo->getDirection() == WDT_LONG)
451 {
452 if (ordInfo->getOffsetType() == WOT_OPEN)
453 {
454 pItem._long._volume += curVol;
455 }
456 else
457 {
458 pItem._long._volume -= curVol;
459 pItem._long._frozen -= curVol;
460 }
461 }
462 else
463 {
464 if (ordInfo->getOffsetType() == WOT_OPEN)
465 {
466 pItem._short._volume += curVol;
467 }
468 else
469 {
470 pItem._short._volume -= curVol;
471 pItem._short._frozen -= curVol;
472 }
473 }
474 }
475
476
477 if (_listener)
478 {
479 StdUniqueLock lock(_mutex_api);
480 _listener->onPushOrder(ordInfo);
481 _listener->onPushTrade(trade);
482 }
483
484 if (_trades == NULL)
485 _trades = WTSArray::create();
486
487 _trades->append(trade, false);
488 }
489 }
490
491 if (count > 0)
492 {
493 //write_log(_listener,LL_INFO, "[TraderMocker]触发 %s.%s 开多 %u 条,价格:%u", tick->exchg(), tick->code(), iCount, uPrice);
494 for (const std::string& oid : to_erase)
495 {
496 _awaits->remove(oid);
497 }
498 }
499 }
500 }
501
502 if(curTick)
503 curTick->release();
504 }
505
506 _last_match_time = _max_tick_time;
507
508 _ticks->clear();
509
510 if(count > 0)
511 save_positions();
512
513 return count;
514}

750int TraderMocker::orderAction(WTSEntrustAction* action)
751{
752 action->retain();
753
754 _io_service.post([this, action](){
755 StdUniqueLock lck(_mtx_awaits); //一定要把awaits锁起来,不然可能会导致一边撮合一边撤单
756 WTSOrderInfo* ordInfo = (WTSOrderInfo*)_awaits->grab(action->getOrderID());
757
758 /*
759 * 撤单也要考虑几个问题
760 * 1、是否处于可以撤销的状态
761 * 2、如果是开仓,则直接撤销
762 * 3、如果是平仓,要释放冻结
763 */
764 if(ordInfo == NULL)
765 {
766 write_log(_listener,LL_ERROR, "订单{}不存在或者已完成", action->getOrderID());
767 WTSError* err = WTSError::create(WEC_ORDERCANCEL, "订单不存在或者处于不可撤销状态");
768 if (_listener)
769 _listener->onTraderError(err);
770 err->release();
771 return;
772 }
773
774 WTSContractInfo* ct = ordInfo->getContractInfo();
775 WTSCommodityInfo* commInfo = ct->getCommInfo();
776
777 bool bPass = false;
778 do
779 {
780 //开仓委托直接撤单
