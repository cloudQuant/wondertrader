231static __inline char*
232SStr_Strsep2(char **ppString, char **ppTokenEnd, const char *pDelim) {
233 char *pBegin = (char *) NULL;
234 char *pEnd = (char *) NULL;
235
236 SLOG_ASSERT(ppString && pDelim && *pDelim);
237
238 pBegin = *ppString;
239 if (__spk_unlikely(! pBegin || ! *pBegin)) {
240 *ppString = NULL;
241 if (ppTokenEnd) {
242 *ppTokenEnd = NULL;
243 }
244 return (char *) pBegin;
245 }
246
247 if (pDelim[1] == '\0') {
248 if (*pBegin == pDelim[0]) {
249 pEnd = pBegin;
250 } else {
251 pEnd = strchr(pBegin + 1, pDelim[0]);
252 }
253 } else {
254 pEnd = SStr_Strpbrk(pBegin, pDelim);
255 }
256
257 if (pEnd) {
258 *ppString = pEnd + 1;
259
260 if (ppTokenEnd) {
261 *ppTokenEnd = pEnd;
262 } else {
263 *pEnd = '\0';
264 }
265 } else {
266 *ppString = NULL;
267
268 if (ppTokenEnd) {
269 *ppTokenEnd = NULL;
270 }
271 }
272
273 return pBegin;
274}

415static __inline int32
416SStr_CheckPasswdStrength(const char *pStr, int32 strength, int32 minLength) {
417 static const int32 _STRONGEST_PASSWD_STRENGTH = 4;
418 int32 uppercaseCnt = 0;
419 int32 lowercaseCnt = 0;
420 int32 digitCnt = 0;
421 int32 symbolCnt = 0;
422 int32 totalLength = 0;
423 int32 strengthLevel = 0;
424
425 if (__spk_unlikely(SStr_IsEmpty(pStr))) {
426 return SPK_NEG(EINVAL);
427 }
428
429 while (*pStr) {
430 if (SPK_ISUPPER(*pStr)) {
431 uppercaseCnt++;
432 } else if (SPK_ISLOWER(*pStr)) {
433 lowercaseCnt++;
434 } else if (SPK_ISDIGIT(*pStr)) {
435 digitCnt++;
436 } else if (SStr_IsAnyChar(*pStr, "@._-")) {
437 symbolCnt++;
438 } else {
439 return SPK_NEG(EINVAL);
440 }
441 pStr++;
442 }
443
444 totalLength = uppercaseCnt + lowercaseCnt + digitCnt + symbolCnt;
445 strengthLevel = (uppercaseCnt > 0 ? 1 : 0)
446 + (lowercaseCnt > 0 ? 1 : 0)
447 + (digitCnt > 0 ? 1 : 0)
448 + (symbolCnt > 0 ? 1 : 0);
449 if (__spk_unlikely((minLength > 0 && totalLength < minLength)
450 || strengthLevel < SPK_MIN(strength, _STRONGEST_PASSWD_STRENGTH))) {
451 return EINVAL;
452 }
453
454 return 0;
455}

494 char ch = 0;
495 char prev = 0;

498 return (const char *) NULL;

938static __inline char*
939SStr_StrRightAlignmentCopy(char *pBuf, const char *pStr, int32 alignedLen,
940 char leftFiller) {
941 const char *pPtr = (const char *) NULL;
942 int32 len = alignedLen;
943
944 if (__spk_unlikely(alignedLen <= 0)) {
945 return SStr_StrTrimCopy(pBuf, pStr, alignedLen);
946 }
947
948 SLOG_ASSERT(pBuf);
949 if (__spk_unlikely(! pStr)) {
950 *pBuf = '\0';
951 return pBuf;
952 }
953
954 while (SPK_ISSPACE(*pStr)) {
955 pStr++;
956 }
957
958 pPtr = pStr;
959 while (*pPtr && len != 0) {
960 pPtr++;
961 len--;
962 }
963
964 do {
965 pPtr--;
966 len++;
967 } while (len <= alignedLen && SPK_ISSPACE(*pPtr));
968 len--;
969
970 memset(pBuf, leftFiller, len);
971 memcpy(pBuf + len, pStr, alignedLen - len);
972
973 pBuf[alignedLen] = '\0';
974 return pBuf;
975}

1110static __inline int32
1111SStr_ReplaceCharReverse(char *pStr, const char c1, const char c2, int32 n) {
1112 int32 endPos = 0;
1113 int32 count = 0;
1114
1115 SLOG_ASSERT(pStr);
1116
1117 endPos = strlen(pStr) - 1;
1118 while (endPos >= 0 && n) {
