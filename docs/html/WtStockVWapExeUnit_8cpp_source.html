98inline double calTmStamp(uint32_t actiontime) //计算tick时间属于哪个时间单元
99{
100 string timestamp = to_string(actiontime);
101 int hour = stoi(timestamp.substr(0, 2));
102 int minute = stoi(timestamp.substr(2, 2));
103 double total_minute = 0;
104
105 // 开盘前
106 if (hour < 9 || (hour == 9 && minute < 30)) {
107 total_minute = 0;
108 }
109 // 上午交易时段：9:30-11:30
110 else if (hour < 11 || (hour == 11 && minute <= 30)) {
111 total_minute = (hour - 9) * 60 + minute - 30;
112 }
113 // 中午休市
114 else if (hour < 13 || (hour == 13 && minute < 30)) {
115 total_minute = 120 + (hour - 11) * 60 + minute;
116 }
117 // 下午交易时段：13:00-15:00
118 else if (hour < 15 || (hour == 15 && minute <= 0)) {
119 total_minute = 240 + (hour - 13) * 60 + minute - 30;
120 }
121 // 收盘后
122 else {
123 total_minute = 240;
124 }
125
126 // 中午休市时间特殊处理
127 if (timestamp >= "113000000" && timestamp < "130000000") {
128 total_minute = 120;
129 }
130
131 // 添加秒和毫秒的分数
132 total_minute += stoi(timestamp.substr(4, 2)) / 60;
133 total_minute += stoi(timestamp.substr(6, 3)) / 60000;
134
135 return total_minute; // 这里应该+1，对应vector 所以再-1

162void WtStockVWapExeUnit::init(ExecuteContext * ctx, const char * stdCode, WTSVariant * cfg)
163{

165 ExecuteUnit::init(ctx, stdCode, cfg);
166

172 // 获取交易时段信息并保持引用
173 _sess_info = ctx->getSessionInfo(stdCode);//获取交易时间模板信息
174 if (_sess_info)
175 _sess_info->retain();
176
177 // 从配置中读取执行参数
178 _begin_time = cfg->getUInt32("begin_time"); // 开始时间
179 _end_time = cfg->getUInt32("end_time"); // 结束时间

258void WtStockVWapExeUnit::on_order(uint32_t localid, const char * stdCode, bool isBuy, double leftover, double price, bool isCanceled)
259{
260 // 检查订单是否在监控器中
261 if (!_orders_mon.has_order(localid))
262 return;
263
264 // 处理订单完成或撤销的情况
265 if (isCanceled || leftover == 0)
266 {
267 // 从监控器中移除订单
268 _orders_mon.erase_order(localid);
269 // 如果有撤单计数，减少并记录日志
270 if (_cancel_cnt > 0)
271 {
272 _cancel_cnt--;
273 _ctx->writeLog(fmtutil::format("[{}@{}] Order of {} cancelling done, cancelcnt -> {}", __FILE__, __LINE__, _code.c_str(), _cancel_cnt));
274 }
275 }
276
277 // 如果订单全部成交（非撤销），重置撤单次数计数
278 if (leftover == 0 && !isCanceled) {
279 _cancel_times = 0;
280 _ctx->writeLog(fmtutil::format("Order {} has filled", localid));
281 }
282
283 // 如果全部订单已撤销，这个时候一般是遇到要超时撤单（挂单超时）
284 if (isCanceled && _cancel_cnt == 0)
285 {
286 // 获取当前实际持仓
287 double realPos = _ctx->getPosition(stdCode);
288 // 如果实际持仓与目标不一致，需要重新发单
289 if (!decimal::eq(realPos, _this_target))
290 {
291 _ctx->writeLog(fmtutil::format("Order {} of {} canceled, re_fire will be done", localid, stdCode));
292 _cancel_times++;
293 // 撤单以后重发，一般是加点重发；对最小下单量的校验
294 fire_at_once(max(_min_open_lots, _this_target - realPos));
295 }
296 }
297
298 // 处理异常情况：订单未撤销但撤单计数不为0
299 if (!isCanceled && _cancel_cnt != 0) { // 一般出现问题，需要返回检查 触发撤单 cnt++,onorder响应处理才会--
300 _ctx->writeLog(fmtutil::format("Order {} of {} hasn't canceled, error will be return ", localid, stdCode));
301 return;
302 }

313void WtStockVWapExeUnit::on_channel_ready()
314{
315 // 标记交易通道为就绪状态
316 _channel_ready = true;
317
318 // 获取未完成的订单数量
319 double undone = _ctx->getUndoneQty(_code.c_str());
320
321 // 情况1: 如果有未完成订单但本地没有订单记录，需要清除未被管理的订单
322 if (!decimal::eq(undone, 0) && !_orders_mon.has_order())
323 {
324 /*
325 * 如果未完成单不为0，而OMS没有订单
326 * 这说明有未完成单不在监控之中,全部撤销掉
327 * 因为这些订单没有本地订单号，无法直接进行管理
328 * 这种情况，就是刚启动的时候，上次的未完成单或者外部的挂单
329 */
330 _ctx->writeLog(fmt::format("{} unmanaged orders of {},cancel all", undone, _code).c_str());
331
332 // 根据未完成订单的正负判断是买入还是卖出订单
333 bool isBuy = (undone > 0);
334
335 // 撤销对应方向的所有未完成订单
336 OrderIDs ids = _ctx->cancel(_code.c_str(), isBuy);
337
338 // 将撤销的订单添加到监控器中进行跟踪
339 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime());
340
341 // 更新撤单计数
342 _cancel_cnt += ids.size();
343
344 // 记录撤单日志
345 _ctx->writeLog(fmtutil::format("[{}@{}]cancelcnt -> {}", __FILE__, __LINE__, _cancel_cnt));
346 }
347 // 情况2: 如果没有未完成订单但本地有订单记录，需要清除本地错误订单
348 else if (decimal::eq(undone, 0) && _orders_mon.has_order())
349 { /*
350 * By Wesey @ 2021.12.13
351 * 如果未完成单为0，但是OMS中是有订单的
352 * 说明OMS中是错单，需要清理掉，不然超时撤单就会出错
353 * 这种情况，一般是断线重连以后，之前下出去的订单，并没有真正发送到柜台
354 * 所以这里需要清理掉本地订单
355 */
356 // 记录日志并清除所有本地订单
357 _ctx->writeLog(fmtutil::format("Local orders of {} not confirmed in trading channel, clear all", _code.c_str()));
358 _orders_mon.clear_orders();
359 }
360 // 情况3: 其他异常情况处理
361 else
362 {
363 // 记录日志，显示当前未完成订单和本地监控状态
364 _ctx->writeLog(fmtutil::format("Unrecognized condition while channle ready, {:.2f} live orders of {} exists, local orders {}exist",
365 undone, _code.c_str(), _orders_mon.has_order() ? "" : "not "));
366 }
367
368 // 通道就绪后触发一次计算，重新调整仓位
369 do_calc();
370}

378void WtStockVWapExeUnit::on_tick(WTSTickData * newTick)
379{
380 // 检查行情数据是否有效且属于当前合约
381 if (newTick == NULL || _code.compare(newTick->code()) != 0)
382 return;
383
384 // 标记是否为第一笔行情
385 bool isFirstTick = false;
386
387 // 释放旧的行情数据
388 if (_last_tick) {
389 _last_tick->release();
390 }
391 else {
392 // 如果之前没有行情数据，这是第一笔行情
393 isFirstTick = true;
394 // 如果行情时间不在交易时间内（如集合竞价时段），则过滤掉该行情
395 if (_sess_info != NULL && !_sess_info->isInTradingTime(newTick->actiontime() / 100000))
396 return;
397 }
398 // 保存新的行情数据并增加引用计数
399 _last_tick = newTick;
400 _last_tick->retain();
401
402 // 如果是第一笔行情，需要检查目标仓位是否需要调整
403 if (isFirstTick)
404 {
405 // 获取当前目标仓位、未完成订单和实际仓位
406 double newVol = _target_pos;
407 const char* stdCode = _code.c_str();
408 double undone = _ctx->getUndoneQty(stdCode);
409 double realPos = _ctx->getPosition(stdCode);
410
411 // 如果目标仓位不等于未完成订单加实际仓位，需要调整
412 if (!decimal::eq(newVol, undone + realPos))
413 {
414 // 触发计算逻辑进行仓位调整
415 do_calc();
416 }
417 }
418 else // 非第一笔行情处理
419 {
420 // 获取当前时间
421 uint64_t now = TimeUtils::getLocalTimeNow();
422 bool hasCancel = false;
423
424 // 检查是否有超时订单需要撤销
425 if (_ord_sticky != 0 && _orders_mon.has_order())
426 {
427 // 检查订单是否超时，如果超时则撤销
428 _orders_mon.check_orders(_ord_sticky, now, [this, &hasCancel](uint32_t localid) {
429 if (_ctx->cancel(localid))
430 {
431 _cancel_cnt++;
432 _ctx->writeLog(fmt::format("Order expired, cancelcnt updated to {}", _cancel_cnt).c_str());
433 hasCancel = true;
434 }
435 });
436 }
437
438 // 如果没有撤单操作，且距离上次发单时间超过了发单间隔，则触发新的计算
439 if (!hasCancel && (now - _last_fire_time >= _fire_span * 1000))
440 {
441 do_calc();
442 }
443 }
444}

466void WtStockVWapExeUnit::do_calc()
467{
468 // 防止重入计算
469 CalcFlag flag(&_in_calc);
470 if (flag)
471 return;
472
473 // 加锁保护共享数据
474 StdUniqueLock lock(_mtx_calc);
475 const char* code = _code.c_str();
476 double undone = _ctx->getUndoneQty(code);
477 double newVol = get_real_target(_target_pos);
478 double realPos = _ctx->getPosition(code);//总仓位，昨仓 + 今仓买入的
479 double vailyPos = _ctx->getPosition(code,false);//可用仓位 即昨仓
480 double diffQty = newVol - realPos;//目标差
481 if (!_channel_ready)
482 return;
483 if (_is_finish)
484 return;
485 if (_is_t0)
486 vailyPos = realPos;
487 double target_pos = max(realPos - vailyPos, _target_pos);
488 if (!decimal::eq(target_pos, _target_pos))
489 {
490 _ctx->writeLog(fmtutil::format("{} can sell hold pos not enough, target adjust {}->{}", code, _target_pos, target_pos));
491 _target_pos = target_pos;
492 }
493 //有正在撤销的订单,则不能进行下一轮计算
494 if (_cancel_cnt != 0)
495 {
496 _ctx->writeLog(fmt::format("{}尚有未完成的撤单指令，暂时退出本轮执行", _code).c_str());
497 return;
498 }
499 if (decimal::eq(diffQty, 0))
500 return;
501 if (decimal::ge(_start_price, 0)) {//补一次
502 _start_price = _last_tick->price();
503 }
504 // 在判断的时候，要两边四舍五入，防止一些碎股导致一直无法完成执行
505 if (decimal::eq(round_hands(target_pos, _min_hands), round_hands(realPos, _min_hands)) && !(target_pos == 0 && realPos < _min_hands && realPos>target_pos))
506 {
507 _ctx->writeLog(fmtutil::format("{}: target position {} set finish", _code.c_str(), _target_pos));
508 _is_finish = true;
509 return;
510 }
511 //每一次发单要保障成交,所以如果有未完成单,说明上一轮没完成
512 //有未完成订单&&与实际仓位变动方向相反
513 //则需要撤销现有订单
514 bool isBuy = decimal::gt(undone, 0);//undone>0,isbuy=1
515 if (decimal::lt(diffQty*undone, 0))
516 {
517 OrderIDs ids = _ctx->cancel(code, isBuy);
518 if (!ids.empty())
519 {
520 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime());
521 _cancel_cnt += ids.size();
522 _ctx->writeLog(fmtutil::format("[{}@{}] live opposite order of {} canceled, cancelcnt -> {}", __FILE__, __LINE__, _code.c_str(), _cancel_cnt));//相反的订单取消
523 }
524 return;
525 }
526 if (!decimal::eq(undone, 0))
527 {
528 _ctx->writeLog(fmt::format("{}上一轮有挂单未完成,暂时退出本轮执行", _code).c_str());
529 return;
530 }
531 if (_last_tick == NULL)
532 {
533 _ctx->writeLog(fmt::format("{}没有最新的tick数据，退出执行逻辑", _code).c_str());
534 return;
535 }
536 double curPos = realPos;
537
538 if (decimal::eq(curPos, newVol))
539 {
540 //当前仓位和最新目标仓位匹配时，如果不是全部清仓的需求，则直接退出计算了
541 if (!is_clear(_target_pos))
542 return;
543
544 //如果是清仓的需求，还要再进行对比
545 //如果多头为0，说明已经全部清理掉了，则直接退出
546 double lPos = _ctx->getPosition(code, true, 1);
547 if (decimal::eq(lPos, 0))
548 return;
549 //如果还有多头仓位，则将目标仓位设置为非0，强制触发
550 newVol = -min(lPos, _order_lots);
551 _ctx->writeLog(fmtutil::format("Clearing process triggered, target position of {} has been set to {}", _code.c_str(), newVol));
552 }
553 //如果相比上次没有更新的tick进来，则先不下单，防止开盘前集中下单导致通道被封
554 uint64_t curTickTime = (uint64_t)_last_tick->actiondate() * 1000000000 + _last_tick->actiontime();
555 if (curTickTime <= _last_tick_time)
556 {
557 _ctx->writeLog(fmtutil::format("No tick of {} updated, {} <= {}, execute later", _code, curTickTime, _last_tick_time));
558 return;
559 }
560 _last_tick_time = curTickTime;
561 double InminsTm = calTmStamp(_last_tick->actiontime());//当前tick属于vwap240分钟内的第几(-1)分钟
562 double aimQty = VwapAim[InminsTm];//取到对应时刻的目标vwapaim （递增）
563
564 uint32_t leftTimes = _total_times - _fired_times;
565 _ctx->writeLog(fmt::format("第 {} 次发单", _fired_times + 1).c_str());
566 _Vwap_vol = aimQty - curPos;//若在本分钟发单，对应的VWapVol
567 bool bNeedShowHand = false;
568 double curQty = 0;
569 if (leftTimes == 0 && !decimal::eq(diffQty, 0))
570 {
571 bNeedShowHand = true;
572 curQty = max(diffQty, _min_open_lots);
573 }
574 else {
575 curQty = max(_Vwap_vol, _min_open_lots)* abs(diffQty) / diffQty;//curqty=单位预测量sum
576 }
577
578 //买
579 if (isBuy) {
580 //如果是买的话，要考虑取整和
581 curQty = round_hands(curQty, _min_open_lots);
582 }
583 // 卖要对碎股做检查
584 else {
585 if (decimal::lt(vailyPos, _min_open_lots))
586 {
587 curQty = vailyPos;
588 }
589 else
590 {
591 curQty = round_hands(curQty, _min_open_lots);
592 }
593 curQty = min(vailyPos, _min_open_lots);
594 }
595
596 //设定本轮目标仓位
597 _this_target = realPos + curQty;
598
599 WTSTickData* curTick = _last_tick;
600 uint64_t now = TimeUtils::getLocalTimeNow();
601 isBuy = decimal::gt(diffQty, 0);
602 double targetPx = 0;
603 //根据价格模式设置,确定委托基准价格: 0-最新价,1-最优价,2-对手价
604 if (_price_mode == 0)
605 {
606 targetPx = curTick->price();
607 }
608 else if (_price_mode == 1)
609 {
610 targetPx = isBuy ? curTick->bidprice(0) : curTick->askprice(0);
611 }
612 else
613 {
614 targetPx = isBuy ? curTick->askprice(0) : curTick->bidprice(0);
615 }
616 if (bNeedShowHand) {
617 targetPx += _comm_info->getPriceTick() * 5 * (isBuy ? 1 : -1);
618 }
619 else if (_price_offset != 0)
620 {
621 targetPx += _comm_info->getPriceTick() * _price_offset * (isBuy ? 1 : -1);
622 }
623
624 // 如果最后价格为0，再做一个修正
625 if (decimal::eq(targetPx, 0.0))
626 targetPx = decimal::eq(_last_tick->price(), 0.0) ? _last_tick->preclose() : _last_tick->price();
627
628 //检查涨跌停价
629 isCanCancel = true;

664void WtStockVWapExeUnit::fire_at_once(double qty)
665{
666 // 如果数量为0，直接返回
667 if (decimal::eq(qty, 0))
668 return;
669 // 增加行情数据的引用计数
670 _last_tick->retain();
671 WTSTickData* curTick = _last_tick;
672
673 // 获取合约代码、当前时间
674 const char* code = _code.c_str();
675 uint64_t now = TimeUtils::getLocalTimeNow();
676
677 // 根据数量正负判断是买入还是卖出
678 bool isBuy = decimal::gt(qty, 0);
679 double targetPx = 0;
680
681 // 根据价格模式设置确定委托基准价格:
682 // 0-最新价, 1-最优价, 2-对手价
683 if (_price_mode == 0) {
684 // 使用最新成交价格
685 targetPx = curTick->price();
686 }
687 else if (_price_mode == 1) {
688 // 使用最优价：买入时用卖一价，卖出时用买一价
689 targetPx = isBuy ? curTick->askprice(0) : curTick->bidprice(0);
690 }
691 else // if(_price_mode == 2)
692 {
693 // 使用对手价：买入时用卖一价，卖出时用买一价
694 targetPx = isBuy ? curTick->askprice(0) : curTick->bidprice(0);
695 }
696
697 // 根据撤单次数增加价格偏移，买入时价格上移，卖出时价格下移
698 targetPx += _comm_info->getPriceTick()*_cancel_times *(isBuy ? 1 : -1);
699
700 // 检查并处理涨跌停限制
701 isCanCancel = true; // 默认订单可撤销
702
703 // 如果是买入订单，检查是否超过涨停价
704 if (isBuy && !decimal::eq(_last_tick->upperlimit(), 0) && decimal::gt(targetPx, _last_tick->upperlimit()))
705 {
706 // 记录日志并将价格修正为涨停价
707 _ctx->writeLog(fmt::format("Buy price {} of {} modified to upper limit price", targetPx, _code.c_str(), _last_tick->upperlimit()).c_str());
708 targetPx = _last_tick->upperlimit();
709 isCanCancel = false; // 涨停价订单不可撤销
710 }
711
712 // 如果是卖出订单，检查是否低于跌停价
713 if (isBuy != 1 && !decimal::eq(_last_tick->lowerlimit(), 0) && decimal::lt(targetPx, _last_tick->lowerlimit()))
714 {
715 // 记录日志并将价格修正为跌停价
716 _ctx->writeLog(fmt::format("Sell price {} of {} modified to lower limit price", targetPx, _code.c_str(), _last_tick->lowerlimit()).c_str());
717 targetPx = _last_tick->lowerlimit();
718 isCanCancel = false; // 跌停价订单不可撤销
719 }
720
721 // 创建订单ID容器
722 OrderIDs ids;
723
724 // 根据数量正负发送买入或卖出订单
725 if (qty > 0)
726 // 发送买入订单
727 ids = _ctx->buy(code, targetPx, abs(qty));
728 else
729 // 发送卖出订单
730 ids = _ctx->sell(code, targetPx, abs(qty));
731
732 // 将订单添加到监控器中
733 _orders_mon.push_order(ids.data(), ids.size(), now, isCanCancel);
734
735 // 释放行情数据
736 curTick->release();
737}
