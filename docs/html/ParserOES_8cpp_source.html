291{

295
296 if (_inited)
297 return;
298

300 * By Wesley @ 2022.07.26
301 * UDP模式下会出现多次调用的情况
302 */
303 _inited = true;
304 if (_sink)
305 {
306 _sink->handleEvent(WPE_Connect, 0);

410void ParserOES::doOnMessage(SMsgHeadT *pMsgHead, void *pMsgItem)
411{
412 if (pMsgHead == NULL || pMsgItem == NULL)
413 return;
414
415 if (_bd_mgr == NULL)
416 return;
417
418
419 MdsMktRspMsgBodyT *pRspMsg = (MdsMktRspMsgBodyT *)pMsgItem;
420
421 /* 根据消息类型对行情消息进行处理 */
422 switch (pMsgHead->msgId) {
423 case MDS_MSGTYPE_L2_TRADE:
424 /* 处理Level2逐笔成交消息 @see MdsL2TradeT */
425 {
426 std::string code, exchg;
427 if (pRspMsg->trade.exchId == MDS_EXCH_SSE)
428 {
429 exchg = "SSE";
430 }
431 else
432 {
433 exchg = "SZSE";
434 }
435 code = pRspMsg->trade.SecurityID;
436
437 WTSContractInfo* ct = _bd_mgr->getContract(code.c_str(), exchg.c_str());
438 if (ct == NULL)
439 {
440 return;
441 }
442 WTSCommodityInfo* commInfo = ct->getCommInfo();
443
444 WTSTransData *trans = WTSTransData::create(code.c_str());
445 WTSTransStruct& ts = trans->getTransStruct();
446 strcpy(ts.exchg, commInfo->getExchg());
447
448 ts.trading_date = pRspMsg->trade.tradeDate;
449 ts.action_date = pRspMsg->trade.tradeDate;
450 ts.action_time = pRspMsg->trade.TransactTime;
451
452 ts.index = pRspMsg->trade.ApplSeqNum;
453 ts.side = wrapTransSide(pRspMsg->trade.TradeBSFlag);
454 ts.ttype = pRspMsg->trade.ExecType == 'F' ? TT_Match : TT_Cancel;
455
456 ts.price = wrapPrice(pRspMsg->trade.TradePrice);
457 ts.volume = pRspMsg->trade.TradeQty;
458 ts.bidorder = pRspMsg->trade.BidApplSeqNum;
459 ts.askorder = pRspMsg->trade.OfferApplSeqNum;
460
461 if (_sink)
462 _sink->handleTransaction(trans);
463
464 static uint32_t recv_cnt = 0;
465 recv_cnt++;
466 if (recv_cnt % _gpsize == 0)
467 write_log(_sink, LL_DEBUG, "[ParserOES] {} transactions received in total", recv_cnt);
468 }
469 break;
470
471 case MDS_MSGTYPE_L2_ORDER:
472 case MDS_MSGTYPE_L2_SSE_ORDER:
473 /* 处理Level2逐笔委托消息 @see MdsL2OrderT */
474 {
475 std::string code, exchg;
476 if (pRspMsg->order.exchId == MDS_EXCH_SSE)
477 {
478 exchg = "SSE";
479 }
480 else
481 {
482 exchg = "SZSE";
483 }
484 code = pRspMsg->order.SecurityID;
485
486 WTSContractInfo* ct = _bd_mgr->getContract(code.c_str(), exchg.c_str());
487 if (ct == NULL)
488 {
489 return;
490 }
491 WTSCommodityInfo* commInfo = ct->getCommInfo();
492
493 WTSOrdDtlData *ordDtl = WTSOrdDtlData::create(code.c_str());
494 WTSOrdDtlStruct& ts = ordDtl->getOrdDtlStruct();
495 strcpy(ts.exchg, commInfo->getExchg());
496
497 ts.trading_date = pRspMsg->order.tradeDate;
498 ts.action_date = pRspMsg->order.tradeDate;
499 ts.action_time = pRspMsg->order.TransactTime;
500
501 ts.index = pRspMsg->order.SseOrderNo;
502 ts.side = wrapOrdDtlSide(pRspMsg->order.Side);
503 ts.otype = pRspMsg->order.OrderType;
504
505 ts.price = wrapPrice(pRspMsg->order.Price);
506 ts.volume = pRspMsg->order.OrderQty;
507
508 if (_sink)
509 _sink->handleOrderDetail(ordDtl);
510
511 static uint32_t recv_cnt = 0;
512 recv_cnt++;
513 if (recv_cnt % _gpsize == 0)
514 write_log(_sink, LL_DEBUG, "[ParserOES] {} orders received in total", recv_cnt);
515 }
516 break;
517
518 case MDS_MSGTYPE_L2_MARKET_DATA_SNAPSHOT:
519 /* 处理Level2快照行情消息 @see MdsL2StockSnapshotBodyT */
520 {
521 std::string code, exchg;
522 if (pRspMsg->mktDataSnapshot.head.exchId == MDS_EXCH_SSE)
523 {
524 exchg = "SSE";
525 }
526 else
527 {
528 exchg = "SZSE";
529 }
530 code = pRspMsg->mktDataSnapshot.l2Stock.SecurityID;
531
532 WTSContractInfo* ct = _bd_mgr->getContract(code.c_str(), exchg.c_str());
533 if (ct == NULL)
534 {
535 //if (_sink)
536 // write_log(_sink, LL_ERROR, "[ParserXTP] Instrument {}.{} not exists...", exchg.c_str(), code);
537 return;
538 }
539 WTSCommodityInfo* commInfo = ct->getCommInfo();
540
541 WTSTickData* tick = WTSTickData::create(code.c_str());
542 tick->setContractInfo(ct);
543 WTSTickStruct& quote = tick->getTickStruct();
544 strcpy(quote.exchg, commInfo->getExchg());
545
546 quote.trading_date = pRspMsg->mktDataSnapshot.head.tradeDate;
547 quote.action_date = pRspMsg->mktDataSnapshot.head.tradeDate;
548 quote.action_time = pRspMsg->mktDataSnapshot.head.updateTime;
549
550 quote.price = wrapPrice(pRspMsg->mktDataSnapshot.l2Stock.TradePx);
551 quote.open = wrapPrice(pRspMsg->mktDataSnapshot.l2Stock.OpenPx);
552 quote.high = wrapPrice(pRspMsg->mktDataSnapshot.l2Stock.HighPx);
553 quote.low = wrapPrice(pRspMsg->mktDataSnapshot.l2Stock.LowPx);
554 quote.total_volume = (double)pRspMsg->mktDataSnapshot.l2Stock.TotalVolumeTraded;
555 quote.total_turnover = pRspMsg->mktDataSnapshot.l2Stock.TotalValueTraded / 10000.0;
556
557 quote.pre_close = wrapPrice(pRspMsg->mktDataSnapshot.l2Stock.PrevClosePx);
558
559 //委卖价格
560 for (int i = 0; i < 10; i++)
561 {
562 quote.ask_prices[i] = wrapPrice(pRspMsg->mktDataSnapshot.l2Stock.OfferLevels[i].Price);
563 quote.ask_qty[i] = (double)pRspMsg->mktDataSnapshot.l2Stock.OfferLevels[i].OrderQty;
564
565 quote.bid_prices[i] = wrapPrice(pRspMsg->mktDataSnapshot.l2Stock.BidLevels[i].Price);
566 quote.bid_qty[i] = (double)pRspMsg->mktDataSnapshot.l2Stock.BidLevels[i].OrderQty;
567 }
568
569 if (_sink)
570 _sink->handleQuote(tick, 1);
571
572 tick->release();
573
574 static uint32_t recv_cnt = 0;
575 recv_cnt++;
576 if (recv_cnt % _gpsize == 0)
577 write_log(_sink, LL_DEBUG, "[ParserOES] {} L2 ticks received in total", recv_cnt);
578 }
579 break;
580
581 case MDS_MSGTYPE_L2_BEST_ORDERS_SNAPSHOT:
582 /* 处理Level2委托队列消息(买一／卖一前五十笔委托明细) @see MdsL2BestOrdersSnapshotBodyT */
583 {
584 std::string code, exchg;
585 if (pRspMsg->mktDataSnapshot.head.exchId == MDS_EXCH_SSE)
586 {
587 exchg = "SSE";
588 }
589 else
590 {
591 exchg = "SZSE";
592 }
593 code = pRspMsg->mktDataSnapshot.l2BestOrders.SecurityID;
594
595 WTSContractInfo* ct = _bd_mgr->getContract(code.c_str(), exchg.c_str());
596 if (ct == NULL)
597 {
598 //if (_sink)
599 // write_log(_sink, LL_ERROR, "[ParserXTP] Instrument {}.{} not exists...", exchg.c_str(), code);
600 return;
601 }
602 WTSCommodityInfo* commInfo = ct->getCommInfo();
603
604 WTSOrdQueData* buyQue = WTSOrdQueData::create(code.c_str());
605 buyQue->setContractInfo(ct);
606
607 WTSOrdQueData* sellQue = WTSOrdQueData::create(code.c_str());
608 sellQue->setContractInfo(ct);
609
610 WTSOrdQueStruct& buyOS = buyQue->getOrdQueStruct();
611 strcpy(buyOS.exchg, commInfo->getExchg());
612
613 WTSOrdQueStruct& sellOS = sellQue->getOrdQueStruct();
614 strcpy(sellOS.exchg, commInfo->getExchg());
615
616 buyOS.trading_date = pRspMsg->mktDataSnapshot.head.tradeDate;
617 buyOS.action_date = pRspMsg->mktDataSnapshot.head.tradeDate;
618 buyOS.action_time = pRspMsg->mktDataSnapshot.head.updateTime;
619
620 sellOS.trading_date = pRspMsg->mktDataSnapshot.head.tradeDate;
621 sellOS.action_date = pRspMsg->mktDataSnapshot.head.tradeDate;
622 sellOS.action_time = pRspMsg->mktDataSnapshot.head.updateTime;
623
624 buyOS.side = BDT_Buy;
625 buyOS.price = wrapPrice(pRspMsg->mktDataSnapshot.l2BestOrders.BestBidPrice);
626 buyOS.order_items = pRspMsg->mktDataSnapshot.l2BestOrders.NoBidOrders;
627
628 sellOS.side = BDT_Sell;
629 sellOS.price = wrapPrice(pRspMsg->mktDataSnapshot.l2BestOrders.BestOfferPrice);
630 sellOS.order_items = pRspMsg->mktDataSnapshot.l2BestOrders.NoOfferOrders;
631
632 for(int i = 0; i < 50; i++)
633 {
634 if(pRspMsg->mktDataSnapshot.l2BestOrders.BidOrderQty[i] != 0)
635 {
636 buyOS.volumes[i] = pRspMsg->mktDataSnapshot.l2BestOrders.BidOrderQty[i];
637 buyOS.qsize++;
638 }
639
640 if (pRspMsg->mktDataSnapshot.l2BestOrders.OfferOrderQty[i] != 0)
641 {
642 sellOS.volumes[i] = pRspMsg->mktDataSnapshot.l2BestOrders.OfferOrderQty[i];
643 sellOS.qsize++;
644 }
645 }
646
647 if (_sink)
648 {
649 _sink->handleOrderQueue(buyQue);
650 _sink->handleOrderQueue(sellQue);
651 }
652
653 buyQue->release();
654 sellQue->release();
655
656 static uint32_t recv_cnt = 0;
657 recv_cnt += 2;
658 if (recv_cnt % _gpsize == 0)
659 write_log(_sink, LL_DEBUG, "[ParserOES] {} queues received in total", recv_cnt);
660 }
661 break;
662
663 case MDS_MSGTYPE_MARKET_DATA_SNAPSHOT_FULL_REFRESH:
664 /* 处理Level1快照行情消息 @see MdsStockSnapshotBodyT */
665 {
666 std::string code, exchg;
667 if (pRspMsg->mktDataSnapshot.head.exchId == MDS_EXCH_SSE)
668 {
669 exchg = "SSE";
670 }
671 else
672 {
673 exchg = "SZSE";
674 }
675 code = pRspMsg->mktDataSnapshot.stock.SecurityID;
676
677 WTSContractInfo* ct = _bd_mgr->getContract(code.c_str(), exchg.c_str());
678 if (ct == NULL)
679 {
680 //if (_sink)
681 // write_log(_sink, LL_ERROR, "[ParserXTP] Instrument {}.{} not exists...", exchg.c_str(), code);
682 return;
683 }
