21bool ShareBlocks::init_master(const char* name, const char* path/* = ""*/)
22{
23 ShmPair& shm = (ShmPair&)_shm_blocks[name];
24 if (shm._block != NULL)
25 return true;
26
27 std::string filename = path;
28 if (filename.empty())
29 filename = name;
30
31 if(!StdFile::exists(filename.c_str()))
32 {
33 BoostFile bf;
34 bf.create_new_file(filename.c_str());
35 bf.truncate_file(sizeof(ShmBlock));
36 bf.close_file();
37 }
38
39 shm._domain.reset(new BoostMappingFile);
40 shm._domain->map(filename.c_str());
41 shm._master = true;
42 shm._block = (ShmBlock*)shm._domain->addr();
43
44 /*
45 * By Wesley @ 2023.09.20
46 * 这里要做一个清理，如果state为0，则不再保留
47 */
48 std::vector<SecInfo> aySecs;
49 for (uint32_t i = 0; i < shm._block->_count; i++)
50 {
51 SecInfo& secInfo = shm._block->_sections[i];
52 if (secInfo._count == 0 || secInfo._state != 1)
53 continue;
54
55 aySecs.emplace_back(secInfo);
56 }
57
58 if(aySecs.size() != shm._block->_count)
59 {
60 shm._block->_count = (uint32_t)aySecs.size();
61 memset(shm._block->_sections, 0, sizeof(SecInfo)*MAX_SEC_CNT);
62 if (shm._block->_count > 0)
63 memcpy(shm._block->_sections, aySecs.data(), sizeof(SecInfo)*shm._block->_count);
64
65 shm._blocktime = shm._block->_updatetime;
66 }
67
68 {
69 //这里要做初始化，要把已经有的key加载进去
70 for (uint32_t i = 0; i < shm._block->_count; i++)
71 {
72 SecInfo& secInfo = shm._block->_sections[i];
73 if (secInfo._count == 0)
74 continue;
75
76 //置零的目的看后面会不会用到
77 //如果不会用到，那么就不会变成1
78 //下次启动就会删掉这个section
79 secInfo._state = 0;
80
81 ShmPair::KVPair& kvPair = shm._sections[secInfo._name];
82 kvPair._index = i;
83 for (uint32_t j = 0; j < secInfo._count; j++)
84 {
85 KeyInfo& key = secInfo._keys[j];
86 kvPair._keys[key._key] = &key;
87 }
88 }
89 }
90
91 return true;
92}

293void* ShareBlocks::make_valid(const char* domain, const char* section, const char* key, ValueType vType, SecInfo* &secInfo)
294{
295 auto it = _shm_blocks.find(domain);
296 if (it == _shm_blocks.end())
297 return nullptr;
298
299 std::size_t len = SMVT_SIZES[vType];
300
301 ShmPair& shm = (ShmPair&)it->second;
302 KeyInfo* keyInfo = nullptr;
303 ShmPair::KVPair* kvPair = nullptr;
304 auto sit = shm._sections.find(section);
305 if (sit == shm._sections.end())
306 {
307 //如果不是master，就不能创建
308 if (!shm._master)
309 return nullptr;
310
311 if (shm._block->_count == MAX_SEC_CNT)
312 {
313 //已经没有额外的空间可以分配了
314 return nullptr;
315 }
316
317 secInfo = &shm._block->_sections[shm._block->_count];
318 wt_strcpy(secInfo->_name, section);
319 secInfo->_updatetime = TimeUtils::getLocalTimeNow();
320 kvPair = &shm._sections[section];
321 kvPair->_index = shm._block->_count;
322 shm._block->_count++;
323 }
324 else
325 {
326 kvPair = (ShmPair::KVPair*)&sit->second;
327 }
328
329 secInfo = &shm._block->_sections[kvPair->_index];
330 secInfo->_state = 1;
331
332 auto kit = kvPair->_keys.find(key);
333 if (kit == kvPair->_keys.end())
334 {
335 //如果不是master，就不能创建
336 if (!shm._master)
337 return nullptr;
338
339 if (secInfo->_count == MAX_KEY_CNT)
340 return nullptr;
341
342 if (secInfo->_offset + len > 1024)
343 return nullptr;
344
345 keyInfo = &secInfo->_keys[secInfo->_count];
346 wt_strcpy(keyInfo->_key, key);
347 keyInfo->_updatetime = TimeUtils::getLocalTimeNow();
348 keyInfo->_offset = secInfo->_offset;
349 kvPair->_keys[key] = keyInfo;
350
351 //字符串固定最大长度为64
352 secInfo->_count++;
353 secInfo->_offset += (uint32_t)len;
354 }
355 else
356 {
357 keyInfo = kit->second;
358 }
359
360 return keyInfo;
361}

887bool ShareBlocks::init_cmder(const char* name, bool isCmder /* = false */, const char* path /* = "" */)
888{
889 CmdPair& cmdPair = _cmd_blocks[name];
890 if (cmdPair._block != NULL)
891 return true;
892
893 std::string filename = path;
894 if (filename.empty())
895 filename = ".cmd";
896
897 if (!StdFile::exists(filename.c_str()))
898 {
899 BoostFile bf;
900 bf.create_new_file(filename.c_str());
901 bf.truncate_file(sizeof(CmdBlock));
902 bf.close_file();
903 }
904
905 cmdPair._domain.reset(new BoostMappingFile);
906 cmdPair._domain->map(filename.c_str());
907 cmdPair._cmder = isCmder;
908 cmdPair._block = (CmdBlock*)cmdPair._domain->addr();
909 if(cmdPair._block->_capacity == 0)
910 new(cmdPair._domain->addr()) CmdBlock();
911
912 if(cmdPair._cmder)
913#ifdef _MSC_VER
914 cmdPair._block->_cmdpid = _getpid();
915#else
916 cmdPair._block->_cmdpid = getpid();
917#endif
918
919
920 //启动的时候都做一下偏移
921 cmdPair._block->_writable %= cmdPair._block->_capacity;
922 if(cmdPair._block->_readable != UINT32_MAX)
923 {
924 cmdPair._block->_readable %= cmdPair._block->_capacity;
925 if (cmdPair._block->_readable > cmdPair._block->_writable)
926 cmdPair._block->_writable += cmdPair._block->_capacity;
927 }
928
929 return true;
930}

939bool ShareBlocks::add_cmd(const char* name, const char* cmd)
940{
941 auto it = _cmd_blocks.find(name);
942 if (it == _cmd_blocks.end())
943 return false;
944
945 CmdPair& cmdPair = it->second;
946
947 if (cmdPair._block == NULL)
948 return false;
949
950#ifdef _MSC_VER
951 if (cmdPair._block->_cmdpid != _getpid())
952#else
953 if (cmdPair._block->_cmdpid != getpid())
954#endif
955
956 return false;
957
958 /*
959 * 先移动写的下标，然后写入数据
960 * 写完了以后，再移动读的下标
961 */
962 uint32_t wIdx = cmdPair._block->_writable++;
963 uint32_t realIdx = wIdx % cmdPair._block->_capacity;
964 cmdPair._block->_commands[realIdx]._state = 0;
965 strcpy(cmdPair._block->_commands[realIdx]._command, cmd);
966 cmdPair._block->_readable = wIdx;
967 return true;
968}

977const char* ShareBlocks::get_cmd(const char* name, uint32_t& lastIdx)
978{
979 auto it = _cmd_blocks.find(name);
980 if (it == _cmd_blocks.end())
981 return "";
982
983 CmdPair& cmdPair = it->second;
