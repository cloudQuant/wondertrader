512bool CtaMocker::init_cta_factory(WTSVariant* cfg)
513{
514 if (cfg == NULL)
515 return false;
516
517 const char* module = cfg->getCString("module");
518
519 DllHandle hInst = DLLHelper::load_library(module);
520 if (hInst == NULL)
521 return false;
522
523 FuncCreateStraFact creator = (FuncCreateStraFact)DLLHelper::get_symbol(hInst, "createStrategyFact");
524 if (creator == NULL)
525 {
526 DLLHelper::free_library(hInst);
527 return false;
528 }
529
530 _factory._module_inst = hInst;
531 _factory._module_path = module;
532 _factory._creator = creator;
533 _factory._remover = (FuncDeleteStraFact)DLLHelper::get_symbol(hInst, "deleteStrategyFact");
534 _factory._fact = _factory._creator();
535
536 WTSVariant* cfgStra = cfg->get("strategy");
537 if (cfgStra)
538 {
539 _strategy = _factory._fact->createStrategy(cfgStra->getCString("name"), cfgStra->getCString("id"));
540 if(_strategy)
541 {
542 WTSLogger::info("Strategy {}.{} is created,strategy ID: {}", _factory._fact->getName(), _strategy->getName(), _strategy->id());
543 }
544 _strategy->init(cfgStra->get("params"));
545 _name = _strategy->id();
546 }
547
548 return true;
549}

853void CtaMocker::proc_tick(const char* stdCode, double last_px, double cur_px)
854{
855 {
856 auto it = _sig_map.find(stdCode);
857 if (it != _sig_map.end())
858 {
859 //if (sInfo->isInTradingTime(_replayer->get_raw_time(), true))
860 {
861 const SigInfo& sInfo = it->second;
862 double price;
863 if (decimal::eq(sInfo._desprice, 0.0))
864 price = cur_px;
865 else
866 price = sInfo._desprice;
867 do_set_position(stdCode, sInfo._volume, price, sInfo._usertag.c_str());
868
869 //如果是条件单触发，则回调on_condition_triggered
870 if (sInfo._sigtype == 2)
871 on_condition_triggered(stdCode, sInfo._volume, cur_px, sInfo._usertag.c_str());
872 _sig_map.erase(it);
873 }
874 }
875 }
876
877 update_dyn_profit(stdCode, cur_px);
878
880 //检查条件单
881 if (!_condtions.empty())
882 {
883 auto it = _condtions.find(stdCode);
884 if (it == _condtions.end())
885 return;
886
887 const CondList& condList = it->second;
888 double curPrice = cur_px;
889 const CondEntrust* matchedEntrust = NULL;
890 for (const CondEntrust& entrust : condList)
891 {
892 /*
893 * 如果开启了tick模式，就正常比较
894 * 但是如果没有开启tick模式，逻辑就非常复杂
895 * 因为不开回测的时候tick是用开高低收模拟出来的，如果直接按照目标价格触发，可能是有问题的
896 * 首先要拿到上一笔价格，和当前最新价格做一个比价，得到左边界和右边界
897 * 这里只能假设前后两笔价格之间是连续的，这样需要将两笔价格都加入判断
898 * 当条件是等于时，如果目标价格在左右边界之间，说明目标价格在这期间是出现过的，则认为价格匹配
899 * 当条件是大于的时候，我们需要判断右边界，即稍大的值是否满足条件，并取左边界与目标价中稍大的作为当前价
900 * 当条件是小于的时候，我们需要判断左边界，即稍小的值是否满足条件，并取右边界与目标价中稍小的作为当前价
901 */
902
903 double left_px = min(last_px, cur_px);
904 double right_px = max(last_px, cur_px);
905
906 bool isMatched = false;
907 if (!_replayer->is_tick_simulated())
908 {
909 //如果tick数据不是模拟的，则使用最新价格
910 switch (entrust._alg)
911 {
912 case WCT_Equal:
913 isMatched = decimal::eq(curPrice, entrust._target);
914 break;
915 case WCT_Larger:
916 isMatched = decimal::gt(curPrice, entrust._target);
917 break;
918 case WCT_LargerOrEqual:
919 isMatched = decimal::ge(curPrice, entrust._target);
920 break;
921 case WCT_Smaller:
922 isMatched = decimal::lt(curPrice, entrust._target);
923 break;
924 case WCT_SmallerOrEqual:
925 isMatched = decimal::le(curPrice, entrust._target);
926 break;
927 default:
928 break;
929 }
930
931 if (isMatched)
932 {
933 matchedEntrust = &entrust;
934 break;
935 }
936 }
937 else
938 {
939 //如果tick数据是模拟的，则要处理一下
940 switch (entrust._alg)
941 {
942 case WCT_Equal:
943 isMatched = decimal::le(left_px, entrust._target) && decimal::ge(right_px, entrust._target);
944 break;
945 case WCT_Larger:
946 isMatched = decimal::gt(right_px, entrust._target);
947 break;
948 case WCT_LargerOrEqual:
949 isMatched = decimal::ge(right_px, entrust._target);
950 break;
951 case WCT_Smaller:
952 isMatched = decimal::lt(left_px, entrust._target);
953 break;
954 case WCT_SmallerOrEqual:
955 isMatched = decimal::le(left_px, entrust._target);
956 break;
957 default:
958 break;
959 }
960
961 if (isMatched)
962 {
963 /*
964 * By HeJ @ 2023.02.27
965 * 在bar回测中，经常会出现同一个价格触发了多个条件单时，要选出一个作为最终的触发价，遵循以下规则：
966 * 1 alg不同的条件单，或者alg为WCT_Equal，以最先设置的那个为准
967 * 2 alg一样的调价单，如果是WCT_Larger与WCT_LargerOrEqual，取触发价较小的，WCT_Smaller与WCT_SmallerOrEqual，取触发价较大的
968 */
969 if (matchedEntrust == NULL)
970 {
971 matchedEntrust = &entrust;
972 if (entrust._alg == WCT_Larger || entrust._alg == WCT_LargerOrEqual)
973 curPrice = max(left_px, entrust._target);
974 else if (entrust._alg == WCT_Smaller || entrust._alg == WCT_SmallerOrEqual)
975 curPrice = min(right_px, entrust._target);
976 else
977 curPrice = entrust._target;
978 }
979 else if (matchedEntrust->_alg == entrust._alg)
980 {
981 if (entrust._alg == WCT_Larger || entrust._alg == WCT_LargerOrEqual)
982 {
983 if (entrust._target < matchedEntrust->_target)
984 {
985 matchedEntrust = &entrust;
986 curPrice = max(left_px, entrust._target);
987 }
988 }
989 else if (entrust._alg == WCT_Smaller || entrust._alg == WCT_SmallerOrEqual)
990 {
991 if (entrust._target > matchedEntrust->_target)
992 {
993 matchedEntrust = &entrust;
994 curPrice = min(right_px, entrust._target);
995 }
996 }
997 }
998 }
999 }
1000 }
1001
1002 if (matchedEntrust != NULL)
1003 {
1004 const CondEntrust& entrust = *matchedEntrust;
1005 double price = curPrice;
1006 double curQty = stra_get_position(stdCode);
1007 //_replayer->is_tick_enabled() ? newTick->price() : entrust._target; //如果开启了tick回测,则用tick数据的价格,如果没有开启,则只能用条件单价格
1008 WTSLogger::log_dyn("strategy", _name.c_str(), LL_INFO,
1009 "Condition order triggered[newprice: {}{}{}], instrument: {}, {} {}",
1010 curPrice, CMP_ALG_NAMES[entrust._alg], entrust._target, stdCode, ACTION_NAMES[entrust._action], entrust._qty);
1011 switch (entrust._action)
1012 {
1013 case COND_ACTION_OL:
1014 {
1015 if (decimal::lt(curQty, 0))
1016 append_signal(stdCode, entrust._qty, entrust._usertag, price, 2);
1017 else
1018 append_signal(stdCode, curQty + entrust._qty, entrust._usertag, price, 2);
1019 }
1020 break;
1021 case COND_ACTION_CL:
1022 {
1023 double maxQty = min(curQty, entrust._qty);
1024 append_signal(stdCode, curQty - maxQty, entrust._usertag, price, 2);
1025 }
1026 break;
1027 case COND_ACTION_OS:
1028 {
1029 if (decimal::gt(curQty, 0))
1030 append_signal(stdCode, -entrust._qty, entrust._usertag, price, 2);
1031 else
1032 append_signal(stdCode, curQty - entrust._qty, entrust._usertag, price, 2);
1033 }
1034 break;
1035 case COND_ACTION_CS:
1036 {
1037 double maxQty = min(abs(curQty), entrust._qty);
1038 append_signal(stdCode, curQty + maxQty, entrust._usertag, price, 2);
1039 }
1040 break;
1041 case COND_ACTION_SP:
1042 {
1043 append_signal(stdCode, entrust._qty, entrust._usertag, price, 2);
1044 }
1045 default: break;
1046 }
1047
1048 //同一个bar设置针对同一个合约的条件单,只可能触发一条
1049 //所以这里直接清理掉即可
1050 _condtions.erase(it);
1051 }
1052 }
1053}

1888{
1889 PosInfo& pInfo = _pos_map[stdCode];
1890 double curPx = price;
1891 if (decimal::eq(price, 0.0))
1892 curPx = _price_map[stdCode];
1893 uint64_t curTm = (uint64_t)_replayer->get_date() * 10000 + _replayer->get_min_time();
1894 uint32_t curTDate = _replayer->get_trading_date();
1895
1896 //手数相等则不用操作了
1897 if (decimal::eq(pInfo._volume, qty))
1898 return;
1899
1900 WTSCommodityInfo* commInfo = _replayer->get_commodity_info(stdCode);
1901 if (commInfo == NULL)
1902 return;
1903
1904 //成交价
1905 double trdPx = curPx;
1906
1907 double diff = qty - pInfo._volume;
1908 bool isBuy = decimal::gt(diff, 0.0);
1909 if (decimal::gt(pInfo._volume*diff, 0))//当前持仓和仓位变化方向一致, 增加一条明细, 增加数量即可
1910 {
1911 pInfo._volume = qty;
1912
1913 //如果T+1，则冻结仓位要增加
1914 if (commInfo->isT1())
1915 {
1916 //ASSERT(diff>0);
1917 pInfo._frozen += diff;
1918 log_debug("{} frozen position up to {}", stdCode, pInfo._frozen);
1919 }
1920
1921 if (_slippage != 0)
1922 {
1923 if (_ratio_slippage)
1924 {
1925 //By Wesley @ 2023.05.05
1926 //如果是比率滑点，则要根据目标成交价计算
1927 //得到滑点以后，再根据pricetick做一个修正
1928 double slp = (_slippage * trdPx / 10000.0);
1929 slp = round(slp / commInfo->getPriceTick())*commInfo->getPriceTick();
1930
1931 trdPx += slp * (isBuy ? 1 : -1);
1932 }
1933 else
1934 trdPx += _slippage * commInfo->getPriceTick()*(isBuy ? 1 : -1);
1935 }
1936
1937 DetailInfo dInfo;
1938 dInfo._long = decimal::gt(qty, 0);
1939 dInfo._price = trdPx;
1940 dInfo._max_price = trdPx;
1941 dInfo._min_price = trdPx;
1942 dInfo._volume = abs(diff);
1943 dInfo._opentime = curTm;
1944 dInfo._opentdate = curTDate;
1945 strcpy(dInfo._opentag, userTag);
1946 dInfo._open_barno = _schedule_times;
1947 pInfo._details.emplace_back(dInfo);
1948 pInfo._last_entertime = curTm;
1949
1950 double fee = _replayer->calc_fee(stdCode, trdPx, abs(diff), 0);
1951 _fund_info._total_fees += fee;
1952
1953 log_trade(stdCode, dInfo._long, true, curTm, trdPx, abs(diff), userTag, fee, _schedule_times);
1954 }
1955 else
1956 {//持仓方向和仓位变化方向不一致,需要平仓
1957 double left = abs(diff);
1958 if (_slippage != 0)
1959 {
1960 if (_ratio_slippage)
1961 {
1962 //By Wesley @ 2023.05.05
1963 //如果是比率滑点，则要根据目标成交价计算
1964 //得到滑点以后，再根据pricetick做一个修正
1965 double slp = (_slippage * trdPx / 10000.0);
1966 slp = round(slp / commInfo->getPriceTick())*commInfo->getPriceTick();
1967
1968 trdPx += slp * (isBuy ? 1 : -1);
1969 }
1970 else
1971 trdPx += _slippage * commInfo->getPriceTick()*(isBuy ? 1 : -1);
1972 }
1973
1974 pInfo._volume = qty;
1975 if (decimal::eq(pInfo._volume, 0))
1976 pInfo._dynprofit = 0;
1977 uint32_t count = 0;
1978 for (auto it = pInfo._details.begin(); it != pInfo._details.end(); it++)
1979 {
1980 DetailInfo& dInfo = *it;
1981 double maxQty = min(dInfo._volume, left);
1982 if (decimal::eq(maxQty, 0))
1983 continue;
1984
1985 double maxProf = dInfo._max_profit * maxQty / dInfo._volume;
1986 double maxLoss = dInfo._max_loss * maxQty / dInfo._volume;
1987
1988 dInfo._volume -= maxQty;
1989 left -= maxQty;
1990
1991 if (decimal::eq(dInfo._volume, 0))
1992 count++;
1993
1994 double profit = (trdPx - dInfo._price) * maxQty * commInfo->getVolScale();
1995 if (!dInfo._long)
1996 profit *= -1;
1997 pInfo._closeprofit += profit;
1998 _total_closeprofit += profit;
1999 pInfo._dynprofit = pInfo._dynprofit*dInfo._volume / (dInfo._volume + maxQty);//浮盈也要做等比缩放
2000 pInfo._last_exittime = curTm;
2001 _fund_info._total_profit += profit;
2002
2003 double fee = _replayer->calc_fee(stdCode, trdPx, maxQty, dInfo._opentdate == curTDate ? 2 : 1);
2004 _fund_info._total_fees += fee;
2005 //这里写成交记录
2006 log_trade(stdCode, dInfo._long, false, curTm, trdPx, maxQty, userTag, fee, _schedule_times);
2007 //这里写平仓记录
2008 log_close(stdCode, dInfo._long, dInfo._opentime, dInfo._price, curTm, trdPx, maxQty, profit, maxProf, maxLoss,
2009 _total_closeprofit - _fund_info._total_fees, dInfo._opentag, userTag, dInfo._open_barno, _schedule_times);
2010
2011 if (left == 0)
2012 break;
2013 }
2014
2015 //需要清理掉已经平仓完的明细
2016 while (count > 0)
2017 {
2018 auto it = pInfo._details.begin();
2019 pInfo._details.erase(it);
2020 count--;
2021 }
2022
2023 //最后,如果还有剩余的,则需要反手了
2024 if (left > 0)
2025 {
2026 left = left * qty / abs(qty);
2027
2028 //如果T+1，则冻结仓位要增加
2029 if (commInfo->isT1())
2030 {
2031 pInfo._frozen += left;
2032 log_debug("{} frozen position up to {}", stdCode, pInfo._frozen);
2033 }
2034
2035 DetailInfo dInfo;
2036 dInfo._long = decimal::gt(qty, 0);
2037 dInfo._price = trdPx;
2038 dInfo._max_price = trdPx;
2039 dInfo._min_price = trdPx;
2040 dInfo._volume = abs(left);
2041 dInfo._opentime = curTm;
2042 dInfo._opentdate = curTDate;
2043 dInfo._open_barno = _schedule_times;
2044 strcpy(dInfo._opentag, userTag);
2045 pInfo._details.emplace_back(dInfo);
2046
2047 //这里还需要写一笔成交记录
2048 double fee = _replayer->calc_fee(stdCode, trdPx, abs(left), 0);
2049 _fund_info._total_fees += fee;
2050 log_trade(stdCode, dInfo._long, true, curTm, trdPx, abs(left), userTag, fee, _schedule_times);
