175void WTSLogger::initLogger(const char* catName, WTSVariant* cfgLogger)
176{
177 bool bAsync = cfgLogger->getBoolean("async");
178 const char* level = cfgLogger->getCString("level");
179
180 WTSVariant* cfgSinks = cfgLogger->get("sinks");
181 std::vector<spdlog::sink_ptr> sinks;
182 for (uint32_t idx = 0; idx < cfgSinks->size(); idx++)
183 {
184 WTSVariant* cfgSink = cfgSinks->get(idx);
185 const char* type = cfgSink->getCString("type");
186 if (strcmp(type, "daily_file_sink") == 0)
187 {
188 std::string filename = cfgSink->getString("filename");
189 StrUtil::replace(filename, "%s", catName);
190 checkDirs(filename.c_str());
191 auto sink = std::make_shared<spdlog::sinks::daily_file_sink_mt>(filename, 0, 0);
192 sink->set_pattern(cfgSink->getCString("pattern"));
193 sinks.emplace_back(sink);
194 }
195 else if (strcmp(type, "basic_file_sink") == 0)
196 {
197 std::string filename = cfgSink->getString("filename");
198 StrUtil::replace(filename, "%s", catName);
199 checkDirs(filename.c_str());
200 auto sink = std::make_shared<spdlog::sinks::basic_file_sink_mt>(filename, cfgSink->getBoolean("truncate"));
201 sink->set_pattern(cfgSink->getCString("pattern"));
202 sinks.emplace_back(sink);
203 }
204 else if (strcmp(type, "console_sink") == 0)
205 {
206 auto sink = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
207 sink->set_pattern(cfgSink->getCString("pattern"));
208 sinks.emplace_back(sink);
209 }
210 else if (strcmp(type, "ostream_sink") == 0)
211 {
212 auto sink = std::make_shared<spdlog::sinks::ostream_sink_mt>(std::cout, true);
213 sink->set_pattern(cfgSink->getCString("pattern"));
214 sinks.emplace_back(sink);
215 }
216 }
217
218 if (!bAsync)
219 {
220 auto logger = std::make_shared<spdlog::logger>(catName, sinks.begin(), sinks.end());
221 logger->set_level(str_to_level(cfgLogger->getCString("level")));
222 spdlog::register_logger(logger);
223 }
224 else
225 {
226 if(!m_bTpInited)
227 {
228 spdlog::init_thread_pool(8192, 2);
229 m_bTpInited = true;
230 }
231
232 auto logger = std::make_shared<spdlog::async_logger>(catName, sinks.begin(), sinks.end(), spdlog::thread_pool(), spdlog::async_overflow_policy::block);
233 logger->set_level(str_to_level(cfgLogger->getCString("level")));
234 spdlog::register_logger(logger);
235 }
236
237 if(strcmp(catName, "root")==0)
238 {
239 m_logLevel = str_to_ll(cfgLogger->getCString("level"));
240 }
241}

249void WTSLogger::init(const char* propFile /* = "logcfg.json" */, bool isFile /* = true */, ILogHandler* handler /* = NULL */)
250{
251 if (m_bInited)
252 return;
253
254 if (isFile && !StdFile::exists(propFile))
255 return;
256
257 WTSVariant* cfg = isFile ? WTSCfgLoader::load_from_file(propFile) : WTSCfgLoader::load_from_content(propFile, false);
258 if (cfg == NULL)
259 return;
260
261 auto keys = cfg->memberNames();
262 for (std::string& key : keys)
263 {
264 WTSVariant* cfgItem = cfg->get(key.c_str());
265 if (key == DYN_PATTERN)
266 {
267 auto pkeys = cfgItem->memberNames();
268 for(std::string& pkey : pkeys)
269 {
270 WTSVariant* cfgPattern = cfgItem->get(pkey.c_str());
271 if (m_mapPatterns == NULL)
272 m_mapPatterns = LogPatterns::create();
273
274 m_mapPatterns->add(pkey.c_str(), cfgPattern, true);
275 }
276 continue;
277 }
278
279 initLogger(key.c_str(), cfgItem);
280 }
281
282 m_rootLogger = getLogger("root");
283 if(m_rootLogger == NULL)
284 {
285 throw std::runtime_error("root logger can not be null, please check the config file");
286 }
287 spdlog::set_default_logger(m_rootLogger);
288 spdlog::flush_every(std::chrono::seconds(2));
289
290 m_logHandler = handler;
291
292 m_bInited = true;
293}

407void WTSLogger::log_raw(WTSLogLevel ll, const char* message)
408{
409 if (m_logLevel > ll || m_bStopped)
410 return;
411
412 if (!m_bInited)
413 {
414 print_message(message);
415 return;
416 }
417
418 auto logger = m_rootLogger;
419
420 if (logger)
421 {
422 switch (ll)
423 {
424 case LL_DEBUG:
425 debug_imp(logger, message); break;
426 case LL_INFO:
427 info_imp(logger, message); break;
428 case LL_WARN:
429 warn_imp(logger, message); break;
430 case LL_ERROR:
431 error_imp(logger, message); break;
432 case LL_FATAL:
433 fatal_imp(logger, message); break;
434 default:
435 break;
436 }
437 }

545SpdLoggerPtr WTSLogger::getLogger(const char* logger, const char* pattern /* = "" */)
546{
547 SpdLoggerPtr ret = spdlog::get(logger);
548 if (ret == NULL && strlen(pattern) > 0)
549 {
550 //当成动态的日志来处理
551 if (m_mapPatterns == NULL)
552 return SpdLoggerPtr();
553
554 WTSVariant* cfg = (WTSVariant*)m_mapPatterns->get(pattern);
555 if (cfg == NULL)
556 return SpdLoggerPtr();
557
558 initLogger(logger, cfg);
559
560 m_setDynLoggers.insert(logger);
561
562 return spdlog::get(logger);
563 }
564
565 return ret;
566}
