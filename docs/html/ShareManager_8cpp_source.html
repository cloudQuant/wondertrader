78bool ShareManager::start_watching(uint32_t microsecs)
79{
80 if (!_inited)
81 return false;
82
83 if (_inited && !_stopped && _worker == nullptr)
84 {
85 _worker.reset(new StdThread([this, microsecs]() {
86 while (!_stopped)
87 {
88 for(auto& v : _secnames)
89 {
90 if(_stopped)
91 break;
92
93 const char* section = v.first.c_str();
94 uint64_t& udtTime = (uint64_t&)v.second;
95
96 uint64_t lastUdtTime = _get_section_updatetime(_exchg.c_str(), section);
97 if(lastUdtTime > v.second)
98 {
99 //触发通知
100 _engine->notify_params_update(section);
101 udtTime = lastUdtTime;
102 }
103 }
104
105 //如果等待时间为0，则进入无限循环的检查中
106 if(microsecs > 0 && !_stopped)
107 std::this_thread::sleep_for(std::chrono::microseconds(microsecs));
108 }
109 }));
110
111 WTSLogger::info("Share domain is on watch");
112 }
113
114 return true;
115}
