64 inline uint32_t strToTime(const char* strTime)
65 {
66 static char str[10] = { 0 };
67 const char *pos = strTime;
68 int idx = 0;
69 auto len = strlen(strTime);
70 // 移除字符串中的冒号
71 for (std::size_t i = 0; i < len; i++)
72 {
73 if (strTime[i] != ':')
74 {
75 str[idx] = strTime[i];
76 idx++;
77 }
78 }
79 str[idx] = '\0';
80
81 // 将字符串转换为整数
82 return strtoul(str, NULL, 10);

349 bool UftLatencyTool::init()
350 {
351 // 初始化日志系统
352 WTSLogger::init("logcfg.yaml");
353
354 // 设置配置文件路径
355 std::string cfgFile = "config.yaml";
356
357 // 加载配置文件
358 WTSVariant* _config = WTSCfgLoader::load_from_file(cfgFile);
359 if (_config == NULL)
360 {
361 WTSLogger::error("Loading config file {} failed", cfgFile);
362 return false;
363 }
364
365 // 加载基础数据文件
366 WTSVariant* cfgBF = _config->get("basefiles");
367 // 加载交易时间段数据
368 if (cfgBF->get("session"))
369 _bd_mgr.loadSessions(cfgBF->getCString("session"));
370
371 // 加载商品数据
372 WTSVariant* cfgItem = cfgBF->get("commodity");
373 if (cfgItem)
374 {
375 // 单个商品文件
376 if (cfgItem->type() == WTSVariant::VT_String)
377 {
378 _bd_mgr.loadCommodities(cfgItem->asCString());
379 }
380 // 多个商品文件
381 else if (cfgItem->type() == WTSVariant::VT_Array)
382 {
383 for (uint32_t i = 0; i < cfgItem->size(); i++)
384 {
385 _bd_mgr.loadCommodities(cfgItem->get(i)->asCString());
386 }
387 }
388 }
389
390 // 加载合约数据
391 cfgItem = cfgBF->get("contract");
392 if (cfgItem)
393 {
394 // 单个合约文件
395 if (cfgItem->type() == WTSVariant::VT_String)
396 {
397 _bd_mgr.loadContracts(cfgItem->asCString());
398 }
399 // 多个合约文件
400 else if (cfgItem->type() == WTSVariant::VT_Array)
401 {
402 for (uint32_t i = 0; i < cfgItem->size(); i++)
403 {
404 _bd_mgr.loadContracts(cfgItem->get(i)->asCString());
405 }
406 }
407 }
408
409 // 设置模拟tick个数
410 _times = _config->getUInt32("times");
411 WTSLogger::warn("{} ticks will be simulated", _times);
412
413 // 设置测试线程绑定的CPU核心
414 _core = _config->getUInt32("core");
415 WTSLogger::warn("Testing thread will be bind to core {}", _core);
416
417 // 初始化交易引擎
418 initEngine(_config->get("env"));
419 // 初始化交易和解析模块
420 initModules();
421 // 初始化测试策略
422 initStrategies();
423
424 // 释放配置对象内存
425 _config->release();
426 return true;
427 }

511 void UftLatencyTool::run()
512 {
513 // 如果指定了CPU核心，则绑定当前线程到指定核心

515 {
516 // 注意核心编号从0开始，所以需要-1

519 WTSLogger::error("Binding to core {} failed", _core);

521 }
522
523 try
524 {
525 // 启动解析器和交易器
526 _parsers.run();
527 _traders.run();
528
529 // 启动UFT引擎
530 _engine.run();
531
532 // 运行测试解析器，生成指定数量的tick数据并计算延迟
533 theParser->run(_times);
534 }

537 // 捕捉并忽略所有异常
