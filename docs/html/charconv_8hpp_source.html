85 void init(const char *utf8_string)

87 if (0 == utf8_string)
88 t_string = 0;

91 needFree = false;

95 {
96 needFree = false;
97 t_string = (char *)utf8_string;
98 }
99 else
100 {
101 // Either TCHAR = Unicode (2 bytes), or utf8_string contains non-ASCII characters.

103 needFree = true;

106 std::size_t string_len = strlen(utf8_string);
107 std::size_t dst_len = string_len * 2 + 2;
108#ifdef _MSC_VER
109 wchar_t *buffer = new wchar_t[string_len + 1];
110 MultiByteToWideChar(CP_UTF8, 0, utf8_string, -1, buffer, (int)string_len + 1);
111 buffer[string_len] = 0;
112
113 t_string = new char[string_len * 2 + 2];
114 WideCharToMultiByte(CP_ACP, 0, buffer, -1, t_string, (int)dst_len, 0, 0);
115 t_string[string_len * 2 + 1] = 0;
116 delete[] buffer;
117#else
118 iconv_t cd;
119 t_string = new char[dst_len];
120 char* p = t_string;
121 cd = iconv_open("gb2312", "utf-8");
122 if (cd != 0)
123 {
124 memset(t_string, 0, dst_len);
125 iconv(cd, (char**)&utf8_string, &string_len, &p, &dst_len);

635 static bool isUtf8(unsigned char* data, std::size_t len) {
636 int num = 0;
637 std::size_t i = 0;
638 while (i < len) {
639 if ((data[i] & 0x80) == 0x00)
640 {
641 // 0XXX_XXXX
642 i++;
643 continue;
644 }
645 else if ((num = preNUm(data[i])) > 2)
646 {
647 // 110X_XXXX 10XX_XXXX
648 // 1110_XXXX 10XX_XXXX 10XX_XXXX
649 // 1111_0XXX 10XX_XXXX 10XX_XXXX 10XX_XXXX
650 // 1111_10XX 10XX_XXXX 10XX_XXXX 10XX_XXXX 10XX_XXXX
651 // 1111_110X 10XX_XXXX 10XX_XXXX 10XX_XXXX 10XX_XXXX 10XX_XXXX
652 // preNUm() 返回首个字节8个bits中首bit前面1bit的个数，该数量也是该字符所使用的字节数
653 i++;
654 for (int j = 0; j < num - 1; j++) {
655 //判断后面num - 1 个字节是不是都是10开
656 if ((data[i] & 0xc0) != 0x80) {
657 return false;
658 }
659 i++;
660 }
661 }
662 else
663 {
664 //其他情况说明不是utf-8
665 return false;
666 }
667 }
668 return true;
