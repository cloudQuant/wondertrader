110void WtStraDualThrust::on_session_begin(ICtaStraCtx* ctx, uint32_t uTDate)
111{
112 // 获取当前的主力合约代码
113 std::string newMonCode = ctx->stra_get_rawcode(_code.c_str());
114 // 如果主力合约发生变化
115 if(newMonCode!=_moncode)
116 {
117 // 如果已有之前的主力合约
118 if(!_moncode.empty())
119 {
120 // 获取当前持仓
121 double curPos = ctx->stra_get_position(_moncode.c_str());
122 // 如果有持仓，需要转移到新主力合约
123 if (!decimal::eq(curPos, 0))
124 {
125 ctx->stra_log_info(fmt::format("主力换月, 老主力{}[{}]将会被清理", _moncode, curPos).c_str());
126 // 将旧主力合约持仓清零
127 ctx->stra_set_position(_moncode.c_str(), 0, "switchout");
128 // 将相同持仓转移到新主力合约
129 ctx->stra_set_position(newMonCode.c_str(), curPos, "switchin");
130 }
131 }
132
133 // 更新当前主力合约代码
134 _moncode = newMonCode;
135 }

151void WtStraDualThrust::on_schedule(ICtaStraCtx* ctx, uint32_t curDate, uint32_t curTime)
152{
153 // 获取交易的合约代码
154 std::string code = _code;
155
156 // 获取K线数据
157 WTSKlineSlice *kline = ctx->stra_get_bars(code.c_str(), _period.c_str(), _count, true);
158 if(kline == NULL)
159 {
160 //这里可以输出一些日志
161 return;
162 }
163
164 // 检查K线数据是否为空
165 if (kline->size() == 0)
166 {
167 kline->release();
168 return;
169 }
170
171 // 设置交易单位，股票为100股，其他为1手
172 uint32_t trdUnit = 1;
173 if (_isstk)
174 trdUnit = 100;
175
176
177 // 转换回看天数为有符号整数
178 int32_t days = (int32_t)_days;
179
180 // 计算指定天数内的最高价和最低价
181 double hh = kline->maxprice(-days, -2);
182 double ll = kline->minprice(-days, -2);
183
184 // 计算指定天数内的最高收盘价和最低收盘价
185 WTSValueArray* closes = kline->extractData(KFT_CLOSE);
186 double hc = closes->maxvalue(-days, -2);
187 double lc = closes->minvalue(-days, -2);
188 double curPx = closes->at(-1); // 当前收盘价
189 closes->release();
190
191 // 获取当前周期K线的开盘价
192 double openPx = kline->at(-1)->open;
193
194 // 计算上下轨道
195 double upper_bound = openPx + _k1 * (std::max(hh - lc, hc - ll));
196 double lower_bound = openPx - _k2 * std::max(hh - lc, hc - ll);
197
198 //设置指标值，用于图表显示
199 ctx->set_index_value("DualThrust", "upper_bound", upper_bound);
200 ctx->set_index_value("DualThrust", "lower_bound", lower_bound);
201
202 // 获取合约信息
203 WTSCommodityInfo* commInfo = ctx->stra_get_comminfo(_code.c_str());
204
205 // 获取当前持仓并转换为手数
206 double curPos = ctx->stra_get_position(_moncode.c_str()) / trdUnit;
207
208 // 无持仓时的入场逻辑
209 if(decimal::eq(curPos,0))
210 {
211 // 如果当前价格突破上轨，做多
212 if(curPx >= upper_bound)
213 {
214 ctx->stra_enter_long(_moncode.c_str(), 2 * trdUnit, "DT_EnterLong");
215 //向上突破
216 ctx->stra_log_info(fmt::format("向上突破{}>={},多仓进场", curPx, upper_bound).c_str());
217
218 //添加图表标记
219 ctx->add_chart_mark(curPx, "wt-mark-buy", "DT_EnterLong");
220 }
221 // 如果当前价格突破下轨且不是股票，做空
222 else if (curPx <= lower_bound && !_isstk)
223 {
224 ctx->stra_enter_short(_moncode.c_str(), 2 * trdUnit, "DT_EnterShort");
225 //向下突破
226 ctx->stra_log_info(fmt::format("向下突破{}<={},空仓进场", curPx, lower_bound).c_str());
227
228 //添加图表标记
229 ctx->add_chart_mark(curPx, "wt-mark-sell", "DT_EnterShort");
230 }
231 }
232 // 持有多头仓位时的出场逻辑
233 else if (decimal::gt(curPos, 0))
234 {
235 // 如果当前价格突破下轨，多头出场
236 if(curPx <= lower_bound)
237 {
238 //多仓出场
239 ctx->stra_exit_long(_moncode.c_str(), 2 * trdUnit, "DT_ExitLong");
240 ctx->stra_log_info(fmt::format("向下突破{}<={},多仓出场", curPx, lower_bound).c_str());
241
242 //添加图表标记
243 ctx->add_chart_mark(curPx, "wt-mark-sell", "DT_ExitLong");
244 }
245 }
246 // 持有空头仓位时的出场逻辑
247 else if (decimal::lt(curPos, 0))
248 {
249 // 如果当前价格突破上轨且不是股票，空头出场
250 if (curPx >= upper_bound && !_isstk)
251 {
252 //空仓出场
253 ctx->stra_exit_short(_moncode.c_str(), 2 * trdUnit, "DT_ExitShort");
254 ctx->stra_log_info(fmt::format("向上突破{}>={},空仓出场", curPx, upper_bound).c_str());
255
256 //添加图表标记
257 ctx->add_chart_mark(curPx, "wt-mark-buy", "DT_ExitShort");
258 }
259 }
260
261 //这个释放一定要做
262 kline->release();
263}
