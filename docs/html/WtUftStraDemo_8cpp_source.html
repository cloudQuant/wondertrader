144void WtUftStraDemo::on_tick(IUftStraCtx* ctx, const char* code, WTSTickData* newTick)
145{
146 // 如果不是监控的品种，则直接返回
147 if (_code.compare(code) != 0)
148 return;
149
150 // 如果有待成交的订单，先检查是否超时
151 if (!_orders.empty())
152 {
153 check_orders();
154 return;
155 }
156
157 // 如果交易通道未就绪，则等待
158 if (!_channel_ready)
159 return;
160
161 // 获取当前品种的最新Tick，这里只是为了测试接口
162 WTSTickData* curTick = ctx->stra_get_last_tick(code);
163 if (curTick)
164 curTick->release(); // 释放内存
165
166 // 提取Tick数据的分钟时间
167 uint32_t curMin = newTick->actiontime() / 100000; //actiontime是带毫秒的,要取得分钟,则需要除以10w
168 if (curMin > _last_calc_time)
169 {//如果spread上次计算的时候小于当前分钟,则重算spread
170 //WTSKlineSlice* kline = ctx->stra_get_bars(code, "m5", 30);
171 //if (kline)
172 // kline->release();
173
174 //重算晚了以后,更新计算时间
175 _last_calc_time = curMin;
176 }
177
178 // 频率控制，要求相邻入场时间间隔至少为_freq毫秒
179 uint64_t now = TimeUtils::makeTime(ctx->stra_get_date(), ctx->stra_get_time() * 100000 + ctx->stra_get_secs());
180 if(now - _last_entry_time <= _freq * 1000)
181 {
182 return;
183 }
184
185 // 交易信号，1为看多，-1为看空，0为没有信号
186 int32_t signal = 0;
187 // 获取当前最新价
188 double price = newTick->price();
189 // 计算理论价格，这里采用加权平均价格计算方法
190 double pxInThry = (newTick->bidprice(0)*newTick->askqty(0) + newTick->askprice(0)*newTick->bidqty(0)) / (newTick->bidqty(0) + newTick->askqty(0));
191
192 // 当理论价格大于实际价格时，生成做多信号
193 if (pxInThry > price)
194 {
195 // 正向信号（做多）
196 signal = 1;
197 }
198 // 当理论价格小于实际价格时，生成做空信号
199 else if (pxInThry < price)
200 {
201 // 反向信号（做空）
202 signal = -1;
203 }
204
205 // 如果有交易信号生成则处理
206 if (signal != 0)
207 {
208 // 获取当前品种的持仓
209 double curPos = ctx->stra_get_position(code);
210
211 // 获取品种信息，用于查询价格步长等数据
212 WTSCommodityInfo* cInfo = ctx->stra_get_comminfo(code);
213
214 // 处理做多信号
215 if(signal > 0 && decimal::le(curPos, 0))
216 {//正向信号,且当前仓位小于等于0
217 // 计算目标下单价格，在最新价基础上加上偏移跳数
218 double targetPx = price + cInfo->getPriceTick() * _offset;
219
220 // 发送买入指令
221 auto ids = ctx->stra_buy(code, targetPx, _lots, UFT_OrderFlag_Nor);
222
223 {
224 // 将新生成的订单ID加入监控列表
225 _mtx_ords.lock();
226 for (uint32_t localid : ids)
227 _orders.insert(localid);
228 _mtx_ords.unlock();
229 // 更新最后入场时间
230 _last_entry_time = now;
231 }
232
233 }
234 // 处理做空信号
235 else if (signal < 0 && decimal::ge(curPos, 0))
236 {//反向信号,且当前仓位大于0,或者仓位为0但不是股票,或者仓位为0但是基础仓位有修正
237 // 计算目标下单价格，在最新价基础上减去偏移跳数
238 double targetPx = price - cInfo->getPriceTick()*_offset;
239
240 // 发送卖出指令
241 auto ids = ctx->stra_sell(code, targetPx, _lots, UFT_OrderFlag_Nor);
242
243 {
244 // 将新生成的订单ID加入监控列表
245 _mtx_ords.lock();
246 for (uint32_t localid : ids)
247 _orders.insert(localid);
248 _mtx_ords.unlock();
249 // 更新最后入场时间
250 _last_entry_time = now;
251 }
252 }
253 }

260void WtUftStraDemo::check_orders()
261{
262 // 如果有待成交的订单且最后入场时间有效
263 if (!_orders.empty() && _last_entry_time != UINT64_MAX)
264 {
265 // 获取当前时间
266 uint64_t now = TimeUtils::makeTime(_ctx->stra_get_date(), _ctx->stra_get_time() * 100000 + _ctx->stra_get_secs());
267 // 如果超过设定的超时时间（_secs秒），则撤销所有订单
268 if (now - _last_entry_time >= _secs * 1000) //如果超过一定时间没有成交完,则撤销
269 {
270 // 加锁以保护并发安全
271 _mtx_ords.lock();
272 // 遍历并撤销所有订单
273 for (auto localid : _orders)
274 {
275 _ctx->stra_cancel(localid); // 撤销订单
276 _cancel_cnt++; // 撤单计数器增加
277 // 输出日志，记录撤单信息
278 _ctx->stra_log_info(fmt::format("Order expired, cancelcnt updated to {}", _cancel_cnt).c_str());
279 }
280 // 释放锁
281 _mtx_ords.unlock();
282 }
283 }

352void WtUftStraDemo::on_order(IUftStraCtx* ctx, uint32_t localid, const char* stdCode, bool isLong, uint32_t offset, double totalQty, double leftQty, double price, bool isCanceled)
353{
354 //如果不是我发出去的订单,我就不管了
355 auto it = _orders.find(localid);
356 if (it == _orders.end())
357 return;
358
359 //如果已撤销或者剩余数量为0,则清除掉原有的id记录
360 if(isCanceled || leftQty == 0)
361 {
362 // 加锁以保护并发安全
363 _mtx_ords.lock();
364 // 从订单集合中移除该订单
365 _orders.erase(it);
366 // 如果有撤单计数，则减少计数并输出日志
367 if (_cancel_cnt > 0)
368 {
369 _cancel_cnt--;
370 _ctx->stra_log_info(fmt::format("cancelcnt -> {}", _cancel_cnt).c_str());
371 }
372 // 释放锁
373 _mtx_ords.unlock();
374 }
375}

382void WtUftStraDemo::on_channel_ready(IUftStraCtx* ctx)
383{
384 // 获取当前品种的未完成订单数量
385 double undone = _ctx->stra_get_undone(_code.c_str());
386 // 如果有未完成订单但不在当前监控列表中，则需要处理
387 if (!decimal::eq(undone, 0) && _orders.empty())
388 {
389 //这说明有未完成单不在监控之中,先撤掉
390 _ctx->stra_log_info(fmt::format("{}有不在管理中的未完成单 {} 手,全部撤销", _code, undone).c_str());
391
392 // 撤销所有未完成订单并获取撤单ID列表
393 OrderIDs ids = _ctx->stra_cancel_all(_code.c_str());
394 // 将撤单ID添加到监控列表中
395 for (auto localid : ids)
396 {
397 _orders.insert(localid);
398 }
399 // 更新撤单计数器
400 _cancel_cnt += ids.size();
401
402 // 输出日志记录撤单信息
403 _ctx->stra_log_info(fmt::format("cancelcnt -> {}", _cancel_cnt).c_str());
404 }
405
406 // 标记交易通道为就绪状态
407 _channel_ready = true;
408}
