53const char* getModuleName()

115const char* get_version()
116{
117 // 使用静态字符串缓存版本信息，避免重复构建
118 static std::string _ver;
119 if (_ver.empty())
120 {
121 // 构建版本信息字符串，包含平台名称、版本号和编译日期时间
122 _ver = PLATFORM_NAME;
123 _ver += " ";
124 _ver += WT_VERSION;
125 _ver += " Build@";
126 _ver += __DATE__;
127 _ver += " ";
128 _ver += __TIME__;
129 }
130 return _ver.c_str();
131}
