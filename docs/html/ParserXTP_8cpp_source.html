201void ParserXTP::OnTickByTick(XTPTBT *tbt_data)
202{
203 if (m_pBaseDataMgr == NULL)
204 {
205 return;
206 }
207
208 std::string exchg;
209 if (tbt_data->exchange_id == XTP_EXCHANGE_SH)
210 {
211 exchg = "SSE";
212 }
213 else
214 {
215 exchg = "SZSE";
216 }
217
218 WTSContractInfo* ct = m_pBaseDataMgr->getContract(tbt_data->ticker, exchg.c_str());
219 if (ct == NULL)
220 {
221 if (m_sink)
222 write_log(m_sink, LL_ERROR, "[ParserXTP] Instrument {}.{} not exists...", exchg.c_str(), tbt_data->ticker);
223 return;
224 }
225 WTSCommodityInfo* commInfo = ct->getCommInfo();
226
227 uint32_t actDate = (uint32_t)(tbt_data->data_time / 1000000000);
228 uint32_t actTime = tbt_data->data_time % 1000000000;
229 uint32_t actHour = actTime / 10000000;
230
231 if(tbt_data->type == XTP_TBT_ENTRUST)
232 {
233 WTSOrdDtlData *ordDtl = WTSOrdDtlData::create(tbt_data->ticker);
234 ordDtl->setContractInfo(ct);
235 WTSOrdDtlStruct& ts = ordDtl->getOrdDtlStruct();
236 strcpy(ts.exchg, commInfo->getExchg());
237
238 const XTPTickByTickEntrust& eInfo = tbt_data->entrust;
239
240 ts.trading_date = actDate;
241 ts.action_date = actDate;
242 ts.action_time = actTime;
243
244 ts.index = eInfo.seq;
245 ts.side = eInfo.side;
246 ts.otype = eInfo.ord_type;
247
248 ts.price = eInfo.price;
249 ts.volume = (uint32_t)eInfo.qty;
250
251 if (m_sink)
252 m_sink->handleOrderDetail(ordDtl);
253 }
254 else if (tbt_data->type == XTP_TBT_TRADE)
255 {
256 WTSTransData *trans = WTSTransData::create(tbt_data->ticker);
257 trans->setContractInfo(ct);
258 WTSTransStruct& ts = trans->getTransStruct();
259 strcpy(ts.exchg, commInfo->getExchg());
260
261 const XTPTickByTickTrade& tInfo = tbt_data->trade;
262
263 ts.trading_date = actDate;
264 ts.action_date = actDate;
265 ts.action_time = actTime;
266
267 ts.index = tInfo.seq;
268 ts.side = tInfo.trade_flag;
269 ts.ttype = tInfo.trade_flag == '4' ? TT_Cancel : TT_Match;
270
271 ts.price = tInfo.price;
272 ts.volume = (uint32_t)tInfo.qty;
273 ts.bidorder = tInfo.bid_no;
274 ts.askorder = tInfo.ask_no;
275
276 if (m_sink)
277 m_sink->handleTransaction(trans);
278 }

441void ParserXTP::DoLogin()
442{
443 if(m_pUserAPI == NULL)
444 {
445 return;
446 }
447
448 m_pUserAPI->SetHeartBeatInterval(m_uHBInterval);
449 m_pUserAPI->SetUDPBufferSize(m_uBuffSize);
450 int iResult = m_pUserAPI->Login(m_strHost.c_str(), m_iPort, m_strUser.c_str(), m_strPass.c_str(), m_iProtocol, m_strLocalIP.c_str());
451 if(iResult != 0)
452 {
453 if (m_sink)
454 {
455 auto error_info = m_pUserAPI->GetApiLastError();
456 if(iResult == -1)
457 {
458 _asyncio.post([this, iResult] {
459 m_sink->handleEvent(WPE_Connect, iResult);
460 });
461
462 write_log(m_sink, LL_ERROR, "[ParserXTP] Connecting server failed: {}", error_info->error_msg);
463 }
464 else
465 {
466 m_sink->handleEvent(WPE_Connect, 0);
467 _asyncio.post([this, iResult] {
468 m_sink->handleEvent(WPE_Connect, 0);
469 });
470
471 write_log(m_sink, LL_ERROR, "[ParserXTP] Sending login request failed: {}", error_info->error_msg);
472 }
473 }
474 }
475 else
476 {
477 m_uTradingDate = strToTime(m_pUserAPI->GetTradingDay());
478 _asyncio.post([this] {
479 if (m_sink)
480 {
481 m_sink->handleEvent(WPE_Connect, 0);
482 m_sink->handleEvent(WPE_Login, 0);
483 }
484 });
485
486
487 write_log(m_sink, LL_INFO, "[ParserXTP] Connecting server successed: {}, begin to subscibe data ...", iResult);
488
489 DoSubscribeMD();
490 }
