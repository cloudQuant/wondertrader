157void WtDiffMinImpactExeUnit::on_channel_ready()
158{
159 double undone = _ctx->getUndoneQty(_code.c_str());
160
161 if(!decimal::eq(undone, 0) && !_orders_mon.has_order())
162 {
163 /*
164 * 如果未完成单不为0，而OMS没有订单
165 * 这说明有未完成单不在监控之中,全部撤销掉
166 * 因为这些订单没有本地订单号，无法直接进行管理
167 * 这种情况，就是刚启动的时候，上次的未完成单或者外部的挂单
168 */
169 _ctx->writeLog(fmtutil::format("Unmanaged live orders with qty {} of {} found, cancel all", undone, _code.c_str()));
170
171 bool isBuy = (undone > 0);
172 OrderIDs ids = _ctx->cancel(_code.c_str(), isBuy);
173 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime());
174 _cancel_cnt += ids.size();
175
176 _ctx->writeLog(fmtutil::format("[{}@{}]cancelcnt -> {}", __FILE__, __LINE__, _cancel_cnt));
177 }
178 else if (decimal::eq(undone, 0) && _orders_mon.has_order())
179 {
180 /*
181 * By Wesey @ 2021.12.13
182 * 如果未完成单为0，但是OMS中是有订单的
183 * 说明OMS中是错单，需要清理掉，不然超时撤单就会出错
184 * 这种情况，一般是断线重连以后，之前下出去的订单，并没有真正发送到柜台
185 * 所以这里需要清理掉本地订单
186 */
187 _ctx->writeLog(fmtutil::format("Local orders of {} not confirmed in trading channel, clear all", _code.c_str()));
188 _orders_mon.clear_orders();
189 }
190 else
191 {
192 _ctx->writeLog(fmtutil::format("Unrecognized condition while channle ready, {:.2f} live orders of {} exists, local orders {}exist",
193 undone, _code.c_str(), _orders_mon.has_order() ? "" : "not "));
194 }
195
196
197 do_calc();
198}

214void WtDiffMinImpactExeUnit::on_tick(WTSTickData* newTick)
215{
216 if (newTick == NULL || _code.compare(newTick->code()) != 0)
217 return;
218
219 //如果原来的tick不为空,则要释放掉
220 if (_last_tick)
221 {
222 _last_tick->release();
223 }
224 else
225 {
226 //如果行情时间不在交易时间,这种情况一般是集合竞价的行情进来,下单会失败,所以直接过滤掉这笔行情
227 if (_sess_info != NULL && !_sess_info->isInTradingTime(newTick->actiontime() / 100000))
228 return;
229 }
230
231 //新的tick数据,要保留
232 _last_tick = newTick;
233 _last_tick->retain();
234
235 /*
236 * 这里可以考虑一下
237 * 如果写的上一次丢出去的单子不够达到目标仓位
238 * 那么在新的行情数据进来的时候可以再次触发核心逻辑
239 */
240
241 if(_expire_secs != 0 && _orders_mon.has_order() && _cancel_cnt==0)
242 {
243 uint64_t now = _ctx->getCurTime();
244
245 _orders_mon.check_orders(_expire_secs, now, [this](uint32_t localid) {
246 if (_ctx->cancel(localid))
247 {
248 _cancel_cnt++;
249 _ctx->writeLog(fmtutil::format("[{}@{}] Expired order of {} canceled, cancelcnt -> {}", __FILE__, __LINE__, _code.c_str(), _cancel_cnt));
250 }
251 });
252 }
253
254 do_calc();
255}

309void WtDiffMinImpactExeUnit::do_calc()
310{
311 CalcFlag flag(&_in_calc);
312 if (flag)
313 {
314 _ctx->writeLog(fmtutil::format("Duplicated calculating, DiffMinImpactExeUnit of {}", _code));
315 return;
316 }
317
318 if (_cancel_cnt != 0)
319 {
320 _ctx->writeLog(fmtutil::format("In Cancelling, DiffMinImpactExeUnit of {}", _code));
321 return;
322 }
323
324 //这里加一个锁，主要原因是实盘过程中发现
325 //在修改目标仓位的时候，会触发一次do_calc
326 //而ontick也会触发一次do_calc，两次调用是从两个线程分别触发的，所以会出现同时触发的情况
327 //如果不加锁，就会引起问题
328 //这种情况在原来的SimpleExecUnit没有出现，因为SimpleExecUnit只在set_position的时候触发
329 StdUniqueLock lock(_mtx_calc);
330
331 const char* stdCode = _code.c_str();
332
333 double undone = _ctx->getUndoneQty(stdCode);
334 double diffPos = _left_diff;
335
336 //因为是逐笔发单，所以如果有不需要撤销的未完成单，则暂不发单
337 if (!decimal::eq(undone, 0))
338 {
339 _ctx->writeLog(fmtutil::format("Live orders exist, DiffMinImpactExeUnit of {}", _code));
340 return;
341 }
342
343 if (decimal::eq(diffPos, 0))
344 return;
345
346 if (_last_tick == NULL)
347 {
348 _ctx->writeLog(fmtutil::format("No lastest tick data of {}, execute later", _code.c_str()));
349 return;
350 }
351
352 //检查下单时间间隔
353 uint64_t now = TimeUtils::makeTime(_last_tick->actiondate(), _last_tick->actiontime());
354 if (now - _last_place_time < _entrust_span)
355 return;
356
357 bool isBuy = decimal::gt(diffPos, 0);
358
359 //如果相比上次没有更新的tick进来，则先不下单，防止开盘前集中下单导致通道被封
360 uint64_t curTickTime = (uint64_t)_last_tick->actiondate() * 1000000000 + _last_tick->actiontime();
361 if (curTickTime <= _last_tick_time)
362 {
363 _ctx->writeLog(fmtutil::format("No tick of {} updated, {} <= {}, execute later",
364 _code, curTickTime, _last_tick_time));
365 return;
366 }
367
368 _last_tick_time = curTickTime;
369
370 double this_qty = _order_lots;
371 if (_by_rate)
372 {
373 this_qty = isBuy ? _last_tick->askqty(0) : _last_tick->bidqty(0);
374 this_qty = round(this_qty*_qty_rate);
375 if (decimal::lt(this_qty, 1))
376 this_qty = 1;
377 }
378
379 //By Wesley @ 2022.09.13
380 //这里要对下单数量做一个修正
381 this_qty = min(this_qty, abs(diffPos));
382
383 //如果买入且有空头持仓，或者卖出且有多头持仓
384 //对单次下单做一个修正，保证平仓和开仓不会同时下单
385 double curPos = _ctx->getPosition(stdCode);
386 if((isBuy && decimal::lt(curPos, 0)) || (!isBuy && decimal::gt(curPos, 0)))
387 {
388 this_qty = min(this_qty, abs(curPos));
389 }
390
391 double buyPx, sellPx;
392 if (_price_mode == 2)
393 {
394 double mp = (_last_tick->bidqty(0) - _last_tick->askqty(0))*1.0 / (_last_tick->bidqty(0) + _last_tick->askqty(0));
395 bool isUp = (mp > 0);
396 if (isUp)
397 {
398 buyPx = _last_tick->askprice(0);
399 sellPx = _last_tick->askprice(0);
400 }
401 else
402 {
403 buyPx = _last_tick->bidprice(0);
404 sellPx = _last_tick->bidprice(0);
405 }
406
407 /*
408 * By Wesley @ 2022.03.07
409 * 如果最后价格为0，再做一个修正
410 */
411 if (decimal::eq(buyPx, 0.0))
412 buyPx = decimal::eq(_last_tick->price(), 0.0) ? _last_tick->preclose() : _last_tick->price();
413
414 if (decimal::eq(sellPx, 0.0))
415 sellPx = decimal::eq(_last_tick->price(), 0.0) ? _last_tick->preclose() : _last_tick->price();
416
417 buyPx += _comm_info->getPriceTick() * _cancel_times;
418 sellPx -= _comm_info->getPriceTick() * _cancel_times;
419 }
420 else
421 {
422 if (_price_mode == -1)
423 {
424 buyPx = _last_tick->bidprice(0);
425 sellPx = _last_tick->askprice(0);
426 }
427 else if (_price_mode == 0)
428 {
429 buyPx = _last_tick->price();
430 sellPx = _last_tick->price();
431 }
432 else if (_price_mode == 1)
433 {
434 buyPx = _last_tick->askprice(0);
435 sellPx = _last_tick->bidprice(0) - _comm_info->getPriceTick() * _price_offset;
436 }
437
438 /*
439 * By Wesley @ 2022.03.07
440 * 如果最后价格为0，再做一个修正
441 */
442 if (decimal::eq(buyPx, 0.0))
443 buyPx = decimal::eq(_last_tick->price(), 0.0)? _last_tick->preclose(): _last_tick->price();
444
445 if (decimal::eq(sellPx, 0.0))
446 sellPx = decimal::eq(_last_tick->price(), 0.0) ? _last_tick->preclose() : _last_tick->price();
447
448 buyPx += _comm_info->getPriceTick() * _price_offset;
449 sellPx -= _comm_info->getPriceTick() * _price_offset;
450 }
451
452
453 //检查涨跌停价
454 bool isCanCancel = true;
455 if (!decimal::eq(_last_tick->upperlimit(), 0) && decimal::gt(buyPx, _last_tick->upperlimit()))
456 {
457 _ctx->writeLog(fmtutil::format("Buy price {} of {} modified to upper limit price", buyPx, _code.c_str(), _last_tick->upperlimit()));
458 buyPx = _last_tick->upperlimit();
459 isCanCancel = false; //如果价格被修正为涨跌停价，订单不可撤销
460 }
461
462 if (!decimal::eq(_last_tick->lowerlimit(), 0) && decimal::lt(sellPx, _last_tick->lowerlimit()))
463 {
464 _ctx->writeLog(fmtutil::format("Sell price {} of {} modified to lower limit price", sellPx, _code.c_str(), _last_tick->lowerlimit()));
465 sellPx = _last_tick->lowerlimit();
466 isCanCancel = false; //如果价格被修正为涨跌停价，订单不可撤销
467 }
468
469 if (isBuy)
470 {
471 OrderIDs ids = _ctx->buy(stdCode, buyPx, this_qty, false);
472 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime(), isCanCancel);
473 }
474 else
475 {
476 OrderIDs ids = _ctx->sell(stdCode, sellPx, this_qty, false);
477 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime(), isCanCancel);
478 }
479
480 _last_place_time = now;
481}
