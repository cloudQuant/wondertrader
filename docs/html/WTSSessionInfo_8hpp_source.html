135 /*
136 * 将时间转换成分钟数
137 * @uTime 当前时间,格式如0910
138 * @autoAdjust 是否自动调整,如果开启,非交易时间内的行情,会自动对齐到下一个交易时间,如8点59分的行情,会自动算作9点00的行情
139 * 会不会有别的影响,暂时无法确定,主要是担心非交易时间里收到错误数据
140 * 但是有接收时间控制,应该没问题
141 */
142 uint32_t timeToMinutes(uint32_t uTime, bool autoAdjust = false)
143 {
144 if(m_tradingTimes.empty())
145 return INVALID_UINT32;
146
147 if(isInAuctionTime(uTime))
148 return 0;
149
150 uint32_t offTime = offsetTime(uTime, true);
151
152 uint32_t offset = 0;
153 bool bFound = false;
154 auto it = m_tradingTimes.begin();
155 for(; it != m_tradingTimes.end(); it++)
156 {
157 TradingSection &section = *it;
158 if (section.first <= offTime && offTime <= section.second)
159 {
160 int32_t hour = offTime / 100 - section.first / 100;
161 int32_t minute = offTime % 100 - section.first % 100;
162 offset += hour*60 + minute;
163 bFound = true;
164 break;
165 }
166 else if(offTime > section.second) //大于上边界
167 {
168 int32_t hour = section.second/100 - section.first/100;
169 int32_t minute = section.second%100 - section.first%100;
170 offset += hour*60 + minute;
171 }
172 else //小于下边界
173 {
174 if(autoAdjust)
175 {
176 bFound = true;
177 }
178 break;
179 }
180 }
181
182 //没找到就返回0
183 if(!bFound)
184 return INVALID_UINT32;
185
186 return offset;
187 }

189 uint32_t minuteToTime(uint32_t uMinutes, bool bHeadFirst = false)
190 {
191 if(m_tradingTimes.empty())
192 return INVALID_UINT32;
193
194 uint32_t offset = uMinutes;
195 TradingTimes::iterator it = m_tradingTimes.begin();
196 for(; it != m_tradingTimes.end(); it++)
197 {
198 TradingSection &section = *it;
199 uint32_t startMin = section.first/100*60 + section.first%100;
200 uint32_t stopMin = section.second/100*60 + section.second%100;
201
202 if(!bHeadFirst)
203 {
204 if (startMin + offset >= stopMin)
205 {
206 offset -= (stopMin - startMin);
207 if (offset == 0)
208 {
209 return originalTime(stopMin / 60 * 100 + stopMin % 60);
210 }
211 }
212 else
213 {
214 //干好位于该区间
215 uint32_t desMin = startMin + offset;
216 if (desMin >= 1440)
217 desMin -= 1440;
218
219 return originalTime(desMin / 60 * 100 + desMin % 60);
220 }
221 }
222 else
223 {
224 if (startMin + offset < stopMin)
225 {
226 //干好位于该区间
227 uint32_t desMin = startMin + offset;
228 if (desMin >= 1440)
229 desMin -= 1440;
230
231 return originalTime(desMin / 60 * 100 + desMin % 60);
232 }
233 else
234 {
235 offset -= (stopMin - startMin);
236 }
237 }
238 }

243 uint32_t timeToSeconds(uint32_t uTime)
244 {
245 if(m_tradingTimes.empty())
246 return INVALID_UINT32;
247
248 //如果是集合竞价的价格,则认为是0秒价格
249 if(isInAuctionTime(uTime/100))
250 return 0;
251
252 uint32_t sec = uTime%100;
253 uint32_t h = uTime/10000;
254 uint32_t m = uTime%10000/100;
255 uint32_t offMin = offsetTime(h*100 + m, true);
256 h = offMin/100;
257 m = offMin%100;
258 uint32_t seconds = h*60*60 + m*60 + sec;
259
260 uint32_t offset = 0;
261 bool bFound = false;
262 TradingTimes::iterator it = m_tradingTimes.begin();
263 for(; it != m_tradingTimes.end(); it++)
264 {
265 TradingSection &section = *it;
266 uint32_t startSecs = (section.first/100*60 + section.first%100)*60;
267 uint32_t stopSecs = (section.second/100*60 + section.second%100)*60;
268 //uint32_t s = section.first;
269 //uint32_t e = section.second;
270 //uint32_t hour = (e/100 - s/100);
271 //uint32_t minute = (e%100 - s%100);
272 if(startSecs <= seconds && seconds <= stopSecs)
273 {
274 offset += seconds-startSecs;
275 if(seconds == stopSecs)
276 offset--;
277 bFound = true;
278 break;
279 }
280 else
281 {
282 offset += stopSecs - startSecs;
283 }
284 }
285
286 //没找到就返回0
287 if(!bFound)
288 return INVALID_UINT32;
289
290 return offset;
291 }

293 uint32_t secondsToTime(uint32_t seconds)
294 {
295 if(m_tradingTimes.empty())
296 return INVALID_UINT32;
297
298 uint32_t offset = seconds;
299 TradingTimes::iterator it = m_tradingTimes.begin();
300 for(; it != m_tradingTimes.end(); it++)
301 {
302 TradingSection &section = *it;
303 uint32_t startSecs = (section.first/100*60 + section.first%100)*60;
304 uint32_t stopSecs = (section.second/100*60 + section.second%100)*60;
305
306 if(startSecs + offset >= stopSecs)
307 {
308 offset -= (stopSecs-startSecs);
309 if(offset == 0)
310 {
311 uint32_t desMin = stopSecs/60;
312 return originalTime((desMin/60*100 + desMin%60))*100 + stopSecs%60;
313 }
314 }
315 else
316 {
317 //干好位于该区间
318 uint32_t desSecs = startSecs+offset;
319 if(desSecs >= 86400)
320 desSecs -= 86400;
321
322 uint32_t desMin = desSecs/60;
323 return originalTime((desMin/60*100 + desMin%60))*100 + desSecs%60;
324 }
325 }
326
327 return INVALID_UINT32;

498 /*
499 * 计算偏移时间
500 * @uTime 原始时间
501 * @bAlignLeft 是否向左对齐，这个主要针对0点结束的情况
502 * 如果向左对齐，则0点就做0点算
503 * 如果向右对齐，则0点就做24点算
504 */
505 inline uint32_t offsetTime(uint32_t uTime, bool bAlignLeft) const
506 {
507 if (m_uOffsetMins == 0)
508 return uTime;
509
510 int32_t curMinute = (uTime/100)*60 + uTime%100;
511 curMinute += m_uOffsetMins;
512 if(bAlignLeft)
513 {
514 if (curMinute >= 1440)
515 curMinute -= 1440;
516 else if (curMinute < 0)
517 curMinute += 1440;
518 }
519 else
520 {
521 if (curMinute > 1440)
522 curMinute -= 1440;
523 else if (curMinute <= 0)
524 curMinute += 1440;
525 }
526
527 return (curMinute/60)*100 + curMinute%60;
528 }

538 curMinute -= 1440;

540 curMinute += 1440;
