80bool EventNotifier::init(WTSVariant* cfg)
81{
82 // 检查是否启用事件通知器
83 if (!cfg->getBoolean("active"))
84 return false;
85
86 // 获取消息队列服务器URL
87 _url = cfg->getCString("url");
88 std::string module = DLLHelper::wrap_module("WtMsgQue", "lib");
89 //先看工作目录下是否有对应模块
90 std::string dllpath = WtHelper::getCWD() + module;
91 //如果没有,则再看模块目录,即dll同目录下
92 if (!StdFile::exists(dllpath.c_str()))
93 dllpath = WtHelper::getInstDir() + module;
94
95 // 加载消息队列动态库
96 DllHandle dllInst = DLLHelper::load_library(dllpath.c_str());
97 if (dllInst == NULL)
98 {
99 WTSLogger::error("MQ module {} loading failed", dllpath.c_str());
100 return false;
101 }
102
103 // 获取创建服务器函数
104 _creator = (FuncCreateMQServer)DLLHelper::get_symbol(dllInst, "create_server");
105 if (_creator == NULL)
106 {
107 DLLHelper::free_library(dllInst);
108 WTSLogger::error("MQ module {} is not compatible", dllpath.c_str());
109 return false;
110 }
111
112 // 获取其他函数指针
113 _remover = (FuncDestroyMQServer)DLLHelper::get_symbol(dllInst, "destroy_server");
114 _publisher = (FundPublishMessage)DLLHelper::get_symbol(dllInst, "publish_message");
115 _register = (FuncRegCallbacks)DLLHelper::get_symbol(dllInst, "regiter_callbacks");
116
117 // 注册日志回调函数
118 _register(on_mq_log);
119
120 // 创建消息队列服务器
121 _mq_sid = _creator(_url.c_str());
122
123 WTSLogger::info("EventNotifier initialized with channel {}", _url.c_str());
124
125 // 创建并启动工作线程
126 if (_worker == NULL)
127 {
128 // 创建一个work对象保证io_service不会因为没有任务而退出
129 boost::asio::io_service::work work(_asyncio);
130 _worker.reset(new StdThread([this]() {
131 while (!_stopped)
132 {
133 // 定期运行一个异步任务
134 std::this_thread::sleep_for(std::chrono::milliseconds(2));
135 _asyncio.run_one();
136 //m_asyncIO.run();
137 }
138 }));
139 }
140
141 return true;
142}

150void EventNotifier::notify_log(const char* tag, const char* message)
151{
152 // 检查消息队列服务器是否初始化
153 if (_mq_sid == 0)
154 return;
155
156 // 复制字符串，避免异步调用中的生命周期问题
157 std::string strTag = tag;
158 std::string strMsg = message;
159 // 将任务添加到异步IO服务中
160 _asyncio.post([this, strTag, strMsg]() {
161 std::string data;
162 {
163 // 创建JSON对象
164 rj::Document root(rj::kObjectType);
165 rj::Document::AllocatorType &allocator = root.GetAllocator();
166
167 // 添加标签、时间和消息内容
168 root.AddMember("tag", rj::Value(strTag.c_str(), allocator), allocator);
169 root.AddMember("time", TimeUtils::getLocalTimeNow(), allocator);
170 root.AddMember("message", rj::Value(strMsg.c_str(), allocator), allocator);
171
172 // 将JSON对象转换为格式化的字符串
173 rj::StringBuffer sb;
174 rj::PrettyWriter<rj::StringBuffer> writer(sb);
175 root.Accept(writer);
176
177 data = sb.GetString();
178 }
179
180 // 发送到消息队列，主题为"LOG"
181 if (_publisher)
182 _publisher(_mq_sid, "LOG", data.c_str(), (unsigned long)data.size());
183 });
184}
