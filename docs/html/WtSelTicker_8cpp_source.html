117void WtSelRtTicker::on_tick(WTSTickData* curTick, uint32_t hotFlag /* = 0 */)
118{
119 // 如果监控线程未启动，直接触发行情处理
120 if (_thrd == NULL)
121 {
122 trigger_price(curTick, hotFlag);
123 return;
124 }
125
126 // 获取行情数据的日期和时间
127 uint32_t uDate = curTick->actiondate();
128 uint32_t uTime = curTick->actiontime();
129
130 // 检查行情时间是否小于当前本地时间，如果是则直接处理行情
131 if (_date != 0 && (uDate < _date || (uDate == _date && uTime < _time)))
132 {
133 //WTSLogger::info("行情时间{}小于本地时间{}", uTime, _time);
134 trigger_price(curTick, hotFlag);
135 return;
136 }
137
138 // 更新当前日期和时间
139 _date = uDate;
140 _time = uTime;
141
142 // 处理分钟级别的时间信息
143 // 将时间拆分为分钟和秒数部分
144 uint32_t curMin = _time / 100000;
145 uint32_t curSec = _time % 100000;
146 // 将当前分钟时间转换为相对分钟数
147 uint32_t minutes = _s_info->timeToMinutes(curMin);
148 // 检查是否是交易时段的最后一分钟
149 bool isSecEnd = _s_info->isLastOfSection(curMin);
150 if (isSecEnd)
151 {
152 minutes--;
153 }
154 minutes++;
155 uint32_t rawMin = curMin; // 保存原始分钟时间
156 // 将相对分钟数转换回分钟时间格式
157 curMin = _s_info->minuteToTime(minutes);
158
159 // 如果当前位置是0，表示第一次设置
160 if (_cur_pos == 0)
161 {
162 // 直接设置当前分钟位置
163 _cur_pos = minutes;
164 }
165 // 如果新的分钟数大于当前分钟数，表示时间前进，需要触发旧分钟的闭合事件
166 else if (_cur_pos < minutes)
167 {
168 // 如果已记录的分钟小于新的分钟，则需要触发闭合事件
169 // 这个时候要先触发闭合，再修改平台时间和价格
170 if (_last_emit_pos < _cur_pos)
171 {
172 // 触发数据回放模块
173 StdUniqueLock lock(_mtx); // 加锁保护
174
175 // 优先修改时间标记
176 _last_emit_pos = _cur_pos;
177
178 // 获取当前分钟的时间格式
179 uint32_t thisMin = _s_info->minuteToTime(_cur_pos);
180
181 // 记录日志，分钟线已经闭合
182 WTSLogger::info("Minute Bar {}.{:04d} Closed by data", _date, thisMin);
183 // 通知数据存储器分钟结束
184 if (_store)
185 _store->onMinuteEnd(_date, thisMin);
186
187 // 触发选股引擎的分钟结束事件
188 _engine->on_minute_end(_date, thisMin);
189
190 // 计算时间偏移量，检查是否是交易时段的结束
191 uint32_t offMin = _s_info->offsetTime(thisMin, true);
192 if (offMin == _s_info->getCloseTime(true))
193 {
194 // 触发交易时段结束事件
195 _engine->on_session_end();
196 }
197 }
198
199 // 触发新行情的处理
200 trigger_price(curTick, hotFlag);
201 // 更新选股引擎的时间和交易日期
202 if (_engine)
203 {
204 _engine->set_date_time(_date, curMin, curSec, rawMin);
205 _engine->set_trading_date(curTick->tradingdate());
206 }
207
208 // 更新当前分钟位置
209 _cur_pos = minutes;
210 }
211 else
212 {
213 // 如果分钟数还是一致的，则直接触发行情和时间即可
214 trigger_price(curTick, hotFlag);
215 if (_engine)
216 _engine->set_date_time(_date, curMin, curSec, rawMin);
217 }
218
219 // 计算秒数和毫秒数
220 uint32_t sec = curSec / 1000;
221 uint32_t msec = curSec % 1000;
222 // 计算距离下一分钟还有多少毫秒
223 uint32_t left_ticks = (60 - sec) * 1000 - msec;
224 // 设置下次检查时间
225 _next_check_time = TimeUtils::getLocalTimeNow() + left_ticks;
226}

236void WtSelRtTicker::run()
237{
238 // 如果线程已经存在，直接返回
239 if (_thrd)
240 return;
241
242 // 从基础数据管理器获取当前交易日期
243 uint32_t curTDate = _engine->get_basedata_mgr()->calcTradingDate(_s_info->id(), _engine->get_date(), _engine->get_min_time(), true);
244 // 设置选股引擎的交易日期
245 _engine->set_trading_date(curTDate);
246
247 // 触发选股引擎的初始化事件
248 _engine->on_init();
249
250 // 触发交易时段开始事件
251 _engine->on_session_begin();
252
253 // 先检查当前时间
254 //uint32_t offTime = _s_info->offsetTime(_engine->get_min_time());
255
256 // 创建并启动监控线程
257 _thrd.reset(new StdThread([this](){
258 // 循环运行直到触发器停止
259 while (!_stopped)
260 {
261 // 如果当前时间有效并且在交易时间内
262 if (_time != UINT_MAX && _s_info->isInTradingTime(_time / 100000, true))
263 {
264 // 等待10毫秒再检查
265 std::this_thread::sleep_for(std::chrono::milliseconds(10));
266 // 获取当前时间
267 uint64_t now = TimeUtils::getLocalTimeNow();
268
269 // 检查是否到达下次检查时间并且最近的分钟结束还未触发
270 if (now >= _next_check_time && _last_emit_pos < _cur_pos)
271 {
272 // 触发数据回放模块
273 StdUniqueLock lock(_mtx);
274
275 // 优先修改时间标记
276 _last_emit_pos = _cur_pos;
277
278 // 获取当前分钟的时间格式
279 uint32_t thisMin = _s_info->minuteToTime(_cur_pos);
280 _time = thisMin;
281
282 // 如果thisMin是0，表示时间进入下一天
283 // 这里是本地计时导致的换日，需要自动将日期+1
284 // 同时因为时间是235959xxx，所以也要手动置为0
285 if (thisMin == 0)
286 {
287 // 记录旧日期
288 uint32_t lastDate = _date;
289 // 获取并设置下一天的日期
290 _date = TimeUtils::getNextDate(_date);
291 // 重置时间为0
292 _time = 0;
293 // 记录日期变更日志
294 WTSLogger::info("Data automatically changed at time 00:00: {} -> {}", lastDate, _date);
295 }
296
297 // 记录分钟线自动闭合日志
298 WTSLogger::info("Minute bar {}.{:04d} closed automatically", _date, thisMin);
299 // 通知数据存储器分钟结束
300 if (_store)
301 _store->onMinuteEnd(_date, thisMin);
302
303 // 触发选股引擎的分钟结束事件
304 _engine->on_minute_end(_date, thisMin);
305
306 // 计算时间偏移量，检查是否到达交易时段结束时间
307 uint32_t offMin = _s_info->offsetTime(thisMin, true);
308 if (offMin >= _s_info->getCloseTime(true))
309 {
310 // 触发交易时段结束事件
311 _engine->on_session_end();
312 }
313
314 // 145959000 - 设置时间，秒数置为0
315 if (_engine)
316 _engine->set_date_time(_date, thisMin, 0);
317 }
318 }
319 else
320 {// 如果不在交易时间，则每隐10毫秒检查一次，如果分钟发生变化则触发
321 // 等待10毫秒
322 std::this_thread::sleep_for(std::chrono::milliseconds(10));
323 // 获取当前分钟时间
324 uint32_t curTime = TimeUtils::getCurMin();
325 // 检查当前时间是否有效并且分钟发生变化
326 if (_time != UINT_MAX && curTime != _time)
327 {
328 // 触发旧分钟的结束事件
329 _engine->on_minute_end(_date, _time);
330 // 如果当前时间小于旧时间，说明日期已经变化，需要将日期增加1天
331 if (curTime < _time)
332 _date = TimeUtils::getNextDate(_date);
333 // 更新当前时间
334 _time = curTime;
335 }
336 }
337 }
338 }));
339}
