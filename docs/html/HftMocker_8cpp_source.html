520void HftMocker::on_tick(const char* stdCode, WTSTickData* newTick)
521{
522 _price_map[stdCode] = newTick->price();
523 {
524 std::unique_lock<std::recursive_mutex> lck(_mtx_control);
525 }
526
527 update_dyn_profit(stdCode, newTick);
528
529 OrderIDs all_ids;
530 for (auto it = _orders.begin(); it != _orders.end(); it++)
531 all_ids.push_back(it->first);
532 //如果开启了同tick撮合，则先触发策略的ontick，再处理订单
533 //如果没开启同tick撮合，则先处理订单，再触发策略的ontick
534 if (_match_this_tick)
535 {
536 if (_has_hook && _hook_valid)
537 {
538 WTSLogger::log_dyn("strategy", _name.c_str(), LL_DEBUG, "Waiting for resume notify");
539 StdUniqueLock lock(_mtx_calc);
540 _cond_calc.wait(_mtx_calc);
541 WTSLogger::log_dyn("strategy", _name.c_str(), LL_DEBUG, "Calc resumed");
542 _resumed = true;
543 }
544
545 on_tick_updated(stdCode, newTick);
546
547 procTask();
548
549 if (!_orders.empty())
550 {
551 StdLocker<StdRecurMutex> lock(_mtx_ords);
552 OrderIDs ids;
553 for (uint32_t localid : all_ids)
554 {
555 bool bNeedErase = procOrder(localid);
556 if (bNeedErase)
557 ids.emplace_back(localid);
558 }
559
560 for (uint32_t localid : ids)
561 {
562 _orders.erase(localid);
563 }
564 }
565 }
566 else
567 {
568 if (!_orders.empty())
569 {
570 StdLocker<StdRecurMutex> lock(_mtx_ords);
571 OrderIDs ids;
572 for (uint32_t localid : all_ids)
573 {
574 bool bNeedErase = procOrder(localid);
575 if (bNeedErase)
576 ids.emplace_back(localid);
577 }
578
579 for (uint32_t localid : ids)
580 {
581 auto it = _orders.find(localid);
582 _orders.erase(it);
583 }
584 }
585
586 if (_has_hook && _hook_valid)
587 {
588 WTSLogger::log_dyn("strategy", _name.c_str(), LL_DEBUG, "Waiting for resume notify");
589 StdUniqueLock lock(_mtx_calc);
590 _cond_calc.wait(_mtx_calc);
591 WTSLogger::log_dyn("strategy", _name.c_str(), LL_DEBUG, "Calc resumed");
592 _resumed = true;
593 }
594
595 on_tick_updated(stdCode, newTick);
596
597 procTask();
598 }
599
600 if (_has_hook && _hook_valid)
601 {
602 WTSLogger::log_dyn("strategy", _name.c_str(), LL_DEBUG, "Calc done, notify control thread");
603 while (_resumed)
604 _cond_calc.notify_all();
605 }

1026{

1028 if (it != _pos_map.end())

1033 pInfo._dynprofit = 0;
1034 }
1035 else
1036 {

1038 double price = isLong ? newTick->bidprice(0) : newTick->askprice(0);

1046 dInfo._profit = dInfo._volume*(price - dInfo._price)*commInfo->getVolScale()*(dInfo._long ? 1 : -1);
1047 if (dInfo._profit > 0)

1049 else if (dInfo._profit < 0)
1050 dInfo._max_loss = min(dInfo._profit, dInfo._max_loss);
1051
1052 dynprofit += dInfo._profit;
1053 }
1054
1055 pInfo._dynprofit = dynprofit;
1056 }

1073bool HftMocker::procOrder(uint32_t localid)
1074{
1075 auto it = _orders.find(localid);
1076 if (it == _orders.end())
1077 return false;
1078
1079 OrderInfoPtr ordInfo = it->second;
1080
1081 //第一步,如果在撤单概率中,则执行撤单
1082 if(_error_rate>0 && genRand(10000)<=_error_rate)
1083 {
1084 on_order(localid, ordInfo->_code, ordInfo->_isBuy, ordInfo->_total, ordInfo->_left, ordInfo->_price, true, ordInfo->_usertag);
1085 log_info("Random error order: {}", localid);
1086 return true;
1087 }
1088 else if(!ordInfo->_proced_after_placed)
1089 {
1090 //如果下单以后，还没处理过，则触发on_order
1091 on_order(localid, ordInfo->_code, ordInfo->_isBuy, ordInfo->_total, ordInfo->_left, ordInfo->_price, false, ordInfo->_usertag);
1092 ordInfo->_proced_after_placed = true;
1093 }
1094
1095 WTSTickData* curTick = stra_get_last_tick(ordInfo->_code);
1096 if (curTick == NULL)
1097 return false;
1098
1099 double curPx = curTick->price();
1100 double orderQty = ordInfo->_isBuy ? curTick->askqty(0) : curTick->bidqty(0); //看对手盘的数量
1101 if (decimal::eq(orderQty, 0.0))
1102 return false;
1103
1104 if (!_use_newpx)
1105 {
1106 curPx = ordInfo->_isBuy ? curTick->askprice(0) : curTick->bidprice(0);
1107 //if (curPx == 0.0)
1108 if(decimal::eq(curPx, 0.0))
1109 {
1110 curTick->release();
1111 return false;
1112 }
1113 }
1114 curTick->release();
1115
1116 //如果没有成交条件,则退出逻辑
1117 if(!decimal::eq(ordInfo->_price, 0.0))
1118 {
1119 if(ordInfo->_isBuy && decimal::gt(curPx, ordInfo->_price))
1120 {
1121 //买单,但是当前价大于限价,不成交
1122 return false;
1123 }
1124
1125 if (!ordInfo->_isBuy && decimal::lt(curPx, ordInfo->_price))
1126 {
1127 //卖单,但是当前价小于限价,不成交
1128 return false;
1129 }

1751void HftMocker::do_set_position(const char* stdCode, double qty, double price /* = 0.0 */, const char* userTag /*= ""*/)
1752{
1753 PosInfo& pInfo = _pos_map[stdCode];
1754 double curPx = price;
1755 if (decimal::eq(price, 0.0))
1756 curPx = _price_map[stdCode];
1757 uint64_t curTm = (uint64_t)_replayer->get_date() * 1000000000 + (uint64_t)_replayer->get_min_time()*100000 + _replayer->get_secs();
1758 uint32_t curTDate = _replayer->get_trading_date();
1759
1760 //手数相等则不用操作了
1761 if (decimal::eq(pInfo._volume, qty))
1762 return;
1763
1764 log_debug("[{:04d}.{:05d}] {} position updated: {} -> {}", _replayer->get_min_time(), _replayer->get_secs(), stdCode, pInfo._volume, qty);
1765
1766 WTSCommodityInfo* commInfo = _replayer->get_commodity_info(stdCode);
1767 if (commInfo == NULL)
1768 return;
1769
1770 //成交价
1771 double trdPx = curPx;
1772
1773 double diff = qty - pInfo._volume;
1774 bool isBuy = decimal::gt(diff, 0.0);
1775 if (decimal::gt(pInfo._volume*diff, 0))//当前持仓和仓位变化方向一致, 增加一条明细, 增加数量即可
1776 {
1777 pInfo._volume = qty;
1778 //如果T+1，则冻结仓位要增加
1779 if (commInfo->isT1())
1780 {
1781 //ASSERT(diff>0);
1782 pInfo._frozen += diff;
1783 log_debug("{} frozen position up to {}", stdCode, pInfo._frozen);
1784 }
1785
1786 DetailInfo dInfo;
1787 dInfo._long = decimal::gt(qty, 0);
1788 dInfo._price = trdPx;
1789 dInfo._volume = abs(diff);
1790 dInfo._opentime = curTm;
1791 dInfo._opentdate = curTDate;
1792 strcpy(dInfo._usertag, userTag);
1793 pInfo._details.emplace_back(dInfo);
1794
1795 double fee = _replayer->calc_fee(stdCode, trdPx, abs(diff), 0);
1796 _fund_info._total_fees += fee;
1797
1798 log_trade(stdCode, dInfo._long, true, curTm, trdPx, abs(diff), fee, userTag);
1799 }
1800 else
1801 {//持仓方向和仓位变化方向不一致,需要平仓
1802 double left = abs(diff);
1803
1804 pInfo._volume = qty;
1805 if (decimal::eq(pInfo._volume, 0))
1806 pInfo._dynprofit = 0;
1807 uint32_t count = 0;
1808 for (auto it = pInfo._details.begin(); it != pInfo._details.end(); it++)
1809 {
1810 DetailInfo& dInfo = *it;
1811 double maxQty = min(dInfo._volume, left);
1812 if (decimal::eq(maxQty, 0))
1813 continue;
1814
1815 double maxProf = dInfo._max_profit * maxQty / dInfo._volume;
1816 double maxLoss = dInfo._max_loss * maxQty / dInfo._volume;
1817
1818 dInfo._volume -= maxQty;
1819 left -= maxQty;
1820
1821 if (decimal::eq(dInfo._volume, 0))
1822 count++;
1823
1824 double profit = (trdPx - dInfo._price) * maxQty * commInfo->getVolScale();
1825 if (!dInfo._long)
1826 profit *= -1;
1827 pInfo._closeprofit += profit;
1828 pInfo._dynprofit = pInfo._dynprofit*dInfo._volume / (dInfo._volume + maxQty);//浮盈也要做等比缩放
1829 _fund_info._total_profit += profit;
1830
1831 double fee = _replayer->calc_fee(stdCode, trdPx, maxQty, dInfo._opentdate == curTDate ? 2 : 1);
1832 _fund_info._total_fees += fee;
1833 //这里写成交记录
1834 log_trade(stdCode, dInfo._long, false, curTm, trdPx, maxQty, fee, userTag);
1835 //这里写平仓记录
1836 log_close(stdCode, dInfo._long, dInfo._opentime, dInfo._price, curTm, trdPx, maxQty, profit, maxProf, maxLoss, pInfo._closeprofit, dInfo._usertag, userTag);
1837
1838 if (left == 0)
1839 break;
1840 }
1841
1842 //需要清理掉已经平仓完的明细
1843 while (count > 0)
1844 {
1845 auto it = pInfo._details.begin();
1846 pInfo._details.erase(it);
1847 count--;
1848 }
1849
1850 //最后,如果还有剩余的,则需要反手了
1851 if (left > 0)
1852 {
1853 left = left * qty / abs(qty);
1854
1855 //如果T+1，则冻结仓位要增加
1856 if (commInfo->isT1())
1857 {
1858 pInfo._frozen += left;
1859 log_debug("{} frozen position up to {}", stdCode, pInfo._frozen);
1860 }
1861
1862 DetailInfo dInfo;
1863 dInfo._long = decimal::gt(qty, 0);
1864 dInfo._price = trdPx;
1865 dInfo._volume = abs(left);
1866 dInfo._opentime = curTm;
1867 dInfo._opentdate = curTDate;
1868 strcpy(dInfo._usertag, userTag);
1869 pInfo._details.emplace_back(dInfo);
1870
1871 //这里还需要写一笔成交记录
1872 double fee = _replayer->calc_fee(stdCode, trdPx, abs(left), 0);
1873 _fund_info._total_fees += fee;
