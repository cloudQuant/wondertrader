39bool CsvReader::load_from_file(const char* filename)
40{
41 if (!StdFile::exists(filename))
42 return false;
43
44 _ifs.open(filename);
45
46 _ifs.getline(_buffer, 1024);
47 //判断是不是UTF-8BOM 编码
48 static char flag[] = { (char)0xEF, (char)0xBB, (char)0xBF };
49 char* buf = _buffer;
50 if (memcmp(_buffer, flag, sizeof(char) * 3) == 0)
51 buf += 3;
52
53 std::string row = buf;
54
55 //替换掉一些字段的特殊符号
56 StrUtil::replace(row, "<", "");
57 StrUtil::replace(row, ">", "");
58 StrUtil::replace(row, "\"", "");
59 StrUtil::replace(row, "'", "");
60
61 //将字段名转成小写
62 StrUtil::toLowerCase(row);
63
64 StringVector fields = StrUtil::split(row, _item_splitter.c_str());
65 for (uint32_t i = 0; i < fields.size(); i++)
66 {
67 StrUtil::trim(fields[i], " ");
68 StrUtil::trim(fields[i], "\n");
69 StrUtil::trim(fields[i], "\t");
70 StrUtil::trim(fields[i], "\r");
71 if (fields[i].empty())
72 break;
73
74 _fields_map[fields[i]] = i;
75 }
76
77 return true;
78}

86bool CsvReader::next_row()
