27bool IndexFactory::init(WTSVariant* config, IHotMgr* hotMgr, IBaseDataMgr* bdMgr, DataManager* dataMgr)
28{
29 _hot_mgr = hotMgr;

32
33 uint32_t poolsize = config->getUInt32("poolsize");
34 if (poolsize > 0)
35 {
36 _pool.reset(new boost::threadpool::pool(poolsize));
37 }
38
39 WTSVariant* cfgIdx = config->get("indice");
40 if(cfgIdx == NULL || !cfgIdx->isArray())
41 {
42 return false;
43 }
44
45 auto cnt = cfgIdx->size();
46 for(std::size_t i = 0; i < cnt; i++)

49 if(!cfgItem->getBoolean("active"))
50 continue;
51
52 IndexWorkerPtr worker(new IndexWorker(this));
53 if (!worker->init(cfgItem))
54 continue;
55
56 _workers.emplace_back(worker);
57 }
58
59 return true;
60}

69{

73 const char* fullCode = newTick->getContractInfo()->getFullCode();
74 auto it = _subbed.find(fullCode);
75 if (it == _subbed.end())

77
78 newTick->retain();

80 if(_pool)

83
84 for(IndexWorkerPtr& worker : _workers)
85 {
86 worker->handle_quote(newTick);
87 }
88 //这里加一个处理
89 newTick->release();
90 });
91 }
92 else
93 {
94 for (IndexWorkerPtr& worker : _workers)
