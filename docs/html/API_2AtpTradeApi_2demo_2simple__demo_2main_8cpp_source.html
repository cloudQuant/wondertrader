244// 建立连接
245ATPRetCodeType connect(ATPTradeAPI* client, ATPTradeHandler* handler)
246{
247 // 设置连接信息
248 ATPConnectProperty prop;
249 prop.user = "user3"; // 网关用户名
250 prop.password = "password3"; // 网关用户密码
251 prop.locations = { "10.16.131.223:32001", "127.0.0.1:32495" }; // 网关主备节点的地址+端口
252 prop.heartbeat_interval_milli = 5000; // 发送心跳的时间间隔，单位：毫秒
253 prop.connect_timeout_milli = 5000; // 连接超时时间，单位：毫秒
254 prop.reconnect_time = 10; // 重试连接次数
255 prop.client_name = "api_demo"; // 客户端程序名字
256 prop.client_version = "V1.0.0"; // 客户端程序版本
257 prop.report_sync = report_sync; // 回报同步数据分区号+序号，首次是空，断线重连时填入的是接受到的最新分区号+序号
258 prop.mode = 0; // 模式0-同步回报模式，模式1-快速登录模式，不同步回报
259
260 // 建立连接
261 while (!g_connected_flag.load())
262 {
263 // 在连接中
264 if (g_waiting_flag.load())
265 {
266 sleep(0);
267 }
268 else
269 {
270 g_waiting_flag.store(true);
271 // 建立连接
272 ATPRetCodeType ec = client->Connect(prop, handler);
273 if (ec != ErrorCode::kSuccess)
274 {
275 std::cout << "Invoke Connect error:" << ec << std::endl;
276 return ec;
277 }
278 sleep(0);
279 }
280 }
281
282 return kSuccess;
283}
