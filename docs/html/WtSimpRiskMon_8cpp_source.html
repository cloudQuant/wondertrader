84void WtSimpleRiskMon::run()
85{
86 if (_thrd)
87 return;
88
89 _thrd.reset(new std::thread([this](){
90 while (!_stopped)
91 {
92 if (_ctx && _ctx->isInTrading())
93 {
94 WTSPortFundInfo* fundInfo = _ctx->getFundInfo();
95 const WTSFundStruct& fs = fundInfo->fundInfo();
96 /*
97 * 条件1: 整体盘子的浮动收益比上一交易日结束时（收盘价计）, 增长 1% 以上
98 * 组合盘的动态权益 ≥ 上日收盘时的动态权益的 101%
99 * 条件2: 30min以内, 从今日高点回调到 80%以下
100 * 30min以内, 今日收益从高点回调到 80%以下
101 *
102 * 动作:
103 * 方式A: 所有品种减仓（减少到 30% 仓位）, 下一交易日重新按策略新仓位补齐
104 * 方式B: 所有盈利品种都 平仓, 下一交易日重新按策略新仓位补齐
105 */
106
107 if (_inner_day_active && fs._max_dyn_bal != DBL_MAX)
108 {
109 double predynbal = fundInfo->predynbalance() + _base_amount; //上日动态权益
110 double maxBal = fs._max_dyn_bal + _base_amount; //当日最大动态权益
111 double curBal = fs._balance + fs._dynprofit + _base_amount; //当前动态权益
112
113 double rate = 0.0;
114 if(!decimal::eq(maxBal, predynbal))
115 rate = (maxBal - curBal) * 100 / (maxBal - predynbal); //当日盈利回撤比例
116
117 //如果当日最大权益超过止盈边界条件
118 if (maxBal > (_basic_ratio*predynbal / 100.0))
119 {
120
121 /*
122 * 这里要转成日内分钟数处理
123 * 不然如果遇到午盘启动或早盘启动,
124 * 可能会因为中途休息时间过长, 而不触发风控
125 * 导致更大风险的发生
126 */
127 uint32_t maxTime = _ctx->transTimeToMin(fundInfo->max_dynbal_time());
128 uint32_t curTime = _ctx->transTimeToMin(_ctx->getCurTime()); //转成日内分钟数
129
130 if (rate >= _inner_day_fd && curTime - maxTime <= _risk_span && !_limited)
131 {
132 _ctx->writeRiskLog(fmt::format("Current IDD {:.2f}%, ≥MaxIDD {:.2f}%, Position down to {:.1f}%", rate, _inner_day_fd, _risk_scale).c_str());
133 _ctx->setVolScale(_risk_scale);
134 _limited = true;
135 }
136 else
137 {
138 _ctx->writeRiskLog(fmt::format("Current Balance Ratio: {:.2f}%, Current IDD: {:.2f}%", curBal*100.0 / predynbal, rate).c_str());
139 //_limited = false;
140 }
141 }
142 else
143 {
144 //如果当日最大权益没有超过盈利边界条件
145 _ctx->writeRiskLog(fmt::format("Current Balance Ratio: {:.2f}%", curBal*100.0 / predynbal).c_str());
146 //_limited = false;
147 }
148 }
149
150 if (_multi_day_active && fs._max_md_dyn_bal._date != 0)
151 {
152 double maxBal = fs._max_md_dyn_bal._dyn_balance + _base_amount;
153 double curBal = fs._balance + fs._dynprofit + _base_amount;
154
155 if (curBal < maxBal)
156 {
157 double rate = (maxBal - curBal) * 100 / maxBal;
158 if (rate >= _multi_day_fd)
159 {
160 _ctx->writeRiskLog(fmt::format("Current MDD {:.2f}%, >= MaxMDD {:.2f}%, Position down to 0.0%", rate, _multi_day_fd).c_str());
161 _ctx->setVolScale(0.0);
162 }
163 }
164 }
165 }
166
167 _last_time = TimeUtils::getLocalTimeNow();
168
169 while (!_stopped)
170 {
171 std::this_thread::sleep_for(std::chrono::milliseconds(2));
172 uint64_t now = TimeUtils::getLocalTimeNow();;
173 if(now - _last_time >= _calc_span*1000)
174 break;
175 }
176 }
177 }));
178}
