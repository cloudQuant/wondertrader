27/* A function can be defined using prototypes and compile on both ANSI C
28 and traditional C compilers with something like this:
29 extern char *func __P((char *, char *, int)); */
30
31#if defined (__STDC__) || defined(__cplusplus)
32
33# if !defined (__P)
34# define __P(protos) protos
35# endif
36
37# if !defined (__CONCAT1) && !defined (__CONCAT)
38# define __CONCAT1(x,y) x ## y
39# endif
40
41# if !defined (__CONCAT)
42# define __CONCAT(x,y) __CONCAT1(x,y)
43# endif
44
45/* stringify without expanding x */
46# if !defined (__STRING)
47# define __STRING(x) #x
48# endif
49
50/* expand x, then stringify */
51# if !defined (__XSTRING)
52# define __XSTRING(x) __STRING(x)
53# endif
54
55# if !defined (__GNUC__) && !defined(__cplusplus) && !defined(inline)
56# define inline
57# endif
58
59#else /* !__STDC__ */
60
61# if !defined (__P)
62# define __P(protos) ()
63# endif
64
65# if !defined (__CONCAT)
66# define __CONCAT(x,y) xy
67# endif
68
69# if !defined (__STRING)
70# define __STRING(x) "x"
71# endif
72
73# if !defined (__XSTRING)
74# define __XSTRING(x) __STRING(x)
75# endif
76
77#if defined (__GNUC__) /* gcc with -traditional */
78# if !defined (const)

80# endif
81# if !defined (inline)

83# endif
84# if !defined (signed)
85# define signed __signed
86# endif
87# if !defined (volatile)
88# define volatile __volatile
89# endif

91# if !defined (const)

103#endif /* !__GNUC__ */
104
105#endif /* !__STDC__ */
