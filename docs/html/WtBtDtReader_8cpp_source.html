139bool WtBtDtReader::read_raw_bars(const char* exchg, const char* code, WTSKlinePeriod period, std::string& buffer)
140{
141 std::stringstream ss;
142 ss << _base_dir << "his/" << PERIOD_NAME[period] << "/" << exchg << "/" << code << ".dsb";
143 std::string filename = ss.str();
144 if (!StdFile::exists(filename.c_str()))
145 {
146 pipe_btreader_log(_sink, LL_WARN, "Back {} data file {} not exists", PERIOD_NAME[period], filename);
147 return false;
148 }
149
150 pipe_btreader_log(_sink, LL_DEBUG, "Reading back {} bars from file {}...", PERIOD_NAME[period], filename);
151 StdFile::read_file_content(filename.c_str(), buffer);
152 bool bSucc = proc_block_data(buffer, true, false);
153 if(!bSucc)
154 pipe_btreader_log(_sink, LL_ERROR, "Processing back {} data from file {} failed", PERIOD_NAME[period], filename);
155
156 return bSucc;
157}
