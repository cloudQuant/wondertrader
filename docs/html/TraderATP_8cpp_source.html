160inline ATPOrdTypeType wrapOrdType(WTSPriceType priceType, WTSOrderFlag flag)
161{
162
163 if (WPT_LIMITPRICE == priceType && flag == WOF_NOR)
164 return ATPOrdTypeConst::kFixedNew; // 限价委托、增强限价
165
166 if (WPT_LIMITPRICE == priceType && flag == WOF_FOK)
167 return ATPOrdTypeConst::kFixedFullDealOrCancel; // 限价全额成交或撤销
168
169 if (WPT_ANYPRICE == priceType && flag == WOF_NOR)
170 return ATPOrdTypeConst::kMarketTransferFixed; // 市价剩余转限价
171
172 if (WPT_ANYPRICE == priceType && flag == WOF_FAK)
173 return ATPOrdTypeConst::kImmediateDealTransferCancel; // 立即成交剩余撤销
174
175 if (WPT_ANYPRICE == priceType && flag == WOF_FOK)
176 return ATPOrdTypeConst::kFullDealOrCancel; // 全额成交或撤销
177
178 return ATPOrdTypeConst::kDefault;
179}
