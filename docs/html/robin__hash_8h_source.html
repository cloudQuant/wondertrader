879 icloser_bucket = ito_move_closer_value -

881 std::size_t(m_buckets[ito_move_closer_value].dist_from_ideal_bucket()));

1208 const std::size_t hash = hash_key(key);
1209
1210 std::size_t ibucket = bucket_for_hash(hash);
1211 distance_type dist_from_ideal_bucket = 0;
1212
1213 while(dist_from_ideal_bucket <= m_buckets[ibucket].dist_from_ideal_bucket()) {
1214 if((!USE_STORED_HASH_ON_LOOKUP || m_buckets[ibucket].bucket_hash_equal(hash)) &&
1215 compare_keys(KeySelect()(m_buckets[ibucket].value()), key))
1216 {
1217 return std::make_pair(iterator(m_buckets + ibucket), false);
1218 }
1219
1220 ibucket = next_bucket(ibucket);
1221 dist_from_ideal_bucket++;
1222 }
1223
1224 if(rehash_on_extreme_load()) {
1225 ibucket = bucket_for_hash(hash);
1226 dist_from_ideal_bucket = 0;
1227
1228 while(dist_from_ideal_bucket <= m_buckets[ibucket].dist_from_ideal_bucket()) {
1229 ibucket = next_bucket(ibucket);
1230 dist_from_ideal_bucket++;
1231 }
1232 }
1233
1234
1235 if(m_buckets[ibucket].empty()) {
1236 m_buckets[ibucket].set_value_of_empty_bucket(dist_from_ideal_bucket, bucket_entry::truncate_hash(hash),
1237 std::forward<Args>(value_type_args)...);
1238 }

1240 insert_value(ibucket, dist_from_ideal_bucket, bucket_entry::truncate_hash(hash),
