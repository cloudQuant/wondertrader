84bool MQServer::init(const char* url, bool confirm /* = false */)
85{
86 // 如果已经初始化过，直接返回成功
87 if (_sock >= 0)
88 return true;
89
90 _confirm = confirm;
91
92 // 创建nanomsg PUB类型的套接字
93 _sock = nn_socket(AF_SP, NN_PUB);
94 if(_sock < 0)
95 {
96 _mgr->log_server(_id, fmt::format("MQServer {} has an error {} while initializing", _id, _sock).c_str());
97 return false;
98 }
99
100 // 设置发送缓冲区大小为8MB
101 int bufsize = 8 * 1024 * 1024;
102 nn_setsockopt(_sock, NN_SOL_SOCKET, NN_SNDBUF, &bufsize, sizeof(bufsize));
103
104 _url = url;
105 // 绑定套接字到指定的URL
106 if(nn_bind(_sock, url) < 0)
107 {
108 _mgr->log_server(_id, fmt::format("MQServer {} has an error while binding url {}", _id, url).c_str());
109 return false;
110 }
111 else
112 {
113 _mgr->log_server(_id, fmt::format("MQServer {} has binded to {} ", _id, url).c_str());
114 }
115
116 // 设置服务端为就绪状态
117 _ready = true;
118
119 _mgr->log_server(_id, fmt::format("MQServer {} ready", _id).c_str());
120 return true;
121}

131void MQServer::publish(const char* topic, const void* data, uint32_t dataLen)
132{
133 // 检查服务端是否已初始化
134 if(_sock < 0)
135 {
136 _mgr->log_server(_id, fmt::format("MQServer {} has not been initialized yet", _id).c_str());
137 return;
138 }
139
140 // 检查数据是否有效及线程是否已终止
141 if(data == NULL || dataLen == 0 || m_bTerminated)
142 return;
143
144 {
145 // 加锁并将消息添加到队列
146 StdUniqueLock lock(m_mtxCast);
147 m_dataQue.push(PubData(topic, data, dataLen));
148 m_bTimeout = false; // 重置超时标志
149 }
150
151 // 如果发布线程不存在，创建新的发布线程
152 if(m_thrdCast == NULL)
153 {
154 m_thrdCast.reset(new StdThread([this](){
155
156 // 初始化发送缓冲区
157 if (m_sendBuf.empty())
158 m_sendBuf.resize(1024 * 1024, 0); // 初始化为1MB
159
160 // 主循环，直到终止标志被设置
161 while (!m_bTerminated)
162 {
163 // 获取当前连接的客户端数量
164 int cnt = (int)nn_get_statistic(_sock, NN_STAT_CURRENT_CONNECTIONS);
165
166 // 如果队列为空或者在确认模式下没有客户端连接，等待
167 if(m_dataQue.empty() || (cnt == 0 && _confirm))
168 {
169 StdUniqueLock lock(m_mtxCast);
170 m_bTimeout = true;
171 // 等待60秒或者直到被通知
172 m_condCast.wait_for(lock, std::chrono::seconds(60));
173 //如果有新的数据进来，timeout会被改为false
174 //如果没有新的数据进来，timeout会保持为true
175 if (m_bTimeout)
176 {
177 //等待超时以后，广播心跳包
178 m_dataQue.push(PubData("HEARTBEAT", "", 0));
179 }
180 else
181 {
182 continue; // 被通知有新消息，继续循环
183 }
184 }
185
186 // 创建临时队列并交换数据，减少锁的持有时间
187 PubDataQue tmpQue;
188 {
189 StdUniqueLock lock(m_mtxCast);
190 tmpQue.swap(m_dataQue);
191 }
192
193 // 处理队列中的每个消息
194 while(!tmpQue.empty())
195 {
196 const PubData& pubData = tmpQue.front();
197
198 // 只发送非空消息（心跳包可能是空消息）
199 if (!pubData._data.empty())
200 {
201 // 计算消息包总长度
202 std::size_t len = sizeof(MQPacket) + pubData._data.size();
203
204 // 如果缓冲区不足，扩大缓冲区
205 if (m_sendBuf.size() < len)
206 m_sendBuf.resize(m_sendBuf.size() * 2);
207
208 // 构建消息包
209 MQPacket* pack = (MQPacket*)m_sendBuf.data();
210 strncpy(pack->_topic, pubData._topic.c_str(), 32);
211 pack->_length = (uint32_t)pubData._data.size();
212 memcpy(&pack->_data, pubData._data.data(), pubData._data.size());
213
214 // 发送消息，确保完整发送
215 int bytes_snd = 0;
216 for(;;)
217 {
218 int bytes = nn_send(_sock, m_sendBuf.data() + bytes_snd, len - bytes_snd, 0);
219 if (bytes >= 0)
220 {
221 bytes_snd += bytes;
222 if(bytes_snd == len) // 全部发送完成
223 break;
224 }
225 else
226 // 发送失败，等待一小段时间后重试
227 std::this_thread::sleep_for(std::chrono::milliseconds(1));
228 }
229
230 }
231 tmpQue.pop(); // 移除已处理的消息
232 }
233 }
234 }));
235 }
236 else
237 {
238 // 如果发布线程已存在，通知它处理新消息
239 m_condCast.notify_all();
240 }
241}
