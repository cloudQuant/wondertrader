281static __inline long
282SStr_FixedStrtoL10(const char *pValue, int32 len) {
283 char __tmpBuf[32];
284
285 SLOG_ASSERT(pValue && len < (int32) sizeof(__tmpBuf));
286 memcpy(__tmpBuf, pValue, len);
287 __tmpBuf[len] = '\0';
288
289 return strtol(__tmpBuf, (char **) NULL, 10);
290}
