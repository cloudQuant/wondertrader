239int32_t Connect()
240{
241 std::vector<std::string> locations;
242
243 locations.push_back(remote_address_master);
244 locations.push_back(remote_address_backup);
245
246 //设置连接信息
247 ATPConnectProperty prop;
248 prop.user = user;//登录AGW的账户
249 prop.password = password;//登录AGW的密码
250 prop.locations = locations;//AGW IP地址 格式为："ip:port"
251 prop.heartbeat_interval_milli = 5000;//发送心跳的时间间隔,单位：毫秒，默认值：5000
252 prop.connect_timeout_milli = 5000;//连接超时时间，单位：毫秒，默认值：5000
253 prop.reconnect_time = 10;// 重试连接次数，默认值：连接地址数,
254 prop.report_sync = report_sync;//回报同步数据map<int32_t, int32_t>，默认值：空
255
256
257 // API连接
258 ATPRetCodeType ec = atp_trade_api_client->Connect(prop, new DemoATPTradeHandler(g_login_success, g_exit));
259 if (ec != ErrorCode::kSuccess)
260 {
261 std::cout << "Connect error:" << ec << std::endl;
262 return -1;
263 }
264 return 0 ;
265}
