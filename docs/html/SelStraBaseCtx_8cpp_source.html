640void SelStraBaseCtx::update_dyn_profit(const char* stdCode, double price)
641{
642 auto it = _pos_map.find(stdCode);
643 if (it != _pos_map.end())
644 {
645 PosInfo& pInfo = (PosInfo&)it->second;
646 if (pInfo._volume == 0)
647 {
648 pInfo._dynprofit = 0;
649 }
650 else
651 {
652 WTSCommodityInfo* commInfo = _engine->get_commodity_info(stdCode);
653 double dynprofit = 0;
654 for (auto pit = pInfo._details.begin(); pit != pInfo._details.end(); pit++)
655 {
656 DetailInfo& dInfo = *pit;
657 dInfo._profit = dInfo._volume*(price - dInfo._price)*commInfo->getVolScale()*(dInfo._long ? 1 : -1);
658 if (dInfo._profit > 0)
659 dInfo._max_profit = std::max(dInfo._profit, dInfo._max_profit);
660 else if (dInfo._profit < 0)
661 dInfo._max_loss = std::min(dInfo._profit, dInfo._max_loss);
662
663 dInfo._max_price = std::max(dInfo._max_price, price);
664 dInfo._min_price = std::min(dInfo._min_price, price);
665
666 dynprofit += dInfo._profit;
667 }
668
669 pInfo._dynprofit = dynprofit;
670 }
671 }
672
673 double total_dynprofit = 0;
674 for (auto v : _pos_map)
675 {
676 const PosInfo& pInfo = v.second;
677 total_dynprofit += pInfo._dynprofit;
678 }
679
680 _fund_info._total_dynprofit = total_dynprofit;
681}

743bool SelStraBaseCtx::on_schedule(uint32_t curDate, uint32_t curTime, uint32_t fireTime)
744{
745 _schedule_date = curDate;
746 _schedule_time = curTime;
747
748 _is_in_schedule = true;//开始调度, 修改标记
749
750 //主要用于保存浮动盈亏的
751 save_data();
752
753 TimeUtils::Ticker ticker;
754 on_strategy_schedule(curDate, fireTime);
755 log_debug("Strategy {} scheduled @ {}", _context_id, curTime);
756
757 wt_hashset<std::string> to_clear;
758 for (auto& v : _pos_map)
759 {
760 const PosInfo& pInfo = v.second;
761 const char* code = v.first.c_str();
762 if (_sig_map.find(code) == _sig_map.end() && !decimal::eq(pInfo._volume, 0.0))
763 {
764 //新的信号中没有该持仓,则要清空
765 to_clear.insert(code);
766 }
767 }
768
769 for (const std::string& code : to_clear)
770 {
771 append_signal(code.c_str(), 0, "autoexit");
772 }
773
774 _emit_times++;
775 _total_calc_time += ticker.micro_seconds();
776
777 if (_emit_times % 20 == 0)
778 log_info("Strategy has been scheduled {} times, totally taking {} us, {:.3f} us each time",
779 _emit_times, _total_calc_time, _total_calc_time*1.0 / _emit_times);
780
781 if (_ud_modified)
782 {
783 save_userdata();
784 _ud_modified = false;
785 }
786
787 _is_in_schedule = false;//调度结束, 修改标记
788 return true;
789}

1016void SelStraBaseCtx::do_set_position(const char* stdCode, double qty, const char* userTag /* = "" */, bool bTriggered /* = false */)
1017{
1018 PosInfo& pInfo = _pos_map[stdCode];
1019 double curPx = _price_map[stdCode];
1020 uint64_t curTm = (uint64_t)_engine->get_date() * 10000 + _engine->get_min_time();
1021 uint32_t curTDate = _engine->get_trading_date();
1022
1023 if (decimal::eq(pInfo._volume, qty))
1024 return;
1025
1026 double diff = qty - pInfo._volume;
1027
1028 WTSCommodityInfo* commInfo = _engine->get_commodity_info(stdCode);
1029 if (commInfo == NULL)
1030 return;
1031
1032 //成交价
1033 double trdPx = curPx;
1034
1035 bool isBuy = decimal::gt(diff, 0.0);
1036 if (decimal::gt(pInfo._volume*diff, 0))//当前持仓和目标仓位方向一致, 增加一条明细, 增加数量即可
1037 {
1038 pInfo._volume = qty;
1039 //如果T+1，则冻结仓位要增加
1040 if (commInfo->isT1())
1041 {
1042 //ASSERT(diff>0);
1043 pInfo._frozen += diff;
1044 pInfo._frozen_date = curTDate;
1045 log_debug("{} frozen position updated to {}", stdCode, pInfo._frozen);
1046 }
1047
1048 if (_slippage != 0)
1049 {
1050 trdPx += _slippage * commInfo->getPriceTick()*(isBuy ? 1 : -1);
1051 }
1052
1053 DetailInfo dInfo;
1054 dInfo._long = decimal::gt(qty, 0);
1055 dInfo._price = trdPx;
1056 dInfo._max_price = trdPx;
1057 dInfo._min_price = trdPx;
1058 dInfo._volume = abs(diff);
1059 dInfo._opentime = curTm;
1060 dInfo._opentdate = curTDate;
1061 wt_strcpy(dInfo._opentag, userTag);
1062 pInfo._details.push_back(dInfo);
1063 pInfo._last_entertime = curTm;
1064
1065 double fee = commInfo->calcFee(trdPx, abs(qty), 0);
1066 _fund_info._total_fees += fee;
1067 //_engine->mutate_fund(fee, FFT_Fee);
1068 log_trade(stdCode, dInfo._long, true, curTm, trdPx, abs(qty), userTag, fee);
1069 }
1070 else
1071 {//持仓方向和目标仓位方向不一致, 需要平仓
1072 double left = abs(diff);
1073
1074 if (_slippage != 0)
1075 trdPx += _slippage * commInfo->getPriceTick()*(isBuy ? 1 : -1);
1076
1077 pInfo._volume = qty;
1078 if (decimal::eq(pInfo._volume, 0))
1079 pInfo._dynprofit = 0;
1080 uint32_t count = 0;
1081 for (auto it = pInfo._details.begin(); it != pInfo._details.end(); it++)
1082 {
1083 DetailInfo& dInfo = *it;
1084 double maxQty = min(dInfo._volume, left);
1085 //if (maxQty == 0)
1086 if (decimal::eq(maxQty, 0))
1087 continue;
1088
1089 dInfo._volume -= maxQty;
1090 left -= maxQty;
1091
1092 if (decimal::eq(dInfo._volume, 0))
1093 count++;
1094
1095 double profit = (trdPx - dInfo._price) * maxQty * commInfo->getVolScale();
1096 if (!dInfo._long)
1097 profit *= -1;
1098 pInfo._closeprofit += profit;
1099 pInfo._dynprofit = pInfo._dynprofit*dInfo._volume / (dInfo._volume + maxQty);//浮盈也要做等比缩放
1100 pInfo._last_exittime = curTm;
1101 _fund_info._total_profit += profit;
1102
1103 double fee = commInfo->calcFee(trdPx, maxQty, dInfo._opentdate == curTDate ? 2 : 1);
1104 _fund_info._total_fees += fee;
1105 //这里写成交记录
1106 log_trade(stdCode, dInfo._long, false, curTm, trdPx, maxQty, userTag, fee);
1107 //这里写平仓记录
1108 log_close(stdCode, dInfo._long, dInfo._opentime, dInfo._price, curTm, trdPx, maxQty, profit, pInfo._closeprofit, dInfo._opentag, userTag);
1109
1110 //if (left == 0)
1111 if (decimal::eq(left, 0))
1112 break;
1113 }
1114
1115 //需要清理掉已经平仓完的明细
1116 while (count > 0)
1117 {
1118 auto it = pInfo._details.begin();
1119 pInfo._details.erase(it);
1120 count--;
1121 }
1122
1123 //最后, 如果还有剩余的, 则需要反手了
1124 //if (left > 0)
1125 if (decimal::gt(left, 0))
1126 {
1127 left = left * qty / abs(qty);
1128
1129 //如果T+1，则冻结仓位要增加
1130 if (commInfo->isT1())
1131 {
1132 //ASSERT(diff>0);
1133 pInfo._frozen += diff;
1134 pInfo._frozen_date = curTDate;
1135 log_debug("{} frozen position updated to {}", stdCode, pInfo._frozen);
1136 }
1137
1138 DetailInfo dInfo;
1139 dInfo._long = decimal::gt(qty, 0);
1140 dInfo._price = trdPx;
1141 dInfo._max_price = trdPx;
1142 dInfo._min_price = trdPx;
1143 dInfo._volume = abs(left);
1144 dInfo._opentime = curTm;
1145 dInfo._opentdate = curTDate;
1146 wt_strcpy(dInfo._opentag, userTag);
1147 pInfo._details.push_back(dInfo);
1148 pInfo._last_entertime = curTm;
1149
1150 //这里还需要写一笔成交记录
1151 double fee = commInfo->calcFee(trdPx, abs(qty), 0);
1152 _fund_info._total_fees += fee;
1153 //_engine->mutate_fund(fee, FFT_Fee);
1154 log_trade(stdCode, dInfo._long, true, curTm, trdPx, abs(left), userTag, fee);
1155 }
1156 }
1157
1158 //存储数据
1159 save_data();
1160
1161 _engine->handle_pos_change(_name.c_str(), stdCode, diff);
1162}
