111bool ParserAdapter::init(const char* id, WTSVariant* cfg)
112{
113 if (cfg == NULL)
114 return false;
115
116 _id = id;
117
118 if (_cfg != NULL)
119 return false;
120
121 _cfg = cfg;
122 _cfg->retain();
123
124 {
125 //加载模块
126 if (cfg->getString("module").empty())
127 return false;
128
129 std::string module = DLLHelper::wrap_module(cfg->getCString("module"), "lib");;
130
131 if (!StdFile::exists(module.c_str()))
132 {
133 module = WtHelper::get_module_dir();
134 module += "parsers/";
135 module += DLLHelper::wrap_module(cfg->getCString("module"), "lib");
136 }
137
138 DllHandle hInst = DLLHelper::load_library(module.c_str());
139 if (hInst == NULL)
140 {
141 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser module {} loading failed", _id.c_str(), module.c_str());
142 return false;
143 }
144 else
145 {
146 WTSLogger::log_dyn("parser", _id.c_str(), LL_INFO, "[{}] Parser module {} loaded", _id.c_str(), module.c_str());
147 }
148
149 FuncCreateParser pFuncCreateParser = (FuncCreateParser)DLLHelper::get_symbol(hInst, "createParser");
150 if (NULL == pFuncCreateParser)
151 {
152 WTSLogger::log_dyn("parser", _id.c_str(), LL_FATAL, "[{}] Entrance function createParser not found", _id.c_str());
153 return false;
154 }
155
156 _parser_api = pFuncCreateParser();
157 if (NULL == _parser_api)
158 {
159 WTSLogger::log_dyn("parser", _id.c_str(), LL_FATAL, "[{}] Creating parser api failed", _id.c_str());
160 return false;
161 }
162
163 _remover = (FuncDeleteParser)DLLHelper::get_symbol(hInst, "deleteParser");
164 }
165
166
167 const std::string& strFilter = cfg->getString("filter");
168 if (!strFilter.empty())
169 {
170 const StringVector &ayFilter = StrUtil::split(strFilter, ",");
171 auto it = ayFilter.begin();
172 for (; it != ayFilter.end(); it++)
173 {
174 _exchg_filter.insert(*it);
175 }
176 }
177
178 std::string strCodes = cfg->getString("code");
179 if (!strCodes.empty())
180 {
181 const StringVector &ayCodes = StrUtil::split(strCodes, ",");
182 auto it = ayCodes.begin();
183 for (; it != ayCodes.end(); it++)
184 {
185 _code_filter.insert(*it);
186 }
187 }
188
189 if (_parser_api)
190 {
191 _parser_api->registerSpi(this);
192
193 if (_parser_api->init(cfg))
194 {
195 ContractSet contractSet;
196 if (!_code_filter.empty())//优先判断合约过滤器
197 {
198 ExchgFilter::iterator it = _code_filter.begin();
199 for (; it != _code_filter.end(); it++)
200 {
201 //全代码,形式如SSE.600000,期货代码为CFFEX.IF2005
202 std::string code, exchg;
203 auto ay = StrUtil::split((*it).c_str(), ".");
204 if (ay.size() == 1)
205 code = ay[0];
206 else if (ay.size() == 2)
207 {
208 exchg = ay[0];
209 code = ay[1];
210 }
211 else if (ay.size() == 3)
212 {
213 exchg = ay[0];
214 code = ay[2];
215 }
216 WTSContractInfo* contract = _bd_mgr->getContract(code.c_str(), exchg.c_str());
217 if (contract)
218 contractSet.insert(contract->getFullCode());
219 else
220 {
221 //如果是品种ID，则将该品种下全部合约都加到订阅列表
222 WTSCommodityInfo* commInfo = _bd_mgr->getCommodity(exchg.c_str(), code.c_str());
223 if (commInfo)
224 {
225 const auto& codes = commInfo->getCodes();
226 for (const auto& c : codes)
227 {
228 contractSet.insert(fmt::format("{}.{}", exchg, c.c_str()));
229 }
230 }
231 }
232 }
233 }
234 else if (!_exchg_filter.empty())
235 {
236 ExchgFilter::iterator it = _exchg_filter.begin();
237 for (; it != _exchg_filter.end(); it++)
238 {
239 const char* exchg = (*it).c_str();
240 WTSArray* ayContract = _bd_mgr->getContracts(exchg);
241 auto cnt = ayContract->size();
242 WTSArray::Iterator it = ayContract->begin();
243 for (; it != ayContract->end(); it++)
244 {
245 WTSContractInfo* contract = STATIC_CONVERT(*it, WTSContractInfo*);
246 contractSet.insert(contract->getFullCode());
247 }
248
249 ayContract->release();
250
251 WTSLogger::log_dyn("parser", _id.c_str(), LL_INFO, "[{}] {} contracts of {} added to sublist...", _id.c_str(), cnt, exchg);
252 }
253 }
254 else
255 {
256 WTSArray* ayContract = _bd_mgr->getContracts();
257 WTSArray::Iterator it = ayContract->begin();
258 for (; it != ayContract->end(); it++)
259 {
260 WTSContractInfo* contract = STATIC_CONVERT(*it, WTSContractInfo*);
261 contractSet.insert(contract->getFullCode());
262 }
263
264 ayContract->release();
265 }
266
267 _parser_api->subscribe(contractSet);
268 contractSet.clear();
269 }
270 else
271 {
272 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser initializing failed: api initializing failed...", _id.c_str());
273 }
274 }
275 else
276 {
277 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser initializing failed: creating api failed...", _id.c_str());
278 }
279
280 return true;
281}
