138WTSContractInfo* WTSBaseDataMgr::getContract(const char* code, const char* exchg /* = "" */, uint32_t uDate /* = 0 */)
139{
140 //如果直接找到对应的市场代码,则直接
141
142 // 将合约代码转换为字符串作为查询键
143 auto lKey = std::string(code);
144
145 // 如果没有指定交易所，则从合约映射容器中直接查找
146 if (strlen(exchg) == 0)
147 {
148 // 在合约映射容器中查找合约代码
149 auto it = m_mapContracts->find(lKey);
150 if (it == m_mapContracts->end())
151 return NULL;
152
153 // 获取合约数组
154 WTSArray* ayInst = (WTSArray*)it->second;
155 if (ayInst == NULL || ayInst->size() == 0)
156 return NULL;
157
158 // 遍历合约数组，找到符合条件的合约
159 for(std::size_t i = 0; i < ayInst->size(); i++)
160 {
161 WTSContractInfo* cInfo = (WTSContractInfo*)ayInst->at(i);
162 /*
163 * By Wesley @ 2023.10.23
164 * if param uDate is not zero, need to check whether contract is valid
165 */
166 // 如果指定了日期，检查合约在该日期是否有效
167 if (uDate == 0 || (cInfo->getOpenDate() <= uDate && cInfo->getExpireDate() >= uDate))
168 return cInfo;
169 }
170 return NULL;
171 }
172 else
173 {
174 // 如果指定了交易所，则先从交易所-合约映射容器中查找交易所
175 auto sKey = std::string(exchg);
176 auto it = m_mapExchgContract->find(sKey);
177 if (it != m_mapExchgContract->end())
178 {
179 // 获取交易所对应的合约列表
180 WTSContractList* contractList = (WTSContractList*)it->second;
181 // 在合约列表中查找指定的合约代码
182 auto it = contractList->find(lKey);
183 if (it != contractList->end())
184 {
185 WTSContractInfo* cInfo = (WTSContractInfo*)it->second;
186 /*
187 * By Wesley @ 2023.10.23
188 * if param uDate is not zero, need to check whether contract is valid
189 */
190 // 如果指定了日期，检查合约在该日期是否有效
191 if (uDate == 0 || (cInfo->getOpenDate() <= uDate && cInfo->getExpireDate() >= uDate))
192 return cInfo;
193 }
194
195 return NULL;
196 }
197 }
198
199 return NULL;
200}

363{
364 // 先检查是否为周末，周六和周日算作节假日

368

371 if (!isTpl)

377 {
378 // 获取模板并检查指定日期是否在节假日集合中
379 const TradingDayTpl& tpl = it->second;
380 return (tpl._holidays.find(uDate) != tpl._holidays.end());
381 }

425bool WTSBaseDataMgr::loadSessions(const char* filename)
426{
427 // 检查文件是否存在
428 if (!StdFile::exists(filename))
429 {
430 WTSLogger::error("Trading sessions configuration file {} not exists", filename);
431 return false;
432 }
433
434 // 加载配置文件
435 WTSVariant* root = WTSCfgLoader::load_from_file(filename);
436 if (root == NULL)
437 {
438 WTSLogger::error("Loading session config file {} failed", filename);
439 return false;
440 }
441
442 // 遍历所有交易时段定义
443 for(const std::string& id : root->memberNames())
444 {
445 WTSVariant* jVal = root->get(id);
446
447 // 获取时段名称和时区偏移
448 const char* name = jVal->getCString("name");
449 int32_t offset = jVal->getInt32("offset");
450
451 // 创建交易时段对象
452 WTSSessionInfo* sInfo = WTSSessionInfo::create(id.c_str(), name, offset);
453
454 // 处理集合端占配置，支持单个时段和多个时段两种格式
455 if (jVal->has("auction"))
456 {
457 WTSVariant* jAuc = jVal->get("auction");
458 sInfo->setAuctionTime(jAuc->getUInt32("from"), jAuc->getUInt32("to"));
459 }
460 else if (jVal->has("auctions"))
461 {
462 WTSVariant* jAucs = jVal->get("auctions");
463 for (uint32_t i = 0; i < jAucs->size(); i++)
464 {
465 WTSVariant* jSec = jAucs->get(i);
466 sInfo->addAuctionTime(jSec->getUInt32("from"), jSec->getUInt32("to"));
467 }
468 }
469
470 // 处理正常交易时段配置
471 WTSVariant* jSecs = jVal->get("sections");
472 if (jSecs == NULL || !jSecs->isArray())
473 continue;
474
475 // 遍历添加所有交易时段
476 for (uint32_t i = 0; i < jSecs->size(); i++)
477 {
478 WTSVariant* jSec = jSecs->get(i);
479 sInfo->addTradingSection(jSec->getUInt32("from"), jSec->getUInt32("to"));
480 }
481
482 // 将交易时段添加到全局映射中
483 m_mapSessions->add(id.c_str(), sInfo);
484 }
485
486 // 释放配置对象
487 root->release();
488
489 return true;
490}

500void parseCommodity(WTSCommodityInfo* pCommInfo, WTSVariant* jPInfo)
501{
502 // 设置最小变动单位
503 pCommInfo->setPriceTick(jPInfo->getDouble("pricetick"));
504 // 设置合约数量乘数
505 pCommInfo->setVolScale(jPInfo->getUInt32("volscale"));
506
507 // 设置合约类别，如期货、期权等，默认为期货
508 if (jPInfo->has("category"))
509 pCommInfo->setCategory((ContractCategory)jPInfo->getUInt32("category"));
510 else
511 pCommInfo->setCategory(CC_Future);
512
513 // 设置平仓模式
514 pCommInfo->setCoverMode((CoverMode)jPInfo->getUInt32("covermode"));
515 // 设置价格模式
516 pCommInfo->setPriceMode((PriceMode)jPInfo->getUInt32("pricemode"));
517
518 // 设置交易模式，默认为多空都可交易
519 if (jPInfo->has("trademode"))
520 pCommInfo->setTradingMode((TradingMode)jPInfo->getUInt32("trademode"));
521 else
522 pCommInfo->setTradingMode(TM_Both);
523
524 // 设置手数最小变动单位和最小交易手数
525 double lotsTick = 1;
526 double minLots = 1;
527 if (jPInfo->has("lotstick"))
528 lotsTick = jPInfo->getDouble("lotstick");
529 if (jPInfo->has("minlots"))
530 minLots = jPInfo->getDouble("minlots");
531 pCommInfo->setLotsTick(lotsTick);
532 pCommInfo->setMinLots(minLots);
533}

543bool WTSBaseDataMgr::loadCommodities(const char* filename)
544{
545 // 检查配置文件是否存在
546 if (!StdFile::exists(filename))
547 {
548 WTSLogger::error("Commodities configuration file {} not exists", filename);
549 return false;
550 }
551
552 // 加载配置文件
553 WTSVariant* root = WTSCfgLoader::load_from_file(filename);
554 if (root == NULL)
555 {
556 WTSLogger::error("Loading commodities config file {} failed", filename);
557 return false;
558 }
559
560 // 遍历所有交易所
561 for(const std::string& exchg : root->memberNames())
562 {
563 WTSVariant* jExchg = root->get(exchg);
564
565 // 遍历当前交易所下的所有品种
566 for (const std::string& pid : jExchg->memberNames())
567 {
568 WTSVariant* jPInfo = jExchg->get(pid);
569
570 // 获取品种名称、交易时段ID和节假日模板ID
571 const char* name = jPInfo->getCString("name");
572 const char* sid = jPInfo->getCString("session");
573 const char* hid = jPInfo->getCString("holiday");
574
575 // 确保交易时段配置有效
576 if (strlen(sid) == 0)
577 {
578 WTSLogger::warn("No session configured for {}.{}", exchg.c_str(), pid.c_str());
579 continue;
580 }
581
582 // 创建品种信息对象并进行属性设置
583 WTSCommodityInfo* pCommInfo = WTSCommodityInfo::create(pid.c_str(), name, exchg.c_str(), sid, hid);
584 parseCommodity(pCommInfo, jPInfo);
585
586 // 关联交易时段信息
587 WTSSessionInfo* sInfo = getSession(sid);
588 pCommInfo->setSessionInfo(sInfo);
589
590 // 生成品种标准键值并添加到品种映射
591 std::string key = fmt::format("{}.{}", exchg.c_str(), pid.c_str());
592 if (m_mapCommodities == NULL)
593 m_mapCommodities = WTSCommodityMap::create();
594
595 m_mapCommodities->add(key, pCommInfo, false);
596
597 // 建立交易时段和品种的双向关联
598 m_mapSessionCode[sid].insert(key);
599 }
600 }
601
602 WTSLogger::info("Commodities configuration file {} loaded", filename);
603 root->release();
604 return true;
605}

616bool WTSBaseDataMgr::loadContracts(const char* filename)
617{
618 // 检查配置文件是否存在
619 if (!StdFile::exists(filename))
620 {
621 WTSLogger::error("Contracts configuration file {} not exists", filename);
622 return false;
623 }
624
625 // 加载配置文件
626 WTSVariant* root = WTSCfgLoader::load_from_file(filename);
627 if (root == NULL)
628 {
629 WTSLogger::error("Loading contracts config file {} failed", filename);
630 return false;
631 }
632
633 for(const std::string& exchg : root->memberNames())
634 {
635 WTSVariant* jExchg = root->get(exchg);
636
637 for(const std::string& code : jExchg->memberNames())
638 {
639 WTSVariant* jcInfo = jExchg->get(code);
640
641 /*
642 * By Wesley @ 2021.12.28
643 * 这里做一个兼容，如果product为空,先检查是否配置了rules属性，如果配置了rules属性，把合约单独当成品种自动加入
644 * 如果没有配置rules，则直接跳过该合约
645 */
646 WTSCommodityInfo* commInfo = NULL;
647 std::string pid;
648 if(jcInfo->has("product"))
649 {
650 pid = jcInfo->getCString("product");
651 commInfo = getCommodity(jcInfo->getCString("exchg"), pid.c_str());
652 }
653 else if(jcInfo->has("rules"))
654 {
655 pid = code.c_str();
656 WTSVariant* jPInfo = jcInfo->get("rules");
657 const char* name = jcInfo->getCString("name");
658 std::string sid = jPInfo->getCString("session");
659 std::string hid;
660 if(jPInfo->has("holiday"))
661 hid = jPInfo->getCString("holiday");
662
663 //这里不能像解析commodity那样处理，直接赋值为ALLDAY
664 if (sid.empty())
665 sid = "ALLDAY";
666
667 commInfo = WTSCommodityInfo::create(pid.c_str(), name, exchg.c_str(), sid.c_str(), hid.c_str());
668 parseCommodity(commInfo, jPInfo);
669 WTSSessionInfo* sInfo = getSession(sid.c_str());
670 commInfo->setSessionInfo(sInfo);
671
672 std::string key = fmt::format("{}.{}", exchg.c_str(), pid.c_str());
673 if (m_mapCommodities == NULL)
674 m_mapCommodities = WTSCommodityMap::create();
675
676 m_mapCommodities->add(key, commInfo, false);
677
678 m_mapSessionCode[sid].insert(key);
679
680 WTSLogger::debug("Commodity {} has been automatically added", key.c_str());
681 }
682
683 if (commInfo == NULL)
684 {
685 WTSLogger::warn("Commodity {}.{} not found, contract {} skipped", jcInfo->getCString("exchg"), jcInfo->getCString("product"), code.c_str());
686 continue;
687 }
688
689 WTSContractInfo* cInfo = WTSContractInfo::create(code.c_str(),
690 jcInfo->getCString("name"),
691 jcInfo->getCString("exchg"),
692 pid.c_str());
693
694 cInfo->setCommInfo(commInfo);
695
696 uint32_t maxMktQty = 1000000;
697 uint32_t maxLmtQty = 1000000;
698 uint32_t minMktQty = 1;
699 uint32_t minLmtQty = 1;
700 if (jcInfo->has("maxmarketqty"))
701 maxMktQty = jcInfo->getUInt32("maxmarketqty");
702 if (jcInfo->has("maxlimitqty"))
703 maxLmtQty = jcInfo->getUInt32("maxlimitqty");
704 if (jcInfo->has("minmarketqty"))
705 minMktQty = jcInfo->getUInt32("minmarketqty");
706 if (jcInfo->has("minlimitqty"))
707 minLmtQty = jcInfo->getUInt32("minlimitqty");
708 cInfo->setVolumeLimits(maxMktQty, maxLmtQty, minMktQty, minLmtQty);
709
710 uint32_t opendate = 0;
711 uint32_t expiredate = 0;
712 if (jcInfo->has("opendate"))
713 opendate = jcInfo->getUInt32("opendate");
714 if (jcInfo->has("expiredate"))
715 expiredate = jcInfo->getUInt32("expiredate");
716 cInfo->setDates(opendate, expiredate);
717
718 double lMargin = 0;
719 double sMargin = 0;
720 if (jcInfo->has("longmarginratio"))
721 lMargin = jcInfo->getDouble("longmarginratio");
722 if (jcInfo->has("shortmarginratio"))
723 sMargin = jcInfo->getDouble("shortmarginratio");
724 cInfo->setMarginRatios(lMargin, sMargin);
725
726 WTSContractList* contractList = (WTSContractList*)m_mapExchgContract->get(std::string(cInfo->getExchg()));
727 if (contractList == NULL)
728 {
729 contractList = WTSContractList::create();
730 m_mapExchgContract->add(std::string(cInfo->getExchg()), contractList, false);
731 }
732 contractList->add(std::string(cInfo->getCode()), cInfo, false);
733
734 commInfo->addCode(code.c_str());
735
736 std::string key = std::string(cInfo->getCode());
737 WTSArray* ayInst = (WTSArray*)m_mapContracts->get(key);
738 if(ayInst == NULL)
739 {
740 ayInst = WTSArray::create();
741 m_mapContracts->add(key, ayInst, false);
742 }
743
744 ayInst->append(cInfo, true);
745 }
746 }
747
748 WTSLogger::info("Contracts configuration file {} loaded, {} exchanges", filename, m_mapExchgContract->size());
749 root->release();
750 return true;
751}

815uint64_t WTSBaseDataMgr::getBoundaryTime(const char* stdPID, uint32_t tDate, bool isSession /* = false */, bool isStart /* = true */)

821 // 设置节假日模板ID和待查询的交易时段信息
822 std::string tplid = stdPID;
823 bool isTpl = false;
824 WTSSessionInfo* sInfo = NULL;
825
826 // 如果传入的是交易时段ID
827 if (isSession)
828 {
829 sInfo = getSession(stdPID);
830 tplid = DEFAULT_HOLIDAY_TPL; // 使用默认的节假日模板
831 isTpl = true;
832 }
833 else // 如果传入的是标准品种ID
834 {
835 // 获取品种信息并检查其有效性
836 WTSCommodityInfo* cInfo = getCommodity(stdPID);
837 if (cInfo == NULL)
838 return 0;
839
840 // 从品种信息中获取相关的交易时段信息
841 sInfo = cInfo->getSessionInfo();
842 }
843
844 // 检查交易时段信息是否有效
845 if (sInfo == NULL)
846 return 0;
847
848 // 处理周末情况，周六(6)和周日(0)需要调整到最近的交易日
849 uint32_t weekday = TimeUtils::getWeekDay(tDate);
850 if (weekday == 6 || weekday == 0)
851 {
852 // 根据是获取开始时间还是结束时间，分别调整到下一个或上一个交易日
853 if (isStart)
854 tDate = getNextTDate(tplid.c_str(), tDate, 1, isTpl);

861 if (sInfo->getOffsetMins() == 0)

863 if (isStart)
864 return (uint64_t)tDate * 10000 + sInfo->getOpenTime();
865 else
866 return (uint64_t)tDate * 10000 + sInfo->getCloseTime();
867 }
868
869 // 情况2：负偏移的交易时段（往前偏移，一般用于外盘交易所）
870 if(sInfo->getOffsetMins() < 0)
871 {
872 // 往前偏移,就是交易日推后,一般用于外盘
873 // 这个比较简单,只需要按自然日取即可
874 if (isStart)
875 return (uint64_t)tDate * 10000 + sInfo->getOpenTime();
876 else
877 // 收盘时间在下一个自然日
878 return (uint64_t)TimeUtils::getNextDate(tDate) * 10000 + sInfo->getCloseTime();
879 }
880 // 情况3：正偏移的交易时段（往后偏移，如国内期货夜盘）
881 else
882 {
883 // 往后偏移,一般国内期货夜盘都是这个,即夜盘算是第二个交易日
884 // 这个比较复杂,主要节假日后第一天的边界很麻烦（如一般情况的周一）
885 // 这种情况唯一方便的就是,收盘时间不需要处理
886 if(!isStart)
887 return (uint64_t)tDate * 10000 + sInfo->getCloseTime();
888
889 // 想到一个简单的办法,就是不管怎么样,开始时间一定是上一个交易日的晚上
890 // 所以我只需要拿到上一个交易日即可
891 tDate = getPrevTDate(tplid.c_str(), tDate, 1, isTpl);
892 return (uint64_t)tDate * 10000 + sInfo->getOpenTime();

908uint32_t WTSBaseDataMgr::calcTradingDate(const char* stdPID, uint32_t uDate, uint32_t uTime, bool isSession /* = false */)
909{
910 // 如果没有提供日期，则使用当前日期和时间
911 if (uDate == 0)
912 {
913 TimeUtils::getDateTime(uDate, uTime);
914 uTime /= 100000; // 将毫秒级时间转换为不带秒的时间
915 }
916
917 // 设置节假日模板ID和是否直接使用模板的标志
918 std::string tplid = stdPID;
919 bool isTpl = false;
920 WTSSessionInfo* sInfo = NULL;
921
922 // 根据参数确定使用交易时段ID还是品种ID
923 if(isSession)
924 {
925 // 如果使用交易时段ID，直接获取交易时段信息
926 sInfo = getSession(stdPID);
927 tplid = DEFAULT_HOLIDAY_TPL; // 使用默认节假日模板
928 isTpl = true;
929 }
930 else
931 {
932 // 如果使用品种ID，需要先获取品种信息
933 WTSCommodityInfo* cInfo = getCommodity(stdPID);
934 if (cInfo == NULL)
935 return uDate; // 如果品种不存在，返回原始日期
936
937 // 从品种信息中获取交易时段
938 sInfo = cInfo->getSessionInfo();
939 }
940
941 // 如果未找到交易时段信息，返回原始日期
942 if (sInfo == NULL)
943 return uDate;
944
945 // 计算时间偏移点，用于判断交易日的边界
946 uint32_t offMin = sInfo->offsetTime(uTime, true);
947
948 // 特殊情况7*24交易时间单独处理（全天候交易的设置）
949 uint32_t total_mins = sInfo->getTradingMins();
950 if(total_mins == 1440) // 1440分钟 = 24小时
951 {
952 // 正偏移情况下，如果当前时间大于偏移时间，交易日向后偏移
953 if(sInfo->getOffsetMins() > 0 && uTime > offMin)
954 {
955 return TimeUtils::getNextDate(uDate, 1);
956 }
957 // 负偏移情况下，如果当前时间小于偏移时间，交易日向前偏移
958 else if (sInfo->getOffsetMins() < 0 && uTime < offMin)
959 {
960 return TimeUtils::getNextDate(uDate, -1);
961 }
962
963 // 其他情况下交易日就是自然日
964 return uDate;
965 }
966
967 // 获取当前日期的星期几，用于判断周末情况
968 uint32_t weekday = TimeUtils::getWeekDay(uDate);
969
970 // 情况1：正偏移情况（如国内期货夜盘）
971 if (sInfo->getOffsetMins() > 0)
972 {
973 // 如果向后偏移,且当前时间大于偏移时间,说明向后跨日了
974 // 这时交易日=下一个交易日
975 if (uTime > offMin)
976 {
977 // 例如，20151016 23:00,偏移300分钟,时间边界为5:00
978 // 因为23:00大于5:00，所以已经属于下一个交易日
979 return getNextTDate(tplid.c_str(), uDate, 1, isTpl);
980 }
981 else if (weekday == 6 || weekday == 0)
982 {
983 // 如果是周末，例如，20151017 1:00,周六,交易日为20151019
984 // 周末需要跳过到下一个交易日
985 return getNextTDate(tplid.c_str(), uDate, 1, isTpl);
986 }
987 }
988 // 情况2：负偏移情况（如一些外盘交易所）
989 else if (sInfo->getOffsetMins() < 0)
990 {
991 // 如果向前偏移,且当前时间小于偏移时间,说明还是前一个交易日
992 // 这时交易日=前一个交易日
993 if (uTime < offMin)
994 {
995 // 例如20151017 1:00,偏移-300分钟,时间边界为20:00
996 // 因1:00小于20:00，所以还属于前一个交易日
997 return getPrevTDate(tplid.c_str(), uDate, 1, isTpl);
998 }
999 else if (weekday == 6 || weekday == 0)
1000 {
1001 // 因为向前偏移,如果在周末,则直接到下一个交易日
1002 // 周末无论如何都需要跳到下一个有效交易日
1003 return getNextTDate(tplid.c_str(), uDate, 1, isTpl);
1004 }
1005 }
1006 // 情况3：无偏移情况下的周末处理
1007 else if (weekday == 6 || weekday == 0)
1008 {
1009 // 如果没有偏移,且在周末,则直接读取下一个交易日
1010 // 周末需要跳过到下一个交易日
1011 return getNextTDate(tplid.c_str(), uDate, 1, isTpl);;
1012 }
1013
1014 // 其他所有情况,交易日就等于自然日
1015 return uDate;
1016}

1082uint32_t WTSBaseDataMgr::getNextTDate(const char* pid, uint32_t uDate, int days /* = 1 */, bool isTpl /* = false */)
1083{
1084 // 初始化当前日期为起始日期
1085 uint32_t curDate = uDate;
1086 // 记录还需要偏移的有效交易日数量
1087 int left = days;
1088 while (true)
1089 {
1090 // 将当前日期转换为tm结构体
1091 tm t;
1092 memset(&t, 0, sizeof(tm));
1093 t.tm_year = curDate / 10000 - 1900;
1094 t.tm_mon = (curDate % 10000) / 100 - 1;
1095 t.tm_mday = curDate % 100;
1096 //t.tm_isdst
1097 // 转换为time_t类型并向后偏移一天(86400秒)
1098 time_t ts = mktime(&t);
1099 ts += 86400;
1100
1101 // 将偏移后的时间转回日期格式
1102 tm* newT = localtime(&ts);
1103 curDate = (newT->tm_year + 1900) * 10000 + (newT->tm_mon + 1) * 100 + newT->tm_mday;
1104 // 判断新日期是否为有效交易日（非周末且非节假日）
1105 if (newT->tm_wday != 0 && newT->tm_wday != 6 && !isHoliday(pid, curDate, isTpl))
1106 {
1107 // 如果不是周末,也不是节假日,则剩余的天数-1
1108 left--;
1109 // 当已经找到足够数量的交易日后返回结果
1110 if (left == 0)
1111 return curDate;
1112 }
1113 }
