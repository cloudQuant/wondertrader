64bool WtSimpDataMgr::initStore(WTSVariant* cfg)
65{
66 // 检查配置是否为空
67 if (cfg == NULL)
68 return false;
69
70 // 获取数据存储模块名称
71 std::string module = cfg->getCString("module");
72 if (module.empty())
73 // 如果模块名称为空，使用默认的WtDataStorage
74 module = WtHelper::getInstDir() + DLLHelper::wrap_module("WtDataStorage");
75 else
76 // 否则使用指定的模块
77 module = WtHelper::getInstDir() + DLLHelper::wrap_module(module.c_str());
78
79 // 加载数据存储模块动态库
80 DllHandle hInst = DLLHelper::load_library(module.c_str());
81 if (hInst == NULL)
82 {
83 WTSLogger::error("Data reader {} loading failed", module.c_str());
84 return false;
85 }
86
87 // 获取创建数据读取器的函数
88 FuncCreateDataReader funcCreator = (FuncCreateDataReader)DLLHelper::get_symbol(hInst, "createDataReader");
89 if (funcCreator == NULL)
90 {
91 WTSLogger::error("Data reader {} loading failed: entrance function createDataReader not found", module.c_str());
92 DLLHelper::free_library(hInst);
93 return false;
94 }
95
96 // 创建数据读取器
97 _reader = funcCreator();
98 if (_reader == NULL)
99 {
100 WTSLogger::error("Data reader {} creating api failed", module.c_str());
101 DLLHelper::free_library(hInst);
102 return false;
103 }
104
105 // 初始化数据读取器
106 _reader->init(cfg, this);
107
108 // 获取交易时段信息
109 _s_info = _runner->get_session_info(cfg->getCString("session"), false);
110
111 return true;
112}

224void WtSimpDataMgr::handle_push_quote(const char* stdCode, WTSTickData* curTick)
225{
226 // 检查Tick数据是否有效
227 if (curTick == NULL)
228 return;
229
230 // 如果实时Tick缓存不存在，则创建
231 if (_rt_tick_map == NULL)
232 _rt_tick_map = DataCacheMap::create();
233
234 // 将新的Tick数据添加到缓存中，true表示如果已存在则覆盖
235 _rt_tick_map->add(stdCode, curTick, true);
236
237 // 获取Tick数据中的日期和时间
238 uint32_t uDate = curTick->actiondate();
239 uint32_t uTime = curTick->actiontime();
240
241 // 如果新收到的数据时间早于当前已存储的时间，则忽略
242 if (_cur_date != 0 && (uDate < _cur_date || (uDate == _cur_date && uTime < _cur_act_time)))
243 {
244 return;
245 }
246
247 // 更新当前日期和完整时间
248 _cur_date = uDate;
249 _cur_act_time = uTime;
250
251 // 计算当前原始时间和秒数
252 uint32_t _cur_raw_time = _cur_act_time / 100000; // 提取小时分钟部分
253 uint32_t _cur_secs = _cur_act_time % 100000; // 提取秒数部分
254
255 // 将时间转换为分钟数
256 uint32_t minutes = _s_info->timeToMinutes(_cur_raw_time);
257 // 检查是否是交易时段的最后一分钟
258 bool isSecEnd = _s_info->isLastOfSection(_cur_raw_time);
259 if (isSecEnd)
260 {
261 // 如果是交易时段的最后一分钟，则减去1分钟
262 minutes--;
263 }
264 // 增加1分钟，确保分钟数正确
265 minutes++;
266 // 将分钟数转换回时间格式
267 _cur_min_time = _s_info->minuteToTime(minutes);
268 // 设置当前交易日
269 _cur_tdate = curTick->tradingdate();
270}

324WTSKlineSlice* WtSimpDataMgr::get_kline_slice(const char* stdCode, WTSKlinePeriod period, uint32_t times, uint32_t count, uint64_t etime /*= 0*/)
325{
326 // 检查数据读取器是否存在
327 if (_reader == NULL)
328 return NULL;
329
330 // 生成缓存键
331 std::string key = StrUtil::printf("%s-%u", stdCode, period);
332
333 // 如果是基础周期（倍数为1），直接从数据读取器中读取
334 if (times == 1)
335 {
336 return _reader->readKlineSlice(stdCode, period, count, etime);
337 }
338
339 // 只有非基础周期的会进到下面的步骤
340 // 获取交易时段信息
341 WTSSessionInfo* sInfo = _runner->get_session_info(stdCode, true);
342
343 // 如果K线缓存不存在，则创建
344 if (_bars_cache == NULL)
345 _bars_cache = DataCacheMap::create();
346
347 // 生成包含周期倍数的缓存键
348 key = StrUtil::printf("%s-%u-%u", stdCode, period, times);
349
350 // 从缓存中获取K线数据
351 WTSKlineData* kData = (WTSKlineData*)_bars_cache->get(key);
352 // 如果缓存中的数据不存在或条数不足，需要重新读取
353 if (kData == NULL || kData->size() < count)
354 {
355 // 计算需要读取的实际条数，考虑周期倍数
356 uint32_t realCount = count * times + times;
357 // 读取原始K线数据
358 WTSKlineSlice* rawData = _reader->readKlineSlice(stdCode, period, realCount, etime);
359 if (rawData != NULL)
360 {
361 // 从原始K线数据中提取指定周期倍数的K线数据
362 kData = g_dataFact.extractKlineData(rawData, period, times, sInfo, true);
363 // 释放原始K线数据
364 rawData->release();
365 }
366 else
367 {
368 return NULL;
369 }
370
371 // 将提取的K线数据添加到缓存中，false表示不覆盖已存在的数据
372 if (kData)
373 _bars_cache->add(key, kData, false);
374 }
375
376 // 计算返回数据的起始索引和实际条数
377 int32_t sIdx = 0;
378 // 实际返回条数为请求条数和缓存条数的最小值
379 uint32_t rtCnt = min(kData->size(), count);
380 // 计算起始索引，保证返回最新的数据
381 sIdx = kData->size() - rtCnt;
382 // 获取数据头指针
383 WTSBarStruct* rtHead = kData->at(sIdx);
384 // 创建并返回K线切片
385 WTSKlineSlice* slice = WTSKlineSlice::create(stdCode, period, times, rtHead, rtCnt);
386 return slice;
387}
