308 static inline int32_t getTZOffset()
309 {
310 static int32_t offset = 99;
311 if(offset == 99)
312 {
313 time_t now = time(NULL);
314 tm tm_ltm = *localtime(&now);
315 tm tm_gtm = *gmtime(&now);
316
317 time_t _gt = mktime(&tm_gtm);
318 tm _gtm2 = *localtime(&_gt);
319
320 offset = (uint32_t)(((now - _gt) + (_gtm2.tm_isdst ? 3600 : 0)) / 60);
321 offset /= 60;
322 }

340 static inline int64_t makeTime(long lDate, long lTimeWithMs, bool isToUTC = false)
341 {
342 tm t;
343 memset(&t,0,sizeof(tm));
344 t.tm_year = lDate/10000 - 1900;
345 t.tm_mon = (lDate%10000)/100 - 1;
346 t.tm_mday = lDate % 100;
347 t.tm_hour = lTimeWithMs/10000000;
348 t.tm_min = (lTimeWithMs%10000000)/100000;
349 t.tm_sec = (lTimeWithMs%100000)/1000;

355 if (isToUTC)
356 ts -= getTZOffset() * 3600;
357 return ts * 1000+ millisec;
