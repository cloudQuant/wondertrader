285static __inline struct tm*
286__STime_FastSecondToDate2(const int64 unixSecs, struct tm *pResult,
287 int32 timeZone) {
288# define K_HOURS_IN_DAY (24)
289# define K_MINUTES_IN_HOUR (60)
290# define K_DAYS_FROM_UNIXTIME (2472632)
291# define K_DAYS_FROM_YEAR (153)
292# define K_MAGIC_UNKONWN_FIRST (146097)
293# define K_MAGIC_UNKONWN_SEC (1461)
294
295 int32 hours;
296 int32 a, b, c, d, e, m;
297
298 SLOG_ASSERT(pResult);
299 pResult->tm_sec = unixSecs % K_MINUTES_IN_HOUR;
300
301 hours = (int32) (unixSecs / K_MINUTES_IN_HOUR);
302 pResult->tm_min = hours % K_MINUTES_IN_HOUR;
303
304 hours /= K_MINUTES_IN_HOUR;
305 pResult->tm_hour = (hours + timeZone) % K_HOURS_IN_DAY;
306 pResult->tm_mday = (hours + timeZone) / K_HOURS_IN_DAY;
307
308 a = pResult->tm_mday + K_DAYS_FROM_UNIXTIME;
309 b = (a * 4 + 3) / K_MAGIC_UNKONWN_FIRST;
310 c = (-b * K_MAGIC_UNKONWN_FIRST) / 4 + a;
311 d = ((c * 4 + 3) / K_MAGIC_UNKONWN_SEC);
312 e = -d * K_MAGIC_UNKONWN_SEC;
313 e = e / 4 + c;
314
315 m = (5 * e + 2) / K_DAYS_FROM_YEAR;
316 pResult->tm_mday = -(K_DAYS_FROM_YEAR * m + 2) / 5 + e + 1;
317 pResult->tm_mon = (-m / 10) * 12 + m + 2;
318 pResult->tm_year = b * 100 + d - 6700 + (m / 10);
319
320 return pResult;
321}

738STime_GetMonthDays(int32 year, int32 month) {
739 static const int32 normalDays[] = {
740 0, \

995static __inline int32
996STime_DiffIntDays(int32 iBeginDate, int32 iEndDate) {
997 int32 beginYear = 0;
998 int32 beginMonth = 0;
999 int32 beginDay = 0;
1000 int32 endYear = 0;
1001 int32 endMonth = 0;
1002 int32 endDay = 0;
1003 int32 y1 = 0;
1004 int32 m1 = 0;
1005 int32 d1 = 0;
1006 int32 y2 = 0;
1007 int32 m2 = 0;
1008 int32 d2 = 0;
1009
1010 beginYear = STime_GetYearOfIntDate(iBeginDate);
1011 beginMonth = STime_GetMonthOfIntDate(iBeginDate);
1012 beginDay = STime_GetDayOfIntDate(iBeginDate);
1013
1014 endYear = STime_GetYearOfIntDate(iEndDate);
1015 endMonth = STime_GetMonthOfIntDate(iEndDate);
1016 endDay = STime_GetDayOfIntDate(iEndDate);
1017
1018 m1 = (beginMonth + 9) % 12;
1019 y1 = beginYear - m1 / 10;
1020 d1 = 365 * y1 + y1 / 4 - y1 / 100 + y1 / 400 + (m1 * 306 + 5) / 10
1021 + (beginDay - 1);
1022
1023 m2 = (endMonth + 9) % 12;
1024 y2 = endYear - m2 / 10;
1025 d2 = 365 * y2 + y2 / 4 - y2 / 100 + y2 / 400 + (m2 * 306 + 5) / 10
1026 + (endDay - 1);
1027
1028 return (d2 - d1);
1029}
