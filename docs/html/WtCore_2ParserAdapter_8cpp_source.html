66bool ParserAdapter::initExt(const char* id, IParserApi* api, IParserStub* stub, IBaseDataMgr* bgMgr, IHotMgr* hotMgr/* = NULL*/)
67{
68 // 检查API是否为空
69 if (api == NULL)
70 return false;
71
72 // 保存参数
73 _parser_api = api;
74 _stub = stub;
75 _bd_mgr = bgMgr;
76 _hot_mgr = hotMgr;
77 _id = id;
78
79 if (_parser_api)

81 // 注册回调接口

99 contractSet.clear();
100 }
101 else
102 {
103 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser initializing failed: api initializing failed...", _id.c_str());

120bool ParserAdapter::init(const char* id, WTSVariant* cfg, IParserStub* stub, IBaseDataMgr* bgMgr, IHotMgr* hotMgr/* = NULL*/)
121{
122 // 检查配置是否为空
123 if (cfg == NULL)
124 return false;
125
126 // 保存参数
127 _stub = stub;
128 _bd_mgr = bgMgr;
129 _hot_mgr = hotMgr;
130 _id = id;
131
132 // 检查是否已初始化
133 if (_cfg != NULL)
134 return false;
135
136 // 保存配置并增加引用计数
137 _cfg = cfg;
138 _cfg->retain();
139
140 // 获取时间检查标志
141 _check_time = cfg->getBoolean("check_time");
142
143 {
144 //加载解析器模块
145 if (cfg->getString("module").empty())
146 return false;
147
148 std::string module = DLLHelper::wrap_module(cfg->getCString("module"), "lib");;
149
150 //先看工作目录下是否有交易模块
151 std::string dllpath = WtHelper::getModulePath(module.c_str(), "parsers", true);
152 //如果没有,则再看模块目录,即dll同目录下
153 if (!StdFile::exists(dllpath.c_str()))
154 dllpath = WtHelper::getModulePath(module.c_str(), "parsers", false);
155
156 DllHandle hInst = DLLHelper::load_library(dllpath.c_str());
157 if (hInst == NULL)
158 {
159 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser module {} loading failed", _id.c_str(), dllpath.c_str());
160 return false;
161 }
162 else
163 {
164 WTSLogger::log_dyn("parser", _id.c_str(), LL_INFO, "[{}] Parser module {} loaded", _id.c_str(), dllpath.c_str());
165 }
166
167 FuncCreateParser pFuncCreateParser = (FuncCreateParser)DLLHelper::get_symbol(hInst, "createParser");
168 if (NULL == pFuncCreateParser)
169 {
170 WTSLogger::log_dyn("parser", _id.c_str(), LL_FATAL, "[{}] Entrance function createParser not found", _id.c_str());
171 return false;
172 }
173
174 _parser_api = pFuncCreateParser();
175 if (NULL == _parser_api)
176 {
177 WTSLogger::log_dyn("parser", _id.c_str(), LL_FATAL, "[{}] Creating parser api failed", _id.c_str());
178 return false;
179 }
180
181 _remover = (FuncDeleteParser)DLLHelper::get_symbol(hInst, "deleteParser");
182 }
183
184
185 const std::string& strFilter = cfg->getString("filter");
186 if (!strFilter.empty())
187 {
188 const StringVector &ayFilter = StrUtil::split(strFilter, ",");
189 auto it = ayFilter.begin();
190 for (; it != ayFilter.end(); it++)
191 {
192 _exchg_filter.insert(*it);
193 }
194 }
195
196 std::string strCodes = cfg->getString("code");
197 if (!strCodes.empty())
198 {
199 const StringVector &ayCodes = StrUtil::split(strCodes, ",");
200 auto it = ayCodes.begin();
201 for (; it != ayCodes.end(); it++)
202 {
203 _code_filter.insert(*it);
204 }
205 }
206
207 if (_parser_api)
208 {
209 _parser_api->registerSpi(this);
210
211 if (_parser_api->init(cfg))
212 {
213 ContractSet contractSet;
214 if (!_code_filter.empty())//优先判断合约过滤器
215 {
216 ExchgFilter::iterator it = _code_filter.begin();
217 for (; it != _code_filter.end(); it++)
218 {
219 //全代码,形式如SSE.600000,期货代码为CFFEX.IF2005
220 std::string code, exchg;
221 auto ay = StrUtil::split((*it).c_str(), ".");
222 if (ay.size() == 1)
223 code = ay[0];
224 else if (ay.size() == 2)
225 {
226 exchg = ay[0];
227 code = ay[1];
228 }
229 else if (ay.size() == 3)
230 {
231 exchg = ay[0];
232 code = ay[2];
233 }
234 WTSContractInfo* contract = _bd_mgr->getContract(code.c_str(), exchg.c_str());
235 if(contract)
236 contractSet.insert(contract->getFullCode());
237 else
238 {
239 //如果是品种ID，则将该品种下全部合约都加到订阅列表
240 WTSCommodityInfo* commInfo = _bd_mgr->getCommodity(exchg.c_str(), code.c_str());
241 if(commInfo)
242 {
243 const auto& codes = commInfo->getCodes();
244 for(const auto& c : codes)
245 {
246 contractSet.insert(fmt::format("{}.{}", exchg, c.c_str()));
247 }
248 }
249 }
250 }
251 }
252 else if (!_exchg_filter.empty())
253 {
254 ExchgFilter::iterator it = _exchg_filter.begin();
255 for (; it != _exchg_filter.end(); it++)
256 {
257 WTSArray* ayContract =_bd_mgr->getContracts((*it).c_str());
258 WTSArray::Iterator it = ayContract->begin();
259 for (; it != ayContract->end(); it++)
260 {
261 WTSContractInfo* contract = STATIC_CONVERT(*it, WTSContractInfo*);
262 contractSet.insert(contract->getFullCode());
263 }
264
265 ayContract->release();
266 }
267 }
268 else
269 {
270 WTSArray* ayContract =_bd_mgr->getContracts();
271 WTSArray::Iterator it = ayContract->begin();
272 for (; it != ayContract->end(); it++)
273 {
274 WTSContractInfo* contract = STATIC_CONVERT(*it, WTSContractInfo*);
275 contractSet.insert(contract->getFullCode());
276 }
277
278 ayContract->release();
279 }
280
281 _parser_api->subscribe(contractSet);
282 contractSet.clear();
283 }
284 else
285 {
286 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser initializing failed: api initializing failed...", _id.c_str());
287 }
288 }
289 else
290 {
291 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser initializing failed: creating api failed...", _id.c_str());
292 }
293
294 WTSLogger::log_dyn("parser", _id.c_str(), LL_INFO, "[{}] Parser initialzied, check_time: {}", _id.c_str(), _check_time);
295
296 return true;

343void ParserAdapter::handleQuote(WTSTickData *quote, uint32_t procFlag)
344{
345 // 检查数据有效性
346 if (quote == NULL || _stopped || quote->actiondate() == 0 || quote->tradingdate() == 0)
347 return;
348
349 // 交易所过滤
350 if (!_exchg_filter.empty() && (_exchg_filter.find(quote->exchg()) == _exchg_filter.end()))
351 return;
352
353 // 获取合约信息
354 WTSContractInfo* cInfo = quote->getContractInfo();
355 if (cInfo == NULL)
356 {
357 cInfo = _bd_mgr->getContract(quote->code(), quote->exchg());
358 quote->setContractInfo(cInfo);
359 }
360
361 if (cInfo == NULL)
362 return;
363
364 // 获取品种信息
365 WTSCommodityInfo* commInfo = cInfo->getCommInfo();
366 WTSSessionInfo* sInfo = commInfo->getSessionInfo();
367
368 // 时间检查
369 if (_check_time)
370 {
371 int64_t tick_time = TimeUtils::makeTime(quote->actiondate(), quote->actiontime());
372 int64_t local_time = TimeUtils::getLocalTimeNow();
373
374 /*
375 * By Wesley @ 2022.04.20
376 * 如果最新的tick时间，和本地时间相差太大
377 * 则认为tick的时间戳是错误的
378 * 这里要求本地时间是要时常进行校准的
379 */
380 if (tick_time - local_time > RESONABLE_MILLISECS)
381 {
382 WTSLogger::warn("Tick of {} with wrong timestamp {}.{} received, skipped", cInfo->getFullCode(), quote->actiondate(), quote->actiontime());
383 return;
384 }
385 }
386
387 // 代码转换
388 std::string stdCode;
389 if (commInfo->getCategoty() == CC_FutOption || commInfo->getCategoty() == CC_SpotOption)
390 {
391 // 期权代码转换
392 stdCode = CodeHelper::rawFutOptCodeToStdCode(cInfo->getCode(), cInfo->getExchg());
393 }
394 else if(CodeHelper::isMonthlyCode(quote->code()))
395 {
396 //分月合约代码转换
397 stdCode = CodeHelper::rawMonthCodeToStdCode(cInfo->getCode(), cInfo->getExchg());
398 }
399 else
400 {
401 // 普通代码转换
402 stdCode = CodeHelper::rawFlatCodeToStdCode(cInfo->getCode(), cInfo->getExchg(), cInfo->getProduct());
403 }
404 quote->setCode(stdCode.c_str());
405
406 // 转发行情数据
407 _stub->handle_push_quote(quote);
408}
