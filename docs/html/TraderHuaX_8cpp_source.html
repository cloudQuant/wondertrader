105 entrust_info->OrderPriceType,

107 pRet->setContractInfo(ct);
108 pRet->setDirection(wrapDirectionType(entrust_info->Direction));
109 pRet->setPriceType(wrapPriceType(entrust_info->OrderPriceType));
110 if (pRet->getDirection() == WDT_LONG)

112 else if (pRet->getDirection() == WDT_SHORT)
113 pRet->setOffsetType(WOT_CLOSEYESTERDAY);

115 pRet->setOrderFlag(WOF_NOR);
116
117 genEntrustID(pRet->getEntrustID(), entrust_info->OrderRef);
118
119 const char* usertag = _eidCache.get(pRet->getEntrustID());

126WTSOrderInfo* TraderHuaX::makeOrderInfo(CTORATstpOrderField* order_info)
127{
128 std::string code, exchg;
129 if (order_info->ExchangeID == TORA_TSTP_EXD_SSE)
130 exchg = WT_MKT_SH_A;
131 else
132 exchg = WT_MKT_SZ_A;
133 code = order_info->SecurityID;
134 WTSContractInfo* contract = _bd_mgr->getContract(code.c_str(), exchg.c_str());
135 if (contract == NULL)
136 return NULL;
137
138 WTSOrderInfo* pRet = WTSOrderInfo::create();
139 pRet->setContractInfo(contract);
140 pRet->setPrice(order_info->LimitPrice);
141 pRet->setVolume((uint32_t)order_info->VolumeTotalOriginal);
142 pRet->setDirection(wrapDirectionType(order_info->Direction));
143 pRet->setPriceType(wrapPriceType(order_info->OrderPriceType));
144 pRet->setOrderFlag(WOF_NOR);
145 if (pRet->getDirection() == WDT_LONG)
146 pRet->setOffsetType(WOT_OPEN);
147 else if (pRet->getDirection() == WDT_SHORT)
148 pRet->setOffsetType(WOT_CLOSEYESTERDAY);
149
150 pRet->setVolTraded((uint32_t)order_info->VolumeTraded);
151 pRet->setVolLeft((uint32_t)order_info->VolumeTotalOriginal - order_info->VolumeTraded);
152
153 pRet->setCode(code.c_str());
154 pRet->setExchange(contract->getExchg());
155
156 pRet->setOrderDate((uint32_t)std::stoi(order_info->InsertDate));
157 auto time_vector = StrUtil::split(order_info->InsertTime,":");
158 uint32_t uTime{ 0 };
159 for (std::string time_str : time_vector)
160 {
161 uTime = uTime * 100;
162 uTime += std::stoi(time_str);
163 }
164 pRet->setOrderTime(TimeUtils::makeTime(pRet->getOrderDate(), uTime*1000));
165
166 pRet->setOrderState(wrapOrderState(order_info->OrderStatus));
167 if (order_info->OrderStatus == TORA_TSTP_OST_Rejected)
168 pRet->setError(true);
169
170 genEntrustID(pRet->getEntrustID(), order_info->OrderRef);
171 pRet->setOrderID(order_info->OrderSysID);
172
173 pRet->setStateMsg("");
174
175 const char* usertag = _eidCache.get(pRet->getEntrustID());
176 if (strlen(usertag) == 0)
177 {
178 pRet->setUserTag(pRet->getEntrustID());
179 }
180 else
181 {
182 pRet->setUserTag(usertag);
183
184 if (strlen(pRet->getOrderID()) > 0)
185 {
186 _oidCache.put(StrUtil::trim(order_info->OrderSysID).c_str(), usertag, 0, [this](const char* message) {
187 write_log(_sink, LL_ERROR, message);
188 });
189 }
190 }
191 return pRet;
