83bool EventNotifier::init(WTSVariant* cfg)
84{
85 // 如果不激活消息队列，直接返回失败
86 if (!cfg->getBoolean("active"))
87 return false;
88
89 // 获取消息队列服务器URL
90 _url = cfg->getCString("url");
91 std::string module = DLLHelper::wrap_module("WtMsgQue", "lib");
92 //先看工作目录下是否有对应模块
93 std::string dllpath = WtHelper::getCWD() + module;
94 //如果没有,则再看模块目录,即dll同目录下
95 if (!StdFile::exists(dllpath.c_str()))
96 dllpath = WtHelper::getInstDir() + module;
97
98 DllHandle dllInst = DLLHelper::load_library(dllpath.c_str());
99 if (dllInst == NULL)
100 {
101 WTSLogger::error("MQ module {} loading failed", dllpath.c_str());
102 return false;
103 }
104
105 _creator = (FuncCreateMQServer)DLLHelper::get_symbol(dllInst, "create_server");
106 if (_creator == NULL)
107 {
108 DLLHelper::free_library(dllInst);
109 WTSLogger::error("MQ module {} is not compatible", dllpath.c_str());
110 return false;
111 }
112
113 _remover = (FuncDestroyMQServer)DLLHelper::get_symbol(dllInst, "destroy_server");
114 _publisher = (FundPublishMessage)DLLHelper::get_symbol(dllInst, "publish_message");
115 _register = (FuncRegCallbacks)DLLHelper::get_symbol(dllInst, "regiter_callbacks");
116
117 //注册回调函数
118 _register(on_mq_log);
119
120 //创建一个MQServer
121 _mq_sid = _creator(_url.c_str());
122
123 WTSLogger::info("EventNotifier initialized with channel {}", _url.c_str());
124
125 if (_worker == NULL)
126 {
127 boost::asio::io_service::work work(_asyncio);
128 _worker.reset(new StdThread([this]() {
129 while (!_stopped)
130 {
131 std::this_thread::sleep_for(std::chrono::milliseconds(2));
132 _asyncio.run_one();
133 //m_asyncIO.run();
134 }
135 }));
136 }
137
138 return true;
139}

149void EventNotifier::notify_log(const char* tag, const char* message)
150{
151 // 如果消息队列服务器不可用，直接返回
152 if (_mq_sid == 0)
153 return;
154
155 // 复制参数，防止异步调用时参数已被释放
156 std::string strTag = tag;
157 std::string strMsg = message;
158 // 添加异步任务
159 _asyncio.post([this, strTag, strMsg]() {
160 std::string data;
161 {
162 rj::Document root(rj::kObjectType);
163 rj::Document::AllocatorType &allocator = root.GetAllocator();
164
165 root.AddMember("tag", rj::Value(strTag.c_str(), allocator), allocator);
166 root.AddMember("time", TimeUtils::getLocalTimeNow(), allocator);
167 root.AddMember("message", rj::Value(strMsg.c_str(), allocator), allocator);
168
169 rj::StringBuffer sb;
170 rj::PrettyWriter<rj::StringBuffer> writer(sb);
171 root.Accept(writer);
172
173 data = sb.GetString();
174 }
175
176 if (_publisher)
177 _publisher(_mq_sid, "LOG", data.c_str(), (unsigned long)data.size());
178 });
179}
