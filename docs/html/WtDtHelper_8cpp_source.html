45bool proc_block_data(std::string& content, bool isBar, bool bKeepHead /* = true */)
46{
47 BlockHeader* header = (BlockHeader*)content.data();
48
49 bool bCmped = header->is_compressed();
50 bool bOldVer = header->is_old_version();
51
52 //如果既没有压缩，也不是老版本结构体，则直接返回
53 if (!bCmped && !bOldVer)
54 {
55 if (!bKeepHead)
56 content.erase(0, BLOCK_HEADER_SIZE);
57 return true;
58 }
59
60 std::string buffer;
61 if (bCmped)
62 {
63 BlockHeaderV2* blkV2 = (BlockHeaderV2*)content.c_str();
64
65 if (content.size() != (sizeof(BlockHeaderV2) + blkV2->_size))
66 {
67 return false;
68 }
69
70 //将文件头后面的数据进行解压
71 buffer = WTSCmpHelper::uncompress_data(content.data() + BLOCK_HEADERV2_SIZE, blkV2->_size);
72 }
73 else
74 {
75 if (!bOldVer)
76 {
77 //如果不是老版本，直接返回
78 if (!bKeepHead)
79 content.erase(0, BLOCK_HEADER_SIZE);
80 return true;
81 }
82 else
83 {
84 buffer.append(content.data() + BLOCK_HEADER_SIZE, content.size() - BLOCK_HEADER_SIZE);
85 }
86 }
87
88 if (bOldVer)
89 {
90 if (isBar)
91 {
92 std::string bufV2;
93 uint32_t barcnt = buffer.size() / sizeof(WTSBarStructOld);
94 bufV2.resize(barcnt * sizeof(WTSBarStruct));
95 WTSBarStruct* newBar = (WTSBarStruct*)bufV2.data();
96 WTSBarStructOld* oldBar = (WTSBarStructOld*)buffer.data();
97 for (uint32_t idx = 0; idx < barcnt; idx++)
98 {
99 newBar[idx] = oldBar[idx];
100 }
101 buffer.swap(bufV2);
102 }
103 else
104 {
105 uint32_t tick_cnt = buffer.size() / sizeof(WTSTickStructOld);
106 std::string bufv2;
107 bufv2.resize(sizeof(WTSTickStruct)*tick_cnt);
108 WTSTickStruct* newTick = (WTSTickStruct*)bufv2.data();
109 WTSTickStructOld* oldTick = (WTSTickStructOld*)buffer.data();
110 for (uint32_t i = 0; i < tick_cnt; i++)
111 {
112 newTick[i] = oldTick[i];
113 }
114 buffer.swap(bufv2);
115 }
116 }
117
118 if (bKeepHead)
119 {
120 content.resize(BLOCK_HEADER_SIZE);
121 content.append(buffer);
122 header = (BlockHeader*)content.data();
123 header->_version = BLOCK_VERSION_RAW_V2;
124 }
125 else
126 {
127 content.swap(buffer);
128 }
129
130 return true;
131}

144uint32_t strToTime(const char* strTime, bool bKeepSec = false)
145{
146 std::string str;
147 const char *pos = strTime;
148 while (strlen(pos) > 0)
149 {
150 if (pos[0] != ':')
151 {
152 str.append(pos, 1);
153 }
154 pos++;
155 }
156
157 uint32_t ret = strtoul(str.c_str(), NULL, 10);
158 if (ret > 10000 && !bKeepSec)
159 ret /= 100;
160
161 return ret;
162}

173uint32_t strToDate(const char* strDate)
174{
175 StringVector ay = StrUtil::split(strDate, "/");
176 if (ay.size() == 1)
177 ay = StrUtil::split(strDate, "-");
178 std::stringstream ss;
179 if (ay.size() > 1)
180 {
181 auto pos = ay[2].find(" ");
182 if (pos != std::string::npos)
183 ay[2] = ay[2].substr(0, pos);
184 ss << ay[0] << (ay[1].size() == 1 ? "0" : "") << ay[1] << (ay[2].size() == 1 ? "0" : "") << ay[2];
185 }
186 else
187 ss << ay[0];
188
189 return strtoul(ss.str().c_str(), NULL, 10);
190}

436void trans_csv_bars(WtString csvFolder, WtString binFolder, WtString period, FuncLogCallback cbLogger /* = NULL */)
437{
438 if (!BoostFile::exists(csvFolder))
439 return;
440
441 if (!BoostFile::exists(binFolder))
442 BoostFile::create_directories(binFolder);
443
444 WTSKlinePeriod kp = KP_DAY;
445 if (wt_stricmp(period, "m1") == 0)
446 kp = KP_Minute1;
447 else if (wt_stricmp(period, "m5") == 0)
448 kp = KP_Minute5;
449 else
450 kp = KP_DAY;
451
452 boost::filesystem::path myPath(csvFolder);
453 boost::filesystem::directory_iterator endIter;
454 for (boost::filesystem::directory_iterator iter(myPath); iter != endIter; iter++)
455 {
456 if (boost::filesystem::is_directory(iter->path()))
457 continue;
458
459 if (iter->path().extension() != ".csv")
460 continue;
461
462 const std::string& path = iter->path().string();
463
464 if(cbLogger)
465 cbLogger(StrUtil::printf("正在读取数据文件%s...", path.c_str()).c_str());
466
467 CsvReader reader(",");
468 if(!reader.load_from_file(path.c_str()))
469 {
470 if (cbLogger)
471 cbLogger(StrUtil::printf("读取数据文件%s失败...", path.c_str()).c_str());
472 continue;
473 }
474
475 std::vector<WTSBarStruct> bars;
476
477 while(reader.next_row())
478 {
479 //逐行读取
480 WTSBarStruct bs;
481 bs.date = strToDate(reader.get_string("date"));
482 if(kp != KP_DAY)
483 bs.time = TimeUtils::timeToMinBar(bs.date, strToTime(reader.get_string("time")));
484 bs.open = reader.get_double("open");
485 bs.high = reader.get_double("high");
486 bs.low = reader.get_double("low");
487 bs.close = reader.get_double("close");
488 bs.vol = reader.get_double("volume");
489 bs.money = reader.get_double("turnover");
490 bs.hold = reader.get_double("open_interest");
491 bs.add = reader.get_double("diff_interest");
492 bs.settle = reader.get_double("settle");
493 bars.emplace_back(bs);
494
495 if (bars.size() % 1000 == 0)
496 {
497 if (cbLogger)
498 cbLogger(StrUtil::printf("已读取数据%u条", bars.size()).c_str());
499 }
500 }
501 if (cbLogger)
502 cbLogger(StrUtil::printf("数据文件%s全部读取完成,共%u条", path.c_str(), bars.size()).c_str());
503
504 BlockType btype;
505 switch (kp)
506 {
507 case KP_Minute1: btype = BT_HIS_Minute1; break;
508 case KP_Minute5: btype = BT_HIS_Minute5; break;
509 default: btype = BT_HIS_Day; break;
510 }
511
512 HisKlineBlockV2 kBlock;
513 strcpy(kBlock._blk_flag, BLK_FLAG);
514 kBlock._type = btype;
515 kBlock._version = BLOCK_VERSION_CMP_V2;
516
517 std::string cmprsData = WTSCmpHelper::compress_data(bars.data(), sizeof(WTSBarStruct)*bars.size());
518 kBlock._size = cmprsData.size();
519
520 std::string filename = StrUtil::standardisePath(binFolder);
521 filename += iter->path().stem().string();
522 filename += ".dsb";
523
524 BoostFile bf;
525 if (bf.create_new_file(filename.c_str()))
526 {
527 bf.write_file(&kBlock, sizeof(HisKlineBlockV2));
528 }
529 bf.write_file(cmprsData);
530 bf.close_file();
531 if (cbLogger)
532 cbLogger(StrUtil::printf("数据已转储至%s", filename.c_str()).c_str());
533 }
534}
