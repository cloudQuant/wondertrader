104void WtHftRtTicker::on_tick(WTSTickData* curTick)
105{
106 // 如果定时器线程未启动，直接触发行情处理
107 if (_thrd == NULL)
108 {
109 trigger_price(curTick);
110 return;
111 }
112
113 // 获取行情的日期和时间
114 uint32_t uDate = curTick->actiondate();
115 uint32_t uTime = curTick->actiontime();
116
117 // 如果行情时间早于当前时间，直接触发行情处理
118 if (_date != 0 && (uDate < _date || (uDate == _date && uTime < _time)))
119 {
120 //WTSLogger::info("行情时间{}小于本地时间{}", uTime, _time);
121 trigger_price(curTick);
122 return;
123 }
124
125 // 更新当前日期和时间
126 _date = uDate;
127 _time = uTime;
128
129 // 分离分钟和秒数
130 uint32_t curMin = _time / 100000;
131 uint32_t curSec = _time % 100000;
132 // 将时间转换为分钟序号
133 uint32_t minutes = _s_info->timeToMinutes(curMin);
134 // 判断是否是交易时段的最后一分钟
135 bool isSecEnd = _s_info->isLastOfSection(curMin);
136 if (isSecEnd)
137 {
138 minutes--;
139 }
140 minutes++;
141 uint32_t rawMin = curMin;
142 // 将分钟序号转换回时间格式
143 curMin = _s_info->minuteToTime(minutes);
144
145 if (_cur_pos == 0)
146 {
147 // 如果当前时间是0, 则直接赋值即可
148 _cur_pos = minutes;
149 }
150 else if (_cur_pos < minutes)
151 {
152 // 如果已记录的分钟小于新的分钟, 则需要触发闭合事件
153 // 这个时候要先触发闭合, 再修改平台时间和价格
154 if (_last_emit_pos < _cur_pos)
155 {
156 // 触发数据回放模块
157 StdUniqueLock lock(_mtx);
158
159 // 优先修改时间标记
160 _last_emit_pos = _cur_pos;
161
162 uint32_t thisMin = _s_info->minuteToTime(_cur_pos);
163
164 WTSLogger::info("Minute Bar {}.{:04d} Closed by data", _date, thisMin);
165 // 通知数据存储器分钟线结束
166 if (_store)
167 _store->onMinuteEnd(_date, thisMin);
168
169 // 通知引擎分钟线结束
170 _engine->on_minute_end(_date, thisMin);
171
172 // 判断是否是交易时段结束
173 uint32_t offMin = _s_info->offsetTime(thisMin, true);
174 if (offMin == _s_info->getCloseTime(true))
175 {
176 // 触发交易时段结束事件
177 _engine->on_session_end();
178 }
179 }
180
181 // 触发行情处理
182 trigger_price(curTick);
183 if (_engine)
184 {
185 // 设置引擎当前日期时间
186 _engine->set_date_time(_date, curMin, curSec, rawMin);
187 // 设置引擎交易日期
188 _engine->set_trading_date(curTick->tradingdate());
189 }
190
191 // 更新当前分钟位置
192 _cur_pos = minutes;
193 }
194 else
195 {
196 // 如果分钟数还是一致的, 则直接触发行情和时间即可
197 trigger_price(curTick);
198 if (_engine)
199 _engine->set_date_time(_date, curMin, curSec, rawMin);
200 }
201
202 // 计算距离下一分钟的时间
203 uint32_t sec = curSec / 1000;
204 uint32_t msec = curSec % 1000;
205 uint32_t left_ticks = (60 - sec) * 1000 - msec;
206 // 设置下次检查时间
207 _next_check_time = TimeUtils::getLocalTimeNow() + left_ticks;
208}

214void WtHftRtTicker::run()
215{
216 // 如果线程已存在，直接返回
217 if (_thrd)
218 return;
219
220 // 计算当前交易日期
221 uint32_t curTDate = _engine->get_basedata_mgr()->calcTradingDate(_s_info->id(), _engine->get_date(), _engine->get_min_time(), true);
222 // 设置引擎的交易日期
223 _engine->set_trading_date(curTDate);
224
225 // 触发引擎初始化事件
226 _engine->on_init();
227
228 // 触发交易时段开始事件
229 _engine->on_session_begin();
230
231 // 先检查当前时间
232 uint32_t offTime = _s_info->offsetTime(_engine->get_min_time(), true);
233
234 // 创建并启动定时器线程
235 _thrd.reset(new StdThread([this, offTime](){
236 while (!_stopped)
237 {
238 // 如果当前时间在交易时间内
239 if (_time != UINT_MAX && _s_info->isInTradingTime(_time / 100000, true))
240 {
241 // 短暂停以减少CPU占用
242 std::this_thread::sleep_for(std::chrono::milliseconds(10));
243 // 获取当前时间
244 uint64_t now = TimeUtils::getLocalTimeNow();
245
246 // 如果到达下次检查时间且最后触发位置小于当前位置
247 if (now >= _next_check_time && _last_emit_pos < _cur_pos)
248 {
249 // 触发数据回放模块
250 StdUniqueLock lock(_mtx);
251
252 // 优先修改时间标记
253 _last_emit_pos = _cur_pos;
254
255 // 获取当前分钟时间
256 uint32_t thisMin = _s_info->minuteToTime(_cur_pos);
257 _time = thisMin;
258
259 // 如果thisMin是0, 说明换日了
260 // 这里是本地计时导致的换日, 说明日期其实还是老日期, 要自动+1
261 // 同时因为时间是235959xxx, 所以也要手动置为0
262 if (thisMin == 0)
263 {
264 uint32_t lastDate = _date;
265 // 获取下一日日期
266 _date = TimeUtils::getNextDate(_date);
267 _time = 0;
268 WTSLogger::info("Data automatically changed at time 00:00: {} -> {}", lastDate, _date);
269 }
270
271 WTSLogger::info("Minute bar {}.{:04d} closed automatically", _date, thisMin);
272 // 通知数据存储器分钟线结束
273 if (_store)
274 _store->onMinuteEnd(_date, thisMin);
275
276 // 通知引擎分钟线结束
277 _engine->on_minute_end(_date, thisMin);
278
279 // 判断是否到达交易时段结束时间
280 uint32_t offMin = _s_info->offsetTime(thisMin, true);
281 if (offMin >= _s_info->getCloseTime(true))
282 {
283 // 触发交易时段结束事件
284 _engine->on_session_end();
285 }
286
287 // 设置引擎当前日期时间
288 if (_engine)
289 _engine->set_date_time(_date, thisMin, 0);
290 }
291 }
292 else // 非交易时间处理
293 {
294 // 收盘以后，如果发现上次触发的位置不等于总的分钟数，说明少了最后一分钟的闭合逻辑
295 uint32_t total_mins = _s_info->getTradingMins();
296 if (_time != UINT_MAX && _last_emit_pos != 0 && _last_emit_pos < total_mins && offTime >= _s_info->getCloseTime(true))
297 {
298 WTSLogger::warn("Tradingday {} will be ended forcely, last_emit_pos: {}, time: {}", _engine->getTradingDate(), _last_emit_pos.fetch_add(0), _time);
299
300 // 触发数据回放模块
301 StdUniqueLock lock(_mtx);
302
303 // 优先修改时间标记
304 _last_emit_pos = total_mins;
305
306 bool bEndingTDate = true;
307 // 获取收盘时间
308 uint32_t thisMin = _s_info->getCloseTime(false);
309 uint32_t offMin = _s_info->getCloseTime(true);
310
311 WTSLogger::info("Minute bar {}.{:04d} closed automatically", _date, thisMin);
312 // 通知数据存储器分钟线结束
313 if (_store)
314 _store->onMinuteEnd(_date, thisMin, _engine->getTradingDate());
315
316 // 触发交易时段结束事件
317 _engine->on_session_end();
318
319 }
320 else
321 {
322 // 非交易时间内休眠时间更长
323 std::this_thread::sleep_for(std::chrono::seconds(10));
324 }
325 }
326
327 }
328 }));
329}
