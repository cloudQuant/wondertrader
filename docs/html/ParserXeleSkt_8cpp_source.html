35inline uint32_t strToTime(const char* strTime)
36{
37 static char str[10] = { 0 };
38 const char *pos = strTime;
39 int idx = 0;
40 auto len = strlen(strTime);
41 for (std::size_t i = 0; i < len; i++)
42 {
43 if (strTime[i] != ':')
44 {
45 str[idx] = strTime[i];
46 idx++;

154bool ParserXeleSkt::prepare()
155{
156 boost::asio::io_context io_context;
157 ip::tcp::socket s(io_context);
158 boost::system::error_code ec;
159 s.connect(_tcp_ep, ec);
160 if(ec)
161 {
162 write_log(_sink, LL_ERROR, "[ParserXeleSkt] Preparing snapshots failed, connecting tcp tunnel {}:{} error: {}", _tcp_host, _tcp_port, ec.message());
163 return false;
164 }
165
166 if (_tick_cache == NULL)
167 _tick_cache = TickCache::create();
168
169 write_log(_sink, LL_INFO, "[ParserXeleSkt] Preparing snapshots via tcp tunnel {}:{}...", _tcp_host, _tcp_port);
170
171 boost::array<char, 4096> buffer;
172 std::string content;
173 auto snap_size = sizeof(CXeleShfeSnapShot) + sizeof(CXeleShfeMarketHead);
174 for (;;)
175 {
176 std::size_t reply_length = boost::asio::read(s, boost::asio::buffer(buffer), ec);
177 if (ec)
178 {
179 //接收错误
180 break;
181 }
182
183 if (reply_length > 0)
184 content.append(buffer.data(), reply_length);
185 }
186
187 while (content.size() > snap_size)
188 {
189 CXeleShfeMarketHead *mh = (CXeleShfeMarketHead *)(content.data());
190 int8_t version = mh->Version;
191 if (version != XELE_MD_DATA_VERSION)
192 break;
193
194 int8_t type = mh->Type;
195 if (type != MESSAGE_SNAP_SHOT)
196 break;
197
198 CXeleShfeSnapShot *p = (CXeleShfeSnapShot *)(content.data() + sizeof(CXeleShfeMarketHead));
199 int instrumentNo = p->InstrumentNo;
200 WTSContractInfo* ct = _bd_mgr->getContract(p->InstrumentID);
201 if (ct != NULL)
202 {
203 auto it = _set_subs.find(ct->getFullCode());
204 if (it != _set_subs.end())
205 {
206
207 _price_scales[instrumentNo] = p->PriceTick;
208
209 WTSTickData* tick = WTSTickData::create(p->InstrumentID);
210 tick->setContractInfo(ct);
211
212 WTSTickStruct& quote = tick->getTickStruct();
213 wt_strcpy(quote.exchg, ct->getExchg());
214
215 quote.action_date = strToTime(p->ActionDay);
216 quote.action_time = strToTime(p->UpdateTime) * 1000 + p->UpdateMilliSec;
217
218 quote.price = checkValid(p->LastPrice);
219 quote.open = checkValid(p->OpenPrice);
220 quote.high = checkValid(p->HighestPrice);
221 quote.low = checkValid(p->LowestPrice);
222 quote.total_volume = p->Volume;
223 quote.trading_date = 0;
224 quote.settle_price = checkValid(p->SettlementPrice);
225 quote.total_turnover = p->Turnover;
226
227 quote.open_interest = p->OpenInterest;
228
229 quote.upper_limit = p->UpperLimitPrice;
230 quote.lower_limit = p->LowerLimitPrice;
231
232 quote.pre_close = p->PreClosePrice;
233 quote.pre_settle = p->PreSettlementPrice;
234 quote.pre_interest = p->PreOpenInterest;
235
236 //委卖价格
237 quote.ask_prices[0] = p->AskPrice;
238 //委买价格
239 quote.bid_prices[0] = p->BidPrice;
240 //委卖量
241 quote.ask_qty[0] = p->AskVolume;
242 //委买量
243 quote.bid_qty[0] = p->BidVolume;
244
245 _tick_cache->add(instrumentNo, tick, false);
246
247 //if (_sink)
248 // _sink->handleQuote(tick, 1);
249 }
250 }
251
252 content.erase(0, snap_size);
253 }
254
255 s.close(ec);
256
257 write_log(_sink, LL_INFO, "[ParserXeleSkt] All snapshots synced");
258
259 return true;
260}

372 int8_t version = mh->Version;
