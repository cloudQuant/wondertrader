658 _api = NULL;
659 }
660
661 std::stringstream ss;
662 ss << _flowdir << "flows/" << _user << "/";

665 if (_api == NULL)

668 _sink->handleEvent(WTE_Connect, -1);
669 write_log(_sink,LL_ERROR, "[TraderrXTP] Module initializing failed");
670
671 StdThreadPtr thrd(new StdThread([this](){
672 std::this_thread::sleep_for(std::chrono::seconds(2));
673 write_log(_sink,LL_WARN, "[TraderrXTP] {} reconnecting...", _user.c_str());
674 reconnect();
675 }));
676 return;
677 }
678
679 _api->SubscribePublicTopic(_quick ? XTP_TERT_QUICK : XTP_TERT_RESUME);
680 _api->SetSoftwareVersion("1.0.0"); //设定此软件的开发版本号,用户自定义
681 _api->SetSoftwareKey(_acckey.c_str());//设定用户的开发代码,在XTP申请开户时,由xtp人员提供
