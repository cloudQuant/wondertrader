119bool ParserUDP::reconnect(uint32_t flag /* = 3 */)
120{
121 if(flag & 1)
122 {//建立广播通道
123 if (_b_socket != NULL)
124 {
125 _b_socket->close();
126 delete _b_socket;
127 _b_socket = NULL;
128 }
129
130 _b_socket = new ip::udp::socket(_io_service);
131
132 _b_socket->open(_broad_ep.protocol());
133 _b_socket->set_option(ip::udp::socket::reuse_address(true));
134 _b_socket->set_option(ip::udp::socket::broadcast(true));
135 _b_socket->set_option(ip::udp::socket::receive_buffer_size(8 * 1024 * 1024));
136 _b_socket->bind(_broad_ep);
137
138 _b_socket->async_receive_from(buffer(_b_buffer), _broad_ep,
139 boost::bind(&ParserUDP::handle_read, this,
140 boost::asio::placeholders::error,
141 boost::asio::placeholders::bytes_transferred, true));
142 }
143
144 if (flag & 2)
145 {
146 std::queue<std::string> emptyQue;
147 {
148 StdUniqueLock lock(_mtx_queue);
149 _send_queue.swap(emptyQue);
150
151 //建立订阅通道
152 if (_s_socket != NULL)
153 {
154 _s_socket->close();
155 delete _s_socket;
156 _s_socket = NULL;
157 }
158
159 _s_inited = false;
160 _s_socket = new ip::udp::socket(_io_service, ip::udp::endpoint(ip::udp::v4(), 0));
161 }
162
163 subscribe();
164 }
165 return true;
166}

168void ParserUDP::subscribe()
169{
170 std::string data;
171 data.resize(sizeof(UDPReqPacket), 0);
172 UDPReqPacket* req = (UDPReqPacket*)data.data();
173 req->_type = UDP_MSG_SUBSCRIBE;
174 uint32_t length = 0;
175 for (auto& code : _set_subs)
176 {
177 if (length > 0)
178 {
179 req->_data[length] = ',';
180 length++;
181 }
182
183 std::size_t pos = code.find('.');
184 if (pos != std::string::npos)
185 strcpy(req->_data + length, (char*)code.c_str() + pos + 1);
186 else
187 strcpy(req->_data + length, code.c_str());
188
189 length += code.size();
190
191 if (length > 1000)
192 {
193 StdUniqueLock lock(_mtx_queue);
194 _send_queue.push(data);
195 }
196
197 data.resize(sizeof(UDPReqPacket), 0);
198 req = (UDPReqPacket*)data.data();
199 req->_type = UDP_MSG_SUBSCRIBE;
200 length = 0;
201 }
202
203 do_send();
204}

206void ParserUDP::do_send()
207{
208 StdUniqueLock lock(_mtx_queue);
209 if (_send_queue.empty() || _s_socket == NULL)
210 return;
211
212 write_log(_sink, LL_INFO, "[ParserUDP] {} Ticks subscribing packets still await", _send_queue.size());
213 std::string& data = _send_queue.front();
214
215 _s_socket->async_send_to(boost::asio::buffer(data, data.size()), _server_ep,
216 boost::bind(&ParserUDP::handle_write, this, boost::asio::placeholders::error));
217
218 if(!_s_inited)
219 {
220 _s_inited = true;
221 _s_socket->async_receive_from(buffer(_s_buffer), _server_ep,
222 boost::bind(&ParserUDP::handle_read, this,
223 boost::asio::placeholders::error,
224 boost::asio::placeholders::bytes_transferred, false));
225 }

308void ParserUDP::handle_read(const boost::system::error_code& e, std::size_t bytes_transferred, bool isBroad /* = true */)
309{
310 if(e)
311 {
312 if(_sink)
313 _sink->handleEvent(WPE_Close, 0);
314
315 write_log(_sink, LL_ERROR, "[ParserUDP] Error occured while receiving from {}: {}({})", isBroad?"broad port":"subscribe port", e.message().c_str(), e.value());
316
317 if (!_stopped && !_connecting)
318 {
319 std::this_thread::sleep_for(std::chrono::seconds(2));
320 reconnect(isBroad?1:2);
321 return;
322 }
323 }
324
325 if(_stopped || bytes_transferred<=0)
326 return;
327
328 extract_buffer(bytes_transferred, isBroad);
329
330 if (isBroad && _b_socket)
331 {
332 _b_socket->async_receive_from(buffer(_b_buffer), _broad_ep,
333 boost::bind(&ParserUDP::handle_read, this,
334 boost::asio::placeholders::error,
335 boost::asio::placeholders::bytes_transferred, true));
336 }
337 else if(!isBroad && _s_socket)
338 {
339 _s_socket->async_receive_from(buffer(_s_buffer), _server_ep,
340 boost::bind(&ParserUDP::handle_read, this,
341 boost::asio::placeholders::error,

347void ParserUDP::extract_buffer(uint32_t length, bool isBroad /* = true */)
348{
349 UDPPacketHead* header = NULL;
350 if(isBroad)
351 header = (UDPTickPacket*)_b_buffer.data();
352 else
353 header = (UDPTickPacket*)_s_buffer.data();
354
355 if (header->_type == UDP_MSG_PUSHTICK || header->_type == UDP_MSG_SUBSCRIBE)
356 {
357 UDPTickPacket* packet = (UDPTickPacket*)header;
358 const char* fullCode = fmtutil::format("{}.{}", packet->_data.exchg, packet->_data.code);
359 auto it = _set_subs.find(fullCode);
360 if (it != _set_subs.end())
361 {
362 WTSTickData* curTick = WTSTickData::create(packet->_data);
363 if (_sink)
364 _sink->handleQuote(curTick, 0);
365
366 curTick->release();
367
368 static uint32_t recv_cnt = 0;
369 recv_cnt++;
370 if (recv_cnt % _gpsize == 0)
371 write_log(_sink, LL_DEBUG, "[ParserUDP] {} ticks received in total", recv_cnt);
372 }
373 }
374 else if (header->_type == UDP_MSG_PUSHORDDTL)
375 {
376 UDPOrdDtlPacket* packet = (UDPOrdDtlPacket*)header;
377 const char* fullCode = fmtutil::format("{}.{}", packet->_data.exchg, packet->_data.code);
378 auto it = _set_subs.find(fullCode);
379 if (it != _set_subs.end())
380 {
381 WTSOrdDtlData* curData = WTSOrdDtlData::create(packet->_data);
382 if (_sink)
383 _sink->handleOrderDetail(curData);
384
385 curData->release();
386
387 static uint32_t recv_cnt = 0;
388 recv_cnt++;
389 if (recv_cnt % _gpsize == 0)
390 write_log(_sink, LL_DEBUG, "[ParserUDP] {} order details received in total", recv_cnt);
391 }
392 }
393 else if (header->_type == UDP_MSG_PUSHORDQUE)
394 {
395 UDPOrdQuePacket* packet = (UDPOrdQuePacket*)header;
396 const char* fullCode = fmtutil::format("{}.{}", packet->_data.exchg, packet->_data.code);
397 auto it = _set_subs.find(fullCode);
398 if (it != _set_subs.end())
399 {
400 WTSOrdQueData* curData = WTSOrdQueData::create(packet->_data);
401 if (_sink)
402 _sink->handleOrderQueue(curData);
403
404 curData->release();
405
406 static uint32_t recv_cnt = 0;
407 recv_cnt++;
408 if (recv_cnt % _gpsize == 0)
409 write_log(_sink, LL_DEBUG, "[ParserUDP] {} order queues received in total", recv_cnt);
410 }
411 }
412 else if (header->_type == UDP_MSG_PUSHTRANS)
413 {
414 UDPTransPacket* packet = (UDPTransPacket*)header;
415 const char* fullCode = fmtutil::format("{}.{}", packet->_data.exchg, packet->_data.code);
416 auto it = _set_subs.find(fullCode);
