551 bool execute_task() volatile
552 {
553 function0<void> task;
554
555 { // fetch task
556 pool_type* lockedThis = const_cast<pool_type*>(this);
557 recursive_mutex::scoped_lock lock(lockedThis->m_monitor);
558
559 // decrease number of threads if necessary
560 if(m_worker_count > m_target_worker_count)
561 {
562 return false; // terminate worker
563 }
564
565
566 // wait for tasks
567 while(lockedThis->m_scheduler.empty())
568 {
569 // decrease number of workers if necessary
570 if(m_worker_count > m_target_worker_count)
571 {
572 return false; // terminate worker
573 }
574 else
575 {
576 m_active_worker_count--;
577 lockedThis->m_worker_idle_or_terminated_event.notify_all();
578 lockedThis->m_task_or_terminate_workers_event.wait(lock);
579 m_active_worker_count++;
580 }
581 }
582
583 task = lockedThis->m_scheduler.top();
584 lockedThis->m_scheduler.pop();
585 }
586
587 // call task function
588 if(task)
589 {
590 task();
591 }
592
593 //guard->disable();
594 return true;
595 }
