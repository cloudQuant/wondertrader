387template<typename HeaderType, typename T>
388void* WtDataWriterAD::resizeRTBlock(BoostMFPtr& mfPtr, uint32_t nCount)
389{
390 // 检查映射文件指针是否有效
391 if (mfPtr == NULL)
392 return NULL;
393
394 //调用该函数之前,应该已经申请了写锁了
395 // 获取块头部指针
396 RTBlockHeader* tBlock = (RTBlockHeader*)mfPtr->addr();
397 // 如果当前容量足够，直接返回
398 if (tBlock->_capacity >= nCount)
399 return mfPtr->addr();
400
401 // 获取文件名和计算新旧大小
402 const char* filename = mfPtr->filename();
403 uint64_t uOldSize = sizeof(HeaderType) + sizeof(T)*tBlock->_capacity;
404 uint64_t uNewSize = sizeof(HeaderType) + sizeof(T)*nCount;
405 // 创建空数据块用于扩展文件
406 std::string data;
407 data.resize((std::size_t)(uNewSize - uOldSize), 0);
408 try
409 {
410 // 打开文件并在尾部添加空间
411 BoostFile f;
412 f.open_existing_file(filename);
413 f.seek_to_end();
414 f.write_file(data.c_str(), data.size());
415 f.close_file();
416 }
417 catch(std::exception& ex)
418 {
419 // 扩展文件失败时记录错误并返回原指针
420 pipe_writer_log(_sink, LL_ERROR, "Exception occured while expanding RT cache file of {}[{}]: {}", filename, uNewSize, ex.what());
421 return mfPtr->addr();
422 }
423
424 // 创建新的映射文件对象
425 BoostMappingFile* pNewMf = new BoostMappingFile();
426 if (!pNewMf->map(filename))
427 {
428 // 映射失败时清理资源并返回NULL
429 delete pNewMf;
430 return NULL;
431 }
432
433 // 更新智能指针
434 mfPtr.reset(pNewMf);
435
436 // 更新块头部并返回新的内存地址
437 tBlock = (RTBlockHeader*)mfPtr->addr();
438 tBlock->_capacity = nCount;
439 return mfPtr->addr();
440}

449bool WtDataWriterAD::writeTick(WTSTickData* curTick, uint32_t procFlag)
450{
451 // 检查输入参数是否有效
452 if (curTick == NULL)
453 return false;
454
455 // 增加引用计数，确保在异步任务中不会被释放
456 curTick->retain();
457 // 将写入任务添加到异步任务队列
458 pushTask([this, curTick, procFlag](){
459
460 do
461 {
462 // 获取合约信息
463 WTSContractInfo* ct = curTick->getContractInfo();
464 if(ct == NULL)
465 break;
466
467 // 获取商品信息
468 WTSCommodityInfo* commInfo = ct->getCommInfo();
469
470 // 根据交易时段过滤
471 if (!_sink->canSessionReceive(commInfo->getSession()))
472 break;
473
474 // 先更新Tick缓存，如果更新失败则终止处理
475 if (!updateTickCache(ct, curTick, procFlag))
476 break;
477
478 // 如果未禁用Tick存储，则将数据写入数据库
479 if(!_disable_tick)
480 pipeToTicks(ct, curTick);
481
482 // 更新K线缓存，生成各周期K线
483 updateBarCache(ct, curTick);
484
485 // 广播Tick数据给监听器
486 _sink->broadcastTick(curTick);
487
488 // 统计并定期输出日志
489 static wt_hashmap<std::string, uint64_t> _tcnt_map;
490 _tcnt_map[curTick->exchg()]++;
491 if (_tcnt_map[curTick->exchg()] % _log_group_size == 0)
492 {
493 pipe_writer_log(_sink, LL_INFO, "{} ticks received from exchange {}",_tcnt_map[curTick->exchg()], curTick->exchg());
494 }
495 } while (false);
496
497 // 释放Tick数据的引用
498 curTick->release();
499 });
500 return true;
501}

509void WtDataWriterAD::pushTask(TaskInfo task)
510{
511 // 如果开启了异步任务模式
512 if(_async_task)
513 {
514 // 加锁并将任务添加到队列
515 StdUniqueLock lck(_task_mtx);
516 _tasks.push(task);
517 // 通知等待中的任务处理线程
518 _task_cond.notify_all();
519 }
520 else
521 {
522 // 如果不是异步模式，直接执行任务
523 task();
524 return;
525 }
526
527 // 如果任务线程还没有创建，则创建一个新的任务处理线程
528 if(_task_thrd == NULL)
529 {
530 _task_thrd.reset(new StdThread([this](){
531 // 线程主循环，直到收到终止信号
532 while (!_terminated)
533 {
534 // 如果任务队列为空，等待新任务
535 if(_tasks.empty())
536 {
537 StdUniqueLock lck(_task_mtx);
538 _task_cond.wait(_task_mtx);
539 continue;
540 }
541
542 // 创建临时队列并交换任务，减少锁的持有时间
543 std::queue<TaskInfo> tempQueue;
544 {
545 StdUniqueLock lck(_task_mtx);
546 tempQueue.swap(_tasks);
547 }
548
549 // 处理临时队列中的所有任务
550 while(!tempQueue.empty())
551 {
552 TaskInfo& curTask = tempQueue.front();
553 curTask();
554 tempQueue.pop();
555 }
556 }
557 }));
558 }

567void WtDataWriterAD::pipeToTicks(WTSContractInfo* ct, WTSTickData* curTick)
568{
569 // 获取对应的Tick数据库
570 WtLMDBPtr db = get_t_db(ct->getExchg(), ct->getCode());
571 if (db)
572 {
573 // 将时间转换为偏移时间，便于根据交易日筛选历史数据
574 // 这里将时间转成偏移时间，并用交易日作为日期
575 uint32_t actTime = curTick->actiontime();
576 // 计算偏移时间：将小时分钟部分转换为偏移时间，保留毫秒部分
577 uint32_t offTime = ct->getCommInfo()->getSessionInfo()->offsetTime(actTime / 100000, true) * 100000 + actTime % 100000;
578
579 // 创建LMDB键值，包含交易所、合约代码、交易日和偏移时间
580 LMDBHftKey key(ct->getExchg(), ct->getCode(), curTick->tradingdate(), offTime);
581 // 创建查询对象并将数据写入数据库
582 WtLMDBQuery query(*db);
583 if (!query.put_and_commit((void*)&key, sizeof(key), &curTick->getTickStruct(), sizeof(WTSTickStruct)))
584 {
585 // 写入失败时记录错误日志
586 pipe_writer_log(_sink, LL_ERROR, "pipe tick of {} to db failed: {}", ct->getFullCode(), db->errmsg());
587 }
588 }
589
590 // 遍历所有外部数据导出器
591 for(auto& item : _dumpers)
592 {
593 // 获取导出器ID和对象

596 if (dumper == NULL)
597 continue;
598
599 // 调用导出器将Tick数据导出
600 bool bSucc = dumper->dumpHisTicks(ct->getFullCode(), curTick->tradingdate(), &curTick->getTickStruct(), 1);
601 if (!bSucc)

603 // 导出失败时记录错误日志

1258 if (decimal::eq(newTick.high, 0))
1259 newTick.high = newTick.price;
1260
1261 // 如果最低价为0，则使用当前价格作为最低价
1262 if (decimal::eq(newTick.low, 0))
1263 newTick.low =newTick.price;

1267 // 设置前收盘价和前持仓量
1268 item._tick.pre_close = pre_close;
1269 item._tick.pre_interest = pre_interest;
1270 }
1271
1272 // 记录新交易日的第一个Tick数据的日志
1273 pipe_writer_log(_sink, LL_INFO, "First tick of new tradingday {},{}.{},{},{},{},{},{}",
1274 newTick.trading_date, curTick->exchg(), curTick->code(), curTick->price(), curTick->volume(),
1275 curTick->turnover(), curTick->openinterest(), curTick->additional());
1276 }
1277 else // 如果是同一交易日的数据
1278 {
1279 // 如果缓存里的数据日期大于最新行情的日期
1280 // 或者缓存里的时间大于等于最新行情的时间,数据就不需要处理

1293 {

1295 curTick->exchg(), curTick->code(), curTick->actiondate(), curTick->actiontime(), curTick->totalvolume(), item._tick.total_volume);
1296 return false;
1297 }

1300 // 这种情况一般是郑商所的特殊情况，处理方式是时间戳+200毫秒

1302 // 由于一秒内可能出现多笔数据，将时间增量从500毫秒改为200毫秒
1303 if(newTick.action_date == item._tick.action_date && newTick.action_time <= item._tick.action_time && newTick.total_volume >= item._tick.total_volume)
1304 {
1305 // 增加200毫秒以区分不同的Tick
1306 newTick.action_time += 200;
1307 }
1308
1309 // 根据处理标志进行不同的处理
1310 if(procFlag == 0) // 不预处理，直接复制
1311 {
1312 memcpy(&item._tick, &newTick, sizeof(WTSTickStruct));
1313 }
1314 else if (procFlag == 1) // 预处理成交量和成交额
1315 {
1316 // 计算单笔成交量、成交额和持仓量变化
1317 newTick.volume = newTick.total_volume - item._tick.total_volume;
1318 newTick.turn_over = newTick.total_turnover - item._tick.total_turnover;
1319 newTick.diff_interest = newTick.open_interest - item._tick.open_interest;
1320
1321 // 复制新的Tick数据到缓存
1322 memcpy(&item._tick, &newTick, sizeof(WTSTickStruct));
1323 }
1324 else if (procFlag == 2) // 特殊处理模式
1325 {
1326 // 自动累加处理
1327 // 如果总成交量为0，则需要累加上一笔的总成交量
1328 if(decimal::eq(newTick.total_volume, 0))
1329 newTick.total_volume = newTick.volume + item._tick.total_volume;
1330
1331 // 如果总成交额为0，则需要累加上一笔的总成交额
1332 if (decimal::eq(newTick.total_turnover, 0))
1333 newTick.total_turnover = newTick.turn_over + item._tick.total_turnover;
1334
1335 // 如果开盘价为0，则使用当前价格作为开盘价
1336 if (decimal::eq(newTick.open, 0))
1337 newTick.open = newTick.price;
1338
1339 // 如果最高价为0，则取当前价格和缓存中最高价的最大值
1340 if (decimal::eq(newTick.high, 0))
1341 newTick.high = max(newTick.price, item._tick.high);
1342
1343 // 如果最低价为0，则取当前价格和缓存中最低价的最大值
1344 // 注意：这里可能有错误，应该是取最小值而非最大值
1345 if (decimal::eq(newTick.low, 0))
1346 newTick.low = max(newTick.price, item._tick.low);
1347
1348 // 复制新的Tick数据到缓存
1349 memcpy(&item._tick, &newTick, sizeof(WTSTickStruct));
1350 }
1351 }
1352
1353 // 更新成功
1354 return true;
