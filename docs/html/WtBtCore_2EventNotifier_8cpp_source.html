66bool EventNotifier::init(WTSVariant* cfg)
67{
68 // 检查是否启用了事件通知器
69 if (!cfg->getBoolean("active"))
70 return false;
71
72 // 获取服务器URL
73 m_strURL = cfg->getCString("url");
74 // 构造消息队列模块的文件名，根据不同系统进行包装
75 std::string module = DLLHelper::wrap_module("WtMsgQue", "lib");
76 // 先在当前工作目录下查找模块
77 std::string dllpath = WtHelper::getCWD() + module;
78 // 如果没有，则在安装目录下查找
79 if (!StdFile::exists(dllpath.c_str()))
80 dllpath = WtHelper::getInstDir() + module;
81
82 // 加载消息队列模块动态库
83 DllHandle dllInst = DLLHelper::load_library(dllpath.c_str());
84 if (dllInst == NULL)
85 {
86 // 加载失败，记录错误日志
87 WTSLogger::error("MQ module %{} loading failed", dllpath.c_str());
88 return false;
89 }
90
91 // 获取创建服务器的函数指针
92 _creator = (FuncCreateMQServer)DLLHelper::get_symbol(dllInst, "create_server");
93 if (_creator == NULL)
94 {
95 // 获取函数指针失败，释放动态库并记录错误日志
96 DLLHelper::free_library(dllInst);
97 WTSLogger::error("MQ module {} is not compatible", dllpath.c_str());
98 return false;
99 }
100
101 // 获取其他必要的函数指针
102 _remover = (FuncDestroyMQServer)DLLHelper::get_symbol(dllInst, "destroy_server");
103 _publisher = (FundPublishMessage)DLLHelper::get_symbol(dllInst, "publish_message");
104 _register = (FuncRegCallbacks)DLLHelper::get_symbol(dllInst, "regiter_callbacks");
105
106 // 注册日志回调函数
107 _register(on_mq_log);
108
109 // 创建消息队列服务器，true表示服务器模式
110 _mq_sid = _creator(m_strURL.c_str(), true);
111
112 // 记录初始化成功的日志
113 WTSLogger::info("EventNotifier initialized with channel {}", m_strURL.c_str());
114
115 return true;
116}

157void EventNotifier::notifyFund(const char* topic, uint32_t uDate, double total_profit, double dynprofit, double dynbalance, double total_fee)
158{
159 // 创建输出字符串
160 std::string output;
161 {
162 // 创建 JSON 对象
163 rj::Document root(rj::kObjectType);
164 rj::Document::AllocatorType &allocator = root.GetAllocator();
165
166 // 添加各个资金信息字段
167 root.AddMember("date", uDate, allocator);
168 root.AddMember("total_profit", total_profit, allocator);
169 root.AddMember("dynprofit", dynprofit, allocator);
170 root.AddMember("dynbalance", dynbalance, allocator);
171 root.AddMember("total_fee", total_fee, allocator);
172
173 // 将 JSON 对象转换为格式化的字符串
174 rj::StringBuffer sb;
175 rj::PrettyWriter<rj::StringBuffer> writer(sb);
176 root.Accept(writer);
177
178 // 获取最终的 JSON 字符串
179 output = sb.GetString();
180 }
181
182 // 如果发布者函数存在，则发布消息
183 if (_publisher)
184 _publisher(_mq_sid, topic, (const char*)output.c_str(), (unsigned long)output.size());
185}
