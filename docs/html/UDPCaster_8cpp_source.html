107bool UDPCaster::init(WTSVariant* cfg, WTSBaseDataMgr* bdMgr, DataManager* dtMgr)
108{
109 // 保存基础数据管理器和数据管理器指针
110 m_bdMgr = bdMgr;
111 m_dtMgr = dtMgr;
112
113 // 如果广播器未激活，直接返回失败
114 if (!cfg->getBoolean("active"))
115 return false;
116
117 // 处理普通广播配置
118 WTSVariant* cfgBC = cfg->get("broadcast");
119 if (cfgBC)
120 {
121 for (uint32_t idx = 0; idx < cfgBC->size(); idx++)
122 {
123 WTSVariant* cfgItem = cfgBC->get(idx);
124 // 添加广播接收者，包含主机、端口和数据类型

126 }
127 }
128
129 // 处理组播配置
130 WTSVariant* cfgMC = cfg->get("multicast");
131 if (cfgMC)
132 {
133 for (uint32_t idx = 0; idx < cfgMC->size(); idx++)
134 {
135 WTSVariant* cfgItem = cfgMC->get(idx);
136 // 添加组播接收者，包含主机、端口、发送端口和数据类型
137 addMRecver(cfgItem->getCString("host"), cfgItem->getInt32("port"), cfgItem->getInt32("sendport"), cfgItem->getUInt32("type"));
138 }
139 }

159void UDPCaster::start(int sport)
160{
161 // 如果有任何类型的接收者，初始化广播套接字
162 if (!m_listFlatRecver.empty() || !m_listJsonRecver.empty() || !m_listRawRecver.empty())
163 {
164 // 创建UDP广播套接字，使用随机端口
165 m_sktBroadcast.reset(new UDPSocket(m_ioservice, boost::asio::ip::udp::endpoint(boost::asio::ip::udp::v4(), 0)));
166 // 允许广播模式
167 boost::asio::socket_base::broadcast option(true);
168 m_sktBroadcast->set_option(option);
169 }
170
171 try
172 {
173 // 创建UDP订阅套接字，绑定到指定端口
174 m_sktSubscribe.reset(new UDPSocket(m_ioservice, boost::asio::ip::udp::endpoint(boost::asio::ip::udp::v4(), sport)));
175 }
176 catch(...)
177 {
178 // 捕获并记录创建订阅套接字时的异常
179 WTSLogger::error("Exception raised while start subscribing service @ port {}", sport);
180 }
181
182 // 开始异步接收数据
183 do_receive();
184
185 // 启动IO服务线程
186 m_thrdIO.reset(new StdThread([this](){
187 try
188 {
189 // 运行Boost.Asio事件循环
190 m_ioservice.run();
191 }
192 catch(...)
193 {
194 // 出现异常时停止IO服务
195 m_ioservice.stop();
196 }
197 }));
198}

204void UDPCaster::stop()
205{
206 // 设置终止标志，让广播循环退出
207 m_bTerminated = true;
208 // 停止IO服务
209 m_ioservice.stop();
210 // 等待IO线程结束
211 if (m_thrdIO)
212 m_thrdIO->join();
213
214 // 唯醒所有等待广播的线程，让它们检测到终止标志
215 m_condCast.notify_all();
216 // 等待广播线程结束
217 if (m_thrdCast)
218 m_thrdCast->join();

225void UDPCaster::do_receive()
226{
227 // 开始异步接收数据，并指定接收完成后的处理函数
228 m_sktSubscribe->async_receive_from(boost::asio::buffer(m_data, max_length), m_senderEP,
229 [this](boost::system::error_code ec, std::size_t bytes_recvd)
230 {
231 // 如果发生错误，重新开始接收
232 if(ec)
233 {
234 do_receive();
235 return;
236 }
237
238 // 检查接收的数据是否为完整的请求包
239 if (bytes_recvd == sizeof(UDPReqPacket))
240 {
241 // 将接收的数据转换为请求包格式
242 UDPReqPacket* req = (UDPReqPacket*)m_data;
243
244 std::string data;
245 // 处理订阅类型的请求
246 if (req->_type == UDP_MSG_SUBSCRIBE)
247 {
248 // 使用逗号分割多个代码
249 const StringVector& ay = StrUtil::split(req->_data, ",");
250 std::string code, exchg;
251 // 逐个处理代码
252 for(const std::string& fullcode : ay)
253 {
254 // 处理带有交易所前缀的完整代码格式 (exchg.code)
255 auto pos = fullcode.find(".");
256 if (pos == std::string::npos)
257 code = fullcode; // 如果没有点，则全部作为代码
258 else
259 {
260 // 如果有点，则分离交易所和代码
261 code = fullcode.substr(pos + 1);
262 exchg = fullcode.substr(0, pos);
263 }
264 // 根据代码和交易所获取合约信息
265 WTSContractInfo* ct = m_bdMgr->getContract(code.c_str(), exchg.c_str());
266 if (ct == NULL)
267 continue; // 无效合约，跳过
268
269 // 获取当前诸如数据
270 WTSTickData* curTick = m_dtMgr->getCurTick(code.c_str(), exchg.c_str());
271 if(curTick == NULL)
272 continue; // 无效诸如数据，跳过
273
274 // 创建并填充回复数据包
275 std::string* data = new std::string();
276 data->resize(sizeof(UDPTickPacket), 0);
277 UDPTickPacket* pkt = (UDPTickPacket*)data->data();
278 pkt->_type = req->_type;
279 // 将诸如数据结构体复制到数据包中
280 memcpy(&pkt->_data, &curTick->getTickStruct(), sizeof(WTSTickStruct));
281 // 释放诸如数据对象
282 curTick->release();
283 // 异步发送回复数据
284 m_sktSubscribe->async_send_to(
285 boost::asio::buffer(*data, data->size()), m_senderEP,
286 [this, data](const boost::system::error_code& ec, std::size_t /*bytes_sent*/)
287 {
288 // 发送完成后释放数据
289 delete data;
290 // 如果发送发生错误则记录
291 if (ec)
292 {
293 WTSLogger::error("Sending data on UDP failed: {}", ec.message().c_str());
294 }
295 });
296 }
297 }
298 }
299 else
300 {
301 // 如果数据不是预期的标准请求包格式，发送错误信息
302 std::string* data = new std::string("Can not indentify the command");
303 // 异步发送错误消息
304 m_sktSubscribe->async_send_to(
305 boost::asio::buffer(*data, data->size()), m_senderEP,
306 [this, data](const boost::system::error_code& ec, std::size_t /*bytes_sent*/)
307 {
308 // 发送完成后释放数据
309 delete data;
310 // 如果发送发生错误则记录
311 if (ec)
312 {
313 WTSLogger::error("Sending data on UDP failed: {}", ec.message().c_str());
314 }
315 });
316 }
317
318 // 继续接收下一个数据包
319 do_receive();
320 });
321}

442void UDPCaster::do_broadcast(WTSObject* data, uint32_t dataType)
443{
444 // 初始条件检查，确保广播套接字和数据对象存在，且终止标志未设置
445 if(m_sktBroadcast == NULL || data == NULL || m_bTerminated)
446 return;
447
448 {
449 // 加锁后将数据压入队列
450 StdUniqueLock lock(m_mtxCast);
451 m_dataQue.push(CastData(data, dataType));
452 }
453
454 // 如果广播线程尚未创建，则创建新线程
455 if(m_thrdCast == NULL)
456 {
457 m_thrdCast.reset(new StdThread([this](){
458
459 // 广播线程主循环
460 while (!m_bTerminated)
461 {
462 // 如果队列为空，则等待条件变量通知
463 if(m_dataQue.empty())
464 {
465 StdUniqueLock lock(m_mtxCast);
466 m_condCast.wait(lock);
467 continue;
468 }
469
470 // 创建临时队列并交换数据，减少锁的持有时间
471 std::queue<CastData> tmpQue;
472 {
473 StdUniqueLock lock(m_mtxCast);
474 tmpQue.swap(m_dataQue);
475 }
476
477 // 处理临时队列中的所有数据
478 while(!tmpQue.empty())
479 {
480 // 获取队列首部数据
481 const CastData& castData = tmpQue.front();
482
483 // 检查数据有效性
484 if (castData._data == NULL)
485 break;
486
487 // 如果有原始格式的接收者或组播组，则处理原始格式广播
488 if (!m_listRawGroup.empty() || !m_listRawRecver.empty())
489 {
490 // 初始化数据缓冲区
491 std::string buf_raw;
492 // 根据数据类型选择不同的处理方式
493 if (castData._datatype == UDP_MSG_PUSHTICK)
494 {
495 // 准备适价数据包
496 buf_raw.resize(sizeof(UDPTickPacket));
497 UDPTickPacket* pack = (UDPTickPacket*)buf_raw.data();
498 pack->_type = castData._datatype;
499 WTSTickData* curObj = (WTSTickData*)castData._data;
500 memcpy(&pack->_data, &curObj->getTickStruct(), sizeof(WTSTickStruct));
501 }
502 else if (castData._datatype == UDP_MSG_PUSHORDDTL)
503 {
504 // 准备委托明细数据包
505 buf_raw.resize(sizeof(UDPOrdDtlPacket));
506 UDPOrdDtlPacket* pack = (UDPOrdDtlPacket*)buf_raw.data();
507 pack->_type = castData._datatype;
508 WTSOrdDtlData* curObj = (WTSOrdDtlData*)castData._data;
509 memcpy(&pack->_data, &curObj->getOrdDtlStruct(), sizeof(WTSOrdDtlStruct));
510 }
511 else if (castData._datatype == UDP_MSG_PUSHORDQUE)
512 {
513 // 准备委托队列数据包
514 buf_raw.resize(sizeof(UDPOrdQuePacket));
515 UDPOrdQuePacket* pack = (UDPOrdQuePacket*)buf_raw.data();
516 pack->_type = castData._datatype;
517 WTSOrdQueData* curObj = (WTSOrdQueData*)castData._data;
518 memcpy(&pack->_data, &curObj->getOrdQueStruct(), sizeof(WTSOrdQueStruct));
519 }
520 else if (castData._datatype == UDP_MSG_PUSHTRANS)
521 {
522 // 准备成交数据包
523 buf_raw.resize(sizeof(UDPTransPacket));
524 UDPTransPacket* pack = (UDPTransPacket*)buf_raw.data();
525 pack->_type = castData._datatype;
526 WTSTransData* curObj = (WTSTransData*)castData._data;
527 memcpy(&pack->_data, &curObj->getTransStruct(), sizeof(WTSTransStruct));
528 }
529 else
530 {
531 // 不支持的数据类型，跳过处理
532 break;
533 }
534
535 // 向普通广播接收者发送数据
536 boost::system::error_code ec;
537 for (auto it = m_listRawRecver.begin(); it != m_listRawRecver.end(); it++)
538 {
539 const UDPReceiverPtr& receiver = (*it);
540 m_sktBroadcast->send_to(boost::asio::buffer(buf_raw), receiver->_ep, 0, ec);
541 // 出错时记录错误日志
542 if (ec)
543 {
544 WTSLogger::error("Error occured while sending to ({}:{}): {}({})",
545 receiver->_ep.address().to_string(), receiver->_ep.port(), ec.value(), ec.message());
546 }
547 }
548
549 // 向组播组发送数据
550 for (auto it = m_listRawGroup.begin(); it != m_listRawGroup.end(); it++)
551 {
552 const MulticastPair& item = *it;
553 it->first->send_to(boost::asio::buffer(buf_raw), item.second->_ep, 0, ec);
554 // 出错时记录错误日志
555 if (ec)
556 {
557 WTSLogger::error("Error occured while sending to ({}:{}): {}({})",
558 item.second->_ep.address().to_string(), item.second->_ep.port(), ec.value(), ec.message());
559 }
560 }
561 }
562
563 // 移除已处理的数据
564 tmpQue.pop();
565 }
566 }
567 }));
568 }
569 else
570 {
571 // 如果广播线程已存在，则唯醒它处理新数据
572 m_condCast.notify_all();
573 }
574}
