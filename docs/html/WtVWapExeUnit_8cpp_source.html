101inline double calTmStamp(uint32_t actiontime) //计算tick时间属于哪个时间单元
102{
103 string timestamp = to_string(actiontime);
104 int hour = stoi(timestamp.substr(0, 2));
105 int minute = stoi(timestamp.substr(2, 2));
106 double total_minute = 0;
107 if (hour < 9 || (hour == 9 && minute < 30)) {
108 total_minute = 0;
109 }
110 else if (hour < 11 || (hour == 11 && minute <= 30)) {
111 total_minute = (hour - 9) * 60 + minute - 30;
112 }
113 else if (hour < 13 || (hour == 13 && minute < 30)) {
114 total_minute = 120 + (hour - 11) * 60 + minute;
115 }
116 else if (hour < 15 || (hour == 15 && minute <= 0)) {
117 total_minute = 240 + (hour - 13) * 60 + minute - 30;
118 }
119 else {
120 total_minute = 240;
121 }
122 if (timestamp >= "113000000" && timestamp < "130000000") {
123 total_minute = 120;
124 }
125 total_minute += stoi(timestamp.substr(4, 2)) / 60;
126 total_minute += stoi(timestamp.substr(6, 3)) / 60000;
127 return total_minute ;//这里应该+1，对应vector 所以再-1

221{
222 if (!_orders_mon.has_order(localid))

245 _cancel_times++;
246 //撤单以后重发,一般是加点重发;对最小下单量的校验
247 fire_at_once(max(_min_open_lots, _this_target - realPos));
248 }
249 }
250
251 if (!isCanceled&&_cancel_cnt != 0) {//一般出现问题，需要返回检查 触发撤单 cnt++,onorder响应处理才会--
252 _ctx->writeLog(fmtutil::format("Order {} of {} hasn't canceled, error will be return ", localid, stdCode));

266void WtVWapExeUnit::on_channel_ready()
267{
268 _channel_ready = true;
269 double undone = _ctx->getUndoneQty(_code.c_str());
270 if (!decimal::eq(undone, 0) && !_orders_mon.has_order())
271 {//未完成单不在监控中，撤单
272 /*
273 * 如果未完成单不为0，而OMS没有订单
274 * 这说明有未完成单不在监控之中,全部撤销掉
275 * 因为这些订单没有本地订单号，无法直接进行管理
276 * 这种情况，就是刚启动的时候，上次的未完成单或者外部的挂单
277 */
278 _ctx->writeLog(fmt::format("{} unmanaged orders of {},cancel all", undone, _code).c_str());
279
280 bool isBuy = (undone > 0);
281 OrderIDs ids =_ctx->cancel(_code.c_str(), isBuy);
282 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime());
283 _cancel_cnt += ids.size();
284
285 _ctx->writeLog(fmtutil::format("[{}@{}]cancelcnt -> {}", __FILE__, __LINE__, _cancel_cnt));
286 }
287 else if (decimal::eq(undone, 0) && _orders_mon.has_order())
288 { /*
289 * By Wesey @ 2021.12.13
290 * 如果未完成单为0，但是OMS中是有订单的
291 * 说明OMS中是错单，需要清理掉，不然超时撤单就会出错
292 * 这种情况，一般是断线重连以后，之前下出去的订单，并没有真正发送到柜台
293 * 所以这里需要清理掉本地订单
294 */
295 _ctx->writeLog(fmtutil::format("Local orders of {} not confirmed in trading channel, clear all", _code.c_str()));
296 _orders_mon.clear_orders();
297 }
298 else
299 {
300 _ctx->writeLog(fmtutil::format("Unrecognized condition while channle ready, {:.2f} live orders of {} exists, local orders {}exist",
301 undone, _code.c_str(), _orders_mon.has_order() ? "" : "not "));
302 }
303 do_calc();
304}

316void WtVWapExeUnit::on_tick(WTSTickData * newTick)
317{
318 if (newTick == NULL || _code.compare(newTick->code()) != 0)
319 return;
320
321 bool isFirstTick = false;
322 //原来tick不为空 则要释放掉
323 if (_last_tick) {
324 _last_tick->release();
325 }
326 else {
327 isFirstTick = true;
328 //如果行情时间不在交易时间,这种情况一般是集合竞价的行情进来,下单会失败,所以直接过滤掉这笔行情
329 if (_sess_info != NULL && !_sess_info->isInTradingTime(newTick->actiontime() / 100000))
330 return;
331 }
332 //新的tick数据需要保留
333 _last_tick = newTick;
334 _last_tick->retain();
335
336 if (isFirstTick)//如果是第一笔tick,则检查目标仓位,不符合则下单
337 {
338 double newVol = _target_pos;
339 const char* stdCode = _code.c_str();
340 double undone = _ctx->getUndoneQty(stdCode);
341 double realPos = _ctx->getPosition(stdCode);
342 if (!decimal::eq(newVol, undone + realPos))
343 {//如果是第一笔TICK，且目标量==未完成+仓位，退出
344 do_calc();
345 }
346 }
347 else
348 {
349 uint64_t now = TimeUtils::getLocalTimeNow();
350 bool hasCancel = false;
351 if (_ord_sticky != 0 && _orders_mon.has_order())
352 {
353 _orders_mon.check_orders(_ord_sticky, now, [this, &hasCancel](uint32_t localid) {
354 if (_ctx->cancel(localid))
355 {
356 _cancel_cnt++;
357 _ctx->writeLog(fmt::format("Order expired, cancelcnt updated to {}", _cancel_cnt).c_str());//订单过期，撤单量更新
358 hasCancel = true;
359 }
360 });
361 }
362 if (!hasCancel && (now - _last_fire_time >= _fire_span * 1000))
363 {
364 do_calc();
365 }
366 }

412void WtVWapExeUnit::do_calc()
413{
414 CalcFlag flag(&_in_calc);
415 if (flag)
416 return;
417
418 StdUniqueLock lock(_mtx_calc);
419 const char* code = _code.c_str();
420 double undone = _ctx->getUndoneQty(code);
421 double newVol = get_real_target(_target_pos);
422 double realPos = _ctx->getPosition(code);
423 double diffQty = newVol - realPos;//目标差
424 if (!_channel_ready)
425 return;
426 //有正在撤销的订单,则不能进行下一轮计算
427 if (_cancel_cnt != 0)
428 {
429 _ctx->writeLog(fmt::format("{}尚有未完成的撤单指令，暂时退出本轮执行", _code).c_str());
430 return;
431 }
432 if (decimal::eq(diffQty, 0))
433 return;
434 //每一次发单要保障成交,所以如果有未完成单,说明上一轮没完成
435 //有未完成订单&&与实际仓位变动方向相反
436 //则需要撤销现有订单
437 if (decimal::lt(diffQty*undone, 0))
438 {
439 bool isBuy = decimal::gt(undone, 0);//undone>0,isbuy=1
440 OrderIDs ids = _ctx->cancel(code, isBuy);
441 if (!ids.empty())
442 {
443 _orders_mon.push_order(ids.data(), ids.size(), _ctx->getCurTime());
444 _cancel_cnt += ids.size();
445 _ctx->writeLog(fmtutil::format("[{}@{}] live opposite order of {} canceled, cancelcnt -> {}", __FILE__, __LINE__, _code.c_str(), _cancel_cnt));//相反的订单取消
446 }
447 return;
448 }
449 if (!decimal::eq(undone, 0))
450 {
451 _ctx->writeLog(fmt::format("{}上一轮有挂单未完成,暂时退出本轮执行", _code).c_str());
452 return;
453 }
454 if (_last_tick == NULL)
455 {
456 _ctx->writeLog(fmt::format("{}没有最新的tick数据，退出执行逻辑", _code).c_str());
457 return;
458 }
459 double curPos = realPos;
460
461 if (decimal::eq(curPos, newVol))
462 {
463 //当前仓位和最新目标仓位匹配时，如果不是全部清仓的需求，则直接退出计算了
464 if (!is_clear(_target_pos))
465 return;
466
467 //如果是清仓的需求，还要再进行对比
468 //如果多头为0，说明已经全部清理掉了，则直接退出
469 double lPos = _ctx->getPosition(code, true, 1);
470 if (decimal::eq(lPos, 0))
471 return;
472 //如果还有多头仓位，则将目标仓位设置为非0，强制触发
473 newVol = -min(lPos, _order_lots);
474 _ctx->writeLog(fmtutil::format("Clearing process triggered, target position of {} has been set to {}", _code.c_str(), newVol));
475 }
476 //如果相比上次没有更新的tick进来，则先不下单，防止开盘前集中下单导致通道被封
477 uint64_t curTickTime = (uint64_t)_last_tick->actiondate() * 1000000000 + _last_tick->actiontime();
478 if (curTickTime <= _last_tick_time)
479 {
480 _ctx->writeLog(fmtutil::format("No tick of {} updated, {} <= {}, execute later", _code, curTickTime, _last_tick_time));
481 return;
482 }
483 _last_tick_time = curTickTime;
484 double InminsTm = calTmStamp(_last_tick->actiontime());//当前tick属于vwap240分钟内的第几(-1)分钟
485 double aimQty = VwapAim[InminsTm];//取到对应时刻的目标vwapaim （递增）
486
487 uint32_t leftTimes = _total_times - _fired_times;
488 _ctx->writeLog(fmt::format("第 {} 次发单", _fired_times+1).c_str());
489 _Vwap_vol = aimQty - curPos;//若在本分钟发单，对应的VWapVol
490 bool bNeedShowHand = false;
491 double curQty = 0;
492 if (leftTimes == 0 && !decimal::eq(diffQty, 0))
493 {
494 bNeedShowHand = true;
495 curQty = max(diffQty, _min_open_lots);
496 }
497 else {
498 curQty = max(_Vwap_vol, _min_open_lots)* abs(diffQty) / diffQty;//curqty=单位预测量sum
499 }
500 //设定本轮目标仓位
501 _this_target = realPos + curQty;
502
503 WTSTickData* curTick = _last_tick;
504 uint64_t now = TimeUtils::getLocalTimeNow();
505 bool isBuy = decimal::gt(diffQty, 0);
506 double targetPx = 0;
507 //根据价格模式设置,确定委托基准价格: 0-最新价,1-最优价,2-对手价
508 if (_price_mode == 0)
509 {
510 targetPx = curTick->price();
511 }
512 else if (_price_mode == 1)
513 {
514 targetPx = isBuy ? curTick->bidprice(0) : curTick->askprice(0);
515 }
516 else
517 {
518 targetPx = isBuy ? curTick->askprice(0) : curTick->bidprice(0);
519 }
520 if (bNeedShowHand) {
521 targetPx += _comm_info->getPriceTick() * 5 * (isBuy ? 1 : -1);
522 }
523 else if (_price_offset != 0)
524 {
525 targetPx += _comm_info->getPriceTick() * _price_offset * (isBuy ? 1 : -1);
526 }
527
528 // 如果最后价格为0，再做一个修正
529 if (decimal::eq(targetPx, 0.0))
530 targetPx = decimal::eq(_last_tick->price(), 0.0) ? _last_tick->preclose() : _last_tick->price();
531
532 //检查涨跌停价
533 isCanCancel = true;
534 if (isBuy && !decimal::eq(_last_tick->upperlimit(), 0) && decimal::gt(targetPx, _last_tick->upperlimit()))
535 {
536 _ctx->writeLog(fmt::format("Buy price {} of {} modified to upper limit price", targetPx, _code.c_str(), _last_tick->upperlimit()).c_str());
537 targetPx = _last_tick->upperlimit();
538 isCanCancel = false;//如果价格被修正为涨跌停价，订单不可撤销
539 }
540 if (isBuy != 1 && !decimal::eq(_last_tick->lowerlimit(), 0) && decimal::lt(targetPx, _last_tick->lowerlimit()))
541 {
542 _ctx->writeLog(fmt::format("Sell price {} of {} modified to lower limit price", targetPx, _code.c_str(), _last_tick->lowerlimit()).c_str());
543 targetPx = _last_tick->lowerlimit();
544 isCanCancel = false; //如果价格被修正为涨跌停价，订单不可撤销
545 }
546 OrderIDs ids;
547 if (curQty > 0)
548 {
549 ids = _ctx->buy(code, targetPx, abs(curQty));
550 }
551 else {
552 ids = _ctx->sell(code, targetPx, abs(curQty));
553 }
554 _orders_mon.push_order(ids.data(), ids.size(), now, isCanCancel);
555 _last_fire_time = now;
556 _fired_times += 1;
557
558 curTick->release();
559}
