217void ParserCTPOpt::OnRtnDepthMarketData( CThostFtdcDepthMarketDataField *pDepthMarketData )
218{
219 if(m_pBaseDataMgr == NULL)
220 {
221 return;
222 }
223
224 uint32_t actDate, actTime;
225 if (m_bLocalTime)
226 {
227 TimeUtils::getDateTime(actDate, actTime);
228 }
229 else
230 {
231 actDate = strtoul(pDepthMarketData->ActionDay, NULL, 10);
232 actTime = strToTime(pDepthMarketData->UpdateTime) * 1000 + pDepthMarketData->UpdateMillisec;
233 }
234 uint32_t actHour = actTime / 10000000;
235
236 if (actDate == m_uTradingDate && actHour >= 20)
237 {
238 //这样的时间是有问题,因为夜盘时发生日期不可能等于交易日
239 //这就需要手动设置一下
240 uint32_t curDate, curTime;
241 TimeUtils::getDateTime(curDate, curTime);
242 uint32_t curHour = curTime / 10000000;
243
244 //早上启动以后,会收到昨晚12点以前收盘的行情,这个时候可能会有发生日期=交易日的情况出现
245 //这笔数据直接丢掉
246 if (curHour >= 3 && curHour < 9)
247 return;
248
249 actDate = curDate;
250
251 if (actHour == 23 && curHour == 0)
252 {
253 //行情时间慢于系统时间
254 actDate = TimeUtils::getNextDate(curDate, -1);
255 }
256 else if (actHour == 0 && curHour == 23)
257 {
258 //系统时间慢于行情时间
259 actDate = TimeUtils::getNextDate(curDate, 1);
260 }
261 }
262
263 WTSContractInfo* cInfo = m_pBaseDataMgr->getContract(pDepthMarketData->InstrumentID, pDepthMarketData->ExchangeID);
264 if (cInfo == NULL)
265 return;
266
267 WTSCommodityInfo* pCommInfo = cInfo->getCommInfo();
268
269 //if (strcmp(contract->getExchg(), "CZCE") == 0)
270 //{
271 // actTime += (uint32_t)(TimeUtils::getLocalTimeNow() % 1000);
272 //}
273
274 WTSTickData* tick = WTSTickData::create(pDepthMarketData->InstrumentID);
275 tick->setContractInfo(cInfo);
276 WTSTickStruct& quote = tick->getTickStruct();
277 strcpy(quote.exchg, pCommInfo->getExchg());
278
279 quote.action_date = actDate != 0 ? actDate : m_uTradingDate;
280 quote.action_time = actTime;
281
282 quote.price = checkValid(pDepthMarketData->LastPrice);
283 quote.open = checkValid(pDepthMarketData->OpenPrice);
284 quote.high = checkValid(pDepthMarketData->HighestPrice);
285 quote.low = checkValid(pDepthMarketData->LowestPrice);
286 quote.total_volume = pDepthMarketData->Volume;
287 quote.trading_date = m_uTradingDate;
288 if(pDepthMarketData->SettlementPrice != DBL_MAX)
289 quote.settle_price = checkValid(pDepthMarketData->SettlementPrice);
290 if(strcmp(quote.exchg, "CZCE") == 0)
291 {
292 quote.total_turnover = pDepthMarketData->Turnover*pCommInfo->getVolScale();
293 }
294 else
295 {
296 if(pDepthMarketData->Turnover != DBL_MAX)
297 quote.total_turnover = pDepthMarketData->Turnover;
298 }
299
300 quote.open_interest = (uint32_t)pDepthMarketData->OpenInterest;
301
302 quote.upper_limit = checkValid(pDepthMarketData->UpperLimitPrice);
303 quote.lower_limit = checkValid(pDepthMarketData->LowerLimitPrice);
304
305 quote.pre_close = checkValid(pDepthMarketData->PreClosePrice);
306 quote.pre_settle = checkValid(pDepthMarketData->PreSettlementPrice);
307 quote.pre_interest = (uint32_t)pDepthMarketData->PreOpenInterest;
308
309 //委卖价格
310 quote.ask_prices[0] = checkValid(pDepthMarketData->AskPrice1);
311 quote.ask_prices[1] = checkValid(pDepthMarketData->AskPrice2);
312 quote.ask_prices[2] = checkValid(pDepthMarketData->AskPrice3);
313 quote.ask_prices[3] = checkValid(pDepthMarketData->AskPrice4);
314 quote.ask_prices[4] = checkValid(pDepthMarketData->AskPrice5);
315
316 //委买价格
317 quote.bid_prices[0] = checkValid(pDepthMarketData->BidPrice1);
318 quote.bid_prices[1] = checkValid(pDepthMarketData->BidPrice2);
319 quote.bid_prices[2] = checkValid(pDepthMarketData->BidPrice3);
320 quote.bid_prices[3] = checkValid(pDepthMarketData->BidPrice4);
321 quote.bid_prices[4] = checkValid(pDepthMarketData->BidPrice5);
322
323 //委卖量
324 quote.ask_qty[0] = pDepthMarketData->AskVolume1;
325 quote.ask_qty[1] = pDepthMarketData->AskVolume2;
326 quote.ask_qty[2] = pDepthMarketData->AskVolume3;
327 quote.ask_qty[3] = pDepthMarketData->AskVolume4;
328 quote.ask_qty[4] = pDepthMarketData->AskVolume5;
329

332 quote.bid_qty[1] = pDepthMarketData->BidVolume2;
333 quote.bid_qty[2] = pDepthMarketData->BidVolume3;
334 quote.bid_qty[3] = pDepthMarketData->BidVolume4;
335 quote.bid_qty[4] = pDepthMarketData->BidVolume5;
336
337 if(m_sink)
338 m_sink->handleQuote(tick, 1);

382void ParserCTPOpt::SubscribeMarketData()
383{
384 CodeSet codeFilter = m_filterSubs;
385 if(codeFilter.empty())
386 {//如果订阅礼包只空的,则取出全部合约列表
387 return;
388 }
389
390 char ** subscribe = new char*[codeFilter.size()];
391 int nCount = 0;
392 for(auto& code : codeFilter)
393 {
394 std::size_t pos = code.find('.');
395 if (pos != std::string::npos)
396 subscribe[nCount++] = (char*)code.c_str() + pos + 1;
397 else
398 subscribe[nCount++] = (char*)code.c_str();
399 }
400
401 if(m_pUserAPI && nCount > 0)
402 {
403 int iResult = m_pUserAPI->SubscribeMarketData(subscribe, nCount);
404 if (iResult != 0)
405 {
406 if (m_sink)
407 write_log(m_sink, LL_ERROR, "[ParserCTPOpt] Sending md subscribe request failed: {}", iResult);
408 }
409 else
410 {
411 if (m_sink)
412 write_log(m_sink, LL_INFO, "[ParserCTPOpt] Market data of {} contracts subscribed in total", nCount);
413 }
414 }
415 codeFilter.clear();
416 delete[] subscribe;
417}
