120void initParsers(WTSVariant* cfg)
121{
122 for (uint32_t idx = 0; idx < cfg->size(); idx++)
123 {
124 WTSVariant* cfgItem = cfg->get(idx);
125 // 跳过未激活的解析器
126 if (!cfgItem->getBoolean("active"))
127 continue;
128
129 const char* id = cfgItem->getCString("id");
130 // By Wesley @ 2021.12.14
131 // 如果id为空，则生成自动id
132 std::string realid = id;
133 if (realid.empty())
134 {
135 // 使用静态变量生成自增的解析器ID
136 static uint32_t auto_parserid = 1000;
137 realid = StrUtil::printf("auto_parser_%u", auto_parserid++);
138 }
139
140 // 创建解析器适配器并初始化
141 ParserAdapterPtr adapter(new ParserAdapter(&g_baseDataMgr, &g_dataMgr, &g_idxFactory));
142 adapter->init(realid.c_str(), cfgItem);
143 // 将适配器添加到管理器中
144 g_parsers.addAdapter(realid.c_str(), adapter);
145 }
146
147 // 输出已加载的解析器数量
148 WTSLogger::info("{} market data parsers loaded in total", g_parsers.size());
149}

159void initialize(const std::string& filename)
160{
161 // 设置模块目录
162 WtHelper::set_module_dir(getBinDir());
163
164 // 加载配置文件
165 WTSVariant* config = WTSCfgLoader::load_from_file(filename.c_str());
166 if(config == NULL)
167 {
168 WTSLogger::error("Loading config file {} failed", filename);
169 return;
170 }
171
172 //加载市场信息
173 WTSVariant* cfgBF = config->get("basefiles");
174 if (cfgBF->get("session"))
175 {
176 g_baseDataMgr.loadSessions(cfgBF->getCString("session"));
177 WTSLogger::info("Trading sessions loaded");
178 }
179
180 WTSVariant* cfgItem = cfgBF->get("commodity");
181 if (cfgItem)
182 {
183 if (cfgItem->type() == WTSVariant::VT_String)
184 {
185 g_baseDataMgr.loadCommodities(cfgItem->asCString());
186 }
187 else if (cfgItem->type() == WTSVariant::VT_Array)
188 {
189 for (uint32_t i = 0; i < cfgItem->size(); i++)
190 {
191 g_baseDataMgr.loadCommodities(cfgItem->get(i)->asCString());
192 }
193 }
194 }
195
196 cfgItem = cfgBF->get("contract");
197 if (cfgItem)
198 {
199 if (cfgItem->type() == WTSVariant::VT_String)
200 {
201 g_baseDataMgr.loadContracts(cfgItem->asCString());
202 }
203 else if (cfgItem->type() == WTSVariant::VT_Array)
204 {
205 for (uint32_t i = 0; i < cfgItem->size(); i++)
206 {
207 g_baseDataMgr.loadContracts(cfgItem->get(i)->asCString());
208 }
209 }
210 }
211
212 if (cfgBF->get("holiday"))
213 {
214 g_baseDataMgr.loadHolidays(cfgBF->getCString("holiday"));
215 WTSLogger::info("Holidays loaded");
216 }
217 if (cfgBF->get("hot"))
218 {
219 g_hotMgr.loadHots(cfgBF->getCString("hot"));
220 WTSLogger::log_raw(LL_INFO, "Hot rules loaded");
221 }
222
223 if (cfgBF->get("second"))
224 {
225 g_hotMgr.loadSeconds(cfgBF->getCString("second"));
226 WTSLogger::log_raw(LL_INFO, "Second rules loaded");
227 }
228
229 if (cfgBF->has("rules"))
230 {
231 auto cfgRules = cfgBF->get("rules");
232 auto tags = cfgRules->memberNames();
233 for (const std::string& ruleTag : tags)
234 {
235 g_hotMgr.loadCustomRules(ruleTag.c_str(), cfgRules->getCString(ruleTag.c_str()));
236 WTSLogger::info("{} rules loaded from {}", ruleTag, cfgRules->getCString(ruleTag.c_str()));
237 }
238 }
239
240 if (config->has("shmcaster"))
241 {
242 g_shmCaster.init(config->get("shmcaster"));
243 g_dataMgr.add_caster(&g_shmCaster);
244 }
245
246 if (config->has("broadcaster"))
247 {
248 g_udpCaster.init(config->get("broadcaster"), &g_baseDataMgr, &g_dataMgr);
249 g_dataMgr.add_caster(&g_udpCaster);
250 }
251
252
253 //By Wesley @ 2021.12.27
254 //全天候模式，不需要再使用状态机
255 bool bAlldayMode = config->getBoolean("allday");
256 if (!bAlldayMode)
257 {
258 g_stateMon.initialize(config->getCString("statemonitor"), &g_baseDataMgr, &g_dataMgr);
259 }
260 else
261 {
262 WTSLogger::info("QuoteFactory will run in allday mode");
263 }
264 initDataMgr(config->get("writer"), bAlldayMode);
265
266 if(config->has("index"))
267 {
268 //如果存在指数模块要，配置指数
269 const char* filename = config->getCString("index");
270 WTSLogger::info("Reading index config from {}...", filename);
271 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
272 if (var)
273 {
274 g_idxFactory.init(var, &g_hotMgr, &g_baseDataMgr, &g_dataMgr);
275 var->release();
276 }
277 else
278 {
279 WTSLogger::error("Loading index config {} failed", filename);
280 }
281 }
282
283 WTSVariant* cfgParser = config->get("parsers");
284 if (cfgParser)
285 {
286 if (cfgParser->type() == WTSVariant::VT_String)
287 {
288 const char* filename = cfgParser->asCString();
289 if (StdFile::exists(filename))
290 {
291 WTSLogger::info("Reading parser config from {}...", filename);
292 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
293 if (var)
294 {
295 initParsers(var->get("parsers"));
296 var->release();
297 }
298 else
299 {
300 WTSLogger::error("Loading parser config {} failed", filename);
301 }
302 }
303 else
304 {
305 WTSLogger::error("Parser configuration {} not exists", filename);
306 }
307 }
308 else if (cfgParser->type() == WTSVariant::VT_Array)
309 {
310 initParsers(cfgParser);
311 }
312 }
313
314 config->release();
315
316 g_parsers.run();
317
318 //全天候模式，不启动状态机
319 if(!bAlldayMode)
320 {
321 std::this_thread::sleep_for(std::chrono::milliseconds(5));
322 g_stateMon.run();
323 }
324}

336int main(int argc, char* argv[])
337{
338 // 初始化命令行参数解析器
339 cppcli::Option opt(argc, argv);
340
341 // 定义命令行参数
342 auto cParam = opt("-c", "--config", "configure filepath, dtcfg.yaml as default", false);
343 auto lParam = opt("-l", "--logcfg", "logging configure filepath, logcfgdt.yaml as default", false);
344
345 auto hParam = opt("-h", "--help", "gain help doc", false)->asHelpParam();
346
347 // 解析命令行参数
348 opt.parse();
349
350 // 如果指定了帮助参数，则显示帮助信息并退出
351 if (hParam->exists())
352 return 0;
353
354 // 处理日志配置文件路径
355 std::string filename;
356 if (lParam->exists())
357 filename = lParam->get<std::string>();
358 else
359 filename = "./logcfgdt.yaml"; // 使用默认日志配置文件
360 // 初始化日志系统
361 WTSLogger::init(filename.c_str());
362
363#ifdef _MSC_VER
364 _CrtSetReportMode(_CRT_WARN, _CRTDBG_MODE_DEBUG);
365 _CrtSetReportMode(_CRT_ERROR, _CRTDBG_MODE_DEBUG);
366 _CrtSetReportMode(_CRT_ASSERT, _CRTDBG_MODE_DEBUG);
367
368 _set_error_mode(_OUT_TO_STDERR);
369 _set_abort_behavior(0, _WRITE_ABORT_MSG);
370
371 g_dwMainThreadId = GetCurrentThreadId();
372 SetConsoleCtrlHandler(ConsoleCtrlhandler, TRUE);
373
374 CMiniDumper::Enable("QuoteFactory.exe", true);
375#endif
376
377 // 初始化退出标志
378 bool bExit = false;
379 // 安装信号处理钩子，用于处理系统信号（如Ctrl+C）
380 install_signal_hooks([&bExit](const char* message) {
381 // 错误信息处理回调
382 if(!bExit)
383 WTSLogger::error(message);
384 }, [&bExit](bool toExit) {
385 // 退出信号处理回调
386 if (bExit)
387 return;
388
389 // 设置退出标志并记录日志
390 bExit = toExit;
391 WTSLogger::info("Exit flag is {}", bExit);
392 });
393
394 // 处理主配置文件路径
395 if (cParam->exists())
396 filename = cParam->get<std::string>();
397 else
398 filename = "./dtcfg.yaml"; // 使用默认配置文件
399
400 // 检查配置文件是否存在
401 if(!StdFile::exists(filename.c_str()))
402 {
403 fmt::print("confiture {} not exists", filename);
404 return 0;
405 }
406
407 // 初始化行情工厂
408 initialize(filename);
409
410 // 主循环，等待退出信号
411 while (!bExit)
412 {
413 // 小休眠减少CPU占用
414 std::this_thread::sleep_for(std::chrono::milliseconds(10));
415 }
416
417 // 正常退出
418 return 0;
419}
