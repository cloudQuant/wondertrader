70bool ExecMocker::init(WTSVariant* cfg)
71{
72 // 从配置中读取基本参数
73 const char* module = cfg->getCString("module"); // 动态库模块路径
74 _code = cfg->getCString("code"); // 合约代码
75 _period = cfg->getCString("period"); // K线周期
76 _volunit = cfg->getDouble("volunit"); // 数量单位
77 _volmode = cfg->getInt32("volmode"); // 数量模式：0-反复正负，-1-一直卖，+1-一直买
78
79 // 初始化撮合引擎
80 _matcher.regisSink(this); // 注册当前对象为撮合引擎的回调接收者
81 _matcher.init(cfg->get("matcher")); // 使用配置初始化撮合引擎
82
83 // 加载执行器工厂动态库
84 DllHandle hInst = DLLHelper::load_library(module);
85 if (hInst == NULL)
86 return false; // 加载失败返回失败
87
88 // 获取创建执行器工厂的函数指针
89 FuncCreateExeFact creator = (FuncCreateExeFact)DLLHelper::get_symbol(hInst, "createExecFact");
90 if (creator == NULL)
91 {
92 DLLHelper::free_library(hInst); // 获取失败时释放动态库句柄
93 return false;
94 }
95
96 // 设置执行器工厂相关信息
97 _factory._module_inst = hInst;
98 _factory._module_path = module;
99 _factory._creator = creator;
100 _factory._remover = (FuncDeleteExeFact)DLLHelper::get_symbol(hInst, "deleteExecFact");
101 _factory._fact = _factory._creator(); // 创建执行器工厂实例
102
103 // 读取执行器配置并创建执行单元
104 WTSVariant* cfgExec = cfg->get("executer");
105 if (cfgExec)
106 {
107 // 创建并初始化执行单元
108 _exec_unit = _factory._fact->createExeUnit(cfgExec->getCString("name"));
109 _exec_unit->init(this, _code.c_str(), cfgExec->get("params"));
110 _id = cfgExec->getCString("id"); // 设置执行单元ID
111 }
112
113 return true;
114}

196void ExecMocker::handle_tick(const char* stdCode, WTSTickData* curTick, uint32_t pxType)

221void ExecMocker::handle_init()

225 basePeriod[0] = _period[0]; // 获取基础周期（如m、d等）

258void ExecMocker::handle_schedule(uint32_t uDate, uint32_t uTime)
259{
260 // 如果时间是15:00，则不处理
261 if (uTime == 1500)
262 return;
263
264 // 获取当前信号价格，使用最新价格或昨收价
265 _sig_px = _last_tick->price();
266 if (_sig_px == DBL_MAX || _sig_px == FLT_MAX) // 如果价格无效，使用昨收价
267 _sig_px = _last_tick->preclose();
268
269 // 记录信号生成时间
270 _sig_time = (uint64_t)uDate * 10000 + uTime;
271
272 // 根据仓位模式计算目标仓位
273 if(_volmode == 0) // 反复正负模式，持仓为负时则做多，持仓为正时则做空
274 {
275 if (_position <= 0)
276 _target = _volunit; // 如果当前空仓或无仓，则做多
277 else
278 _target = -_volunit; // 如果当前多仓，则做空
279 }
280 else if (_volmode == -1) // 一直卖模式，目标仓位不断减少
281 {
282 _target -= _volunit;
283 }
284 else if (_volmode == 1) // 一直买模式，目标仓位不断增加
285 {
286 _target += _volunit;
287 }
288
289 // 设置执行单元的目标仓位
290 _exec_unit->set_position(_code.c_str(), _target);
291 WTSLogger::info("Target position updated @{}.{}: {}", uDate, uTime, _volunit);
292 // 信号计数增加
293 _sig_cnt++;
294}

419bool ExecMocker::cancel(uint32_t localid)
420{
421 // 调用撮合引擎的取消方法，返回取消的数量
422 double change = _matcher.cancel(localid);
423 // 如果没有取消任何数量，返回false
424 if (decimal::eq(change, 0))
425 return false;
426
427 // 更新未完成数量
428 _undone -= change;
429 // 更新取消统计信息
430 _cacl_cnt++; // 增加取消计数
431 _cacl_qty += abs(change); // 增加取消总量
432 WTSLogger::info("{}, undone orders updated: {}", __FUNCTION__, _undone);
433
434 return true;

497void ExecMocker::handle_order(uint32_t localid, const char* stdCode, bool isBuy, double leftover, double price, bool isCanceled, uint64_t ordTime)
498{
499 // 获取当前时间
500 uint64_t curTime = (uint64_t)_last_tick->actiondate() * 1000000000 + _last_tick->actiontime();
501 uint64_t curUnixTime = TimeUtils::makeTime(_last_tick->actiondate(), _last_tick->actiontime());
502
503 // 计算信号时间和委托时间的Unix时间戳
504 uint64_t sigUnixTime = TimeUtils::makeTime((uint32_t)(_sig_time / 10000), _sig_time % 10000 * 100000);
505 uint64_t ordUnixTime = TimeUtils::makeTime((uint32_t)(ordTime / 1000000000), ordTime % 1000000000);
506
507 // 如果订单被取消，记录取消日志
508 if(isCanceled)
509 {
510 // 如果信号价格无效，使用昨收价
511 if (_sig_px == DBL_MAX)
512 _sig_px = _last_tick->preclose();
513
514 // 记录取消订单的交易日志
515 _trade_logs << localid << ","
516 << _sig_time << ","
517 << ordTime << ","
518 << (isBuy ? "B" : "S") << ","
519 << _sig_px << ","
520 << 0 << ","
521 << price << ","
522 << curTime << ","
523 << price << ","
524 << 0 << ","
525 << curUnixTime - sigUnixTime << ","
526 << curUnixTime - ordUnixTime << ","
527 << "true" << std::endl;
528
529 // 更新未完成数量
530 _undone -= leftover * (isBuy ? 1 : -1);
531 WTSLogger::info("{}, undone orders updated: {}", __FUNCTION__, _undone);
532 }
533
534 // 将订单状态变化转发给执行单元
535 _exec_unit->on_order(localid, stdCode, isBuy, leftover, price, isCanceled);
536}
