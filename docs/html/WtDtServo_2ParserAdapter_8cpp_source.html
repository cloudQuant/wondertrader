69bool ParserAdapter::initExt(const char* id, IParserApi* api)
70{
71 if (api == NULL)
72 return false;
73
74 _parser_api = api;
75 _id = id;
76
77 if (_parser_api)
78 {
79 _parser_api->registerSpi(this);
80
81 if (_parser_api->init(NULL))
82 {
83 ContractSet contractSet;
84 WTSArray* ayContract = _bd_mgr->getContracts();
85 WTSArray::Iterator it = ayContract->begin();

95 contractSet.clear();
96 }
97 else

99 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser initializing failed: api initializing failed...", _id.c_str());

117bool ParserAdapter::init(const char* id, WTSVariant* cfg)
118{
119 if (cfg == NULL)
120 return false;
121
122 _id = id;
123
124 if (_cfg != NULL)
125 return false;
126
127 _cfg = cfg;
128 _cfg->retain();
129
130 {
131 //加载模块
132 if (cfg->getString("module").empty())
133 return false;
134
135 std::string module = DLLHelper::wrap_module(cfg->getCString("module"), "lib");;
136
137 if (!StdFile::exists(module.c_str()))
138 {
139 module = WtHelper::get_module_dir();
140 module += "parsers/";
141 module += DLLHelper::wrap_module(cfg->getCString("module"), "lib");
142 }
143
144 DllHandle hInst = DLLHelper::load_library(module.c_str());
145 if (hInst == NULL)
146 {
147 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser module {} loading failed", _id.c_str(), module.c_str());
148 return false;
149 }
150 else
151 {
152 WTSLogger::log_dyn("parser", _id.c_str(), LL_INFO, "[{}] Parser module {} loaded", _id.c_str(), module.c_str());
153 }
154
155 FuncCreateParser pFuncCreateParser = (FuncCreateParser)DLLHelper::get_symbol(hInst, "createParser");
156 if (NULL == pFuncCreateParser)
157 {
158 WTSLogger::log_dyn("parser", _id.c_str(), LL_FATAL, "[{}] Entrance function createParser not found", _id.c_str());
159 return false;
160 }
161
162 _parser_api = pFuncCreateParser();
163 if (NULL == _parser_api)
164 {
165 WTSLogger::log_dyn("parser", _id.c_str(), LL_FATAL, "[{}] Creating parser api failed", _id.c_str());
166 return false;
167 }
168
169 _remover = (FuncDeleteParser)DLLHelper::get_symbol(hInst, "deleteParser");
170 }
171
172
173 const std::string& strFilter = cfg->getString("filter");
174 if (!strFilter.empty())
175 {
176 const StringVector &ayFilter = StrUtil::split(strFilter, ",");
177 auto it = ayFilter.begin();
178 for (; it != ayFilter.end(); it++)
179 {
180 _exchg_filter.insert(*it);
181 }
182 }
183
184 std::string strCodes = cfg->getString("code");
185 if (!strCodes.empty())
186 {
187 const StringVector &ayCodes = StrUtil::split(strCodes, ",");
188 auto it = ayCodes.begin();
189 for (; it != ayCodes.end(); it++)
190 {
191 _code_filter.insert(*it);
192 }
193 }
194
195 if (_parser_api)
196 {
197 _parser_api->registerSpi(this);
198
199 if (_parser_api->init(cfg))
200 {
201 ContractSet contractSet;
202 if (!_code_filter.empty())//优先判断合约过滤器
203 {
204 ExchgFilter::iterator it = _code_filter.begin();
205 for (; it != _code_filter.end(); it++)
206 {
207 //全代码,形式如SSE.600000,期货代码为CFFEX.IF2005
208 std::string code, exchg;
209 auto ay = StrUtil::split((*it).c_str(), ".");
210 if (ay.size() == 1)
211 code = ay[0];
212 else if (ay.size() == 2)
213 {
214 exchg = ay[0];
215 code = ay[1];
216 }
217 else if (ay.size() == 3)
218 {
219 exchg = ay[0];
220 code = ay[2];
221 }
222 WTSContractInfo* contract = _bd_mgr->getContract(code.c_str(), exchg.c_str());
223 if (contract)
224 contractSet.insert(contract->getFullCode());
225 else
226 {
227 //如果是品种ID，则将该品种下全部合约都加到订阅列表
228 WTSCommodityInfo* commInfo = _bd_mgr->getCommodity(exchg.c_str(), code.c_str());
229 if (commInfo)
230 {
231 const auto& codes = commInfo->getCodes();
232 for (const auto& c : codes)
233 {
234 contractSet.insert(fmt::format("{}.{}", exchg, c.c_str()));
235 }
236 }
237 }
238 }
239 }
240 else if (!_exchg_filter.empty())
241 {
242 ExchgFilter::iterator it = _exchg_filter.begin();
243 for (; it != _exchg_filter.end(); it++)
244 {
245 const char* exchg = (*it).c_str();
246 WTSArray* ayContract = _bd_mgr->getContracts(exchg);
247 auto cnt = ayContract->size();
248 WTSArray::Iterator it = ayContract->begin();
249 for (; it != ayContract->end(); it++)
250 {
251 WTSContractInfo* contract = STATIC_CONVERT(*it, WTSContractInfo*);
252 contractSet.insert(contract->getFullCode());
253 }
254
255 ayContract->release();
256
257 WTSLogger::log_dyn("parser", _id.c_str(), LL_INFO, "[{}] {} contracts of {} added to sublist...", _id.c_str(), cnt, exchg);
258 }
259 }
260 else
261 {
262 WTSArray* ayContract = _bd_mgr->getContracts();
263 WTSArray::Iterator it = ayContract->begin();
264 for (; it != ayContract->end(); it++)
265 {
266 WTSContractInfo* contract = STATIC_CONVERT(*it, WTSContractInfo*);
267 contractSet.insert(contract->getFullCode());
268 }
269
270 ayContract->release();
271 }
272
273 _parser_api->subscribe(contractSet);
274 contractSet.clear();
275 }
276 else
277 {
278 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser initializing failed: api initializing failed...", _id.c_str());
279 }
280 }
281 else
282 {
283 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser initializing failed: creating api failed...", _id.c_str());
284 }
285
286 return true;
287}
