67static uint32_t reverseEndian(uint32_t src)
68{
69 uint32_t x = (src & 0x000000FF) << 24;
70 uint32_t y = (src & 0x0000FF00) << 8;
71 uint32_t z = (src & 0x00FF0000) >> 8;
72 uint32_t w = (src & 0xFF000000) >> 24;
73 return x + y + z + w;
74}
