215void ParserCTPMini::OnRtnDepthMarketData( CThostFtdcDepthMarketDataField *pDepthMarketData )
216{
217 if(m_pBaseDataMgr == NULL)
218 {
219 return;
220 }
221
222 uint32_t actDate = strtoul(pDepthMarketData->ActionDay, NULL, 10);
223 uint32_t actTime = strToTime(pDepthMarketData->UpdateTime) * 1000 + pDepthMarketData->UpdateMillisec;
224 uint32_t actHour = actTime / 10000000;
225
226 if (actDate == 0)
227 actDate = m_uTradingDate;
228
229 if (actDate == m_uTradingDate && actHour >= 20)
230 {
231 //这样的时间是有问题,因为夜盘时发生日期不可能等于交易日
232 //这就需要手动设置一下
233 uint32_t curDate, curTime;
234 TimeUtils::getDateTime(curDate, curTime);
235 uint32_t curHour = curTime / 10000000;
236
237 //早上启动以后,会收到昨晚12点以前收盘的行情,这个时候可能会有发生日期=交易日的情况出现
238 //这笔数据直接丢掉
239 if (curHour >= 3 && curHour < 9)
240 return;
241
242 actDate = curDate;
243
244 if (actHour == 23 && curHour == 0)
245 {
246 //行情时间慢于系统时间
247 actDate = TimeUtils::getNextDate(curDate, -1);
248 }
249 else if (actHour == 0 && curHour == 23)
250 {
251 //系统时间慢于行情时间
252 actDate = TimeUtils::getNextDate(curDate, 1);
253 }
254 }
255
256 WTSContractInfo* contract = m_pBaseDataMgr->getContract(pDepthMarketData->InstrumentID, pDepthMarketData->ExchangeID);
257 if (contract == NULL)
258 return;
259
260 WTSCommodityInfo* pCommInfo = contract->getCommInfo();
261
262 //if (strcmp(contract->getExchg(), "CZCE") == 0)
263 //{
264 // actTime += (uint32_t)(TimeUtils::getLocalTimeNow() % 1000);
265 //}
266
267 WTSTickData* tick = WTSTickData::create(pDepthMarketData->InstrumentID);
268 tick->setContractInfo(contract);
269 WTSTickStruct& quote = tick->getTickStruct();
270 strcpy(quote.exchg, pCommInfo->getExchg());
271
272 quote.action_date = actDate;
273 quote.action_time = actTime;
274
275 quote.price = checkValid(pDepthMarketData->LastPrice);
276 quote.open = checkValid(pDepthMarketData->OpenPrice);
277 quote.high = checkValid(pDepthMarketData->HighestPrice);
278 quote.low = checkValid(pDepthMarketData->LowestPrice);
279 quote.total_volume = pDepthMarketData->Volume;
280 quote.trading_date = m_uTradingDate;
281 if(pDepthMarketData->SettlementPrice != DBL_MAX)
282 quote.settle_price = checkValid(pDepthMarketData->SettlementPrice);
283 if(strcmp(quote.exchg, "CZCE") == 0)
284 {
285 quote.total_turnover = pDepthMarketData->Turnover*pCommInfo->getVolScale();
286 }
287 else
288 {
289 if(pDepthMarketData->Turnover != DBL_MAX)
290 quote.total_turnover = pDepthMarketData->Turnover;
291 }
292
293 quote.open_interest = (uint32_t)pDepthMarketData->OpenInterest;
294
295 quote.upper_limit = checkValid(pDepthMarketData->UpperLimitPrice);
296 quote.lower_limit = checkValid(pDepthMarketData->LowerLimitPrice);
297
298 quote.pre_close = checkValid(pDepthMarketData->PreClosePrice);
299 quote.pre_settle = checkValid(pDepthMarketData->PreSettlementPrice);
300 quote.pre_interest = (uint32_t)pDepthMarketData->PreOpenInterest;
301
302 //委卖价格
303 quote.ask_prices[0] = checkValid(pDepthMarketData->AskPrice1);
304 quote.ask_prices[1] = checkValid(pDepthMarketData->AskPrice2);
305 quote.ask_prices[2] = checkValid(pDepthMarketData->AskPrice3);
306 quote.ask_prices[3] = checkValid(pDepthMarketData->AskPrice4);
307 quote.ask_prices[4] = checkValid(pDepthMarketData->AskPrice5);
308
309 //委买价格
310 quote.bid_prices[0] = checkValid(pDepthMarketData->BidPrice1);
311 quote.bid_prices[1] = checkValid(pDepthMarketData->BidPrice2);
312 quote.bid_prices[2] = checkValid(pDepthMarketData->BidPrice3);
313 quote.bid_prices[3] = checkValid(pDepthMarketData->BidPrice4);
314 quote.bid_prices[4] = checkValid(pDepthMarketData->BidPrice5);
315
316 //委卖量
317 quote.ask_qty[0] = pDepthMarketData->AskVolume1;
318 quote.ask_qty[1] = pDepthMarketData->AskVolume2;
319 quote.ask_qty[2] = pDepthMarketData->AskVolume3;
320 quote.ask_qty[3] = pDepthMarketData->AskVolume4;
321 quote.ask_qty[4] = pDepthMarketData->AskVolume5;
322
323 //委买量
324 quote.bid_qty[0] = pDepthMarketData->BidVolume1;
325 quote.bid_qty[1] = pDepthMarketData->BidVolume2;
326 quote.bid_qty[2] = pDepthMarketData->BidVolume3;
327 quote.bid_qty[3] = pDepthMarketData->BidVolume4;
328 quote.bid_qty[4] = pDepthMarketData->BidVolume5;
329
330 if(m_sink)
331 m_sink->handleQuote(tick, 1);

333 tick->release();

375void ParserCTPMini::SubscribeMarketData()
376{
377 CodeSet codeFilter = m_filterSubs;
378 if(codeFilter.empty())
379 {//如果订阅礼包只空的,则取出全部合约列表
380 return;
381 }
382
383 char ** subscribe = new char*[codeFilter.size()];
384 int nCount = 0;
385 for(auto& code : codeFilter)
386 {
387 std::size_t pos = code.find('.');
388 if (pos != std::string::npos)
389 subscribe[nCount++] = (char*)code.c_str() + pos + 1;
390 else
391 subscribe[nCount++] = (char*)code.c_str();
392 }
393
394 if(m_pUserAPI && nCount > 0)
395 {
396 int iResult = m_pUserAPI->SubscribeMarketData(subscribe, nCount);
397 if(iResult != 0)
398 {
399 if(m_sink)
400 write_log(m_sink, LL_ERROR, "[ParserCTPMini] Sending md subscribe request failed: {}", iResult);
401 }
402 else
403 {
404 if(m_sink)
405 write_log(m_sink, LL_INFO, "[ParserCTPMini] Market data of {} contracts subscribed in total", nCount);
406 }
407 }
408 codeFilter.clear();
409 delete[] subscribe;
410}
