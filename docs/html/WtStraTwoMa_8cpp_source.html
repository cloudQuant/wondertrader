112void WtStraTwoMa::on_session_begin(ICtaStraCtx* ctx, uint32_t uTDate)
113{
114 // 获取当前的主力合约代码
115 std::string newMonCode = ctx->stra_get_rawcode(_code.c_str());
116 // 如果主力合约发生变化
117 if(newMonCode!=_moncode)
118 {
119 // 如果已有之前的主力合约
120 if(!_moncode.empty())
121 {
122 // 获取当前持仓
123 double curPos = ctx->stra_get_position(_moncode.c_str());
124 // 如果有持仓，需要转移到新主力合约
125 if (!decimal::eq(curPos, 0))
126 {
127 ctx->stra_log_info(fmt::format("主力换月, 老主力{}[{}]将会被清理", _moncode, curPos).c_str());
128 // 将旧主力合约持仓清零
129 ctx->stra_set_position(_moncode.c_str(), 0, "switchout");
130 // 将相同持仓转移到新主力合约
131 ctx->stra_set_position(newMonCode.c_str(), curPos, "switchin");
132 }
133 }
134
135 // 更新当前主力合约代码
136 _moncode = newMonCode;

149void WtStraTwoMa::on_schedule(ICtaStraCtx* ctx, uint32_t curDate, uint32_t curTime)
150{
151 // 获取交易的合约代码
152 std::string code = _code;
153
154 // 获取K线数据
155 WTSKlineSlice *kline = ctx->stra_get_bars(code.c_str(), _period.c_str(), _count, true);
156 if(kline == NULL)
157 {
158 //这里可以输出一些日志
159 return;
160 }
161
162 // 检查K线数据是否为空
163 if (kline->size() == 0)
164 {
165 kline->release();
166 return;
167 }
168
169 // 设置交易单位，股票为100股，其他为1手
170 uint32_t trdUnit = 1;
171 if (_isstk)
172 trdUnit = 100;
173
174
175 // 转换回看天数为有符号整数
176 //int32_t days = (int32_t)_days;
177 int32_t short_days = (int32_t)_short_days;
178 int32_t long_days = (int32_t)_long_days;
179
180 // 计算指定天数内的最高价和最低价
181 //double hh = kline->maxprice(-days, -2);
182 //double ll = kline->minprice(-days, -2);
183
184 // 计算指定天数内的价格
185 WTSValueArray* closes = kline->extractData(KFT_CLOSE);
186 double pre_short_ma = 0;
187 double pre_long_ma = 0;
188 double now_short_ma = 0;
189 double now_long_ma = 0;
190 for (uint32_t i = short_days+1; i > 1 ; i--){
191 pre_short_ma += closes->at(-1*i);
192 }
193 for (uint32_t i = long_days+1; i > 1 ; i--){
194 pre_long_ma += closes->at(-1*i);
195 }
196 now_short_ma = pre_short_ma - closes->at(-1*(short_days+1))+closes->at(-1);
197 now_long_ma = pre_long_ma - closes->at(-1*(long_days+1)) + closes->at(-1);
198 pre_short_ma = pre_short_ma/short_days;
199 pre_long_ma = pre_long_ma/long_days;
200 now_short_ma = now_short_ma/short_days;
201 now_long_ma = now_long_ma/long_days;
202 //double hc = closes->maxvalue(-days, -2);
203 //double lc = closes->minvalue(-days, -2);
204 double curPx = closes->at(-1); // 当前收盘价
205 closes->release();
206
207 // 获取当前周期K线的开盘价
208 // double openPx = kline->at(-1)->open;
209
210 // 计算上下轨道
211 // double upper_bound = openPx + _k1 * (std::max(hh - lc, hc - ll));
212 // double lower_bound = openPx - _k2 * std::max(hh - lc, hc - ll);
213
214 //设置指标值，用于图表显示
215 // ctx->set_index_value("DualThrust", "upper_bound", upper_bound);
216 // ctx->set_index_value("DualThrust", "lower_bound", lower_bound);
217
218 // 获取合约信息
219 WTSCommodityInfo* commInfo = ctx->stra_get_comminfo(_code.c_str());
220
221 // 获取当前持仓并转换为手数
222 double curPos = ctx->stra_get_position(_moncode.c_str()) / trdUnit;
223
224 // 无持仓时的入场逻辑
225 if(decimal::eq(curPos,0))
226 {
227 // 做多
228 if (pre_short_ma < pre_long_ma and now_short_ma > now_long_ma)
229 {
230 ctx->stra_enter_long(_moncode.c_str(), 2 * trdUnit, "DT_EnterLong");
231 //向上突破
232 ctx->stra_log_info(fmt::format("金叉{}>={},多仓进场", pre_short_ma, now_short_ma).c_str());
233
234 //添加图表标记
235 ctx->add_chart_mark(curPx, "wt-mark-buy", "DT_EnterLong");
236 }
237 // 死叉 不是股票，做空
238 else if ( (pre_short_ma > pre_long_ma and now_short_ma < now_long_ma) && !_isstk)
239 {
240 ctx->stra_enter_short(_moncode.c_str(), 2 * trdUnit, "DT_EnterShort");
241 //向下突破
242 ctx->stra_log_info(fmt::format("死叉{}<={},空仓进场", pre_short_ma, now_short_ma).c_str());
243
244 //添加图表标记
245 ctx->add_chart_mark(curPx, "wt-mark-sell", "DT_EnterShort");
246 }
247 }
248 // 持有多头仓位时的出场逻辑
249 else if (decimal::gt(curPos, 0))
250 {
251 // 如果当前死叉，多头出场
252 if(pre_short_ma > pre_long_ma and now_short_ma < now_long_ma)
253 {
254 //多仓出场
255 ctx->stra_exit_long(_moncode.c_str(), 2 * trdUnit, "DT_ExitLong");
256 ctx->stra_log_info(fmt::format("死叉{}<={},多仓出场", pre_short_ma, now_short_ma).c_str());
257
258 //添加图表标记
259 ctx->add_chart_mark(curPx, "wt-mark-sell", "DT_ExitLong");
260 }
261 }
262 // 持有空头仓位时的出场逻辑
263 else if (decimal::lt(curPos, 0))
264 {
265 // 如果当前价格突破上轨且不是股票，空头出场
266 if ((pre_short_ma < pre_long_ma and now_short_ma > now_long_ma) && !_isstk)
267 {
268 //空仓出场
269 ctx->stra_exit_short(_moncode.c_str(), 2 * trdUnit, "DT_ExitShort");
270 ctx->stra_log_info(fmt::format("金叉{}>={},空仓出场", pre_short_ma, now_short_ma).c_str());
271
272 //添加图表标记
273 ctx->add_chart_mark(curPx, "wt-mark-buy", "DT_ExitShort");
274 }
275 }
276
277 //这个释放一定要做
278 kline->release();
279}
