103std::string extractProductID(const char* instrument)
104{
105 std::string strRet;
106 int nLen = 0;
107 // 提取字母部分作为品种代码
108 while ('A' <= instrument[nLen] && instrument[nLen] <= 'z')
109 {
110 strRet += instrument[nLen];
111 nLen++;
112 }
113
114 return strRet;
115}

126std::string extractProductName(const char* cname)
127{
128 std::string strRet;
129 // 从名称的末尾开始向前查找非数字字符
130 auto idx = strlen(cname) - 1;
131 while (isdigit(cname[idx]) && idx > 0)
132 {
133 idx--;
134 }
135
136 // 复制非数字部分作为品种名称
137 strRet.append(cname, idx + 1);
138 return strRet;
139}

324inline ContractCategory wrapCategory(TThostFtdcProductClassType cType)
325{
326 switch (cType)
327 {
328 case THOST_FTDC_PC_Futures: return CC_Future; // 期货
329 case THOST_FTDC_PC_Options: return CC_FutOption; // 期权
330 case THOST_FTDC_PC_Combination: return CC_Combination; // 组合
331 case THOST_FTDC_PC_Spot: return CC_Spot; // 现货
332 case THOST_FTDC_PC_EFP: return CC_EFP; // 期货期权组合
333 case THOST_FTDC_PC_SpotOption: return CC_SpotOption; // 现货期权
334 default:
335 throw std::runtime_error("non implemented category");
336 }
337}

353void CTraderSpi::OnRspQryInstrument(CThostFtdcInstrumentField *pInstrument, CThostFtdcRspInfoField *pRspInfo, int nRequestID, bool bIsLast)
354{
355 if (!IsErrorRspInfo(pRspInfo))
356 {
357 if (pInstrument)
358 {
359 std::string fullInstId = StrUtil::printf("%s.%s", pInstrument->ExchangeID, pInstrument->InstrumentID);
360 auto it = _contracts.find(fullInstId);
361 if (it != _contracts.end())
362 {
363 std::cerr << "--->>> " << pInstrument->ExchangeID << "." << pInstrument->InstrumentID << " already exists, skipped" << std::endl;
364 return;
365 }
366
367 bool bOption = isOption(pInstrument->ProductClass);
368 bool bFuture = isFuture(pInstrument->ProductClass);
369
370 bool isGranted = false;
371 if (bOption && (CLASSMASK & 2) != 0)
372 isGranted = true;
373 else if (bFuture && (CLASSMASK & 1) != 0)
374 isGranted = true;
375
376 do
377 {
378 if (isGranted)
379 {
380 std::string pname = MAP_NAME[pInstrument->ProductID];
381 if (pname.empty())
382 {
383 std::stringstream ss;
384 ss << pInstrument->ExchangeID << "." << pInstrument->ProductID;
385 pname = MAP_NAME[ss.str()];
386 }
387
388 std::string cname = "";
389 if (pname.empty())
390 {
391 if (ONLYINCFG)
392 {
393 std::cerr << "--->>> OnRspQryInstrument: " << pInstrument->ExchangeID << "." << pInstrument->InstrumentID << std::endl;
394 break;
395 }
396
397 cname = pInstrument->InstrumentName;
398 pname = bFuture ? extractProductName(pInstrument->InstrumentName) : pInstrument->InstrumentName;
399
400 std::cerr << "--->>> Name confirmed automatically: " << pInstrument->ExchangeID << "." << pInstrument->InstrumentID << std::endl;
401 }
402 else
403 {
404 if (bFuture)
405 {
406 std::string month = pInstrument->InstrumentID;
407 month = month.substr(strlen(pInstrument->ProductID));
408 cname = pname + month;
409 }
410 else if (bOption)
411 {
412 std::string underlyPID = extractProductID(pInstrument->UnderlyingInstrID);
413 std::string month = pInstrument->InstrumentID;
414 month = month.substr(underlyPID.size());
415 cname = pname + month;
416 }
417 else
418 {
419 cname = pInstrument->InstrumentName;
420 }
421
422 }
423
424 //合约名称转成UTF8
425 cname = StrUtil::trim(cname.c_str());
426 if (!EncodingHelper::isUtf8((unsigned char*)cname.c_str(), cname.size()))
427 cname = ChartoUTF8(cname);
428
429 Contract contract;
430 contract.m_strCode = pInstrument->InstrumentID;
431 contract.m_strExchg = pInstrument->ExchangeID;
432 contract.m_strName = cname;
433 contract.m_strProduct = pInstrument->ProductID;
434
435 contract.m_maxMktQty = pInstrument->MaxMarketOrderVolume;
436 contract.m_maxLmtQty = pInstrument->MaxLimitOrderVolume;
437 contract.m_minMktQty = pInstrument->MinMarketOrderVolume;
438 contract.m_minLmtQty = pInstrument->MinLimitOrderVolume;
439
440 contract.m_optType = bOption ? (OptionType)pInstrument->OptionsType : OT_None;
441 contract.m_strUnderlying = pInstrument->UnderlyingInstrID;
442 contract.m_strikePrice = pInstrument->StrikePrice;
443 contract.m_dUnderlyingScale = pInstrument->UnderlyingMultiple;
444
445 contract.m_uOpenDate = strtoul(pInstrument->OpenDate, NULL, 10);
446 contract.m_uExpireDate = strtoul(pInstrument->ExpireDate, NULL, 10);
447
448 contract.m_dLongMarginRatio = checkValid(pInstrument->LongMarginRatio);
449 contract.m_dShortMarginRatio = checkValid(pInstrument->ShortMarginRatio);
450
451 std::string fullPid = StrUtil::printf("%s.%s", pInstrument->ExchangeID, pInstrument->ProductID);
452 auto it = _commodities.find(fullPid);
453 if (it == _commodities.end())
454 {
455 //品种名称也转成UTF8
456 pname = StrUtil::trim(pname.c_str());
457 if (!EncodingHelper::isUtf8((unsigned char*)pname.c_str(), pname.size()))
458 pname = ChartoUTF8(pname);
459
460 Commodity commInfo;
461 commInfo.m_strProduct = pInstrument->ProductID;
462 commInfo.m_strName = pname;
463 commInfo.m_strExchg = pInstrument->ExchangeID;
464 commInfo.m_strCurrency = "CNY";
465
466 commInfo.m_strSession = MAP_SESSION[fullPid];
467 commInfo.m_ccCategory = wrapCategory(pInstrument->ProductClass);
468
469 commInfo.m_uVolScale = (pInstrument->VolumeMultiple == 0 ? 1 : pInstrument->VolumeMultiple);
470 commInfo.m_fPriceTick = pInstrument->PriceTick;
471
472 CoverMode cm = CM_OpenCover;
473 /*
474 * By Wesley @ 2023.05.04
475 * 有用户反馈上期所和上能所的期权合约也区分平昨平今
476 * 把这个bFuture的判断去掉
477 */
478 //if (bFuture)
479 {
480 if (strcmp(pInstrument->ExchangeID, "SHFE") == 0 || strcmp(pInstrument->ExchangeID, "INE") == 0)
481 cm = CM_CoverToday;
482 //上期所的就是平今,非上期所的就是开平
483 }
484
485 commInfo.m_coverMode = cm;
486
487 PriceMode pm = PM_Both;
488 /*
489 * By Wesley @ 2023.05.04
490 * 有用户反馈上期所和上能所的期权合约也区分平昨平今
491 * 把这个bFuture的判断去掉
492 */
493 //if (bFuture)
494 {
495 if (strcmp(pInstrument->ExchangeID, "SHFE") == 0 || strcmp(pInstrument->ExchangeID, "INE") == 0)
496 pm = PM_Limit;
497 }
498 commInfo.m_priceMode = pm;
499 commInfo.m_tradeMode = TM_Both;
500
501 if (pInstrument->PriceTick < 0.001)
502 commInfo.m_uPrecision = 4;
503 else if (pInstrument->PriceTick < 0.01)
504 commInfo.m_uPrecision = 3;
505 else if (pInstrument->PriceTick < 0.1)
506 commInfo.m_uPrecision = 2;
507 else if (pInstrument->PriceTick < 1)
508 commInfo.m_uPrecision = 1;
509 else
510 commInfo.m_uPrecision = 0;
511
512 _commodities[fullPid] = commInfo;
513 }
514
515 _contracts[fullInstId] = contract;
516 }
517 } while (false);
518
519 }
520 }
521
522 if (bIsLast)
523 {
524 DumpToJson();
525 exit(0);
526 }

536void CTraderSpi::LoadFromJson()
537{
538 std::string path = SAVEPATH;
539 path += COMM_FILE;
540 WTSVariant* root = WTSCfgLoader::load_from_file(path);
541 if(root)
542 {
543 for(const auto& exchg : root->memberNames())
544 {
545 WTSVariant* jExchg = root->get(exchg.c_str());
546 for(const auto& pid : jExchg->memberNames())
547 {
548 WTSVariant* pComm = jExchg->get(pid.c_str());
549 std::string key = fmt::format("{}.{}", exchg, pid);
550
551 Commodity& commInfo = _commodities[key];
552 commInfo.m_strProduct = pid;
553 commInfo.m_strName = pComm->getCString("name");
554 commInfo.m_strExchg = exchg;
555 commInfo.m_strCurrency = "CNY";
556
557 commInfo.m_strSession = MAP_SESSION[key];
558 commInfo.m_ccCategory = (ContractCategory)pComm->getUInt32("category");
559
560 commInfo.m_uVolScale = pComm->getUInt32("volscale");;
561 commInfo.m_fPriceTick = pComm->getDouble("pricetick");
562
563 commInfo.m_coverMode = (CoverMode)pComm->getUInt32("covermode");
564
565 commInfo.m_priceMode = (PriceMode)pComm->getUInt32("pricemode");;
566 commInfo.m_tradeMode = (TradingMode)pComm->getUInt32("trademode");
567
568 commInfo.m_uPrecision = pComm->getUInt32("precision");
569 }
570 }
571
572 root->release();
573 }
574
575 path = SAVEPATH;
576 path += CONT_FILE;
577 root = WTSCfgLoader::load_from_file(path);
578 if (root)
579 {
580 WTSVariant* root = WTSCfgLoader::load_from_file(path);
581 for (const auto& exchg_id : root->memberNames())
582 {
583 WTSVariant* jExchg = root->get(exchg_id.c_str());
584 for (const auto& inst_id : jExchg->memberNames())
585 {
586 WTSVariant* pCont = jExchg->get(inst_id.c_str());
587 std::string key = fmt::format("{}.{}", exchg_id, inst_id);
588
589 Contract& contract = _contracts[key];
590 contract.m_strCode = inst_id;
591 contract.m_strExchg = exchg_id;
592 contract.m_strName = pCont->getCString("name");
593 contract.m_strProduct = pCont->getCString("product");
594
595 contract.m_maxMktQty = pCont->getUInt32("maxmarketqty");
596 contract.m_maxLmtQty = pCont->getUInt32("maxlimitqty");
597 contract.m_minMktQty = pCont->getUInt32("minmarketqty");
598 contract.m_minLmtQty = pCont->getUInt32("minlimitqty");
599
600 if(pCont->has("option"))
601 {
602 contract.m_optType = (OptionType)pCont->getUInt32("optiontype");
603 contract.m_strUnderlying = pCont->getCString("underlying");
604 contract.m_strikePrice = pCont->getDouble("product");
605 contract.m_dUnderlyingScale = pCont->getDouble("underlyingscale");
606 }
607
608
609 contract.m_uOpenDate = pCont->getUInt32("opendate");
610 contract.m_uExpireDate = pCont->getUInt32("expiredate");
611
612 contract.m_dLongMarginRatio = pCont->getDouble("longmarginratio");
613 contract.m_dShortMarginRatio = pCont->getDouble("shortmarginratio");
614 }
615 }
616
617 root->release();
618 }
619 std::cerr << "--->>> " << "LoadFromJson" << std::endl;
620}

782bool CTraderSpi::IsErrorRspInfo(CThostFtdcRspInfoField *pRspInfo)
783{
784 // 如果ErrorID != 0, 说明收到了错误的响应
