68void WtUftRtTicker::on_tick(WTSTickData* curTick)
69{
70 // 如果线程未初始化，直接将Tick数据转发给引擎并返回
71 if (_thrd == NULL)
72 {
73 if (_engine)
74 _engine->on_tick(curTick->code(), curTick);
75 return;
76 }
77
78 // 获取Tick数据的日期和时间
79 uint32_t uDate = curTick->actiondate();
80 uint32_t uTime = curTick->actiontime();
81
82 // 如果行情时间早于当前时间，直接处理并返回
83 if (_date != 0 && (uDate < _date || (uDate == _date && uTime < _time)))
84 {
85 //WTSLogger::info("行情时间{}小于本地时间{}", uTime, _time);
86 if (_engine)
87 _engine->on_tick(curTick->code(), curTick);
88 return;
89 }
90
91 // 更新当前日期和时间
92 _date = uDate;
93 _time = uTime;
94
95 // 计算当前分钟和秒数
96 uint32_t curMin = _time / 100000;
97 uint32_t curSec = _time % 100000;
98 // 将当前时间转换为交易时段中的分钟位置
99 uint32_t minutes = _s_info->timeToMinutes(curMin);
100 // 判断是否是交易时段的最后一分钟
101 bool isSecEnd = _s_info->isLastOfSection(curMin);
102 if (isSecEnd)
103 {
104 minutes--;
105 }
106 minutes++;
107 // 保存原始分钟时间并将交易时段分钟位置转回实际时间
108 uint32_t rawMin = curMin;
109 curMin = _s_info->minuteToTime(minutes);
110
111 // 如果是首次设置当前分钟位置
112 if (_cur_pos == 0)
113 {
114 //如果当前时间是0, 则直接赋值即可
115 _cur_pos = minutes;
116 }
117 // 如果时间向前跃进，需要触发分钟线结束事件
118 else if (_cur_pos < minutes)
119 {
120 //如果已记录的分钟小于新的分钟, 则需要触发闭合事件
121 //这个时候要先触发闭合, 再修改平台时间和价格
122 if (_last_emit_pos < _cur_pos)
123 {
124 //触发数据回放模块
125 StdUniqueLock lock(_mtx);
126
127 //优先修改时间标记
128 _last_emit_pos = _cur_pos;
129
130 // 获取当前分钟位置对应的实际时间
131 uint32_t thisMin = _s_info->minuteToTime(_cur_pos);
132
133 // 输出日志并触发分钟线结束事件
134 WTSLogger::info("Minute Bar {}.{:04d} Closed by data", _date, thisMin);
135 _engine->on_minute_end(_date, thisMin);
136 }
137
138 // 更新引擎时间并触发Tick数据
139 if (_engine)
140 {
141 _engine->on_tick(curTick->code(), curTick);
142 _engine->set_date_time(_date, curMin, curSec, rawMin);
143 _engine->set_trading_date(curTick->tradingdate());
144 }
145
146 // 更新当前分钟位置
147 _cur_pos = minutes;
148 }
149 // 如果在同一分钟内，直接处理Tick数据
150 else
151 {
152 //如果分钟数还是一致的, 则直接触发行情和时间即可
153 if (_engine)
154 {
155 _engine->on_tick(curTick->code(), curTick);
156 _engine->set_date_time(_date, curMin, curSec, rawMin);
157 }
158 }
159
160 // 计算距离下一分钟的剩余时间（毫秒）
161 uint32_t sec = curSec / 1000;
162 uint32_t msec = curSec % 1000;
163 uint32_t left_ticks = (60 - sec) * 1000 - msec;
164 // 设置下次检查时间
165 _next_check_time = TimeUtils::getLocalTimeNow() + left_ticks;
166}

172void WtUftRtTicker::run()
173{
174 // 如果线程已存在，直接返回
175 if (_thrd)
176 return;
177
178 // 执行引擎初始化
179 _engine->on_init();
180
181 // 计算当前交易日期并设置到引擎中
182 uint32_t curTDate = _engine->get_basedata_mgr()->calcTradingDate(_s_info->id(), _engine->get_date(), _engine->get_min_time(), true);
183 _engine->set_trading_date(curTDate);
184
185 // 触发交易日开始事件
186 _engine->on_session_begin();
187
188 // 获取当前时间在交易时段中的偏移值
189 uint32_t offTime = _s_info->offsetTime(_engine->get_min_time(), true);
190
191 // 创建并启动工作线程
192 _thrd.reset(new StdThread([this, offTime](){
193 // 循环运行直到被停止
194 while (!_stopped)
195 {
196 // 如果当前时间在交易时间内
197 if (_time != UINT_MAX && _s_info->isInTradingTime(_time / 100000, true))
198 {
199 // 在交易时间内每10毫秒检查一次
200 std::this_thread::sleep_for(std::chrono::milliseconds(10));
201 uint64_t now = TimeUtils::getLocalTimeNow();
202
203 // 如果到达下次检查时间且当前分钟位置尚未触发结束事件
204 if (now >= _next_check_time && _last_emit_pos < _cur_pos)
205 {
206 // 触发数据回放模块，加锁保护
207 StdUniqueLock lock(_mtx);
208
209 // 更新上次触发位置
210 _last_emit_pos = _cur_pos;
211
212 // 获取当前分钟位置对应的实际时间
213 uint32_t thisMin = _s_info->minuteToTime(_cur_pos);
214 _time = thisMin;
215
216 // 处理日期切换（如果时间是0点）
217 if (thisMin == 0)
218 {
219 uint32_t lastDate = _date;
220 _date = TimeUtils::getNextDate(_date);
221 _time = 0;
222 WTSLogger::info("Data automatically changed at time 00:00: {} -> {}", lastDate, _date);
223 }
224
225 // 输出日志并触发分钟线结束事件
226 WTSLogger::info("Minute bar {}.{:04d} closed automatically", _date, thisMin);
227 //if (_store)
228 // _store->onMinuteEnd(_date, thisMin);
229
230 _engine->on_minute_end(_date, thisMin);
231
232 // 如果当前时间是交易时段的收盤时间，触发交易日结束事件
233 uint32_t offMin = _s_info->offsetTime(thisMin, true);
234 if (offMin >= _s_info->getCloseTime(true))
235 {
236 _engine->on_session_end();
237 }
238
239 // 更新引擎时间
240 if (_engine)
241 _engine->set_date_time(_date, thisMin, 0);
242 }
243 }
244 else //if (offTime >= _s_info->getOpenTime(true) && offTime <= _s_info->getCloseTime(true))
245 {
246 // 不在交易时间，则休息10s再进行检查
247 // 因为这个逻辑是处理分钟线的，所以休盤时间休息10s，不会引起数据踏空的问题
248 std::this_thread::sleep_for(std::chrono::seconds(10));
249 }
250
251 }
252 }));
253}
