50bool CtaStrategyMgr::loadFactories(const char* path)
51{
52 // 检查目录是否存在
53 if (!StdFile::exists(path))
54 {
55 WTSLogger::error("Directory {} of CTA strategy factory not exists", path);
56 return false;
57 }
58
59 uint32_t count = 0;
60 boost::filesystem::path myPath(path);
61 boost::filesystem::directory_iterator endIter;
62 // 遍历目录中的所有文件
63 for (boost::filesystem::directory_iterator iter(myPath); iter != endIter; iter++)
64 {
65 // 跳过子目录
66 if (boost::filesystem::is_directory(iter->path()))
67 continue;
68
69 // 根据不同平台筛选动态库文件
70#ifdef _WIN32
71 if (iter->path().extension() != ".dll")
72 continue;
73#else //_UNIX
74 if (iter->path().extension() != ".so")
75 continue;
76#endif
77
78 // 加载动态库
79 DllHandle hInst = DLLHelper::load_library(iter->path().string().c_str());
80 if (hInst == NULL)
81 continue;
82
83 // 获取工厂创建函数
84 FuncCreateStraFact creator = (FuncCreateStraFact)DLLHelper::get_symbol(hInst, "createStrategyFact");
85 if (creator == NULL)
86 {
87 DLLHelper::free_library(hInst);
88 continue;
89 }
90
91 // 创建工厂实例
92 ICtaStrategyFact* fact = creator();
93 if(fact != NULL)
94 {
95 // 存储工厂信息
96 StraFactInfo& fInfo = _factories[fact->getName()];
97 fInfo._module_inst = hInst;
98 fInfo._module_path = iter->path().string();
99 fInfo._creator = creator;
100 fInfo._remover = (FuncDeleteStraFact)DLLHelper::get_symbol(hInst, "deleteStrategyFact");
101 fInfo._fact = fact;
102
103 WTSLogger::info("CTA strategy factory[{}] loaded", fact->getName());
104
105 count++;
106 }
107 else
108 {
109 DLLHelper::free_library(hInst);
110 continue;
111 }
112
113 }
114
115 WTSLogger::info("{} CTA strategy factories in directory[{}] loaded", count, path);
116
117 return true;
118}

131CtaStrategyPtr CtaStrategyMgr::createStrategy(const char* factname, const char* unitname, const char* id)
132{
133 // 查找工厂
134 auto it = _factories.find(factname);
135 if (it == _factories.end())
136 return CtaStrategyPtr(); // 如果工厂不存在，返回空指针

138 // 使用工厂创建策略实例
139 StraFactInfo& fInfo = (StraFactInfo&)it->second;

141 // 将策略实例添加到策略映射表
142 _strategies[id] = ret;
143 return ret;
144}
