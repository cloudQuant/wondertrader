78void WtCtaEngine::run()
79{
80 _tm_ticker = new WtCtaRtTicker(this);
81 WTSVariant* cfgProd = _cfg->get("product");
82 _tm_ticker->init(_data_mgr->reader(), cfgProd->getCString("session"));
83
84 //启动之前,先把运行中的策略落地
85 {
86 rj::Document root(rj::kObjectType);
87 rj::Document::AllocatorType &allocator = root.GetAllocator();
88
89 rj::Value jStraList(rj::kArrayType);
90 for (auto& m : _ctx_map)
91 {
92 const CtaContextPtr& ctx = m.second;
93 jStraList.PushBack(rj::Value(ctx->name(), allocator), allocator);
94 }
95
96 root.AddMember("marks", jStraList, allocator);
97
98 rj::Value jChnlList(rj::kArrayType);
99 for (auto& m : _adapter_mgr->getAdapters())
100 {
101 const TraderAdapterPtr& adapter = m.second;
102 jChnlList.PushBack(rj::Value(adapter->id(), allocator), allocator);
103 }
104
105 root.AddMember("channels", jChnlList, allocator);
106
107 rj::Value jExecList(rj::kArrayType);
108 _exec_mgr.enum_executer([&jExecList, &allocator](ExecCmdPtr executer) {
109 if(executer)
110 jExecList.PushBack(rj::Value(executer->name(), allocator), allocator);
111 });
112
113 root.AddMember("executers", jExecList, allocator);
114
115 root.AddMember("engine", rj::Value("CTA", allocator), allocator);
116
117 std::string filename = WtHelper::getBaseDir();
118 filename += "marker.json";
119
120 rj::StringBuffer sb;
121 rj::PrettyWriter<rj::StringBuffer> writer(sb);
122 root.Accept(writer);
123 StdFile::write_file_content(filename.c_str(), sb.GetString());
124 }
125
126 _tm_ticker->run();
127
128 if (_risk_mon)
129 _risk_mon->self()->run();
130}

536void WtCtaEngine::handle_pos_change(const char* straName, const char* stdCode, double diffPos)
537{
538 //这里是持仓增量,所以不用处理未过滤的情况,因为增量情况下,不会改变目标diffQty
539 if(_filter_mgr.is_filtered_by_strategy(straName, diffPos, true))
540 {
541 //输出日志
542 WTSLogger::info("[Filters] Target position of {} of strategy {} ignored by strategy filter", stdCode, straName);
543 return;
544 }
545
546 std::string realCode = stdCode;
547 //const char* ruleTag = _hot_mgr->getRuleTag(stdCode);
548 CodeHelper::CodeInfo cInfo = CodeHelper::extractStdCode(stdCode, _hot_mgr);
549 if (strlen(cInfo._ruletag) > 0)
550 {
551 std::string code = _hot_mgr->getCustomRawCode(cInfo._ruletag, cInfo.stdCommID(), _cur_tdate);
552 realCode = CodeHelper::rawMonthCodeToStdCode(code.c_str(), cInfo._exchg);
553 }
554
555 /*
556 * 这里必须要算一个总的目标仓位
557 */
558 PosInfoPtr& pInfo = _pos_map[realCode];
559 if (pInfo == NULL)
560 pInfo.reset(new PosInfo);
561
562 bool bRiskEnabled = false;
563 if (!decimal::eq(_risk_volscale, 1.0) && _risk_date == _cur_tdate)
564 {
565 WTSLogger::log_by_cat("risk", LL_INFO, "Risk scale of portfolio is {:.2f}", _risk_volscale);
566 bRiskEnabled = true;
567 }
568
569 if (bRiskEnabled && !decimal::eq(diffPos, 0))
570 {
571 double symbol = diffPos / abs(diffPos);
572 diffPos = decimal::rnd(abs(diffPos)*_risk_volscale)*symbol;
573 }
574
575 double targetPos = pInfo->_volume + diffPos;
576
577 append_signal(realCode.c_str(), targetPos, false);
578 save_datas();
579
580 /*
581 * 如果策略绑定了执行通道
582 * 那么就只提交增量
583 * 如果策略没有绑定执行通道，就提交全量
584 */
585 const auto& exec_ids = _exec_mgr.get_route(straName);
586 for(auto& execid : exec_ids)
587 _exec_mgr.handle_pos_change(realCode.c_str(), targetPos, diffPos, execid.c_str());
588}

603void WtCtaEngine::on_tick(const char* stdCode, WTSTickData* curTick)
604{
605 WtEngine::on_tick(stdCode, curTick);
606
607 _data_mgr->handle_push_quote(stdCode, curTick);
608
609 //如果是真实代码, 则要传递给执行器
610 /*
611 * 这里不再做判断，直接全部传递给执行器管理器，因为执行器可能会处理未订阅的合约
612 * 主要场景为主力合约换月期间
613 * By Wesley @ 2021.08.19
614 */
615 {
616 //是否主力合约代码的标记, 主要用于给执行器发数据的
617 _exec_mgr.handle_tick(stdCode, curTick);
618 }
619
620 /*
621 * By Wesley @ 2022.02.07
622 * 这里做了一个彻底的调整
623 * 第一，检查订阅标记，如果标记为0，即无复权模式，则直接按照原始代码触发ontick
624 * 第二，如果标记为1，即前复权模式，则将代码转成xxxx-，再触发ontick
625 * 第三，如果标记为2，即后复权模式，则将代码转成xxxx+，再把tick数据做一个修正，再触发ontick
626 */
627 if(_ready)
628 {
629 auto sit = _tick_sub_map.find(stdCode);
630 if (sit == _tick_sub_map.end())
631 return;
632
633 uint32_t flag = get_adjusting_flag();
634 WTSTickData* adjTick = nullptr;
635
636 //By Wesley
637 //这里做一个拷贝，虽然有点开销，但是可以规避掉一些问题，比如ontick的时候订阅tick
638 SubList sids = sit->second;
639 for (auto it = sids.begin(); it != sids.end(); it++)
640 {
641 uint32_t sid = it->first;
642
643
644 auto cit = _ctx_map.find(sid);
645 if (cit != _ctx_map.end())
646 {
647 CtaContextPtr& ctx = (CtaContextPtr&)cit->second;
648 uint32_t opt = it->second.second;
649
650 if (opt == 0)
651 {
652 /*
653 * By Wesley @ 2023.06.27
654 * 如果使用线程池，则到线程池里去调度
655 */
656 if(_pool)
657 {
658 _pool->schedule([ctx, stdCode, curTick]() {
659 ctx->on_tick(stdCode, curTick);
660 });
661 }
662 else
663 ctx->on_tick(stdCode, curTick);
664 }
665 else
666 {
667 std::string wCode = stdCode;
668 wCode = fmt::format("{}{}", stdCode, opt == 1 ? SUFFIX_QFQ : SUFFIX_HFQ);
669 if (opt == 1)
670 {
671 if (_pool)
672 {
673 _pool->schedule([ctx, wCode, curTick]() {
674 ctx->on_tick(wCode.c_str(), curTick);
675 });
676 }
677 else
678 ctx->on_tick(wCode.c_str(), curTick);
679 }
680 else //(opt == 2)
681 {
682 if (adjTick == nullptr)
683 {
684 WTSTickData* adjTick = WTSTickData::create(curTick->getTickStruct());
685 WTSTickStruct& adjTS = adjTick->getTickStruct();
686 adjTick->setContractInfo(curTick->getContractInfo());
687
688 //这里做一个复权因子的处理
689 double factor = get_exright_factor(stdCode);
690 adjTS.open *= factor;
691 adjTS.high *= factor;
692 adjTS.low *= factor;
693 adjTS.price *= factor;
694
695 adjTS.settle_price *= factor;
696
697 adjTS.pre_close *= factor;
698 adjTS.pre_settle *= factor;
699
700 /*
701 * By Wesley @ 2022.08.15
702 * 这里对tick的复权做一个完善
703 */
704 if (flag & 1)
705 {
706 adjTS.total_volume /= factor;
707 adjTS.volume /= factor;
708 }
709
710 if (flag & 2)
711 {
712 adjTS.total_turnover *= factor;
713 adjTS.turn_over *= factor;
714 }
715
716 if (flag & 4)
717 {
718 adjTS.open_interest /= factor;
719 adjTS.diff_interest /= factor;
720 adjTS.pre_interest /= factor;
721 }
722
723 _price_map[wCode] = adjTS.price;
724 }
725
726 if (_pool)
727 {
728 _pool->schedule([ctx, wCode, adjTick]() {
729 ctx->on_tick(wCode.c_str(), adjTick);
730 });
731 }
732 else
733 ctx->on_tick(wCode.c_str(), adjTick);
734
735 }
736 }
737 }
738 }
739
740 if(nullptr != adjTick)
741 adjTick->release();
742 /*
743 * By Wesley @ 223.06.27
744 * 这里一定要等待线程池全部调度完成
745 */
746 if (_pool)
747 _pool->wait();
748 }

762void WtCtaEngine::on_bar(const char* stdCode, const char* period, uint32_t times, WTSBarStruct* newBar)
763{
764 thread_local static char key[64] = { 0 };
765 fmtutil::format_to(key, "{}-{}-{}", stdCode, period, times);
766
767 const SubList& sids = _bar_sub_map[key];
768 for (auto it = sids.begin(); it != sids.end(); it++)
769 {
770 uint32_t sid = it->first;
771 auto cit = _ctx_map.find(sid);
772 if(cit != _ctx_map.end())
773 {
774 CtaContextPtr& ctx = (CtaContextPtr&)cit->second;
775 if (_pool)
776 {
777 _pool->schedule([ctx, stdCode, period, times, newBar]() {
778 ctx->on_bar(stdCode, period, times, newBar);
779 });
780 }
781 else
782 ctx->on_bar(stdCode, period, times, newBar);
783 }
784 }
785
786 /*
787 * By Wesley @ 223.06.27
788 * 这里一定要等待线程池全部调度完成
789 */
790 if (_pool)
791 _pool->wait();
792
793 WTSLogger::info("KBar [{}] @ {} closed", key, period[0] == 'd' ? newBar->date : newBar->time);
794}
