38bool StateMonitor::initialize(const char* filename, WTSBaseDataMgr* bdMgr, DataManager* dtMgr)
39{
40 _bd_mgr = bdMgr;
41 _dt_mgr = dtMgr;
42
43 if (!StdFile::exists(filename))
44 {
45 WTSLogger::error("State config file {} not exists", filename);
46 return false;
47 }
48
49 WTSVariant* config = WTSCfgLoader::load_from_file(filename);
50 if (config == NULL)
51 {
52 WTSLogger::error("Loading state config failed");
53 return false;
54 }
55
56 auto keys = config->memberNames();
57 for (const std::string& sid : keys)
58 {
59 WTSVariant* jItem = config->get(sid.c_str());
60
61 WTSSessionInfo* ssInfo = _bd_mgr->getSession(sid.c_str());
62 if (ssInfo == NULL)
63 {
64 WTSLogger::error("Trading session template [{}] not exists,state control rule skipped", sid);
65 continue;
66 }
67
68 StatePtr stateInfo(new StateInfo);
69 stateInfo->_sInfo = ssInfo;
70 stateInfo->_init_time = jItem->getUInt32("inittime"); //初始化时间,初始化以后数据才开始接收
71 stateInfo->_close_time = jItem->getUInt32("closetime"); //收盘时间,收盘后数据不再接收了
72 stateInfo->_proc_time = jItem->getUInt32("proctime"); //盘后处理时间,主要把实时数据转到历史去
73
74 strcpy(stateInfo->_session, sid.c_str());
75
76 const auto& auctions = ssInfo->getAuctionSections();//这里面是偏移过的时间,要注意了!!!
77 for(const auto& secInfo : auctions)
78 {
79 uint32_t stime = secInfo.first;
80 uint32_t etime = secInfo.second;
81
82 stime = stime / 100 * 60 + stime % 100;//先将时间转成分钟数
83 etime = etime / 100 * 60 + etime % 100;
84
85 stime = stime / 60 * 100 + stime % 60;//再将分钟数转成时间
86 etime = etime / 60 * 100 + etime % 60;//先不考虑半夜12点的情况,目前看来,几乎没有
87 stateInfo->_sections.emplace_back(StateInfo::Section({ stime, etime }));
88 }
89
90 const auto& sections = ssInfo->getTradingSections();//这里面是偏移过的时间,要注意了!!!
91 for (const auto& secInfo : sections)
92 {
93 uint32_t stime = secInfo.first;
94 uint32_t etime = secInfo.second;
95
96 stime = stime / 100 * 60 + stime % 100;//先将时间转成分钟数
97 etime = etime / 100 * 60 + etime % 100;
98
99 stime--;//开始分钟数-1
100 etime++;//结束分钟数+1
101
102 stime = stime / 60 * 100 + stime % 60;//再将分钟数转成时间
103 etime = etime / 60 * 100 + etime % 60;//先不考虑半夜12点的情况,目前看来,几乎没有
104 stateInfo->_sections.emplace_back(StateInfo::Section({ stime, etime }));
105 }
106
107 _map[stateInfo->_session] = stateInfo;
108
109 CodeSet* pCommSet = _bd_mgr->getSessionComms(stateInfo->_session);
110 if (pCommSet)
111 {
112 uint32_t curDate = TimeUtils::getCurDate();
113 uint32_t curMin = TimeUtils::getCurMin() / 100;
114 uint32_t offDate = ssInfo->getOffsetDate(curDate, curMin);
115 uint32_t offMin = ssInfo->offsetTime(curMin, true);
116
117 //先获取基准的交易日
118
119 for (auto it = pCommSet->begin(); it != pCommSet->end(); it++)
120 {
121 const char* pid = (*it).c_str();
122
123 _bd_mgr->setTradingDate(pid, _bd_mgr->getTradingDate(pid, offDate, offMin, false), false);
124 uint32_t prevDate = TimeUtils::getNextDate(curDate, -1);
125 if ((ssInfo->getOffsetMins() > 0 &&
126 (! _bd_mgr->isTradingDate(pid, curDate) &&
127 !(ssInfo->isInTradingTime(curMin) && _bd_mgr->isTradingDate(pid, prevDate)))) ||
128 (ssInfo->getOffsetMins() <= 0 && ! _bd_mgr->isTradingDate(pid, offDate))
129 )
130 {
131 WTSLogger::info("Instrument {} is in holiday", pid);
132 }
133 }
134 }
135 }
136 return true;
137}

145void StateMonitor::run()
146{
147 if(_thrd == NULL)
148 {
149 // 创建监控线程
150 _thrd.reset(new StdThread([this](){
151
152 // 主循环，直到收到停止信号
153 while (!_stopped)
154 {
155 static uint64_t lastTime = 0; // 上次检查时间戳
156
157 // 控制检查频率为每秒一次
158 while(true)
159 {
160 uint64_t now = TimeUtils::getLocalTimeNow(); // 获取当前时间戳
161 if(now - lastTime >= 1000) // 时间间隔达到1秒
162 break;
163
164 if(_stopped) // 如果收到停止信号，立即退出
165 break;
166
167 // 短暂休眠，避免占用过多CPU资源
168 std::this_thread::sleep_for(std::chrono::milliseconds(2));
169 }
170
171 // 检查是否需要退出线程
172 if(_stopped)
173 break;
174
175 // 获取当前日期和分钟数
176 uint32_t curDate = TimeUtils::getCurDate(); // 当前日期，格式YYYYMMDD
177 uint32_t curMin = TimeUtils::getCurMin() / 100; // 当前分钟数，格式HHMM
178
179 // 遍历所有交易时间段，检查和更新状态
180 auto it = _map.begin();
181 for (; it != _map.end(); it++)
182 {
183 StatePtr& stateInfo = (StatePtr&)it->second; // 获取当前交易时间段的状态信息
184
185 WTSSessionInfo* sInfo = stateInfo->_sInfo; // 获取交易时间段信息
186
187 // 获取偏移后的日期和前一天日期，用于识别夜盘和节假日情况
188 uint32_t offDate = sInfo->getOffsetDate(curDate, curMin); // 计算偏移后的日期
189 uint32_t prevDate = TimeUtils::getNextDate(curDate, -1); // 前一天日期
190
191 // 根据当前状态执行相应的状态转换逻辑
192 // 各个状态有不同的转换条件和目标状态
193 switch(stateInfo->_state)
194 {
195 // 原始状态，初始化后的默认状态
196 case SS_ORIGINAL:
197 {
198 // 计算各个关键时间点，用于状态判断
199 uint32_t offTime = sInfo->offsetTime(curMin, true); // 当前时间的偏移值，格式HHMM
200 uint32_t offInitTime = sInfo->offsetTime(stateInfo->_init_time, true); // 初始化时间的偏移值
201 uint32_t offCloseTime = sInfo->offsetTime(stateInfo->_close_time, false); // 收盘时间的偏移值
202 uint32_t aucStartTime = sInfo->getAuctionStartTime(true); // 集合竹开始时间
203
204 // 检查是否所有合约都处于节假日状态
205 bool isAllHoliday = true;
206 std::stringstream ss_a, ss_b; // 用于收集节假日和非节假日合约ID
207 // 获取当前交易时间段关联的所有合约
208 CodeSet* pCommSet = _bd_mgr->getSessionComms(stateInfo->_session);
209 // 如果存在合约集合，检查每个合约是否处于节假日
210 if (pCommSet)
211 {
212 // 遍历交易时间段关联的所有合约
213 for (auto it = pCommSet->begin(); it != pCommSet->end(); it++)
214 {
215 const char* pid = (*it).c_str(); // 合约ID
216 /*
217 * 节假日判断逻辑：
218 * 1. 如果时间往后偏移（夜盘情况）：
219 * a. 当前日期不是交易日，且
220 * b. 非夜盘后半夜情况（即不满足：当前在交易时间且昨天是交易日）
221 * 2. 如果时间没有往后偏移（白盘情况）：
222 * 状态取决于算出的偏移日期是否为交易日
223 */
224 if ((sInfo->getOffsetMins() > 0 &&
225 (! _bd_mgr->isTradingDate(pid, curDate) && // 当前日期不是交易日
226 !(sInfo->isInTradingTime(curMin) && _bd_mgr->isTradingDate(pid, prevDate)))) || // 当前不在交易时间或昨天不是交易日
227 (sInfo->getOffsetMins() <= 0 && ! _bd_mgr->isTradingDate(pid, offDate)) // 白盘情况下偏移日期不是交易日
228 )
229 {
230 // 当前合约处于节假日
231 ss_a << pid << ","; // 收集节假日合约ID
232 WTSLogger::info("Instrument {} is in holiday", pid);
233 }
234 else
235 {
236 // 当前合约不处于节假日
237 ss_b << pid << ","; // 收集非节假日合约ID
238 isAllHoliday = false; // 至少有1个合约不在节假日
239 }
240 }
241
242 }
243 else
244 {
245 // 如果没有合约与当前交易时间段关联，等同于节假日
246 WTSLogger::info("No corresponding instrument of trading session {}[{}], changed into holiday state", sInfo->name(), stateInfo->_session);
247 stateInfo->_state = SS_Holiday;
248 }
249
250 // 状态转换逻辑：根据合约状态和时间点确定下一个状态
251 if(isAllHoliday) // 全部合约都处于节假日状态
252 {
253 WTSLogger::info("All instruments of trading session {}[{}] are in holiday, changed into holiday state", sInfo->name(), stateInfo->_session);
254 stateInfo->_state = SS_Holiday; // 进入节假日状态
255 }
256 else if (offTime >= offCloseTime) // 当前时间已超过收盘时间
257 {
258 stateInfo->_state = SS_CLOSED; // 进入已收盘状态
259 WTSLogger::info("Trading session {}[{}] stopped receiving data", sInfo->name(), stateInfo->_session);
260 }
261 else if (aucStartTime != -1 && offTime >= aucStartTime) // 集合竹已开始
262 {
263 if (stateInfo->isInSections(offTime)) // 如果当前时间在交易时间区间内
264 {
265 //if(sInfo->_schedule)
266 //{
267 // _dt_mgr->preloadRtCaches();
268 //}
269 stateInfo->_state = SS_RECEIVING; // 进入接收数据状态
270 WTSLogger::info("Trading session {}[{}] started receiving data", sInfo->name(), stateInfo->_session);
271 }
272 else // 如果当前时间不在交易时间区间内
273 {
274 // 小于市场收盘时间且不在交易时间区间内，则为中途休盘时间
275 if(offTime < sInfo->getCloseTime(true))
276 {
277 stateInfo->_state = SS_PAUSED; // 进入休息中状态
278 WTSLogger::info("Trading session {}[{}] paused receiving data", sInfo->name(), stateInfo->_session);
279 }
280 else
281 {// 大于市场收盘时间但小于收盘设定时间，继续接收数据(主要是接收结算价)
282 stateInfo->_state = SS_RECEIVING; // 进入接收数据状态
283 WTSLogger::info("Trading session {}[{}] started receiving data", sInfo->name(), stateInfo->_session);
284 }
285
286 }
287 }
288 else if (offTime >= offInitTime) // 当前时间大于等于初始化时间
289 {
290 stateInfo->_state = SS_INITIALIZED; // 进入已初始化状态
291 WTSLogger::info("Trading session {}[{}] initialized", sInfo->name(), stateInfo->_session);
292 }
293
294
295 }
296 break;
297 // 已初始化状态，系统数据准备就绪后的状态
298 case SS_INITIALIZED:
299 {
300 // 获取当前偏移时间和集合竹开始时间
301 uint32_t offTime = sInfo->offsetTime(curMin, true); // 当前时间的偏移值
302 uint32_t offAucSTime = sInfo->getAuctionStartTime(true); // 集合竹开始时间
303
304 // 如果没有集合竹或已经到达集合竹开始时间
305 if (offAucSTime == -1 || offTime >= sInfo->getAuctionStartTime(true))
306 {
307 // 如果当前时间不在交易时间区间内且小于市场收盘时间，则为中途休盘状态
308 if (!stateInfo->isInSections(offTime) && offTime < sInfo->getCloseTime(true))
309 {
310 //if (sInfo->_schedule)
311 //{
312 // _dt_mgr->preloadRtCaches();
313 //}
314 stateInfo->_state = SS_PAUSED; // 进入休息中状态
315
316 WTSLogger::info("Trading session {}[{}] paused receiving data", sInfo->name(), stateInfo->_session);
317 }
318 else
319 {
320 // 如果当前时间在交易时间区间内或已超过市场收盘时间
321 //if (sInfo->_schedule)
322 //{
323 // _dt_mgr->preloadRtCaches();
324 //}
325 stateInfo->_state = SS_RECEIVING; // 进入接收数据状态
326 WTSLogger::info("Trading session {}[{}] started receiving data", sInfo->name(), stateInfo->_session);
327 }
328
329 }
330 }
331 break;
332 // 交易中状态，正在接收数据
333 case SS_RECEIVING:
334 {
335 // 获取当前偏移时间和收盘时间
336 uint32_t offTime = sInfo->offsetTime(curMin, true); // 当前时间的偏移值
337 uint32_t offCloseTime = sInfo->offsetTime(stateInfo->_close_time, false); // 收盘时间的偏移值
338
339 // 当前时间已超过收盘时间，停止接收数据
340 if (offTime >= offCloseTime)
341 {
342 stateInfo->_state = SS_CLOSED; // 进入已收盘状态
343
344 WTSLogger::info("Trading session {}[{}] stopped receiving data", sInfo->name(), stateInfo->_session);
345 }
346 // 当前时间超过集合竹开始时间
347 else if (offTime >= sInfo->getAuctionStartTime(true))
348 {
349 // 当前时间小于市场收盘时间
350 if (offTime < sInfo->getCloseTime(true))
351 {
352 // 如果当前时间不在交易时间区间内，则为中途休盘时间
353 if (!stateInfo->isInSections(offTime))
354 {
355 //if (sInfo->_schedule)
356 //{
357 // _dt_mgr->preloadRtCaches();
358 //}
359 stateInfo->_state = SS_PAUSED; // 进入休息中状态
360
361 WTSLogger::info("Trading session {}[{}] paused receiving data", sInfo->name(), stateInfo->_session);
362 }
363 }
364 else
365 {
366 // 当前时间大于等于市场收盘时间但小于收盘设定时间
367 // 维持当前状态不变，继续接收数据，主要是接收结算价
368 }
369 }
370 }
371 break;
372 // 休息中状态，交易暂停（如中午休市）
373 case SS_PAUSED:
374 {
375 // 休息状态只能转换为交易状态或节假日状态
376 // 这里使用偏移日期进行判断，避免周末交易时间段判断错误
377 uint32_t weekDay = TimeUtils::getWeekDay(); // 获取当前星期
378
379 // 检查是否所有合约都处于节假日状态
380 bool isAllHoliday = true;
381 // 获取当前交易时间段关联的所有合约
382 CodeSet* pCommSet = _bd_mgr->getSessionComms(stateInfo->_session);
383 if (pCommSet)
384 {
385 // 遍历所有合约检查是否处于节假日
386 for (auto it = pCommSet->begin(); it != pCommSet->end(); it++)
387 {
388 const char* pid = (*it).c_str(); // 合约ID
389 // 使用与原始状态相同的节假日判断逻辑
390 if ((sInfo->getOffsetMins() > 0 &&
391 (! _bd_mgr->isTradingDate(pid, curDate) && // 当前日期不是交易日
392 !(sInfo->isInTradingTime(curMin) && _bd_mgr->isTradingDate(pid, prevDate)))) || // 不处于夜盘后半夜情况
393 (sInfo->getOffsetMins() <= 0 && ! _bd_mgr->isTradingDate(pid, offDate)) // 白盘情况下偏移日期不是交易日
394 )
395 {
396 // 当前合约处于节假日
397 WTSLogger::info("Instrument {} is in holiday", pid);
398 }
399 else
400 {
401 // 当前合约不处于节假日
402 isAllHoliday = false; // 至少有1个合约不在节假日
403 }
404 }
405 }
406
407 // 如果不是所有合约都处于节假日状态
408 if (!isAllHoliday)
409 {
410 uint32_t offTime = sInfo->offsetTime(curMin, true); // 获取当前偏移时间
411 // 检查当前时间是否在交易时间区间内
412 if (stateInfo->isInSections(offTime))
413 {
414 stateInfo->_state = SS_RECEIVING; // 重新进入交易状态
415 WTSLogger::info("Trading session {}[{}] continued to receive data", sInfo->name(), stateInfo->_session);
416 }
417 }
418 else
419 {
420 // 所有合约都处于节假日状态，转入节假日状态
421 WTSLogger::info("All instruments of trading session {}[{}] are in holiday, changed into holiday state", sInfo->name(), stateInfo->_session);
422 stateInfo->_state = SS_Holiday; // 进入节假日状态
423 }
424 }
425 break;
426 // 已收盘状态，交易结束
427 case SS_CLOSED:
428 {
429 // 获取当前偏移时间和相关处理时间
430 uint32_t offTime = sInfo->offsetTime(curMin, true); // 当前时间的偏移值
431 uint32_t offProcTime = sInfo->offsetTime(stateInfo->_proc_time, true); // 相关盘后处理时间
432
433 // 如果当前时间已超过盘后处理时间，则进入盘后处理流程
434 if (offTime >= offProcTime)
435 {
436 // 检查当前交易时间段是否已完成收盘作业
437 if(!_dt_mgr->isSessionProceeded(stateInfo->_session))
438 {
439 stateInfo->_state = SS_PROCING; // 进入收盘作业中状态
440
441 WTSLogger::info("Trading session {}[{}] started processing closing task", sInfo->name(), stateInfo->_session);
442 // 调用数据管理器处理历史数据，将实时数据转入历史数据
443 _dt_mgr->transHisData(stateInfo->_session);
444 }
445 else
446 {
447 // 如果已处理过，直接进入盘后已处理状态
448 stateInfo->_state = SS_PROCED; // 进入盘后已处理状态
449 }
450 }
451 // 当前时间在集合竹开始时间和市场收盘时间之间
452 else if (offTime >= sInfo->getAuctionStartTime(true) && offTime <= sInfo->getCloseTime(true))
453 {
454 // 如果当前时间不在交易时间区间内，进入休息中状态
455 if (!stateInfo->isInSections(offTime))
456 {
457 stateInfo->_state = SS_PAUSED; // 进入休息中状态
458
459 WTSLogger::info("Trading session {}[{}] paused receiving data", sInfo->name(), stateInfo->_session);
460 }
461 }
462 }
463 break;
464 // 收盘作业中状态，正在处理收盘数据
465 case SS_PROCING:
466 stateInfo->_state = SS_PROCED; // 直接转为盘后已处理状态，因为实际处理是异步进行的
467 break;
468 // 盘后已处理状态或节假日状态
469 case SS_PROCED:
470 case SS_Holiday:
471 {
472 // 获取当前偏移时间和初始化时间
473 uint32_t offTime = sInfo->offsetTime(curMin, true); // 当前时间的偏移值
474 uint32_t offInitTime = sInfo->offsetTime(stateInfo->_init_time, true); // 初始化时间
475
476 // 如果当前时间在凌晨和初始化时间之间，考虑重置状态
477 if (offTime >= 0 && offTime < offInitTime)
478 {
479 // 检查是否所有合约都处于节假日状态
480 bool isAllHoliday = true;
481 CodeSet* pCommSet = _bd_mgr->getSessionComms(stateInfo->_session);
482 if (pCommSet)
483 {
484 // 遍历所有合约检查是否处于节假日
485 for (auto it = pCommSet->begin(); it != pCommSet->end(); it++)
486 {
487 const char* pid = (*it).c_str(); // 合约ID
488 // 使用与原始状态相同的节假日判断逻辑
489 if ((sInfo->getOffsetMins() > 0 &&
490 (! _bd_mgr->isTradingDate(pid, curDate) && // 当前日期不是交易日
491 !(sInfo->isInTradingTime(curMin) && _bd_mgr->isTradingDate(pid, prevDate)))) || // 不处于夜盘后半夜情况
492 (sInfo->getOffsetMins() <= 0 && ! _bd_mgr->isTradingDate(pid, offDate)) // 白盘情况下偏移日期不是交易日
493 )
494 {
495 // 当前合约仍然处于节假日状态
496 }
497 else
498 {
499 // 当前合约不为节假日，可以重置状态
500 isAllHoliday = false; // 至少有1个合约不在节假日
501 }
502 }
503 }
504
505 // 如果不是所有合约都处于节假日，重置状态准备新一天的交易
506 if(!isAllHoliday)
507 {
508 stateInfo->_state = SS_ORIGINAL; // 重置为原始状态，开始新一天的状态转换循环
509 WTSLogger::info("Trading session {}[{}] state reset", sInfo->name(), stateInfo->_session);
510 }
511 }
512 }
513 break;
514 }
515
516 } // switch结束
517
518 // 更新上次检查时间
519 lastTime = TimeUtils::getLocalTimeNow();
520
521 // 如果所有非节假日的交易时间段都处于收盘作业中状态，进行缓存清理
522 if (isAllInState(SS_PROCING) && !isAllInState(SS_Holiday))
523 {
524 // 清理缓存数据，避免内存泄漏
525 _dt_mgr->transHisData("CMD_CLEAR_CACHE");
526 }
527 } // 主循环结束
528 }));
529 }
