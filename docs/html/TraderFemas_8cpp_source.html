665void TraderFemas::OnRspQryInvestorPosition(CUstpFtdcRspInvestorPositionField *pRspInvestorPosition, CUstpFtdcRspInfoField *pRspInfo, int nRequestID, bool bIsLast)
666{
667 if (bIsLast)
668 {
669 m_bInQuery = false;
670 triggerQuery();
671 }
672
673 if (!IsErrorRspInfo(pRspInfo) && pRspInvestorPosition)
674 {
675 if(NULL == m_ayPosition)
676 m_ayPosition = WTSArray::create();
677
678 WTSContractInfo* contract = m_bdMgr->getContract(pRspInvestorPosition->InstrumentID, pRspInvestorPosition->ExchangeID);
679 if (contract)
680 {
681 WTSCommodityInfo* commInfo = contract->getCommInfo();
682 WTSPositionItem *pos = WTSPositionItem::create(pRspInvestorPosition->InstrumentID, commInfo->getCurrency(), commInfo->getExchg());
683 pos->setContractInfo(contract);
684 pos->setDirection(wrapPosDirection(pRspInvestorPosition->Direction));
685 pos->setNewPosition(pRspInvestorPosition->Position - pRspInvestorPosition->YdPosition);
686 pos->setPrePosition(pRspInvestorPosition->YdPosition);
687
688 pos->setMargin(pRspInvestorPosition->UsedMargin);
689 pos->setDynProfit(0.0);
690 pos->setPositionCost(pRspInvestorPosition->PositionCost);
691
692 if (pos->getTotalPosition() != 0)
693 {
694 pos->setAvgPrice(pos->getPositionCost() / pos->getTotalPosition() / commInfo->getVolScale());
695 }
696 else
697 {
698 pos->setAvgPrice(0);
699 }
700
701 if (commInfo->getCategoty() != CC_Combination)
702 {
703 int availNew = pRspInvestorPosition->Position - pRspInvestorPosition->YdPosition;
704 availNew -= pRspInvestorPosition->FrozenPosition;
