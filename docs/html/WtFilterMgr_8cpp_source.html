25void WtFilterMgr::load_filters(const char* fileName)
26{
27 // 如果没有设置过滤器文件且也没有提供新的文件名，直接返回
28 if (_filter_file.empty() && (strlen(fileName) == 0))
29 return;
30
31 // 如果提供了新的文件名，则更新过滤器文件路径
32 if(strlen(fileName) > 0)
33 _filter_file = fileName;
34
35 // 检查过滤器文件是否存在
36 if (!StdFile::exists(_filter_file.c_str()))
37 {
38 WTSLogger::debug("Filters configuration file {} not exists", _filter_file);
39 return;
40 }
41
42 // 获取文件的最后修改时间
43 uint64_t lastModTime = boost::filesystem::last_write_time(boost::filesystem::path(_filter_file));
44 // 如果文件没有变化，直接返回
45 if (lastModTime <= _filter_timestamp)
46 return;
47
48 // 如果文件变化了，输出日志并通知
49 if (_filter_timestamp != 0)
50 {
51 WTSLogger::info("Filters configuration file {} modified, will be reloaded", _filter_file);
52 if (_notifier)
53 _notifier->notify_event("Filter file has been reloaded");
54 }
55
56 // 加载过滤器配置文件
57 WTSVariant* cfg = WTSCfgLoader::load_from_file(_filter_file.c_str());
58
59 // 更新文件时间戳
60 _filter_timestamp = lastModTime;
61
62 // 清空所有过滤器
63 _stra_filters.clear();
64 _code_filters.clear();
65 _exec_filters.clear();
66
67 // 读取策略过滤器配置
68 WTSVariant* filterStra = cfg->get("strategy_filters");
69 if (filterStra)
70 {
71 // 遍历所有策略过滤器项
72 auto keys = filterStra->memberNames();
73 for (const std::string& key : keys)
74 {
75 // 获取当前策略过滤器的配置
76 WTSVariant* cfgItem = filterStra->get(key.c_str());
77 const char* action = cfgItem->getCString("action");
78
79 // 解析过滤器动作
80 FilterAction fAct = FA_None;
81 if (wt_stricmp(action, "ignore") == 0)
82 fAct = FA_Ignore;
83 else if (wt_stricmp(action, "redirect") == 0)
84 fAct = FA_Redirect;
85
86 // 如果动作无效，跳过当前项
87 if (fAct == FA_None)
88 {
89 WTSLogger::error("Action {} of strategy filter {} not recognized", action, key);
90 continue;
91 }
92
93 // 创建并设置过滤器项
94 FilterItem& fItem = _stra_filters[key];
95 fItem._key = key;
96 fItem._action = fAct;
97 fItem._target = cfgItem->getDouble("target");
98
99 WTSLogger::info("Strategy filter {} loaded", key);
100 }
101 }
102
103 // 读取合约代码过滤器配置
104 WTSVariant* filterCodes = cfg->get("code_filters");
105 if (filterCodes)
106 {
107 // 遍历所有合约代码过滤器项
108 auto codes = filterCodes->memberNames();
109 for (const std::string& stdCode : codes)
110 {
111 // 获取当前合约代码过滤器的配置
112 WTSVariant* cfgItem = filterCodes->get(stdCode.c_str());
113 const char* action = cfgItem->getCString("action");
114
115 // 解析过滤器动作
116 FilterAction fAct = FA_None;
117 if (wt_stricmp(action, "ignore") == 0)
118 fAct = FA_Ignore;
119 else if (wt_stricmp(action, "redirect") == 0)
120 fAct = FA_Redirect;
121
122 // 如果动作无效，跳过当前项
123 if (fAct == FA_None)
124 {
125 WTSLogger::error("Action {} of code filter {} not recognized", action, stdCode);
126 continue;
127 }
128
129 // 创建并设置过滤器项
130 FilterItem& fItem = _code_filters[stdCode];
131 fItem._key = stdCode;
132 fItem._action = fAct;
133 fItem._target = cfgItem->getDouble("target");
134
135 WTSLogger::info("Code filter {} loaded", stdCode);
136 }
137 }
138
139 // 读取执行器过滤器配置
140 WTSVariant* filterExecuters = cfg->get("executer_filters");
141 if (filterExecuters)
142 {
143 // 遍历所有执行器过滤器项
144 auto executer_ids = filterExecuters->memberNames();
145 for (const std::string& execid : executer_ids)
146 {
147 // 获取当前执行器是否禁用
148 bool bDisabled = filterExecuters->getBoolean(execid.c_str());
149 WTSLogger::info("Executer {} is %s", execid, bDisabled?"disabled":"enabled");
150 _exec_filters[execid] = bDisabled;
151 }
152 }
153
154 // 释放配置对象
155 cfg->release();
156}

164bool WtFilterMgr::is_filtered_by_executer(const char* execid)
165{
166 // 在执行器过滤器映射中查找指定的执行器ID
167 auto it = _exec_filters.find(execid);
168 // 如果没有找到，表示没有过滤该执行器
169 if (it == _exec_filters.end())
170 return false;
171
172 // 返回过滤器设置的值，true表示过滤掉，false表示不过滤
173 return it->second;
174}

195bool WtFilterMgr::is_filtered_by_strategy(const char* straName, double& targetPos, bool isDiff /* = false */)
196{
197 // 在策略过滤器映射中查找指定的策略名称
198 auto it = _stra_filters.find(straName);
199 if (it != _stra_filters.end())
200 {
201 // 获取过滤器项
202 const FilterItem& fItem = it->second;
203
204 // 如果是增量仓位，则直接过滤掉
205 if(isDiff)
206 {
207 // 如果过滤器触发，并且是增量头寿，则直接过滤掉
208 WTSLogger::info("[Filters] Strategy filter {} triggered, the change of position ignored directly", straName);
209 return true;
210 }
211
212 // 输出过滤器触发的日志
213 WTSLogger::info("[Filters] Strategy filter {} triggered, action: {}", straName, fItem._action <= FA_Redirect ? FLTACT_NAMEs[fItem._action] : "Unknown");
214
215 // 如果动作是忽略，返回true表示过滤掉
216 if (fItem._action == FA_Ignore)
217 {
218 return true;
219 }
220 // 如果动作是重定向，则修改目标仓位
221 else if (fItem._action == FA_Redirect)
222 {
223 // 只有不是增量的时候，才有效
224 targetPos = fItem._target;
225 }
226
227 return false;
228 }
229
230 // 如果没有找到过滤器，表示不过滤
231 return false;
232}

244{

252 // 获取过滤器项

254
255 // 输出过滤器触发的日志

257

266 targetPos = fItem._target;
267 }
268
269 return false;
270 }
271
272 // 如果没有找到完整合约代码，则在品种代码过滤器映射中查找
273 cit = _code_filters.find(cInfo.stdCommID());

281

283 if (fItem._action == FA_Ignore)
284 {
285 return true;
286 }
287 // 如果动作是重定向，则修改目标仓位
288 else if (fItem._action == FA_Redirect)
289 {
290 targetPos = fItem._target;
291 }
