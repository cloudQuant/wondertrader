215void ParserFemas::OnRtnDepthMarketData( CUstpFtdcDepthMarketDataField *pDepthMarketData )
216{
217 if(m_pBaseDataMgr == NULL)
218 {
219 return;
220 }
221
222 uint32_t actDate = strtoul(pDepthMarketData->ActionDay, NULL, 10);
223 uint32_t actTime = strToTime(pDepthMarketData->UpdateTime) * 1000 + pDepthMarketData->UpdateMillisec;
224 uint32_t actHour = actTime / 10000000;
225
226 if (actDate == m_uTradingDate && actHour >= 20)
227 {
228 //这样的时间是有问题,因为夜盘时发生日期不可能等于交易日
229 //这就需要手动设置一下
230 uint32_t curDate, curTime;
231 TimeUtils::getDateTime(curDate, curTime);
232 uint32_t curHour = curTime / 10000000;
233
234 //早上启动以后,会收到昨晚12点以前收盘的行情,这个时候可能会有发生日期=交易日的情况出现
235 //这笔数据直接丢掉
236 if (curHour >= 3 && curHour < 9)
237 return;
238
239 actDate = curDate;
240
241 if (actHour == 23 && curHour == 0)
242 {
243 //行情时间慢于系统时间
244 actDate = TimeUtils::getNextDate(curDate, -1);
245 }
246 else if (actHour == 0 && curHour == 23)
247 {
248 //系统时间慢于行情时间
249 actDate = TimeUtils::getNextDate(curDate, 1);
250 }
251 }
252
253 WTSContractInfo* contract = m_pBaseDataMgr->getContract(pDepthMarketData->InstrumentID);
254 if (contract == NULL)
255 return;
256
257 WTSCommodityInfo* pCommInfo = contract->getCommInfo();
258
259 //if (strcmp(contract->getExchg(), "CZCE") == 0)
260 //{
261 // actTime += (uint32_t)(TimeUtils::getLocalTimeNow() % 1000);
262 //}
263
264 WTSTickData* tick = WTSTickData::create(pDepthMarketData->InstrumentID);
265 WTSTickStruct& quote = tick->getTickStruct();
266 strcpy(quote.exchg, pCommInfo->getExchg());
267 tick->setContractInfo(contract);
268
269 quote.action_date = actDate;
270 quote.action_time = actTime;
271
272 quote.price = checkValid(pDepthMarketData->LastPrice);
273 quote.open = checkValid(pDepthMarketData->OpenPrice);
274 quote.high = checkValid(pDepthMarketData->HighestPrice);
275 quote.low = checkValid(pDepthMarketData->LowestPrice);
276 quote.total_volume = pDepthMarketData->Volume;
277 quote.trading_date = m_uTradingDate;
278 if(pDepthMarketData->SettlementPrice != DBL_MAX)
279 quote.settle_price = checkValid(pDepthMarketData->SettlementPrice);
280 if(strcmp(quote.exchg, "CZCE") == 0)
281 {
282 quote.total_turnover = pDepthMarketData->Turnover*pCommInfo->getVolScale();
283 }
284 else
285 {
286 if(pDepthMarketData->Turnover != DBL_MAX)
287 quote.total_turnover = pDepthMarketData->Turnover;
288 }
289
290 quote.open_interest = (uint32_t)pDepthMarketData->OpenInterest;
291
292 quote.upper_limit = checkValid(pDepthMarketData->UpperLimitPrice);
293 quote.lower_limit = checkValid(pDepthMarketData->LowerLimitPrice);
294
295 quote.pre_close = checkValid(pDepthMarketData->PreClosePrice);
296 quote.pre_settle = checkValid(pDepthMarketData->PreSettlementPrice);
297 quote.pre_interest = (uint32_t)pDepthMarketData->PreOpenInterest;
298
299 //委卖价格
300 quote.ask_prices[0] = checkValid(pDepthMarketData->AskPrice1);
301 quote.ask_prices[1] = checkValid(pDepthMarketData->AskPrice2);
302 quote.ask_prices[2] = checkValid(pDepthMarketData->AskPrice3);
303 quote.ask_prices[3] = checkValid(pDepthMarketData->AskPrice4);
304 quote.ask_prices[4] = checkValid(pDepthMarketData->AskPrice5);
305
306 //委买价格
307 quote.bid_prices[0] = checkValid(pDepthMarketData->BidPrice1);
308 quote.bid_prices[1] = checkValid(pDepthMarketData->BidPrice2);
309 quote.bid_prices[2] = checkValid(pDepthMarketData->BidPrice3);
310 quote.bid_prices[3] = checkValid(pDepthMarketData->BidPrice4);
311 quote.bid_prices[4] = checkValid(pDepthMarketData->BidPrice5);
312
313 //委卖量
314 quote.ask_qty[0] = pDepthMarketData->AskVolume1;
315 quote.ask_qty[1] = pDepthMarketData->AskVolume2;
316 quote.ask_qty[2] = pDepthMarketData->AskVolume3;
317 quote.ask_qty[3] = pDepthMarketData->AskVolume4;
318 quote.ask_qty[4] = pDepthMarketData->AskVolume5;
319
320 //委买量
321 quote.bid_qty[0] = pDepthMarketData->BidVolume1;
322 quote.bid_qty[1] = pDepthMarketData->BidVolume2;
323 quote.bid_qty[2] = pDepthMarketData->BidVolume3;
324 quote.bid_qty[3] = pDepthMarketData->BidVolume4;
325 quote.bid_qty[4] = pDepthMarketData->BidVolume5;
326
327 if(m_sink)
328 m_sink->handleQuote(tick, 1);
329
330 tick->release();
331}

371void ParserFemas::SubscribeMarketData()
372{
373 CodeSet codeFilter = m_filterSubs;
374 if(codeFilter.empty())
375 {//如果订阅礼包只空的,则取出全部合约列表
376 return;
377 }
378
379 char ** subscribe = new char*[codeFilter.size()];
380 int nCount = 0;
381 for(auto& code : codeFilter)
382 {
383 std::size_t pos = code.find('.');
384 if (pos != std::string::npos)
385 subscribe[nCount++] = (char*)code.c_str() + pos + 1;
386 else
387 subscribe[nCount++] = (char*)code.c_str();
388 }
389
390 if(m_pUserAPI && nCount > 0)
391 {
392 int iResult = m_pUserAPI->SubMarketData(subscribe, nCount);
393 if(iResult != 0)
394 {
395 if(m_sink)
396 write_log(m_sink, LL_ERROR, "[ParserFemas] Sending md subscribe request failed: {}", iResult);
397 }
398 else
399 {
400 if(m_sink)
401 write_log(m_sink, LL_INFO, "[ParserFemas] Market data of {} contracts subscribed in total", nCount);
402 }
403 }
404 codeFilter.clear();
405 delete[] subscribe;
406}
