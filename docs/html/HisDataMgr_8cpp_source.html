36bool HisDataMgr::init(WTSVariant* cfg)
37{
38 // 获取模块名称，如果没有配置则使用默认的WtDataStorage
39 std::string module = cfg->getCString("module");
40 if (module.empty())
41 module = WtHelper::getInstDir() + DLLHelper::wrap_module("WtDataStorage");
42 else
43 module = WtHelper::getInstDir() + DLLHelper::wrap_module(module.c_str());
44
45 // 加载数据存储模块库
46 DllHandle libParser = DLLHelper::load_library(module.c_str());
47 if (libParser)
48 {
49 // 获取创建读取器的函数指针
50 FuncCreateBtDtReader pFuncCreator = (FuncCreateBtDtReader)DLLHelper::get_symbol(libParser, "createBtDtReader");
51 if (pFuncCreator == NULL)
52 {
53 WTSLogger::error("Initializing of backtest data reader failed: function createBtDtReader not found...");
54 }
55
56 // 如果函数指针获取成功，创建读取器实例
57 if (pFuncCreator)
58 {
59 _reader = pFuncCreator();
60 }
61
62 WTSLogger::debug("Back data storage module {} loaded", module);
63 }
64 else
65 {
66 WTSLogger::error("Loading module back data storage module {} failed", module);
67
68 }
69
70 // 初始化读取器实例，传入配置和当前对象作为回调接收器
71 _reader->init(cfg, this);
72
73 return true;
74}

85bool HisDataMgr::load_raw_bars(const char* exchg, const char* code, WTSKlinePeriod period, FuncLoadDataCallback cb)
86{
87 // 检查读取器是否初始化

92 }
93
94 // 准备接收数据的缓冲区
95 std::string buffer;
96 // 调用读取器加载K线数据
97 bool bSucc = _reader->read_raw_bars(exchg, code, period, buffer);
98 // 如果加载成功，执行回调函数
99 if (bSucc)
100 cb(buffer);
101 return bSucc;
