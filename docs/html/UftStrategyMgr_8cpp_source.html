44bool UftStrategyMgr::loadFactories(const char* path)
45{
46 // 检查路径是否存在
47 if (!StdFile::exists(path))
48 {
49 WTSLogger::error("Directory {} of UFT strategy factory not exists", path);
50 return false;
51 }
52
53 uint32_t count = 0;
54 boost::filesystem::path myPath(path);
55 boost::filesystem::directory_iterator endIter;
56 // 遍历目录中的所有文件
57 for (boost::filesystem::directory_iterator iter(myPath); iter != endIter; iter++)
58 {
59 // 如果是子目录则跳过
60 if (boost::filesystem::is_directory(iter->path()))
61 continue;
62
63 // 检查文件后缀名，Windows下应为.dll，Linux下应为.so
64#ifdef _WIN32
65 if (iter->path().extension() != ".dll")
66 continue;
67#else //_UNIX
68 if (iter->path().extension() != ".so")
69 continue;
70#endif
71
72 // 加载动态库
73 DllHandle hInst = DLLHelper::load_library(iter->path().string().c_str());
74 if (hInst == NULL)
75 continue;
76
77 // 获取创建工厂对象的函数
78 FuncCreateUftStraFact creator = (FuncCreateUftStraFact)DLLHelper::get_symbol(hInst, "createStrategyFact");
79 if (creator == NULL)
80 {
81 DLLHelper::free_library(hInst);
82 continue;
83 }
84
85 // 创建工厂对象
86 IUftStrategyFact* pFact = creator();
87 if (pFact != NULL)
88 {
89 // 创建成功，保存工厂信息
90 StraFactInfo& fInfo = _factories[pFact->getName()];
91
92 fInfo._module_inst = hInst;
93 fInfo._module_path = iter->path().string();
94 fInfo._creator = creator;
95 fInfo._remover = (FuncDeleteUftStraFact)DLLHelper::get_symbol(hInst, "deleteStrategyFact");
96 fInfo._fact = pFact;
97 WTSLogger::info("UFT strategy factory[{}] loaded", pFact->getName());
98
99 count++;
100 }
101 else
102 {
103 // 创建失败，释放动态库
104 DLLHelper::free_library(hInst);
105 continue;
106 }
107 }
108
109 WTSLogger::info("{} UFT strategy factories in directory[{}] loaded", count, path);
110
111 return true;
112}

122UftStrategyPtr UftStrategyMgr::createStrategy(const char* factname, const char* unitname, const char* id)
123{
124 // 查找工厂
125 auto it = _factories.find(factname);
126 if (it == _factories.end())
127 return UftStrategyPtr();
128
129 // 使用工厂创建策略对象
130 StraFactInfo& fInfo = (StraFactInfo&)it->second;
131 UftStrategyPtr ret(new UftStraWrapper(fInfo._fact->createStrategy(unitname, id), fInfo._fact));
132 // 保存策略对象
133 _strategies[id] = ret;
134 return ret;
135}
