406 _db.update_errno(_errno);
407 if (_errno != MDB_SUCCESS)
408 return 0;

437 op = MDB_NEXT;

457 int get_lowers(const std::string& lower_key, const std::string& upper_key, int count, LMDBQueryCallback cb)
458 {
459 MDB_cursor* cursor;
460 int _errno = mdb_cursor_open(_txn, _dbi, &cursor);
461 _db.update_errno(_errno);
462 if (_errno != MDB_SUCCESS)
463 return 0;
464
465 MDB_val rKey, mData;
466 rKey.mv_data = (void*)upper_key.data();
467 rKey.mv_size = upper_key.size();
468
469 int cnt = 0;
470 std::vector<std::string> ayKeys, ayVals;
471 _errno = mdb_cursor_get(cursor, &rKey, &mData, MDB_SET_RANGE);
472 _db.update_errno(_errno);
473
474 if (_errno == MDB_NOTFOUND)
475 {
476 _errno = mdb_cursor_get(cursor, &rKey, &mData, MDB_LAST);
477 _db.update_errno(_errno);
478 }
479
480 for (; _errno != MDB_NOTFOUND;)
481 {
482 //往前查找，所以如果拿到的key，比右边界大，则直接往前退回一条
483 if (memcmp(rKey.mv_data, upper_key.data(), upper_key.size()) > 0)
484 {
485 _errno = mdb_cursor_get(cursor, &rKey, &mData, MDB_PREV);
486 _db.update_errno(_errno);
487 continue;
488 }
489
490 if (memcmp(rKey.mv_data, lower_key.data(), lower_key.size()) < 0)
491 break;
492
493 //回调
494 ayKeys.emplace_back(std::string((char*)rKey.mv_data, rKey.mv_size));
495 ayVals.emplace_back(std::string((char*)mData.mv_data, mData.mv_size));
496 cnt++;
497
498 //如果找到目标数量，则退出
499 if(cnt == count)
500 break;
501
502 _errno = mdb_cursor_get(cursor, &rKey, &mData, MDB_PREV);
503 _db.update_errno(_errno);
504 }
505
506 //向前查找，是逆序的，需要做一个reverse
507 std::reverse(ayKeys.begin(), ayKeys.end());
508 std::reverse(ayVals.begin(), ayVals.end());
509
510 cb(ayKeys, ayVals);
511 mdb_cursor_close(cursor);
512 return cnt;
513 }
