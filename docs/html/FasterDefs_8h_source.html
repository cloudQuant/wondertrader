55struct string_hash
56{
63 std::size_t operator()(const std::string& key) const
64 {
65 size_t seed = 131; // 31 131 1313 13131 131313 etc.. 可选的哈希乘数
66 size_t hash = 0;
67
68 char* str = (char*)key.c_str();
69 while (*str) // 逐字符计算哈希值
70 {
71 hash = hash * seed + (*str++); // 累加当前字符的哈希值
72 }
73
74 return (hash & 0x7FFFFFFF); // 确保哈希值为正数
75 }
76};
