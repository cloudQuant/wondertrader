101bool WtExecRunner::config(const char* cfgFile, bool isFile /* = true */)
102{
103 // 根据参数决定从文件加载还是从内容加载配置
104 _config = isFile ? WTSCfgLoader::load_from_file(cfgFile) : WTSCfgLoader::load_from_content(cfgFile, false);
105 if(_config == NULL)
106 {
107 WTSLogger::log_raw(LL_ERROR, "Loading config file failed");
108 return false;
109 }
110
111 //基础数据文件
112 WTSVariant* cfgBF = _config->get("basefiles");
113 if (cfgBF->get("session"))
114 {
115 _bd_mgr.loadSessions(cfgBF->getCString("session"));
116 WTSLogger::info("Trading sessions loaded");
117 }
118
119 WTSVariant* cfgItem = cfgBF->get("commodity");
120 if (cfgItem)
121 {
122 if (cfgItem->type() == WTSVariant::VT_String)
123 {
124 _bd_mgr.loadCommodities(cfgItem->asCString());
125 }
126 else if (cfgItem->type() == WTSVariant::VT_Array)
127 {
128 for (uint32_t i = 0; i < cfgItem->size(); i++)
129 {
130 _bd_mgr.loadCommodities(cfgItem->get(i)->asCString());
131 }
132 }
133 }
134
135 cfgItem = cfgBF->get("contract");
136 if (cfgItem)
137 {
138 if (cfgItem->type() == WTSVariant::VT_String)
139 {
140 _bd_mgr.loadContracts(cfgItem->asCString());
141 }
142 else if (cfgItem->type() == WTSVariant::VT_Array)
143 {
144 for (uint32_t i = 0; i < cfgItem->size(); i++)
145 {
146 _bd_mgr.loadContracts(cfgItem->get(i)->asCString());
147 }
148 }
149 }
150
151 if (cfgBF->get("holiday"))
152 {
153 _bd_mgr.loadHolidays(cfgBF->getCString("holiday"));
154 WTSLogger::info("Holidays loaded");
155 }
156
157
158 //初始化数据管理
159 initDataMgr();
160
161 //初始化开平策略
162 if (!initActionPolicy())
163 return false;
164
165 //初始化行情通道
166 const char* cfgParser = _config->getCString("parsers");
167 if (StdFile::exists(cfgParser))
168 {
169 WTSLogger::info("Reading parser config from {}...", cfgParser);
170 WTSVariant* var = WTSCfgLoader::load_from_file(cfgParser);
171 if (var)
172 {
173 if (!initParsers(var))
174 WTSLogger::error("Loading parsers failed");
175 var->release();
176 }
177 else
178 {
179 WTSLogger::error("Loading parser config {} failed", cfgParser);
180 }
181 }
182
183 //初始化交易通道
184 const char* cfgTraders = _config->getCString("traders");
185 if (StdFile::exists(cfgTraders))
186 {
187 WTSLogger::info("Reading trader config from {}...", cfgTraders);
188 WTSVariant* var = WTSCfgLoader::load_from_file(cfgTraders);
189 if (var)
190 {
191 if (!initTraders(var))
192 WTSLogger::error("Loading traders failed");
193 var->release();
194 }
195 else
196 {
197 WTSLogger::error("Loading trader config {} failed", cfgTraders);
198 }
199 }
200
201 const char* cfgExecuters = _config->getCString("executers");
202 if (StdFile::exists(cfgExecuters))
203 {
204 WTSLogger::info("Reading executer config from {}...", cfgExecuters);
205 WTSVariant* var = WTSCfgLoader::load_from_file(cfgExecuters);
206 if (var)
207 {
208 if (!initExecuters(var))
209 WTSLogger::error("Loading executers failed");
210 var->release();
211 }
212 else
213 {
214 WTSLogger::error("Loading executer config {} failed", cfgExecuters);
215 }
216 }
217
218 return true;
219}

249bool WtExecRunner::initParsers(WTSVariant* cfgParser)
250{
251 // 获取解析器配置数组
252 WTSVariant* cfg = cfgParser->get("parsers");
253 if (cfg == NULL)
254 return false;
255
256 uint32_t count = 0;
257 for (uint32_t idx = 0; idx < cfg->size(); idx++)
258 {
259 WTSVariant* cfgItem = cfg->get(idx);
260 // 跳过未激活的解析器
261 if (!cfgItem->getBoolean("active"))
262 continue;
263
264 const char* id = cfgItem->getCString("id");
265
266 // By Wesley @ 2021.12.14
267 // 如果id为空，则生成自动id
268 std::string realid = id;
269 if (realid.empty())
270 {
271 static uint32_t auto_parserid = 1000;
272 realid = StrUtil::printf("auto_parser_%u", auto_parserid++);
273 }
274
275 ParserAdapterPtr adapter(new ParserAdapter);
276 adapter->init(realid.c_str(), cfgItem, this, &_bd_mgr, &_hot_mgr);
277 _parsers.addAdapter(realid.c_str(), adapter);
278
279 count++;
280 }
281
282 WTSLogger::info("{} parsers loaded", count);
283
284 return true;
285}

293bool WtExecRunner::initExecuters(WTSVariant* cfgExecuter)
294{
295 // 获取执行器配置数组
296 WTSVariant* cfg = cfgExecuter->get("executers");
297 if (cfg == NULL || cfg->type() != WTSVariant::VT_Array)
298 return false;
299
300 // 先加载自带的执行器工厂
301 std::string path = WtHelper::getInstDir() + "executer//";
302 _exe_factory.loadFactories(path.c_str());
303
304 uint32_t count = 0;
305 for (uint32_t idx = 0; idx < cfg->size(); idx++)
306 {
307 WTSVariant* cfgItem = cfg->get(idx);
308 if (!cfgItem->getBoolean("active"))
309 continue;
310
311 const char* id = cfgItem->getCString("id");
312 std::string name = cfgItem->getCString("name"); //local,diff,dist
313 if (name.empty())
314 name = "local";
315
316 if (name == "local")
317 {
318 WtLocalExecuter* executer = new WtLocalExecuter(&_exe_factory, id, &_data_mgr);
319 if (!executer->init(cfgItem))
320 return false;
321
322 const char* tid = cfgItem->getCString("trader");
323 if (strlen(tid) == 0)
324 {
325 WTSLogger::error("No Trader configured for Executer {}", id);
326 }
327 else
328 {
329 TraderAdapterPtr trader = _traders.getAdapter(tid);
330 if (trader)
331 {
332 executer->setTrader(trader.get());
333 trader->addSink(executer);
334 }
335 else
336 {
337 WTSLogger::error("Trader {} not exists, cannot configured for executer %s", tid, id);
338 }
339 }
340
341 executer->setStub(this);
342 _exe_mgr.add_executer(ExecCmdPtr(executer));
343 }
344 else if (name == "diff")
345 {
346 WtDiffExecuter* executer = new WtDiffExecuter(&_exe_factory, id, &_data_mgr, &_bd_mgr);
347 if (!executer->init(cfgItem))
348 return false;
349
350 const char* tid = cfgItem->getCString("trader");
351 if (strlen(tid) == 0)
352 {
353 WTSLogger::error("No Trader configured for Executer {}", id);
354 }
355 else
356 {
357 TraderAdapterPtr trader = _traders.getAdapter(tid);
358 if (trader)
359 {
360 executer->setTrader(trader.get());
361 trader->addSink(executer);
362 }
363 else
364 {
365 WTSLogger::error("Trader {} not exists, cannot configured for executer %s", tid, id);
366 }
367 }
368
369 executer->setStub(this);
370 _exe_mgr.add_executer(ExecCmdPtr(executer));
371 }
372 else
373 {
374 WtDistExecuter* executer = new WtDistExecuter(id);
375 if (!executer->init(cfgItem))
376 return false;
377
378 executer->setStub(this);
379 _exe_mgr.add_executer(ExecCmdPtr(executer));
380 }
381 count++;
382 }
383
384 WTSLogger::info("{} executers loaded", count);
385
386 return true;
387}

485void WtExecRunner::handle_push_quote(WTSTickData* quote)
486{
487 // 检查行情数据是否有效
488 if (quote == NULL)
489 return;
490
491 // 提取行情数据中的时间信息
492 uint32_t uDate = quote->actiondate();
493 uint32_t uTime = quote->actiontime();
494 uint32_t curMin = uTime / 100000;
495 uint32_t curSec = uTime % 100000;
496 // 设置当前时间和交易日
497 WtHelper::setTime(uDate, curMin, curSec);
498 WtHelper::setTDate(quote->tradingdate());
499
500 // 将行情数据传递给数据管理器
501 _data_mgr.handle_push_quote(quote->code(), quote);
502
503 // 将行情数据传递给执行管理器
504 _exe_mgr.handle_tick(quote->code(), quote);
