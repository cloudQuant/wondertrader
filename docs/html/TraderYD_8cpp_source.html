286 //再查持仓
287 {
288 if (NULL == m_mapPosition)
289 m_mapPosition = DataMap::create();

291 int cnt = m_pUserAPI->getPrePositionCount();
292 for (int i = 0; i < cnt; i++)

294 const YDPrePosition* pInfo = m_pUserAPI->getPrePosition(i);
295 const YDInstrument* instInfo = pInfo->m_pInstrument;
296
297 WTSContractInfo* contract = m_bdMgr->getContract(instInfo->InstrumentID, instInfo->m_pExchange->ExchangeID);
298 if (contract)
299 {
300 WTSCommodityInfo* commInfo = contract->getCommInfo();
301 std::string key = fmt::format("{}-{}", contract->getCode(), wrapPosDirection(pInfo->PositionDirection));

303 if (pos == NULL)

305 pos = WTSPositionItem::create(contract->getCode(), commInfo->getCurrency(), commInfo->getExchg());
306 pos->setDirection(wrapPosDirection(pInfo->PositionDirection));
307 pos->setContractInfo(contract);
308 m_mapPosition->add(key, pos, false);
309 }
310
311 pos->setPrePosition(pInfo->PrePosition);
312 pos->setNewPosition(0);
313
314 pos->setMargin(0);
315 pos->setDynProfit(0);

323 {
324 pos->setAvgPrice(0);

327 pos->setAvailPrePos(pos->getPrePosition());
328 pos->setAvailNewPos(0);
329
330 write_log(m_sink, LL_INFO, "{} PrePosition of {} updated:{}[{}]", pos->getDirection() == WDT_LONG ? "Long" : "Short", contract->getFullCode(), pos->getTotalPosition(), pos->getAvailPosition());

337void TraderYD::notifyOrder(const YDOrder *pOrder, const YDInstrument *pInstrument, const YDAccount *pAccount)
338{
339 WTSOrderInfo *orderInfo = makeOrderInfo(pOrder, pInstrument);
340 if (orderInfo)
341 {
342 //先往缓存里丢
343 if (NULL == m_mapOrders)
344 m_mapOrders = DataMap::create();
345
346 const char* oid = orderInfo->getOrderID();
347 auto it = m_mapOrders->find(oid);
348 if(it == m_mapOrders->end())
349 {
350 //如果该订单是第一次被推送
351 //则检查是否是平仓委托
352 //如果是平仓委托，需要调整冻结手数
353 if (orderInfo->getOffsetType() != WOT_OPEN)
354 {
355 const char* key = fmtutil::format("{}-{}", orderInfo->getCode(), orderInfo->getDirection());
356 WTSPositionItem* pos = (WTSPositionItem*)m_mapPosition->get(key);
357 double preQty = pos->getPrePosition();
358 double newQty = pos->getNewPosition();
359 double availPre = pos->getAvailPrePos();
360 double availNew = pos->getAvailNewPos();
361
362 WTSCommodityInfo* commInfo = orderInfo->getContractInfo()->getCommInfo();
363 if(commInfo->getCoverMode() == CM_CoverToday)
364 {
365 if (orderInfo->getOffsetType() == WOT_CLOSETODAY)
366 availNew -= orderInfo->getVolume();
367 else
368 availPre -= orderInfo->getVolume();
369 }
370 else
371 {
372 //不区分平昨平今，则先冻结昨仓，再冻结今仓
373 double maxQty = min(availPre, orderInfo->getVolume());
374 availPre -= maxQty;
375 if(decimal::lt(orderInfo->getVolume(), maxQty))
376 {
377 availNew -= orderInfo->getVolume() - maxQty;
378 }
379 }
380
381 pos->setAvailPrePos(availPre);
382 pos->setAvailNewPos(availNew);
383 }
384 }
385 else
386 {
387 WTSOrderInfo* preOrd = (WTSOrderInfo*)it->second;
388 //如果订单不是第一次被推送，则看是否是撤单
389 //如果是撤单，并且之间订单状态还是有效的，则对平仓委托要释放冻结的手数
390 if(preOrd->isAlive() && orderInfo->getOrderState() == WOS_Canceled && orderInfo->getOffsetType() != WOT_OPEN)
391 {
392 std::string key = fmt::format("{}-{}", orderInfo->getCode(), orderInfo->getDirection());
393 WTSPositionItem* pos = (WTSPositionItem*)m_mapPosition->get(key);
394 double preQty = pos->getPrePosition();
395 double newQty = pos->getNewPosition();
396 double availPre = pos->getAvailPrePos();
397 double availNew = pos->getAvailNewPos();
398
399 double untrade = orderInfo->getVolume() - orderInfo->getVolTraded();
400
401 WTSCommodityInfo* commInfo = orderInfo->getContractInfo()->getCommInfo();
402 if (commInfo->getCoverMode() == CM_CoverToday)
403 {
404 if (orderInfo->getOffsetType() == WOT_CLOSETODAY)
405 availNew += untrade;
406 else
407 availPre += untrade;
408 }
409 else
410 {
411 //不区分平昨平今，则先释放今仓，再释放昨仓
412 double maxQty = min(newQty-availNew , untrade);
413 availNew += maxQty;
414 if (decimal::lt(untrade, maxQty))
415 {
416 availPre += orderInfo->getVolume() - maxQty;
417 }
418 }
419
420 pos->setAvailPrePos(availPre);
421 pos->setAvailNewPos(availNew);
422 }
423 }
424 m_mapOrders->add(oid, orderInfo, false);
425
426 //如果已经追上了，则直接主推出去
427 if (m_sink && m_bCatchup)
428 {
429 m_sink->onPushOrder(orderInfo);
430 }
431 }

435void TraderYD::notifyTrade(const YDTrade *pTrade, const YDInstrument *pInstrument, const YDAccount *pAccount)
436{
437 WTSTradeInfo *trdInfo = makeTradeRecord(pTrade, pInstrument);
438 if (trdInfo)
439 {
440 //先往缓存里丢
441 if (NULL == m_mapTrades)
442 m_mapTrades = DataMap::create();
443
444 const char* tid = trdInfo->getTradeID();
445 WTSContractInfo* contract = trdInfo->getContractInfo();
446 WTSCommodityInfo* commInfo = contract->getCommInfo();
447 auto it = m_mapTrades->find(tid);
448 if(it == m_mapTrades->end())
449 {
450 m_mapTrades->add(tid, trdInfo, false);
451
452 //成交回报，主要更新持仓
453 std::string key = fmt::format("{}-{}", trdInfo->getCode(), trdInfo->getDirection());
454 WTSPositionItem* pos = (WTSPositionItem*)m_mapPosition->get(key);
455 if(pos == NULL)
456 {
457 pos = WTSPositionItem::create(contract->getCode(), commInfo->getCurrency(), commInfo->getExchg());
458 pos->setContractInfo(contract);
459 pos->setDirection(trdInfo->getDirection());
460 m_mapPosition->add(key, pos, false);
461 }
462
463 double preQty = pos->getPrePosition();
464 double newQty = pos->getNewPosition();
465 double availPre = pos->getAvailPrePos();
466 double availNew = pos->getAvailNewPos();
467
468 double qty = trdInfo->getVolume();
469
470 if(trdInfo->getOffsetType() == WOT_OPEN)
471 {
472 newQty += qty;
473 availNew += qty;
474
475 //开仓一定是今仓
476 pos->setNewPosition(newQty);
477 pos->setAvailNewPos(availNew);
478 }
479 else
480 {
481 //平仓要区分
482 if (commInfo->getCoverMode() == CM_CoverToday)
483 {
484 //平仓不用更新可用持仓
485 //因为可用持仓在订单回报的地方已经更新过了
486 if (trdInfo->getOffsetType() == WOT_CLOSETODAY)
487 newQty -= qty;
488 else
489 preQty -= qty;
490 }
491 else
492 {
493 //不区分平昨平今，则先减掉昨仓，在调整今仓
494 double maxQty = min(preQty, qty);
495 preQty -= maxQty;
496 if (decimal::lt(qty, maxQty))
497 {
498 newQty -= (qty - maxQty);
499 }
500 }
501
502 pos->setNewPosition(newQty);
503 pos->setPrePosition(preQty);
504 }
505 }
506
507 if (m_sink && m_bCatchup)
508 m_sink->onPushTrade(trdInfo);
509 }

690int TraderYD::orderInsert(WTSEntrust* entrust)
691{
692 if (m_pUserAPI == NULL || m_wrapperState != WS_ALLREADY)
693 {
694 return -1;
695 }
696
697 const YDInstrument* pInst = m_pUserAPI->getInstrumentByID(entrust->getCode());
698
699 YDInputOrder req;
700 // inputOrder中的所有不用的字段，应当统一清0
701 memset(&req, 0, sizeof(req));
702
703 if (strlen(entrust->getUserTag()) == 0)
704 {
706 req.OrderRef = m_orderRef.fetch_add(0);
707 }
708 else
709 {
710 uint32_t orderref;
711 extractEntrustID(entrust->getEntrustID(), orderref);
713 req.OrderRef = orderref;
714 }
715
716 if (strlen(entrust->getUserTag()) > 0)
717 {
718 m_eidCache.put(entrust->getEntrustID(), entrust->getUserTag(), 0, [this](const char* message) {
719 write_log(m_sink, LL_WARN, message);
720 });
721 }
722
723 req.Price = entrust->getPrice();
724 req.Direction = wrapDirectionType(entrust->getDirection(), entrust->getOffsetType());
725 req.OffsetFlag = wrapOffsetType(entrust->getOffsetType());
726 req.HedgeFlag = YD_HF_Speculation;
727 req.OrderVolume = (int)entrust->getVolume();
728 // 使用下一个下一个报单引用。YD服务器不检查OrderRef，只是将其用于在报单和成交回报中返回
729 // 用户可以自行选择OrderRef的编码方式
730 // 对于非本系统本次运行产生的报单，系统返回的OrderRef一律是-1
731 // YDClient产生的报单，OrderRef一律是0
732 // 这个例子使用限价单
733 req.OrderType = wrapPriceType(entrust->getPriceType(), entrust->getOrderFlag());
734 // 说明是普通报单
735 req.YDOrderFlag = YD_YOF_Normal;
736 // 说明如何选择连接
737 req.ConnectionSelectionType = YD_CS_Any;
738 // 如果ConnectionSelectionType不是YD_CS_Any，需要指定ConnectionID，范围是0到对应的YDExchange中的ConnectionCount-1
