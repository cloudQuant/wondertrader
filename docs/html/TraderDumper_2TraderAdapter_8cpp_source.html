53bool TraderAdapter::init(const char* id, WTSVariant* params, IBaseDataMgr* bdMgr)
54{
55 if (params == NULL)
56 return false;
57
58 _bd_mgr = bdMgr;
59 _id = id;
60
61 if (_cfg != NULL)
62 return false;
63
64 _cfg = params;
65 _cfg->retain();
66
67
68 if (params->getString("module").empty())
69 return false;
70
71
72 //先看工作目录下是否有交易模块
73 std::string module = DLLHelper::wrap_module(params->getCString("module"), "lib");;
74
75 if (!StdFile::exists(module.c_str()))
76 {
77 module = WtHelper::get_module_dir();
78 module += "traders/";
79 module += DLLHelper::wrap_module(params->getCString("module"), "lib");
80 }
81
82 DllHandle hInst = DLLHelper::load_library(module.c_str());
83 if (hInst == NULL)
84 {
85 WTSLogger::error("[{}]交易模块{}加载失败", _id.c_str(), module.c_str());
86 return false;
87 }
88
89 FuncCreateTrader pFunCreateTrader = (FuncCreateTrader)DLLHelper::get_symbol(hInst, "createTrader");
90 if (NULL == pFunCreateTrader)
91 {
92 WTSLogger::error("[{}]交易接口创建函数读取失败", _id.c_str());
93 return false;
94 }
95
96 _trader_api = pFunCreateTrader();
97 if (NULL == _trader_api)
98 {
99 WTSLogger::error("[{}]交易接口创建失败", _id.c_str());
100 return false;
101 }
102
103 _remover = (FuncDeleteTrader)DLLHelper::get_symbol(hInst, "deleteTrader");
104
105 //这里要强制把quick改成true，不查全部成交和订单
106 params->append("quick", true);
107 if (!_trader_api->init(params))
108 {
109 WTSLogger::error("[{}]交易接口启动失败: 交易接口初始化失败", id);
110 return false;
111 }
112
113 WTSLogger::info("[{}]交易接口初始化成功", id);
114 return true;
115}

138void TraderAdapter::handleEvent(WTSTraderEvent e, int32_t ec)
139{
140 if(e == WTE_Connect)
141 {
142 if(ec == 0)
143 {
144 _trader_api->login(_cfg->getCString("user"), _cfg->getCString("pass"), _cfg->getCString("product"));
145 }
146 else
147 {
148 WTSLogger::error("[{}]交易账号连接失败: {}", _id.c_str(), ec);
149 _mgr->decAlive();

151 }
152 }
153 else if(e == WTE_Close)
154 {
155 WTSLogger::error("[{}]交易账号连接已断开: {}", _id.c_str(), ec);
156 }

289void TraderAdapter::onRspPosition(const WTSArray* ayPositions)
290{
291 if (ayPositions && ayPositions->size() > 0)
292 {
293 for (std::size_t idx = 0; idx < ((WTSArray*)ayPositions)->size(); idx++)
294 {
295 WTSPositionItem* pItem = (WTSPositionItem*)(((WTSArray*)ayPositions)->at(idx));
296 WTSContractInfo* cInfo = _bd_mgr->getContract(pItem->getCode());
297 if (cInfo == NULL)
298 continue;
299 WTSCommodityInfo* commInfo = cInfo->getCommInfo();
300
301 getDumper().on_position(_id.c_str(), cInfo->getExchg(), cInfo->getCode(), _date, (pItem->getDirection() == WDT_LONG ? 0 : 1),
302 pItem->getTotalPosition(), pItem->getPositionCost(), pItem->getMargin(), pItem->getAvgPrice(),
303 pItem->getDynProfit(), commInfo->getVolScale(), idx == ayPositions->size() - 1);
304 }
305 }
306
307 WTSLogger::info("[{}]持仓数据已更新", _id.c_str());
308
309 if (!_done)
310 _trader_api->queryAccount();

383void TraderAdapterMgr::decAlive()
384{
385 _mutex.lock();
386 auto left = _live_cnt.fetch_sub(1);
387 _mutex.unlock();
388 if (left > 0)
389 left--;
390 if(left <= 2 && left > 0)
391 {
392 for (auto it = _adapters.begin(); it != _adapters.end(); it++)
393 {
394 TraderAdapterPtr trader = it->second;
395 if (!trader->isDone())
396 WTSLogger::info("{} is still undone", trader->id());
397 }
398 }
399
400 WTSLogger::info("{}/{}", left, _adapters.size());
401}

408 if (!trader->isDone())
409 continue;
410 trader->queryPosition();
411 trader->queryFund();
412 }
