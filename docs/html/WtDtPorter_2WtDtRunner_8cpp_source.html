58void WtDtRunner::start(bool bAsync /* = false */, bool bAlldayMode /* = false */)
59{
60 _parsers.run();
61
62 if(!bAsync)
63 {
64 install_signal_hooks([this](const char* message) {
65 if(!_to_exit)
66 WTSLogger::error(message);
67 }, [this](bool toExit) {
68 if (_to_exit)
69 return;
70 _to_exit = toExit;
71 WTSLogger::info("Exit flag is {}", _to_exit);
72 });
73
74 _async_io.post([this, bAlldayMode]() {
75 if(!bAlldayMode)
76 {
77 std::this_thread::sleep_for(std::chrono::milliseconds(5));
78 _state_mon.run();
79 }
80 });
81
82 StdThread trd([this] {
83 while (!_to_exit)
84 {
85 std::this_thread::sleep_for(std::chrono::milliseconds(2));
86 _async_io.run_one();
87 }
88 });
89
90 trd.join();
91 }
92 else
93 {
94 if (!bAlldayMode)
95 {
96 std::this_thread::sleep_for(std::chrono::milliseconds(5));
97 _state_mon.run();
98 }
99 }

111void WtDtRunner::initialize(const char* cfgFile, const char* logCfg, const char* modDir /* = "" */, bool bCfgFile /* = true */, bool bLogCfgFile /* = true */)
112{
113 WTSLogger::init(logCfg, bLogCfgFile);
114 WtHelper::set_module_dir(modDir);
115
116 WTSVariant* config = NULL;
117 if (bCfgFile)
118 config = WTSCfgLoader::load_from_file(cfgFile);
119 else
120 config = WTSCfgLoader::load_from_content(cfgFile, false);
121
122 if(config == NULL)
123 {
124 WTSLogger::error("Loading config file {} failed", cfgFile);
125 return;
126 }
127
128 //基础数据文件
129 WTSVariant* cfgBF = config->get("basefiles");
130 if (cfgBF->get("session"))
131 {
132 _bd_mgr.loadSessions(cfgBF->getCString("session"));
133 WTSLogger::info("Trading sessions loaded");
134 }
135
136 WTSVariant* cfgItem = cfgBF->get("commodity");
137 if (cfgItem)
138 {
139 if (cfgItem->type() == WTSVariant::VT_String)
140 {
141 _bd_mgr.loadCommodities(cfgItem->asCString());
142 }
143 else if (cfgItem->type() == WTSVariant::VT_Array)
144 {
145 for (uint32_t i = 0; i < cfgItem->size(); i++)
146 {
147 _bd_mgr.loadCommodities(cfgItem->get(i)->asCString());
148 }
149 }
150 }
151
152 cfgItem = cfgBF->get("contract");
153 if (cfgItem)
154 {
155 if (cfgItem->type() == WTSVariant::VT_String)
156 {
157 _bd_mgr.loadContracts(cfgItem->asCString());
158 }
159 else if (cfgItem->type() == WTSVariant::VT_Array)
160 {
161 for (uint32_t i = 0; i < cfgItem->size(); i++)
162 {
163 _bd_mgr.loadContracts(cfgItem->get(i)->asCString());
164 }
165 }
166 }
167
168 if (cfgBF->get("holiday"))
169 {
170 _bd_mgr.loadHolidays(cfgBF->getCString("holiday"));
171 WTSLogger::info("Holidays loaded");
172 }
173
174
175 if (cfgBF->get("hot"))
176 {
177 _hot_mgr.loadHots(cfgBF->getCString("hot"));
178 WTSLogger::log_raw(LL_INFO, "Hot rules loaded");
179 }
180
181 if (cfgBF->get("second"))
182 {
183 _hot_mgr.loadSeconds(cfgBF->getCString("second"));
184 WTSLogger::log_raw(LL_INFO, "Second rules loaded");
185 }
186
187 if (cfgBF->has("rules"))
188 {
189 auto cfgRules = cfgBF->get("rules");
190 auto tags = cfgRules->memberNames();
191 for (const std::string& ruleTag : tags)
192 {
193 _hot_mgr.loadCustomRules(ruleTag.c_str(), cfgRules->getCString(ruleTag.c_str()));
194 WTSLogger::info("{} rules loaded from {}", ruleTag, cfgRules->getCString(ruleTag.c_str()));
195 }
196 }
197
198 if (config->has("shmcaster"))
199 {
200 _shm_caster.init(config->get("shmcaster"));
201 _data_mgr.add_caster(&_shm_caster);
202 }
203
204 if(config->has("broadcaster"))
205 {
206 _udp_caster.init(config->get("broadcaster"), &_bd_mgr, &_data_mgr);
207 _data_mgr.add_caster(&_udp_caster);
208 }
209
210 //By Wesley @ 2021.12.27
211 //全天候模式，不需要再使用状态机
212 bool bAlldayMode = config->getBoolean("allday");
213 if (!bAlldayMode)
214 {
215 _state_mon.initialize(config->getCString("statemonitor"), &_bd_mgr, &_data_mgr);
216 }
217 else
218 {
219 WTSLogger::info("datakit will run in allday mode");
220 }
221
222 initDataMgr(config->get("writer"), bAlldayMode);
223
224 if (config->has("index"))
225 {
226 //如果存在指数模块要，配置指数
227 const char* filename = config->getCString("index");
228 WTSLogger::info("Reading index config from {}...", filename);
229 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
230 if (var)
231 {
232 _idx_factory.init(var, &_hot_mgr, &_bd_mgr, &_data_mgr);
233 var->release();
234 }
235 else
236 {
237 WTSLogger::error("Loading index config {} failed", filename);
238 }
239 }
240
241 WTSVariant* cfgParser = config->get("parsers");
242 if (cfgParser)
243 {
244 if (cfgParser->type() == WTSVariant::VT_String)
245 {
246 const char* filename = cfgParser->asCString();
247 if (StdFile::exists(filename))
248 {
249 WTSLogger::info("Reading parser config from {}...", filename);
250 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
251 if (var)
252 {
253 initParsers(var->get("parsers"));
254 var->release();
255 }
256 else
257 {
258 WTSLogger::error("Loading parser config {} failed", filename);
259 }
260 }
261 else
262 {
263 WTSLogger::error("Parser configuration {} not exists", filename);
264 }
265 }
266 else if (cfgParser->type() == WTSVariant::VT_Array)
267 {
268 initParsers(cfgParser);
269 }
270 }
271 else
272 WTSLogger::log_raw(LL_WARN, "No parsers config, skipped loading parsers");
273
274 config->release();
275}

293void WtDtRunner::initParsers(WTSVariant* cfg)
294{
295 for (uint32_t idx = 0; idx < cfg->size(); idx++)
296 {
297 WTSVariant* cfgItem = cfg->get(idx);
298 if (!cfgItem->getBoolean("active"))
299 continue;
300
301 const char* id = cfgItem->getCString("id");
302
303 // By Wesley @ 2021.12.14
304 // 如果id为空，则生成自动id
305 std::string realid = id;
306 if (realid.empty())
307 {
308 static uint32_t auto_parserid = 1000;
309 realid = StrUtil::printf("auto_parser_%u", auto_parserid++);
310 }
311
312 ParserAdapterPtr adapter(new ParserAdapter(&_bd_mgr, &_data_mgr, &_idx_factory));
313 adapter->init(realid.c_str(), cfgItem);
314 _parsers.addAdapter(realid.c_str(), adapter);
315 }
316
317 WTSLogger::info("{} market data parsers loaded in total", _parsers.size());
318}
