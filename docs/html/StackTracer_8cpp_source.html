25void print_stack_trace(TracerLogCallback cb) {
26 unsigned int max_frames = 127;
27 // storage array for stack trace address data
28 void *addrlist[max_frames + 1];
29
30 // retrieve current stack addresses
31 unsigned int addrlen = backtrace(addrlist, sizeof(addrlist) / sizeof(void *));
32
33 if (addrlen == 0) {
34 cb("no trace fetched");
35 return;
36 }
37
38 // resolve addresses into strings containing "filename(function+address)",
39 // Actually it will be ## program address function + offset
40 // this array must be free()-ed
41 char **symbollist = backtrace_symbols(addrlist, addrlen);
42
43 // iterate over the returned symbol lines. skip the first, it is the
44 // address of this function.
45 for (unsigned int i = 4; i < addrlen; i++) {
46 char *begin_name = nullptr;
47 char *begin_offset = nullptr;
48 char *end_offset = nullptr;
49
50 // ./module(function+0x15c) [0x8048a6d]
51 for (char *p = symbollist[i]; *p; ++p) {
52 if (*p == '(')
53 begin_name = p;
54 else if (*p == '+')
55 begin_offset = p;
56 else if (*p == ')' && (begin_offset || begin_name))
57 end_offset = p;
58 }
59
60 if (begin_name && end_offset && (begin_name > end_offset)) {
61 *begin_name++ = '\0';
62 *end_offset++ = '\0';
63 if (begin_offset)
64 *begin_offset++ = '\0';
65
66 // mangled name is now in [begin_name, begin_offset) and caller
67 // offset in [begin_offset, end_offset). now apply
68 // __cxa_demangle():
69
70 int status = 0;
71 size_t funcnamesize = 8192;
72 char funcname[8192];
73 char *ret = abi::__cxa_demangle(begin_name, funcname, &funcnamesize, &status);
74 cb(ret);
75 char buf[256] = { 0 };
76 sprintf(buf, "%30s ( %40s + %6s) %s", symbollist[i], status == 0 ? ret : begin_name, begin_offset ? begin_offset : "", end_offset);
77 cb(buf);
78 } else {
79 // couldn't parse the line? print the whole line.
80 cb(symbollist[i]);
81 }
82 }
83 free(symbollist);
84}
85#endif // _WINDOWS
