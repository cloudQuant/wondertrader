69bool ParserAdapter::init(const char* id, WTSVariant* cfg, IParserStub* stub, IBaseDataMgr* bgMgr)
70{
71 if (cfg == NULL)
72 return false;
73
74 _stub = stub;
75 _bd_mgr = bgMgr;
76 _id = id;
77
78 if (_cfg != NULL)
79 return false;
80
81 _cfg = cfg;
82 _cfg->retain();
83
84 {
85 //加载模块
86 if (cfg->getString("module").empty())
87 return false;
88
89 std::string module = DLLHelper::wrap_module(cfg->getCString("module"), "lib");;
90
91 //先看工作目录下是否有交易模块
92 std::string dllpath = WtHelper::getModulePath(module.c_str(), "parsers", true);
93 //如果没有,则再看模块目录,即dll同目录下
94 if (!StdFile::exists(dllpath.c_str()))
95 dllpath = WtHelper::getModulePath(module.c_str(), "parsers", false);
96
97 DllHandle hInst = DLLHelper::load_library(dllpath.c_str());
98 if (hInst == NULL)
99 {
100 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser module {} loading failed", _id.c_str(), dllpath.c_str());
101 return false;
102 }
103 else
104 {
105 WTSLogger::log_dyn("parser", _id.c_str(), LL_INFO, "[{}] Parser module {} loaded", _id.c_str(), dllpath.c_str());
106 }
107
108 FuncCreateParser pFuncCreateParser = (FuncCreateParser)DLLHelper::get_symbol(hInst, "createParser");
109 if (NULL == pFuncCreateParser)
110 {
111 WTSLogger::log_dyn("parser", _id.c_str(), LL_FATAL, "[{}] Entrance function createParser not found", _id.c_str());
112 return false;
113 }
114
115 _parser_api = pFuncCreateParser();
116 if (NULL == _parser_api)
117 {
118 WTSLogger::log_dyn("parser", _id.c_str(), LL_FATAL, "[{}] Creating parser api failed", _id.c_str());
119 return false;
120 }
121
122 _remover = (FuncDeleteParser)DLLHelper::get_symbol(hInst, "deleteParser");
123 }
124
125
126 const std::string& strFilter = cfg->getString("filter");
127 if (!strFilter.empty())
128 {
129 const StringVector &ayFilter = StrUtil::split(strFilter, ",");
130 auto it = ayFilter.begin();
131 for (; it != ayFilter.end(); it++)
132 {
133 _exchg_filter.insert(*it);
134 }
135 }
136
137 std::string strCodes = cfg->getString("code");
138 if (!strCodes.empty())
139 {
140 const StringVector &ayCodes = StrUtil::split(strCodes, ",");
141 auto it = ayCodes.begin();
142 for (; it != ayCodes.end(); it++)
143 {
144 _code_filter.insert(*it);
145 }
146 }
147
148 if (_parser_api)
149 {
150 _parser_api->registerSpi(this);
151
152 if (_parser_api->init(cfg))
153 {
154 ContractSet contractSet;
155 WTSArray* ay = _bd_mgr->getContracts();
156 for(auto it = ay->begin(); it != ay->end(); it++)
157 {
158 WTSContractInfo* cInfo = STATIC_CONVERT(*it, WTSContractInfo*);
159
160 //先检查合约和品种是否符合条件
161 if(!_code_filter.empty())
162 {
163 auto cit = _code_filter.find(cInfo->getFullCode());
164 auto pit = _code_filter.find(cInfo->getFullPid());
165 if (cit != _code_filter.end() || pit != _code_filter.end())
166 {
167 contractSet.insert(cInfo->getFullCode());
168 continue;
169 }
170 }
171
172 //再检查交易所是否符合条件
173 if (!_exchg_filter.empty())
174 {
175 auto eit = _exchg_filter.find(cInfo->getExchg());
176 if (eit != _exchg_filter.end())
177 {
178 contractSet.insert(cInfo->getFullCode());
179 continue;
180 }
181 else
182 {
183 continue;
184 }
185 }
186
187 if(_code_filter.empty() && _exchg_filter.empty())
188 contractSet.insert(cInfo->getFullCode());
189
190 }
191 ay->release();
192
193 _parser_api->subscribe(contractSet);
194 contractSet.clear();
195 }
196 else
197 {
198 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser initializing failed: api initializing failed...", _id.c_str());
199 }
200 }
201 else
202 {
203 WTSLogger::log_dyn("parser", _id.c_str(), LL_ERROR, "[{}] Parser initializing failed: creating api failed...", _id.c_str());
204 }
205
206 return true;
207}
