467void SelMocker::handle_tick(const char* stdCode, WTSTickData* newTick, uint32_t pxType)
468{
469 double cur_px = newTick->price();
470
471 /*
472 * By Wesley @ 2022.04.19
473 * 这里的逻辑改了一下
474 * 如果缓存的价格不存在，则上一笔价格就用最新价
475 * 这里主要是为了应对跨日价格跳空的情况
476 */
477 double last_px = cur_px;
478 if (pxType != 0)
479 {
480 auto it = _price_map.find(stdCode);

491 proc_tick(stdCode, last_px, cur_px);
492
493 on_tick_updated(stdCode, newTick);

496 * By Wesley @ 2022.04.19

498 * 如果是模拟的tick数据，用收盘价模拟tick的时候，isBarEnd才会为true

609void SelMocker::update_dyn_profit(const char* stdCode, double price)
610{
611 auto it = _pos_map.find(stdCode);
612 if (it != _pos_map.end())
613 {
614 PosInfo& pInfo = (PosInfo&)it->second;
615 if (pInfo._volume == 0)
616 {
617 pInfo._dynprofit = 0;
618 }
619 else
620 {
621 WTSCommodityInfo* commInfo = _replayer->get_commodity_info(stdCode);
622 double dynprofit = 0;
623 for (auto pit = pInfo._details.begin(); pit != pInfo._details.end(); pit++)
624 {
625 DetailInfo& dInfo = *pit;
626 dInfo._profit = dInfo._volume*(price - dInfo._price)*commInfo->getVolScale()*(dInfo._long ? 1 : -1);
627 if (dInfo._profit > 0)
628 dInfo._max_profit = max(dInfo._profit, dInfo._max_profit);
629 else if (dInfo._profit < 0)
630 dInfo._max_loss = min(dInfo._profit, dInfo._max_loss);
631
632 dInfo._max_price = std::max(dInfo._max_price, price);
633 dInfo._min_price = std::min(dInfo._min_price, price);
634
635 dynprofit += dInfo._profit;
636 }
637
638 pInfo._dynprofit = dynprofit;
639 }
640 }
641
642 double total_dynprofit = 0;
643 for (auto& v : _pos_map)
644 {
645 const PosInfo& pInfo = v.second;
646 total_dynprofit += pInfo._dynprofit;
647 }
648
649 _fund_info._total_dynprofit = total_dynprofit;
650}

908 }
909
910 double total = stra_get_position(stdCode, false);
911 //如果目标仓位和当前仓位是一致的，直接退出

913 return;
914
915 if (commInfo->isT1())

919 //如果是T+1规则，则目标仓位不能小于冻结仓位
920 if (decimal::lt(qty, frozen))
921 {
922 WTSLogger::log_dyn("strategy", _name.c_str(), LL_ERROR,
923 "New position of {} cannot be set to {} due to {} being frozen", stdCode, qty, frozen);

925 }
926 }

928 _replayer->sub_tick(id(), stdCode);
929 append_signal(stdCode, qty, userTag);

1094 pInfo._last_exittime = curTm;
