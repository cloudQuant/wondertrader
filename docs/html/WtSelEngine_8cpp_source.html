209void WtSelEngine::on_minute_end(uint32_t curDate, uint32_t curTime)
210{
211 //要比较下一分钟的时间
212 uint32_t nextTime = TimeUtils::getNextMinute(curTime, 1);
213 if (nextTime < curTime)
214 curDate = TimeUtils::getNextDate(curDate);
215
216 uint32_t weekDay = TimeUtils::getWeekDay(curDate);
217
218 for (auto& v : _tasks)
219 {
220 TaskInfoPtr& tInfo = (TaskInfoPtr&)v.second;
221 if (tInfo->_time != nextTime)
222 continue;
223
224 uint64_t now = (uint64_t)curDate * 10000 + nextTime;
225 if (tInfo->_last_exe_time >= now)
226 continue;
227
228 if (_base_data_mgr->isHoliday(tInfo->_trdtpl, curDate, true))
229 continue;
230
231 //获取上一个交易日的日期
232 uint32_t preTDate = TimeUtils::getNextDate(_cur_date, -1);
233 bool bHasHoliday = false;
234 uint32_t days = 1;
235 while (_base_data_mgr->isHoliday(tInfo->_trdtpl, preTDate, true))
236 {
237 bHasHoliday = true;
238 preTDate = TimeUtils::getNextDate(preTDate, -1);
239 days++;
240 }
241 uint32_t preWD = TimeUtils::getWeekDay(preTDate);
242
243 WTSSessionInfo* sInfo = get_session_info(tInfo->_session, false);
244
245 bool bIgnore = true;
246 switch (tInfo->_period)
247 {
248 case TPT_Daily:
249 bIgnore = false;
250 break;
251 case TPT_Minute:
252 {
253 uint32_t minutes = sInfo->timeToMinutes(curTime); //先将时间转换成分钟数
254 if(minutes != 0 && (minutes%tInfo->_time == 0)) //如果分钟数能被整除,且不为0,则可以触发
255 {
256 bIgnore = false;
257 }
258 }
259 break;
260 case TPT_Monthly:
261 //if (preTDate % 1000000 < _task->_day && _cur_date % 1000000 >= _task->_day)
262 // fired = true;
263 if (_cur_date % 1000000 == tInfo->_day)
264 bIgnore = false;
265 else if (bHasHoliday)
266 {
267 //上一个交易日在上个月,且当前日期大于触发日期
268 //说明这个月的开始日期在节假日内,顺延到今天
269 if ((preTDate % 10000 / 100 < _cur_date % 10000 / 100) && _cur_date % 1000000 > tInfo->_day)
270 {
271 bIgnore = false;
272 }
273 else if (preTDate % 1000000 < tInfo->_day && _cur_date % 1000000 > tInfo->_day)
274 {
275 //上一个交易日在同一个月,且小于触发日期,但是今天大于触发日期,说明正确触发日期到节假日内,顺延到今天
276 bIgnore = false;
277 }
278 }
279 break;
280 case TPT_Weekly:
281 //if (preWD < _task->_day && weekDay >= _task->_day)
282 // fired = true;
283 if (weekDay == tInfo->_day)
284 bIgnore = false;
285 else if (bHasHoliday)
286 {
287 if (days >= 7 && weekDay > tInfo->_day)
288 {
289 bIgnore = false;
290 }
291 else if (preWD > weekDay && weekDay > tInfo->_day)
292 {
293 //上一个交易日的星期大于今天的星期,说明换了一周了
294 bIgnore = false;
295 }
296 else if (preWD < tInfo->_day && weekDay > tInfo->_day)
297 {
298 bIgnore = false;
299 }
300 }
301 break;
302 case TPT_Yearly:
303 if (preTDate % 10000 < tInfo->_day && _cur_date % 10000 >= tInfo->_day)
304 bIgnore = false;
305 break;
306 }
307
308 if (bIgnore)
309 continue;
310
311 //TODO: 回调任务
312 SelContextPtr ctx = getContext(tInfo->_id);
313 StdThreadPtr thrd(new StdThread([ctx, curDate, curTime, nextTime](){
314 if (ctx)
315 ctx->on_schedule(curDate, curTime, nextTime);
316 }));
317
318 tInfo->_last_exe_time = now;
319 }

322void WtSelEngine::run()
323{
324 WTSVariant* cfgProd = _cfg->get("product");
325 _tm_ticker = new WtSelRtTicker(this);
326 _tm_ticker->init(_data_mgr->reader(), cfgProd->getCString("session"));
327
328 //启动之前,先把运行中的策略落地
329 {
330 rj::Document root(rj::kObjectType);
331 rj::Document::AllocatorType &allocator = root.GetAllocator();
332
333 rj::Value jStraList(rj::kArrayType);
334 for (auto& m : _ctx_map)
335 {
336 const SelContextPtr& ctx = m.second;
337 jStraList.PushBack(rj::Value(ctx->name(), allocator), allocator);
338 }
339
340 root.AddMember("marks", jStraList, allocator);
341
342 rj::Value jChnlList(rj::kArrayType);
343 for (auto& m : _adapter_mgr->getAdapters())
344 {
345 const TraderAdapterPtr& adapter = m.second;
346 jChnlList.PushBack(rj::Value(adapter->id(), allocator), allocator);
347 }
348
349 root.AddMember("channels", jChnlList, allocator);
350
351 root.AddMember("engine", rj::Value("SEL", allocator), allocator);
352
353 std::string filename = WtHelper::getBaseDir();
354 filename += "marker.json";
355
356 rj::StringBuffer sb;
357 rj::PrettyWriter<rj::StringBuffer> writer(sb);
358 root.Accept(writer);
359 StdFile::write_file_content(filename.c_str(), sb.GetString());
360 }
361
362 _tm_ticker->run();
363}
