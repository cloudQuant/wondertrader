288void WtStockMinImpactExeUnit::on_tick(WTSTickData* newTick)
289{
290 // 获取当前时间
291 _now = TimeUtils::getLocalTimeNow();
292 // 检查行情数据是否有效或是否属于当前合约
293 if (newTick == NULL || _code.compare(newTick->code()) != 0)
294 return;
295
296 //如果原来的tick不为空,则要释放掉
297 if (_last_tick)
298 {
299 _last_tick->release();
300 }
301 else
302 {
303 //如果行情时间不在交易时间,这种情况一般是集合竞价的行情进来,下单会失败,所以直接过滤掉这笔行情
304 if (_sess_info != NULL && !_sess_info->isInTradingTime(newTick->actiontime() / 100000))
305 return;
306 }
307
308 //新的tick数据,要保留
309 _last_tick = newTick;
310 _last_tick->retain();
311
312 //如果相比上次没有更新的tick进来，则先不下单，防止开盘前集中下单导致通道被封
313 uint64_t curTickTime = TimeUtils::makeTime(_last_tick->actiondate(), _last_tick->actiontime());
314 if (curTickTime <= _last_tick_time)
315 {
316 _ctx->writeLog(fmt::format("No tick of {} updated, {} <= {}, execute later",
317 _code, curTickTime, _last_tick_time).c_str());
318 return;
319 }
320 _last_tick_time = curTickTime;
321
322 /*
323 * 这里可以考虑一下
324 * 如果写的上一次丢出去的单子不够达到目标仓位
325 * 那么在新的行情数据进来的时候可以再次触发核心逻辑
326 */
327
328 // 输出当前管理的订单状态
329 _orders_mon.enumOrder([this](uint32_t localid, uint64_t entertime, bool cancancel) {
330 _ctx->writeLog(fmtutil::format("[{}]{} entertime:{} cancancel:{} now:{} last_tick_time:{} live_time:{}", _code, localid, entertime, cancancel, _now, _last_tick_time, _now - entertime));
331 });
332
333 // 检查并处理超时订单
334 if (_expire_secs != 0 && _orders_mon.has_order())
335 {
336 _orders_mon.check_orders(_expire_secs, _now, [this](uint32_t localid) {
337 if (_ctx->cancel(localid))
338 {
339 _ctx->writeLog(fmt::format("[{}] Expired order of {} canceled", localid, _code.c_str()).c_str());
340 // 记录撤单次数
341 if (_cancel_map.find(localid) == _cancel_map.end())
342 {
343 _cancel_map[localid] = 0;
344 }
345 _cancel_map[localid] += 1;
346 }
347 });
348 }
349
350 // 处理撤单次数超过限制的订单，可能是错单
351 if (!_cancel_map.empty())
352 {
353 std::vector<uint32_t> erro_cancel_orders{};
354 for (auto item : _cancel_map)
355 {
356 if (item.second > _max_cancel_time)
357 {
358 erro_cancel_orders.push_back(item.first);
359 }
360 }
361 for (uint32_t localid : erro_cancel_orders)
362 {
363 _cancel_map.erase(localid);
364 _orders_mon.erase_order(localid);
365 _ctx->writeLog(fmtutil::format("error order:{} canceled by {} times,erase forcely", localid, _max_cancel_time));
366 }
367 }
368
369 // 触发交易计算
370 do_calc();
371}

501void WtStockMinImpactExeUnit::do_calc()
502{
503 // 如果没有最新行情数据，无法计算
504 if (!_last_tick)
505 return;
506 if (_is_finish)
507 return;
508 if (!_is_ready)
509 {
510 _ctx->writeLog(fmtutil::format("{} wait channel ready", _code));
511 return;
512 }
513
514 //这里加一个锁，主要原因是实盘过程中发现
515 //在修改目标仓位的时候，会触发一次do_calc
516 //而ontick也会触发一次do_calc，两次调用是从两个线程分别触发的，所以会出现同时触发的情况
517 //如果不加锁，就会引起问题
518 //这种情况在原来的SimpleExecUnit没有出现，因为SimpleExecUnit只在set_position的时候触发
519 StdUniqueLock lock(_mtx_calc);
520
521 const char* stdCode = _code.c_str();
522
523 double undone = _ctx->getUndoneQty(stdCode);
524 // 总仓位，等于昨仓 + 今仓买入的
525 double curPos = _ctx->getPosition(stdCode, false);
526 // 可用仓位，即昨仓的
527 double vailyPos = _ctx->getPosition(stdCode, true);
528 if (_is_t0)
529 vailyPos = curPos;
530 double target_pos = max(curPos - vailyPos, _target_pos);
531
532 if (!decimal::eq(target_pos, _target_pos))
533 {
534 _ctx->writeLog(fmtutil::format("{} can sell hold pos not enough, target adjust {}->{}", stdCode, _target_pos, target_pos));
535 _target_pos = target_pos;
536 }
537
538 // 补一次
539 if (decimal::ge(_start_price, 0))
540 {
541 _start_price = _last_tick->price();
542 }
543
544 double diffPos = target_pos - curPos;
545 _ctx->writeLog(fmtutil::format("{}: target: {} hold:{} left {} wait to execute", _code.c_str(), target_pos, curPos, diffPos));
546 // 在判断的时候，要两边四舍五入，防止一些碎股导致一直无法完成执行
547 if (decimal::eq(round_hands(target_pos, _min_hands), round_hands(curPos, _min_hands)) && !(target_pos == 0 && curPos < _min_hands && curPos > target_pos))
548 {
549 _ctx->writeLog(fmtutil::format("{}: target position {} set finish", _code.c_str(), _target_pos));
550 _is_finish = true;
551 return;
552 }
553
554 bool isBuy = decimal::gt(diffPos, 0);
555 //有未完成订单，与实际仓位变动方向相反
556 //则需要撤销现有订单
557 if (decimal::lt(diffPos * undone, 0))
558 {
559 _ctx->writeLog(fmt::format("{} undone:{} diff:{} cancel", stdCode, undone, diffPos).c_str());
560 bool isBuy = decimal::gt(undone, 0);
561 OrderIDs ids = _ctx->cancel(stdCode, isBuy);
562 if (!ids.empty())
563 {
564 _orders_mon.push_order(ids.data(), ids.size(), _now);
565 for (auto localid : ids)
566 {
567 _ctx->writeLog(fmt::format("{} mon push wait cancel order {} enter time:{}", _code.c_str(), localid, _now).c_str());
568 _ctx->writeLog(fmt::format("[{}] live opposite order of {} canceled", localid, _code.c_str()).c_str());
569 }
570 }
571 return;
572 }
573
574 //因为是逐笔发单，所以如果有不需要撤销的未完成单，则暂不发单
575 if (!decimal::eq(undone, 0))
576 {
577 _ctx->writeLog(fmtutil::format("{} undone {} wait...", _code, undone));
578 return;
579 }
580
581 if (_last_tick == NULL)
582 {
583 _ctx->writeLog(fmt::format("No lastest tick data of {}, execute later", _code.c_str()).c_str());
584 return;
585 }
586
587 //检查下单时间间隔;
588 if (_now - _last_place_time < _entrust_span)
589 {
590 _ctx->writeLog(fmtutil::format("entrust span {} last_place_time {} _now {}", _entrust_span, _last_place_time, _now));
591 return;
592 }
593
594 double this_qty = _order_lots;
595 if (_by_rate)
596 {
597 double book_qty = isBuy ? _last_tick->askqty(0) : _last_tick->bidqty(0);
598 book_qty = book_qty * _qty_rate;
599 //book_qty = round_hands(book_qty, _min_hands);
600 book_qty = round_hands(book_qty, _min_order);//2023.6.5-zhaoyk
601 book_qty = max(_min_order, book_qty);
602 this_qty = book_qty;
603 }
604 diffPos = abs(diffPos);
605 this_qty = min(this_qty, diffPos);
606 // 买
607 if (isBuy)
608 {
609 //如果是买的话，要考虑取整和资金余额
610 //this_qty = round_hands(this_qty, _min_hands);
611 this_qty = round_hands(this_qty, _min_order);//2023.6.5-zhaoyk
612 if (_avaliable)
613 {
614 double max_can_buy = _avaliable / _last_tick->price();
615 //max_can_buy = (int)(max_can_buy / _min_hands) * _min_hands;
616 max_can_buy = (int)(max_can_buy / _min_order) * _min_order;//2023.6.5-zhaoyk
617 this_qty = min(max_can_buy, this_qty);
618 }
619 }
620 // 卖要对碎股做检查
621 else
622 {
623 //double chip_stk = vailyPos - int(vailyPos / _min_hands) * _min_hands;
624 //if (decimal::lt(vailyPos, _min_hands))
625 if (decimal::lt(vailyPos, _min_order))//2023.6.5-zhaoyk
626 {
627 this_qty = vailyPos;
628 }
629 else
630 {
631 //this_qty = round_hands(this_qty, _min_hands);
632 this_qty = round_hands(this_qty, _min_order);//2023.6.5-zhaoyk
633
634 }
635 this_qty = min(vailyPos, this_qty);
636 }
637
638 if (decimal::eq(this_qty, 0))
639 return;
640
641 double buyPx, sellPx;
642 if (_price_mode == AUTOPX)
643 {
644 double mp = (_last_tick->bidqty(0) - _last_tick->askqty(0)) * 1.0 / (_last_tick->bidqty(0) + _last_tick->askqty(0));
645 bool isUp = (mp > 0);
646 if (isUp)
647 {
648 buyPx = _last_tick->askprice(0);
649 sellPx = _last_tick->askprice(0);
650 }
651 else
652 {
653 buyPx = _last_tick->bidprice(0);
654 sellPx = _last_tick->bidprice(0);
655 }
656 }
657 else
658 {
659 if (_price_mode == BESTPX)
660 {
661 buyPx = _last_tick->bidprice(0);
662 sellPx = _last_tick->askprice(0);
663 }
664 else if (_price_mode == LASTPX)
665 {
666 buyPx = _last_tick->price();
667 sellPx = _last_tick->price();
668 }
669 else if (_price_mode == MARKET)
670 {
671 buyPx = _last_tick->askprice(0) + _comm_info->getPriceTick() * _price_offset;
672 sellPx = _last_tick->bidprice(0) - _comm_info->getPriceTick() * _price_offset;
673 }
674 }
675
676 // 如果最后价格为0，再做一个修正
677 if (decimal::eq(buyPx, 0.0))
678 buyPx = decimal::eq(_last_tick->price(), 0.0) ? _last_tick->preclose() : _last_tick->price();
679
680 if (decimal::eq(sellPx, 0.0))
