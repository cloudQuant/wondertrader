153template <typename T>
154inline Emitter& Emitter::WriteStreamable(T value) {
155 if (!good())
156 return *this;
157
158 PrepareNode(EmitterNodeType::Scalar);
159
160 std::stringstream stream;
161 SetStreamablePrecision<T>(stream);
162
163 bool special = false;
164 if (std::is_floating_point<T>::value) {
165 if ((std::numeric_limits<T>::has_quiet_NaN ||
166 std::numeric_limits<T>::has_signaling_NaN) &&
167 std::isnan(value)) {
168 special = true;
169 stream << ".nan";
170 } else if (std::numeric_limits<T>::has_infinity && std::isinf(value)) {
171 special = true;
172 if (std::signbit(value)) {
173 stream << "-.inf";
174 } else {
175 stream << ".inf";
176 }
177 }
178 }
179
180 if (!special) {
181 stream << value;
182 }
183 m_stream << stream.str();
184
185 StartedScalar();
186
187 return *this;
188}
