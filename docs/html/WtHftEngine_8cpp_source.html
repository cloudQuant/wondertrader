88void WtHftEngine::run()
89{
90 for (auto it = _ctx_map.begin(); it != _ctx_map.end(); it++)
91 {
92 HftContextPtr& ctx = (HftContextPtr&)it->second;
93 ctx->on_init();
94 }
95
96 _tm_ticker = new WtHftRtTicker(this);
97 WTSVariant* cfgProd = _cfg->get("product");
98 _tm_ticker->init(_data_mgr->reader(), cfgProd->getCString("session"));
99
100 //启动之前,先把运行中的策略落地
101 {
102 rj::Document root(rj::kObjectType);
103 rj::Document::AllocatorType &allocator = root.GetAllocator();
104
105 rj::Value jStraList(rj::kArrayType);
106 for (auto& m : _ctx_map)
107 {
108 const HftContextPtr& ctx = m.second;
109 jStraList.PushBack(rj::Value(ctx->name(), allocator), allocator);
110 }
111
112 root.AddMember("marks", jStraList, allocator);
113
114 rj::Value jChnlList(rj::kArrayType);
115 for (auto& m : _adapter_mgr->getAdapters())
116 {
117 const TraderAdapterPtr& adapter = m.second;
118 jChnlList.PushBack(rj::Value(adapter->id(), allocator), allocator);
119 }
120
121 root.AddMember("channels", jChnlList, allocator);
122
123 root.AddMember("engine", rj::Value("HFT", allocator), allocator);
124
125 std::string filename = WtHelper::getBaseDir();
126 filename += "marker.json";
127
128 rj::StringBuffer sb;
129 rj::PrettyWriter<rj::StringBuffer> writer(sb);
130 root.Accept(writer);
131 StdFile::write_file_content(filename.c_str(), sb.GetString());
132 }
133
134 _tm_ticker->run();
135}

304void WtHftEngine::on_tick(const char* stdCode, WTSTickData* curTick)
305{
306 WtEngine::on_tick(stdCode, curTick);
307
308 _data_mgr->handle_push_quote(stdCode, curTick);
309
310 /*
311 * By Wesley @ 2022.02.07
312 * 这里做了一个彻底的调整
313 * 第一，检查订阅标记，如果标记为0，即无复权模式，则直接按照原始代码触发ontick
314 * 第二，如果标记为1，即前复权模式，则将代码转成xxxx-，再触发ontick
315 * 第三，如果标记为2，即后复权模式，则将代码转成xxxx+，再把tick数据做一个修正，再触发ontick
316 */
317 if(_ready)
318 {
319 auto sit = _tick_sub_map.find(stdCode);
320 if (sit != _tick_sub_map.end())
321 {
322 const SubList& sids = sit->second;
323 for (auto it = sids.begin(); it != sids.end(); it++)
324 {
325 uint32_t sid = it->first;
326
327 auto cit = _ctx_map.find(sid);
328 if (cit != _ctx_map.end())
329 {
330 HftContextPtr& ctx = (HftContextPtr&)cit->second;
331 uint32_t opt = it->second.second;
332
333 if (opt == 0)
334 {
335 ctx->on_tick(stdCode, curTick);
336 }
337 else
338 {
339 std::string wCode = stdCode;
340 wCode = fmt::format("{}{}", stdCode, opt == 1 ? SUFFIX_QFQ : SUFFIX_HFQ);
341 if (opt == 1)
342 {
343 ctx->on_tick(wCode.c_str(), curTick);
344 }
345 else //(opt == 2)
346 {
347 WTSTickData* newTick = WTSTickData::create(curTick->getTickStruct());
348 WTSTickStruct& newTS = newTick->getTickStruct();
349 newTick->setContractInfo(curTick->getContractInfo());
350
351 //这里做一个复权因子的处理
352 double factor = get_exright_factor(stdCode, curTick->getContractInfo()->getCommInfo());
353 newTS.open *= factor;
354 newTS.high *= factor;
355 newTS.low *= factor;
356 newTS.price *= factor;
357
358 _price_map[wCode] = newTS.price;
359
360 ctx->on_tick(wCode.c_str(), newTick);
361 newTick->release();
362 }
363 }
364 }
365 }
366 }
367 }
