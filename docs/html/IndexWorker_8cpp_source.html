64 _trigger = fmt::format("{}.{}", cInfo._exchg, hotMgr->getCustomRawCode(cInfo._ruletag, cInfo.stdCommID()));

128 WTSVariant* cfgItem = cfgCodes->get(i);

178 wFactor._weight = weight;

180 //订阅的时候读取最后的快照，作为基础数据
181 WTSTickData* lastTick = _factor->sub_ticks(fullCode.c_str());

189 }
190 }
191
192 WTSLogger::info("Block index {}.{} initialized，weight algorithm: {}, trigger: {}, timeout: {}", _exchg, _code, WEIGHT_ALGS[_weight_alg], _trigger, _timeout);

194 return true;

204void IndexWorker::handle_quote(WTSTickData* newTick)
205{
206 const char* fullCode = newTick->getContractInfo()->getFullCode();
207
208 {
209 SpinLock lock(_mtx_data);
210 auto it = _weight_scales.find(fullCode);
211 if (it == _weight_scales.end())
212 return;
213
214 WeightFactor& wFactor = (WeightFactor&)it->second;
215 memcpy(&wFactor._tick, &newTick->getTickStruct(), sizeof(WTSTickStruct));
216 }
217
218 //如果使用time，那么当第一个成分合约的行情进来以后，会去更新指数重算时间
219 if(_trigger != "time" && _trigger.compare(fullCode) != 0)
220 return;
221
222 //如果是_trigger，则开始准备触发了
223 if(_timeout == 0)
224 {
225 //如果没有设置超时时间，则直接生成指数
226 generate_tick();
227 }
228 else
229 {
230 if(_thrd_trigger == NULL)
231 {
232 _thrd_trigger.reset(new StdThread([this]() {
233 while(!_stopped)
234 {
235 while(!_process)
236 {
237 StdUniqueLock lck(_mtx_trigger);
238 _cond_trigger.wait(_mtx_trigger);
239 }
240
241 //等待一段时间
242 do
243 {
244 uint64_t now = TimeUtils::getLocalTimeNow();
245 if(now >= _recalc_time)
246 break;
247
248 std::this_thread::sleep_for(std::chrono::milliseconds(5));
249 } while (true);
250
251 //开始生成指数
252 generate_tick();
253 _process = false;
254 }
255 }));
256 }
257
258 if(!_process)
259 {
260 _process = true;
261 _recalc_time = TimeUtils::getLocalTimeNow() + _timeout;
262 _cond_trigger.notify_all();
263 }
264 }
265}

274void IndexWorker::generate_tick()
275{
276 //然后开始计算指数
277 double total_base = 0.0; //权重基数
278 double total_value = 0.0; //数值累加
279 double total_vol = 0.0; //指数总成交量
280 double total_amt = 0.0; //指数总成交额
281 double total_hold = 0.0; //指数总持
282 uint64_t maxTime = 0; //最后一笔tick的时间
283 uint32_t tDate = 0; //交易日
284
285 double total_weight = 0;
286
287 {
288 //先把数据锁住
289 SpinLock lock(_mtx_data);
290 for (const auto& v : _weight_scales)
291 {
292 //如果数据不全，直接退出
293 const WeightFactor& wFactor = v.second;
294 if (wFactor._tick.action_date == 0)
295 return;
296
297 uint64_t curTime = TimeUtils::makeTime(wFactor._tick.action_date, wFactor._tick.action_time);
298 maxTime = std::max(maxTime, curTime);
299 tDate = std::max(tDate, wFactor._tick.trading_date);
300
301 total_vol += wFactor._tick.total_volume;
302 total_amt += wFactor._tick.total_turnover;
303 total_hold += wFactor._tick.open_interest;
304
305 total_weight += wFactor._weight;
306
307 switch (_weight_alg)
308 {
309 case 0://固定权重，只看本身的weight
310 total_base = 1; //因为只看本身weight，所以权重基数为1
311 total_value += wFactor._tick.price * wFactor._weight;
312 break;
313 case 1: //动态总持
314 total_base += wFactor._tick.open_interest; //动态总持为当前总持
315 total_value += wFactor._tick.open_interest * wFactor._tick.price * wFactor._weight;
316 break;
317 case 2: //动态成交量
318 total_base += wFactor._tick.total_volume; //动态成交量
319 total_value += wFactor._tick.total_volume * wFactor._tick.price * wFactor._weight;
320 break;
321 default:
322 break;
323 }
324 }
325 }
326
327 //数据做标准化
328 double index = total_value / total_base / total_weight * _stand_scale;
329
330 //时间做一个修正
331 maxTime += _timeout;
332 TimeUtils::Time32 tm32(maxTime);
333
334 if(_cache.action_time == 0)
335 {
336 strcpy(_cache.exchg, _exchg.c_str());
337 strcpy(_cache.code, _code.c_str());
338 _cache.trading_date = tDate;
339
340 _cache.price = index;
341 _cache.open = index;
342 _cache.high = index;
343 _cache.low = index;
344
345 _cache.action_date = tm32.date();
346 _cache.action_time = tm32.time_ms();
347
348 _cache.total_volume = total_vol;
349 _cache.open_interest = total_hold;
350 _cache.total_turnover = total_amt;
351 }
352 else
353 {
354 _cache.price = index;
355 _cache.high = std::max(_cache.high, index);
356 _cache.low = std::min(_cache.low, index);
357
358 _cache.action_date = tm32.date();
359 _cache.action_time = tm32.time_ms();
360
361 _cache.total_volume = total_vol;
362 _cache.open_interest = total_hold;
363 _cache.total_turnover = total_amt;
364 }
365
366 WTSTickData *newTick = WTSTickData::create(_cache);
367 newTick->setContractInfo(_cInfo);
368 _factor->push_tick(newTick);
369 WTSLogger::debug("{}.{} - {}.{} - {}", _cache.exchg, _cache.code, _cache.action_date, _cache.action_time, _cache.price);
370 newTick->release();
371}
