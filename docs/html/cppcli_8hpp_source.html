280int cppcli::detail::pathUtil::replace_all(std::string &str, const std::string &pattern, const std::string &newpat)
281{
282 // 记录替换次数
283 int count = 0;
284 // 新模式和原模式的长度
285 const size_t nsize = newpat.size();
286 const size_t psize = pattern.size();
287
288 // 查找模式的第一次出现
289 size_t pos = str.find(pattern);
290 while (pos != std::string::npos) {
291 // 替换当前位置的模式
292 str.replace(pos, psize, newpat);
293 // 从替换后的位置继续查找
294 pos = str.find(pattern, pos + nsize);
295 // 替换次数增加
296 ++count;
297 }
298
299 return count;
300}

312std::string cppcli::detail::pathUtil::getFilename(const std::string &filePath)
313{
314 // 创建路径的副本
315 std::string filePathCopy(filePath);
316 // 标准化路径分隔符
317 replace_all(filePathCopy, CPPCLI_SEPARATOR_NO_TYPE, CPPCLI_SEPARATOR_TYPE);

319 // 确保文件存在
320 assert(std::ifstream(filePathCopy.c_str()).good());
321 // 查找最后一个路径分隔符的位置，并从其后一个字符开始提取
322 std::string::size_type pos = filePathCopy.find_last_of(CPPCLI_SEPARATOR_TYPE) + 1;
323 return std::move(filePathCopy.substr(pos, filePathCopy.length() - pos));

394void cppcli::detail::algoUtil::InitCommandMap(int length, char *strArr[], std::map<std::string, std::string> &stringMap)
395{
396 // 初始化临时变量
397 std::string keyTmp; // 当前参数名
398 std::string valueTmp; // 当前参数值
399 int keyIndex = -1; // 当前参数名的索引
400
401 // 遍历命令行参数，从1开始（跳过程序名）
402 for (int currentIndex = 1; currentIndex < length; currentIndex++)
403 {
404 std::string theStr(strArr[currentIndex]);
405 // 如果已找到参数名，并且当前字符串是其后的第一个参数
406 if (keyIndex != -1 && theStr.size() > 0 && currentIndex == keyIndex + 1)
407 {
408 // 如果当前字符串是另一个参数名，则上一个参数的值为空
409 if(theStr.find_first_of('-') == 0 && theStr.size() > 1 && !isdigit(theStr.at(1)))
410 {
411 valueTmp = "";
412 }
413 else
414 {
415 // 否则将当前字符串作为上一个参数的值
416 valueTmp = theStr;
417 }
418
419 // 将参数名和值添加到映射中
420 stringMap.insert(std::make_pair(std::move(keyTmp), std::move(valueTmp)));
421 keyTmp.clear();
422 valueTmp.clear();
423
424 // 重置参数名索引
425 keyIndex = -1;
426 }
427
428 // 检查当前字符串是否是参数名（以短横线开头，且不是负数）
429 if (theStr.find_first_of('-') == 0 && int(std::count(theStr.begin(), theStr.end(), '-')) < theStr.size() && !isdigit(theStr.at(1)))
430 {
431 keyIndex = currentIndex;
432 keyTmp = std::move(theStr);
433 }
434
435 // 如果是最后一个参数且是参数名，则其值为空
436 if (currentIndex == length - 1 && keyIndex != -1)
437 {
438 stringMap.insert(std::make_pair(std::move(keyTmp), std::move("")));
439 }
440 }

451bool cppcli::detail::algoUtil::isInt(const std::string &value)
452{
453 // 空字符串不是整数
454 if (value.empty())
455 {
456 return false;
457 }
458
459 // 如果第一个字符是负号，从第二个字符开始检查
460 int startPos = value.at(0) == '-' ? 1 : 0;
461 for (int i = startPos; i < value.size(); i++)
462 {
463 // 如果有非数字字符，则不是整数
464 if (isdigit(value.at(i)) == 0)
465 return false;
466 }
467 return true;
468}

480bool cppcli::detail::algoUtil::isDouble(const std::string &value)
481{
482 // 空字符串不是浮点数
483 if (value.empty())
484 {
485 return false;
486 }
487 // 浮点数至少需要三个字符（一个数字、小数点、一个数字）
488 if (value.size() < 3)
489 return false;
490 // 处理负数情况
491 std::string tmpValue = value.at(0) == '-' ? value.substr(0, value.npos): value;
492 // 计算数字的数量
493 int numCount = 0;
494 for (char const &c : tmpValue)
495 {
496 if (isdigit(c) != 0)
497 numCount++;
498 }
499
500 // 判断是否为浮点数：
501 // 1. 数字数量比字符串长度少一（小数点）
502 // 2. 小数点位置在中间（不在开头或结尾）
503 if (numCount == tmpValue.size() - 1 && tmpValue.rfind('.') > 0 && tmpValue.rfind('.') < tmpValue.size() - 1)
504 {
505 return true;
506 }
507 return false;
508}

522 if (isInt(value) || isDouble(value))
523 return true;
524 return false;

840const std::string cppcli::Rule::getError(cppcli::detail::ErrorEventType errorEventType)
841{
842 std::ostringstream oss;
843
844 oss << "[";
845 switch (errorEventType)
846 {
847 case cppcli::detail::ErrorEventType::MECESSARY_ERROR: {
848 if (_longParam.empty())
849 oss << _shortParam;
850 else
851 oss << _shortParam << std::move(" | ") << _longParam;
852 break;
853 }
854 case cppcli::detail::ErrorEventType::VALUETYPE_ERROR: {
855 if (_valueType == cppcli::detail::ValueTypeEnum::DOUBLE)
856 oss << std::move(" NUMBER (DOUBLE) ");
857 else if (_valueType == cppcli::detail::ValueTypeEnum::INT)
858 oss << std::move(" NUMBER (INT) ");
859 break;
860 }
861 case cppcli::detail::ErrorEventType::ONEOF_ERROR: {
862 for (int i = 0; i < _limitOneVec.size(); i++)
863 {
864 if (i == (_limitOneVec.size() - 1))
865 {
866 oss << _limitOneVec.at(i);
867 break;
868 }
869 oss << _limitOneVec.at(i) << std::move(" ");
870 }
871
872 break;
873 }
874 case cppcli::detail::ErrorEventType::NUMRANGE_ERROR: {
875 oss << _limitNumRange.first << std::move("(MIN), ") << _limitNumRange.second << std::move("(MAX)");
876 break;
877 }
878 }
879 oss << "]";
880 return std::move(oss.str());
881}

898std::string cppcli::Rule::buildHelpInfoLine()
899{
900 std::ostringstream oss;
901
902 int commandsDis = 28;
903 int helpInfoDis = 36;
904 int necessaryDis = 20;
905 int defaultStrDis = 20;
906 int theDis = 2;
907
908 oss << std::setw(commandsDis) << std::left << (_longParam.empty() ? _shortParam : _shortParam + " | " + _longParam);
909
910 int writeTime = _helpInfo.size() % helpInfoDis - theDis == 0 ? int(_helpInfo.size() / (helpInfoDis - theDis))
911 : int((_helpInfo.size() / (helpInfoDis - theDis))) + 1;
912 std::string necessaryOutStr = _necessary ? "true" : "false";
913 std::string defaultValueOutStr = _default == "[EMPTY]" ? _default : "=" + _default;
914
915 if (writeTime == 1)
916 {
917 oss << std::setw(helpInfoDis) << std::left << _helpInfo;
918 oss << std::setw(necessaryDis) << std::left << "MUST-ENTER[" + necessaryOutStr + "]";
919 oss << std::setw(defaultStrDis) << std::left << "DEFAULT->" + _default;
920 oss << std::endl;
921 return std::move(oss.str());
922 }
923 int pos = 0;
924 for (int i = 0; i < writeTime; i++)
925 {
926 if (i == 0)
927 {
928 oss << std::setw(helpInfoDis) << std::setw(helpInfoDis) << _helpInfo.substr(pos, helpInfoDis - theDis);
929 oss << std::setw(necessaryDis) << std::left << "MUST-ENTER[" + necessaryOutStr + "]";
930 oss << std::setw(defaultStrDis) << std::left << "DEFAULT->" + _default;
931 oss << std::endl;
932 pos += helpInfoDis - theDis;
933 }
934 else
935 {
936 oss << std::setw(commandsDis + 4) << std::left << "";
937 oss << _helpInfo.substr(pos, helpInfoDis - theDis);
938 oss << std::endl;
939 pos += helpInfoDis - theDis;
940 }
941 }
942
943 return std::move(oss.str());
944}
