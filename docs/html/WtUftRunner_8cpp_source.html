96bool WtUftRunner::config(const std::string& filename)
97{
98 // 从文件加载配置
99 _config = WTSCfgLoader::load_from_file(filename.c_str());
100 if(_config == NULL)
101 {
102 WTSLogger::error("Loading config file {} failed", filename);
103 return false;
104 }
105
106 //基础数据文件
107 WTSVariant* cfgBF = _config->get("basefiles");
108 // 加载交易时段配置
109 if (cfgBF->get("session"))
110 _bd_mgr.loadSessions(cfgBF->getCString("session"));
111
112 // 加载品种信息
113 WTSVariant* cfgItem = cfgBF->get("commodity");
114 if (cfgItem)
115 {
116 if (cfgItem->type() == WTSVariant::VT_String)
117 {
118 // 如果是字符串类型，直接加载单个文件
119 _bd_mgr.loadCommodities(cfgItem->asCString());
120 }
121 else if (cfgItem->type() == WTSVariant::VT_Array)
122 {
123 // 如果是数组类型，逐个加载多个文件
124 for (uint32_t i = 0; i < cfgItem->size(); i++)
125 {
126 _bd_mgr.loadCommodities(cfgItem->get(i)->asCString());
127 }
128 }
129 }
130
131 // 加载合约信息
132 cfgItem = cfgBF->get("contract");
133 if (cfgItem)
134 {
135 if (cfgItem->type() == WTSVariant::VT_String)
136 {
137 // 如果是字符串类型，直接加载单个文件
138 _bd_mgr.loadContracts(cfgItem->asCString());
139 }
140 else if (cfgItem->type() == WTSVariant::VT_Array)
141 {
142 // 如果是数组类型，逐个加载多个文件
143 for (uint32_t i = 0; i < cfgItem->size(); i++)
144 {
145 _bd_mgr.loadContracts(cfgItem->get(i)->asCString());
146 }
147 }
148 }
149
150 // 加载节假日信息
151 if (cfgBF->get("holiday"))
152 _bd_mgr.loadHolidays(cfgBF->getCString("holiday"));
153
154 //初始化运行环境
155 initEngine();
156
157 //初始化数据管理
158 initDataMgr();
159
160 // 如果存在共享域配置，初始化共享管理器
161 if (_config->has("share_domain"))
162 {
163 WTSVariant* cfg = _config->get("share_domain");
164 // 设置引擎到共享管理器
165 ShareManager::self().set_engine(&_uft_engine);
166
167 // 初始化共享管理器并设置域名
168 ShareManager::self().initialize(cfg->getCString("module"));
169 ShareManager::self().init_domain(cfg->getCString("name"));
170 }
171
172 // 初始化操作策略管理器
173 if(!_act_policy.init(_config->getCString("bspolicy")))
174 {
175 WTSLogger::error("ActionPolicyMgr init failed, please check config");
176 }
177
178 //初始化行情通道
179 WTSVariant* cfgParser = _config->get("parsers");
180 if (cfgParser)
181 {
182 if (cfgParser->type() == WTSVariant::VT_String)
183 {
184 // 如果是字符串类型，表示指向单独的解析器配置文件
185 const char* filename = cfgParser->asCString();
186 if (StdFile::exists(filename))
187 {
188 WTSLogger::info("Reading parser config from {}...", filename);
189 // 从文件加载解析器配置
190 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
191 if(var)
192 {
193 // 初始化解析器
194 if (!initParsers(var->get("parsers")))
195 WTSLogger::error("Loading parsers failed");
196 var->release();
197 }
198 else
199 {
200 WTSLogger::error("Loading parser config {} failed", filename);
201 }
202 }
203 else
204 {
205 WTSLogger::error("Parser configuration {} not exists", filename);
206 }
207 }
208 else if (cfgParser->type() == WTSVariant::VT_Array)
209 {
210 // 如果是数组类型，直接内嵌了解析器配置
211 initParsers(cfgParser);
212 }
213 }
214
215 //初始化交易通道
216 WTSVariant* cfgTraders = _config->get("traders");
217 if (cfgTraders)
218 {
219 if (cfgTraders->type() == WTSVariant::VT_String)
220 {
221 const char* filename = cfgTraders->asCString();
222 if (StdFile::exists(filename))
223 {
224 WTSLogger::info("Reading trader config from {}...", filename);
225 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
226 if (var)
227 {
228 if (!initTraders(var->get("traders")))
229 WTSLogger::error("Loading traders failed");
230 var->release();
231 }
232 else
233 {
234 WTSLogger::error("Loading trader config {} failed", filename);
235 }
236 }
237 else
238 {
239 WTSLogger::error("Trader configuration {} not exists", filename);
240 }
241 }
242 else if (cfgTraders->type() == WTSVariant::VT_Array)
243 {
244 initTraders(cfgTraders);
245 }
246 }
247
248 initUftStrategies();
249
250 return true;
251}

258bool WtUftRunner::initUftStrategies()
259{
260 // 获取策略配置
261 WTSVariant* cfg = _config->get("strategies");
262 if (cfg == NULL || cfg->type() != WTSVariant::VT_Object)
263 return false;
264
265 // 获取UFT策略具体配置
266 cfg = cfg->get("uft");
267 if (cfg == NULL || cfg->type() != WTSVariant::VT_Array)
268 return false;
269
270 // 构建策略工厂路径并加载策略工厂
271 std::string path = WtHelper::getCWD() + "uft/";
272 _uft_stra_mgr.loadFactories(path.c_str());
273
274 // 遍历所有策略配置
275 for (uint32_t idx = 0; idx < cfg->size(); idx++)
276 {
277 WTSVariant* cfgItem = cfg->get(idx);
278 // 跳过非激活的策略
279 if(!cfgItem->getBoolean("active"))
280 continue;
281 // 获取策略ID和名称
282 const char* id = cfgItem->getCString("id");
283 const char* name = cfgItem->getCString("name");
284 // 创建策略实例
285 UftStrategyPtr stra = _uft_stra_mgr.createStrategy(name, id);
286 if (stra == NULL)
287 {
288 WTSLogger::error("UFT Strategy {} create failed", name);
289 continue;
290 }
291 else
292 {
293 WTSLogger::info("UFT Strategy {}({}) created", name, id);
294 }
295
296 // 使用配置参数初始化策略
297 stra->self()->init(cfgItem->get("params"));
298 // 创建策略上下文
299 UftStraContext* ctx = new UftStraContext(&_uft_engine, id);
300 ctx->set_strategy(stra->self());
301
302 // 绑定交易适配器
303 const char* traderid = cfgItem->getCString("trader");
304 TraderAdapterPtr trader = _traders.getAdapter(traderid);
305 if(trader)
306 {
307 // 设置交易适配器并添加回调接收器
308 ctx->setTrader(trader.get());
309 trader->addSink(ctx);
310 }
311 else
312 {
313 WTSLogger::error("Trader {} not exists, binding trader to UFT strategy failed", traderid);
314 }
315
316 // 将策略上下文添加到引擎
317 _uft_engine.addContext(UftContextPtr(ctx));
318 }
319
320 return true;
321}

371bool WtUftRunner::initParsers(WTSVariant* cfgParser)
372{
373 // 检查配置是否有效
374 if (cfgParser == NULL)
375 return false;
376
377 // 初始化计数器
378 uint32_t count = 0;
379 // 遍历所有解析器配置
380 for (uint32_t idx = 0; idx < cfgParser->size(); idx++)
381 {
382 WTSVariant* cfgItem = cfgParser->get(idx);
383 // 跳过非激活的解析器
384 if(!cfgItem->getBoolean("active"))
385 continue;
386
387 // 获取解析器ID
388 const char* id = cfgItem->getCString("id");
389 // By Wesley @ 2021.12.14
390 // 如果id为空，则生成自动id
391 std::string realid = id;
392 if (realid.empty())
393 {
394 // 使用递增的数字生成自动ID
395 static uint32_t auto_parserid = 1000;
396 realid = StrUtil::printf("auto_parser_%u", auto_parserid++);
397 }
398
399 // 创建并初始化解析器适配器
400 ParserAdapterPtr adapter(new ParserAdapter);
401 adapter->init(realid.c_str(), cfgItem, &_uft_engine, &_bd_mgr);
402 // 将适配器添加到管理器中
403 _parsers.addAdapter(realid.c_str(), adapter);
404
405 count++;
406 }
407
408 // 输出已加载的解析器数量
409 WTSLogger::info("{} parsers loaded", count);
410 return true;
411}
