95bool WtRunner::config(const std::string& filename)
96{
97 _config = WTSCfgLoader::load_from_file(filename);
98 if(_config == NULL)
99 {
100 WTSLogger::error("Loading config file {} failed", filename);
101 return false;
102 }
103
104 //基础数据文件
105 WTSVariant* cfgBF = _config->get("basefiles");
106 if (cfgBF->get("session"))
107 _bd_mgr.loadSessions(cfgBF->getCString("session"));
108
109 WTSVariant* cfgItem = cfgBF->get("commodity");
110 if (cfgItem)
111 {
112 if (cfgItem->type() == WTSVariant::VT_String)
113 {
114 _bd_mgr.loadCommodities(cfgItem->asCString());
115 }
116 else if (cfgItem->type() == WTSVariant::VT_Array)
117 {
118 for (uint32_t i = 0; i < cfgItem->size(); i++)
119 {
120 _bd_mgr.loadCommodities(cfgItem->get(i)->asCString());
121 }
122 }
123 }
124
125 cfgItem = cfgBF->get("contract");
126 if (cfgItem)
127 {
128 if (cfgItem->type() == WTSVariant::VT_String)
129 {
130 _bd_mgr.loadContracts(cfgItem->asCString());
131 }
132 else if (cfgItem->type() == WTSVariant::VT_Array)
133 {
134 for (uint32_t i = 0; i < cfgItem->size(); i++)
135 {
136 _bd_mgr.loadContracts(cfgItem->get(i)->asCString());
137 }
138 }
139 }
140
141 if (cfgBF->get("holiday"))
142 _bd_mgr.loadHolidays(cfgBF->getCString("holiday"));
143
144 if (cfgBF->get("hot"))
145 _hot_mgr.loadHots(cfgBF->getCString("hot"));
146
147 if (cfgBF->get("second"))
148 _hot_mgr.loadSeconds(cfgBF->getCString("second"));
149
150 WTSArray* ayContracts = _bd_mgr.getContracts();
151 for (auto it = ayContracts->begin(); it != ayContracts->end(); it++)
152 {
153 WTSContractInfo* cInfo = (WTSContractInfo*)(*it);
154 bool isHot = _hot_mgr.isHot(cInfo->getExchg(), cInfo->getCode());
155 bool isSecond = _hot_mgr.isSecond(cInfo->getExchg(), cInfo->getCode());
156
157 std::string hotCode = cInfo->getFullPid();
158 if (isHot)
159 hotCode += ".HOT";
160 else if (isSecond)
161 hotCode += ".2ND";
162 else
163 hotCode = "";
164
165 cInfo->setHotFlag(isHot ? 1 : (isSecond ? 2 : 0), hotCode.c_str());
166 }
167 ayContracts->release();
168
169 if (cfgBF->has("rules"))
170 {
171 auto cfgRules = cfgBF->get("rules");
172 auto tags = cfgRules->memberNames();
173 for (const std::string& ruleTag : tags)
174 {
175 _hot_mgr.loadCustomRules(ruleTag.c_str(), cfgRules->getCString(ruleTag.c_str()));
176 WTSLogger::info("{} rules loaded from {}", ruleTag, cfgRules->getCString(ruleTag.c_str()));
177 }
178 }
179
180 //初始化运行环境
181 initEngine();
182
183 //初始化数据管理
184 initDataMgr();
185
186 if (!initActionPolicy())
187 return false;
188
189 //初始化行情通道
190 WTSVariant* cfgParser = _config->get("parsers");
191 if (cfgParser)
192 {
193 if (cfgParser->type() == WTSVariant::VT_String)
194 {
195 const char* filename = cfgParser->asCString();
196 if (StdFile::exists(filename))
197 {
198 WTSLogger::info("Reading parser config from {}...", filename);
199 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
200 if(var)
201 {
202 if (!initParsers(var->get("parsers")))
203 WTSLogger::error("Loading parsers failed");
204 var->release();
205 }
206 else
207 {
208 WTSLogger::error("Loading parser config {} failed", filename);
209 }
210 }
211 else
212 {
213 WTSLogger::error("Parser configuration {} not exists", filename);
214 }
215 }
216 else if (cfgParser->type() == WTSVariant::VT_Array)
217 {
218 initParsers(cfgParser);
219 }
220 }
221
222 //初始化交易通道
223 WTSVariant* cfgTraders = _config->get("traders");
224 if (cfgTraders)
225 {
226 if (cfgTraders->type() == WTSVariant::VT_String)
227 {
228 const char* filename = cfgTraders->asCString();
229 if (StdFile::exists(filename))
230 {
231 WTSLogger::info("Reading trader config from {}...", filename);
232 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
233 if (var)
234 {
235 if (!initTraders(var->get("traders")))
236 WTSLogger::error("Loading traders failed");
237 var->release();
238 }
239 else
240 {
241 WTSLogger::error("Loading trader config {} failed", filename);
242 }
243 }
244 else
245 {
246 WTSLogger::error("Trader configuration {} not exists", filename);
247 }
248 }
249 else if (cfgTraders->type() == WTSVariant::VT_Array)
250 {
251 initTraders(cfgTraders);
252 }
253 }
254
255 initEvtNotifier();
256
257 //如果不是高频引擎,则需要配置执行模块
258 if (!_is_hft)
259 {
260 WTSVariant* cfgExec = _config->get("executers");
261 if (cfgExec != NULL)
262 {
263 if (cfgExec->type() == WTSVariant::VT_String)
264 {
265 const char* filename = cfgExec->asCString();
266 if (StdFile::exists(filename))
267 {
268 WTSLogger::info("Reading executer config from {}...", filename);
269 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
270 if (var)
271 {
272 if (!initExecuters(var->get("executers")))
273 WTSLogger::error("Loading executers failed");
274
275 WTSVariant* c = var->get("routers");
276 if (c != NULL)
277 _cta_engine.loadRouterRules(c);
278
279 var->release();
280 }
281 else
282 {
283 WTSLogger::error("Loading executer config {} failed", filename);
284 }
285 }
286 else
287 {
288 WTSLogger::error("Trader configuration {} not exists", filename);
289 }
290 }
291 else if (cfgExec->type() == WTSVariant::VT_Array)
292 {
293 initExecuters(cfgExec);
294 }
295 }
296
297 WTSVariant* cfgRouter = _config->get("routers");
298 if (cfgRouter != NULL)
299 _cta_engine.loadRouterRules(cfgRouter);
300 }
301
302 if (!_is_hft)
303 initCtaStrategies();
304 else
305 initHftStrategies();
306
307 return true;
308}

315bool WtRunner::initCtaStrategies()
316{
317 WTSVariant* cfg = _config->get("strategies");
318 if (cfg == NULL || cfg->type() != WTSVariant::VT_Object)
319 return false;
320
321 cfg = cfg->get("cta");
322 if (cfg == NULL || cfg->type() != WTSVariant::VT_Array)
323 return false;
324
325 std::string path = WtHelper::getCWD() + "cta/";
326 _cta_stra_mgr.loadFactories(path.c_str());
327
328 for (uint32_t idx = 0; idx < cfg->size(); idx++)
329 {
330 WTSVariant* cfgItem = cfg->get(idx);
331 if (!cfgItem->getBoolean("active"))
332 continue;
333
334 const char* id = cfgItem->getCString("id");
335 const char* name = cfgItem->getCString("name");
336 int32_t slippage = cfgItem->getInt32("slippage");
337 CtaStrategyPtr stra = _cta_stra_mgr.createStrategy(name, id);
338 stra->self()->init(cfgItem->get("params"));
339 CtaStraContext* ctx = new CtaStraContext(&_cta_engine, id, slippage);
340 ctx->set_strategy(stra->self());
341 _cta_engine.addContext(CtaContextPtr(ctx));
342 }
343
344 return true;
345}

352bool WtRunner::initHftStrategies()
353{
354 WTSVariant* cfg = _config->get("strategies");
355 if (cfg == NULL || cfg->type() != WTSVariant::VT_Object)
356 return false;
357
358 cfg = cfg->get("hft");
359 if (cfg == NULL || cfg->type() != WTSVariant::VT_Array)
360 return false;
361
362 std::string path = WtHelper::getCWD() + "hft/";
363 _hft_stra_mgr.loadFactories(path.c_str());
364
365 for (uint32_t idx = 0; idx < cfg->size(); idx++)
366 {
367 WTSVariant* cfgItem = cfg->get(idx);
368 if (!cfgItem->getBoolean("active"))
369 continue;
370
371 const char* id = cfgItem->getCString("id");
372 const char* name = cfgItem->getCString("name");
373 bool agent = cfgItem->getBoolean("agent");
374 int32_t slippage = cfgItem->getInt32("slippage");
375 HftStrategyPtr stra = _hft_stra_mgr.createStrategy(name, id);
376 if (stra == NULL)
377 continue;
378
379 stra->self()->init(cfgItem->get("params"));
380 HftStraContext* ctx = new HftStraContext(&_hft_engine, id, agent,slippage);
381 ctx->set_strategy(stra->self());
382
383 const char* traderid = cfgItem->getCString("trader");
384 TraderAdapterPtr trader = _traders.getAdapter(traderid);
385 if(trader)
386 {
387 ctx->setTrader(trader.get());
388 trader->addSink(ctx);
389 }
390 else
391 {
392 WTSLogger::error("Trader {} not exists, binding trader to HFT strategy failed", traderid);
393 }
394
395 _hft_engine.addContext(HftContextPtr(ctx));
396 }
397
398 return true;

407bool WtRunner::initEngine()
408{
409 WTSVariant* cfg = _config->get("env");
410 if (cfg == NULL)
411 return false;
412
413 const char* name = cfg->getCString("name");
414
415 if (strlen(name) == 0 || wt_stricmp(name, "cta") == 0)
416 {
417 _is_hft = false;
418 _is_sel = false;
419 }
420 else if (wt_stricmp(name, "sel") == 0)
421 {
422 _is_sel = true;
423 }
424 else //if (wt_stricmp(name, "hft") == 0)
425 {
426 _is_hft = true;
427 }
428
429 if (_is_hft)
430 {
431 WTSLogger::info("Trading enviroment initialzied with engine: HFT");
432 _hft_engine.init(cfg, &_bd_mgr, &_data_mgr, &_hot_mgr, &_notifier);
433 _engine = &_hft_engine;
434 }
435 else if (_is_sel)
436 {
437 WTSLogger::info("Trading enviroment initialzied with engine: SEL");
438 _sel_engine.init(cfg, &_bd_mgr, &_data_mgr, &_hot_mgr, &_notifier);
439 _engine = &_sel_engine;
440 }
441 else
442 {
443 WTSLogger::info("Trading enviroment initialzied with engine: CTA");
444 _cta_engine.init(cfg, &_bd_mgr, &_data_mgr, &_hot_mgr, &_notifier);
445 _engine = &_cta_engine;
446 }
447
448 _engine->set_adapter_mgr(&_traders);
449
450 return true;
451}

525bool WtRunner::initExecuters(WTSVariant* cfgExecuter)
526{
527 if (cfgExecuter == NULL || cfgExecuter->type() != WTSVariant::VT_Array)
528 return false;
529
530 std::string path = WtHelper::getCWD() + "executer/";
531 _exe_factory.loadFactories(path.c_str());
532
533 uint32_t count = 0;
534 for (uint32_t idx = 0; idx < cfgExecuter->size(); idx++)
535 {
536 WTSVariant* cfgItem = cfgExecuter->get(idx);
537 if (!cfgItem->getBoolean("active"))
538 continue;
539
540 const char* id = cfgItem->getCString("id");
541 std::string name = cfgItem->getCString("name"); //local,diff,dist
542 if (name.empty())
543 name = "local";
544
545 if (name == "local")
546 {
547 WtLocalExecuter* executer = new WtLocalExecuter(&_exe_factory, id, &_data_mgr);
548 if (!executer->init(cfgItem))
549 return false;
550
551 const char* tid = cfgItem->getCString("trader");
552 if (strlen(tid) == 0)
553 {
554 WTSLogger::error("No Trader configured for Executer {}", id);
555 }
556 else
557 {
558 TraderAdapterPtr trader = _traders.getAdapter(tid);
559 if (trader)
560 {
561 executer->setTrader(trader.get());
562 trader->addSink(executer);
563 }
564 else
565 {
566 WTSLogger::error("Trader {} not exists, cannot configured for executer %s", tid, id);
567 }
568 }
569
570 _cta_engine.addExecuter(ExecCmdPtr(executer));
571 }
572 else if (name == "diff")
573 {
574 WtDiffExecuter* executer = new WtDiffExecuter(&_exe_factory, id, &_data_mgr, &_bd_mgr);
575 if (!executer->init(cfgItem))
576 return false;
577
578 const char* tid = cfgItem->getCString("trader");
579 if (strlen(tid) == 0)
580 {
581 WTSLogger::error("No Trader configured for Executer {}", id);
582 }
583 else
584 {
585 TraderAdapterPtr trader = _traders.getAdapter(tid);
586 if (trader)
587 {
588 executer->setTrader(trader.get());
589 trader->addSink(executer);
590 }
591 else
592 {
593 WTSLogger::error("Trader {} not exists, cannot configured for executer %s", tid, id);
594 }
595 }
596
597 _cta_engine.addExecuter(ExecCmdPtr(executer));
598 }
599 else
600 {
601 WtDistExecuter* executer = new WtDistExecuter(id);
602 if (!executer->init(cfgItem))
603 return false;
604
605 _cta_engine.addExecuter(ExecCmdPtr(executer));
606 }
607 count++;
608 }
609
610 WTSLogger::info("{} executers loaded", count);
611
612 return true;
613}
