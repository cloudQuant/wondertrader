95void UftStraContext::on_tick(const char* stdCode, WTSTickData* newTick)
96{
97 auto it = _positions.find(stdCode);
98 if(it != _positions.end())
99 {
100 WTSCommodityInfo* commInfo = newTick->getContractInfo()->getCommInfo();
101 uint32_t volscale = commInfo->getVolScale();
102 PosInfo& pInfo = it->second;
103
104 for(auto i = pInfo._valid_idx; i < pInfo._details.size(); i++)
105 {
106 auto& ds = pInfo._details[i];
107 if (ds->_volume == 0)
108 ds->_position_profit = 0;
109 else
110 ds->_position_profit = (newTick->price() - ds->_open_price)*ds->_volume*volscale*(ds->_direct == 0 ? 1 : -1);
111 }
112
113 if (decimal::gt(pInfo._volume, 0.0))
114 pInfo._dynprofit = newTick->price()*pInfo._volume*volscale - pInfo._opencost;
115 else if (decimal::lt(pInfo._volume, 0.0))
116 pInfo._dynprofit = newTick->price()*pInfo._volume*volscale + pInfo._opencost;
117 else
118 pInfo._dynprofit = 0;
119 }
120
121 if (_strategy)
122 _strategy->on_tick(this, stdCode, newTick);
123}

690 os._left = leftQty;

694 os._state = 2;

696 os._state = leftQty == 0 ? 1 : 0;
697

699 }

701 if (_strategy)
