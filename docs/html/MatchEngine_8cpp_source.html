75{
76 uint64_t curTime = (uint64_t)curTick->actiondate() * 1000000000 + curTick->actiontime();
77 uint64_t curUnixTime = TimeUtils::makeTime(curTick->actiondate(), curTick->actiontime());
78
79 for (auto& v : _orders)
80 {
81 uint32_t localid = v.first;
82 OrderInfo& ordInfo = (OrderInfo&)v.second;
83
84 if (ordInfo._state == 9)//要撤单
85 {
86 _sink->handle_order(localid, ordInfo._code, ordInfo._buy, 0, ordInfo._limit, true, ordInfo._time);
87 ordInfo._state = 99;
88
89 to_erase.emplace_back(localid);
90
91 WTSLogger::info("订单{}已撤销, 剩余数量: {}", localid, ordInfo._left*(ordInfo._buy ? 1 : -1));
92 ordInfo._left = 0;
93 continue;
94 }
95
96 if (ordInfo._state != 1 || curTick->volume() == 0)
97 continue;
98
99 if (ordInfo._buy)
100 {
101 double price;
102 double volume;
103
104 //主动订单就按照对手价
105 if (ordInfo._positive)
106 {
107 price = curTick->askprice(0);
108 volume = curTick->askqty(0);
109 }
110 else
111 {
112 price = curTick->price();
113 volume = curTick->volume();
114 }
115
116 if (decimal::le(price, ordInfo._limit))
117 {
118 //如果价格相等,需要先看排队位置,如果价格不等说明已经全部被大单吃掉了
119 if (!ordInfo._positive && decimal::eq(price, ordInfo._limit))
120 {
121 double& quepos = ordInfo._queue;
122
123 //如果成交量小于排队位置,则不能成交
124 if (volume <= quepos)
125 {
126 quepos -= volume;
127 continue;
128 }
129 else if (quepos != 0)
130 {
131 //如果成交量大于排队位置,则可以成交
132 volume -= quepos;
133 quepos = 0;
134 }
135 }
136 else if (!ordInfo._positive)
137 {
138 volume = ordInfo._left;
139 }
140
141 double qty = min(volume, ordInfo._left);
142 if (decimal::eq(qty, 0.0))
143 qty = 1;
144
145 _sink->handle_trade(localid, ordInfo._code, ordInfo._buy, qty, ordInfo._price, price, ordInfo._time);
146
147 ordInfo._traded += qty;
148 ordInfo._left -= qty;
149
150 _sink->handle_order(localid, ordInfo._code, ordInfo._buy, ordInfo._left, price, false, ordInfo._time);
151
152 if (ordInfo._left == 0)
153 to_erase.emplace_back(localid);
154 }
155 }
156
157 if (!ordInfo._buy)
158 {
159 double price;
160 double volume;
161
162 //主动订单就按照对手价
163 if (ordInfo._positive)
164 {
165 price = curTick->bidprice(0);
166 volume = curTick->bidqty(0);
167 }

223void MatchEngine::update_lob(WTSTickData* curTick)
224{
225 LmtOrdBook& curBook = _lmt_ord_books[curTick->code()];
226 curBook._cur_px = PRICE_DOUBLE_TO_INT(curTick->price());
227 curBook._ask_px = PRICE_DOUBLE_TO_INT(curTick->askprice(0));
228 curBook._bid_px = PRICE_DOUBLE_TO_INT(curTick->bidprice(0));
229
230 for (uint32_t i = 0; i < 10; i++)
231 {
232 if (PRICE_DOUBLE_TO_INT(curTick->askprice(i)) == 0 && PRICE_DOUBLE_TO_INT(curTick->bidprice(i)) == 0)
233 break;
234
235 uint32_t px = PRICE_DOUBLE_TO_INT(curTick->askprice(i));
236 if (px != 0)
237 {
238 double& volume = curBook._items[px];
239 volume = curTick->askqty(i);
240 }
241
242 px = PRICE_DOUBLE_TO_INT(curTick->bidprice(i));
243 if (px != 0)
244 {
245 double& volume = curBook._items[px];
246 volume = curTick->askqty(i);
247 }
248 }
249
250 //卖一和买一之间的报价必须全部清除掉
251 if (!curBook._items.empty())
252 {
253 auto sit = curBook._items.lower_bound(curBook._bid_px);
254 if (sit->first == curBook._bid_px)
255 sit++;
256
257 auto eit = curBook._items.lower_bound(curBook._ask_px);
258
259 if (sit->first <= eit->first)
260 curBook._items.erase(sit, eit);
261 }
262}

275OrderIDs MatchEngine::buy(const char* stdCode, double price, double qty, uint64_t curTime)
276{
277 WTSTickData* lastTick = grab_last_tick(stdCode);
278 if (lastTick == NULL)
279 return OrderIDs();
280
281 uint32_t localid = makeLocalOrderID();
282 OrderInfo& ordInfo = _orders[localid];
283 strcpy(ordInfo._code, stdCode);
284 ordInfo._buy = true;
285 ordInfo._limit = price;
286 ordInfo._qty = qty;
287 ordInfo._left = qty;
288 ordInfo._price = lastTick->price();
289
290 //订单排队,如果是对手价,则按照对手价的挂单量来排队
291 //如果是最新价,则按照买一卖一的加权平均
292 if (decimal::ge(price, lastTick->askprice(0)))
293 ordInfo._positive = true;
294 else if (decimal::eq(price, lastTick->bidprice(0)))
295 ordInfo._queue = lastTick->bidqty(0);
296 if (decimal::eq(price, lastTick->price()))
297 ordInfo._queue = (uint32_t)round((lastTick->askqty(0)*lastTick->askprice(0) + lastTick->bidqty(0)*lastTick->bidprice(0)) / (lastTick->askprice(0) + lastTick->bidprice(0)));
298
299 //排队位置按照平均撤单率,撤销掉部分
300 ordInfo._queue -= (uint32_t)round(ordInfo._queue*_cancelrate);
301 ordInfo._time = curTime;
302
303 lastTick->release();
304
305 OrderIDs ret;
306 ret.emplace_back(localid);
307 return ret;
308}

364OrderIDs MatchEngine::cancel(const char* stdCode, bool isBuy, double qty, FuncCancelCallback cb)
365{
366 OrderIDs ret;
367 for (auto& v : _orders)
368 {
369 OrderInfo& ordInfo = (OrderInfo&)v.second;
370 if (ordInfo._state != 1)
371 continue;
372
373 double left = qty;
374 if (ordInfo._buy == isBuy)
375 {
376 uint32_t localid = v.first;
377 ret.emplace_back(localid);
378 ordInfo._state = 9;
379 cb(ordInfo._left*(ordInfo._buy ? 1 : -1));
380
381 if (qty != 0)
382 {
383 if ((int32_t)left <= ordInfo._left)
384 break;
385
386 left -= ordInfo._left;
387 }
388 }
389 }
390
391 return ret;

420void MatchEngine::handle_tick(const char* stdCode, WTSTickData* curTick)
421{
422 if (NULL == curTick)
423 return;
424
425 if (NULL == _tick_cache)
426 _tick_cache = WTSTickCache::create();
427
428 _tick_cache->add(stdCode, curTick, true);
429
430 update_lob(curTick);
431
432 OrderIDs to_erase;
433 //检查订单状态
434 fire_orders(stdCode, to_erase);
435
436 //撮合
437 match_orders(curTick, to_erase);
438
439 for (uint32_t localid : to_erase)
440 {
441 auto it = _orders.find(localid);
442 if (it != _orders.end())
443 _orders.erase(it);
444 }
