200 }

205 {

244 for (const std::string& item : ayVals)

246 WTSTickStruct* curTick = (WTSTickStruct*)item.data();

248 }
249 });

254 tickList._last_tick_time = (uint64_t)curTs.trading_date * 1000000000 + sInfo->offsetTime(curTs.action_time / 100000, true) + curTs.action_time % 100000;

265 sTick.action_date = lDate;

282 pTick--;
283 eIdx--;
284 }
285

291 }

481WtRdmDtReaderAD::WtLMDBPtr WtRdmDtReaderAD::get_k_db(const char* exchg, WTSKlinePeriod period)
482{
483 WtLMDBMap* the_map = NULL;
484 std::string subdir;
485 if (period == KP_Minute1)
486 {
487 the_map = &_exchg_m1_dbs;
488 subdir = "min1";
489 }
490 else if (period == KP_Minute5)
491 {
492 the_map = &_exchg_m5_dbs;
493 subdir = "min5";
494 }
495 else if (period == KP_DAY)
496 {
497 the_map = &_exchg_d1_dbs;
498 subdir = "day";
499 }
500 else
501 return std::move(WtLMDBPtr());
502
503 auto it = the_map->find(exchg);
504 if (it != the_map->end())
505 return std::move(it->second);
506
507 WtLMDBPtr dbPtr(new WtLMDB(true));
508 std::string path = StrUtil::printf("%s%s/%s/", _base_dir.c_str(), subdir.c_str(), exchg);
509 boost::filesystem::create_directories(path);
510 if (!dbPtr->open(path.c_str()))
511 {
512 pipe_rdmreader_log(_sink, LL_ERROR, "Opening {} db if {} failed: {}", subdir, exchg, dbPtr->errmsg());
513 return std::move(WtLMDBPtr());
514 }
515 else
516 {
517 pipe_rdmreader_log(_sink, LL_DEBUG, "{} db of {} opened", subdir, exchg);
518 }
519
520 (*the_map)[exchg] = dbPtr;
521 return std::move(dbPtr);
522}
