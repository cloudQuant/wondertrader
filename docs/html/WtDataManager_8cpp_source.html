91 if (libParser)

108 _remover = pFuncDeleteReader;
109 }
110
111 }
112 else
113 {
114 WTSLogger::error("Initializing of random data reader failed: loading module {} failed...", module);
115
116 }

370WTSKlineSlice* WtDataManager::get_kline_slice_by_range(const char* stdCode, WTSKlinePeriod period, uint32_t times,uint64_t stime, uint64_t etime /* = 0 */)
371{
372 if (times == 1)
373 {
374 return _reader->readKlineSliceByRange(stdCode, period, stime, etime);
375 }
376
377 //只有非基础周期的会进到下面的步骤
378 WTSSessionInfo* sInfo = get_session_info(stdCode, true);
379 std::string key = StrUtil::printf("%s-%u-%u", stdCode, period, times);
380 BarCache& barCache = _bars_cache[key];
381 barCache._period = period;
382 barCache._times = times;
383 if(barCache._bars == NULL)
384 {
385 //第一次将全部数据缓存到内存中
386 WTSKlineSlice* rawData = _reader->readKlineSliceByCount(stdCode, period, UINT_MAX, 0);
387 if (rawData != NULL)
388 {
389 WTSKlineData* kData = g_dataFact.extractKlineData(rawData, period, times, sInfo, false);
390 barCache._bars = kData;
391
392 //不管如何，都删除最后一条K线
393 //不能通过闭合标记判断，因为读取的基础周期可能本身没有闭合
394 if (barCache._bars->size() > 0)
395 {
396 auto& bars = barCache._bars->getDataRef();
397 bars.erase(bars.begin() + bars.size() - 1, bars.end());
398 }
399
400 if (period == KP_DAY)
401 barCache._last_bartime = kData->date(-1);
402 else
403 {
404 uint64_t lasttime = kData->time(-1);
405 barCache._last_bartime = 199000000000 + lasttime;
406 }
407
408 rawData->release();
409 }
410 else
411 {
412 return NULL;
413 }
414 }
415 else
416 {
417 //后面则增量更新
418 WTSKlineSlice* rawData = _reader->readKlineSliceByRange(stdCode, period, barCache._last_bartime, 0);
419 if (rawData != NULL)
420 {
421 for(int32_t idx = 0; idx < rawData->size(); idx ++)
422 {
423 uint64_t barTime = 0;
424 if (period == KP_DAY)
425 barTime = rawData->at(0)->date;
426 else
427 barTime = 199000000000 + rawData->at(0)->time;
428
429 //只有时间上次记录的最后一条时间，才可以用于更新K线
430 if(barTime <= barCache._last_bartime)
431 continue;
432
433 g_dataFact.updateKlineData(barCache._bars, rawData->at(idx), sInfo, _align_by_section);
434 }
435
436 //不管如何，都删除最后一条K线
437 //不能通过闭合标记判断，因为读取的基础周期可能本身没有闭合
438 if(barCache._bars->size() > 0)
439 {
440 auto& bars = barCache._bars->getDataRef();
441 bars.erase(bars.begin() + bars.size() - 1, bars.end());
442 }
443
444 if (period == KP_DAY)
445 barCache._last_bartime = barCache._bars->date(-1);
446 else
447 {
448 uint64_t lasttime = barCache._bars->time(-1);
449 barCache._last_bartime = 199000000000 + lasttime;
450 }
451
452
453 rawData->release();
454 }
455 }
456
457 //最后到缓存中定位
458 bool isDay = period == KP_DAY;
459 uint32_t rDate, rTime, lDate, lTime;
460 rDate = (uint32_t)(etime / 10000);
461 rTime = (uint32_t)(etime % 10000);
462 lDate = (uint32_t)(stime / 10000);
463 lTime = (uint32_t)(stime % 10000);
464
465 WTSBarStruct eBar;
466 eBar.date = rDate;
467 eBar.time = (rDate - 19900000) * 10000 + rTime;
468
469 WTSBarStruct sBar;
470 sBar.date = lDate;
471 sBar.time = (lDate - 19900000) * 10000 + lTime;
472
473 uint32_t eIdx, sIdx;
474 auto& bars = barCache._bars->getDataRef();
475 auto eit = std::lower_bound(bars.begin(), bars.end(), eBar, [isDay](const WTSBarStruct& a, const WTSBarStruct& b) {
476 if (isDay)
477 return a.date < b.date;
478 else
479 return a.time < b.time;
480 });
481
482
483 if (eit == bars.end())
484 eIdx = bars.size() - 1;
485 else
486 {
487 if ((isDay && eit->date > eBar.date) || (!isDay && eit->time > eBar.time))
488 {
489 eit--;
490 }
491
492 eIdx = eit - bars.begin();
493 }
494
495 auto sit = std::lower_bound(bars.begin(), eit, sBar, [isDay](const WTSBarStruct& a, const WTSBarStruct& b) {
496 if (isDay)
497 return a.date < b.date;
498 else
499 return a.time < b.time;
500 });
501 sIdx = sit - bars.begin();
502 uint32_t rtCnt = eIdx - sIdx + 1;
503 WTSBarStruct* rtHead = barCache._bars->at(sIdx);
504 WTSKlineSlice* slice = WTSKlineSlice::create(stdCode, period, times, rtHead, rtCnt);
505 return slice;
506}

735void WtDataManager::subscribe_bar(const char* stdCode, WTSKlinePeriod period, uint32_t times)
736{
737 std::string key = fmtutil::format("{}-{}-{}", stdCode, (uint32_t)period, times);
738
739 uint32_t curDate = TimeUtils::getCurDate();
740 uint64_t etime = (uint64_t)curDate * 10000 + 2359;
741
742 if (times == 1)
743 {
744 WTSKlineSlice* slice = _reader->readKlineSliceByCount(stdCode, period, 10, etime);
745 if (slice == NULL)
746 return;
747
748 WTSKlineData* kline = WTSKlineData::create(stdCode, slice->size());
749 kline->setPeriod(period);
750 uint32_t offset = 0;
751 for(uint32_t blkIdx = 0; blkIdx < slice->get_block_counts(); blkIdx++)
752 {
753 memcpy(kline->getDataRef().data() + offset, slice->get_block_addr(blkIdx), sizeof(WTSBarStruct)*slice->get_block_size(blkIdx));
754 offset += slice->get_block_size(blkIdx);
755 }
756
757 {
758 StdUniqueLock lock(_mtx_rtbars);
759 if (_rt_bars == NULL)
760 _rt_bars = RtBarMap::create();
761
762 _rt_bars->add(key, kline, false);
763 }
764
765 slice->release();
766 }
767 else
768 {
769 //只有非基础周期的会进到下面的步骤
770 WTSSessionInfo* sInfo = get_session_info(stdCode, true);
771 WTSKlineSlice* rawData = _reader->readKlineSliceByCount(stdCode, period, 10*times, 0);
772 if (rawData != NULL)
773 {
774 WTSKlineData* kData = g_dataFact.extractKlineData(rawData, period, times, sInfo, true);
775 {
776 StdUniqueLock lock(_mtx_rtbars);
777 if (_rt_bars == NULL)
778 _rt_bars = RtBarMap::create();
779 _rt_bars->add(key, kData, false);
780 }
781 rawData->release();
782 }
783 }
784
785 WTSLogger::info("Realtime bar {} has subscribed", key);
786}

818void WtDataManager::update_bars(const char* stdCode, WTSTickData* newTick)
819{
820 if (_rt_bars == NULL)
821 return;
822
823 StdUniqueLock lock(_mtx_rtbars);
824 auto it = _rt_bars->begin();
825 for(; it != _rt_bars->end(); it++)
826 {
827 WTSKlineData* kData = (WTSKlineData*)it->second;
828 if (strcmp(kData->code(), stdCode) != 0)
829 continue;
830
831 WTSSessionInfo* sInfo = NULL;
832 if (newTick->getContractInfo())
833 sInfo = newTick->getContractInfo()->getCommInfo()->getSessionInfo();
834 else
835 sInfo = get_session_info(kData->code(), true);
836 g_dataFact.updateKlineData(kData, newTick, sInfo, _align_by_section);
837 WTSBarStruct* lastBar = kData->at(-1);
838
839 std::string speriod;
840 uint32_t times = kData->times();
841 switch (kData->period())
842 {
843 case KP_Minute1:
844 speriod = fmtutil::format("m{}", times);
845 break;
846 case KP_Minute5:
847 speriod = fmtutil::format("m{}", times*5);
848 break;
849 default:
850 speriod = fmtutil::format("d{}", times);
851 break;
852 }
853
854 _runner->trigger_bar(stdCode, speriod.c_str(), lastBar);
855 }
