113void WtDiffExecuter::load_data()
114{
115 //读取执行器的理论部位，以及待执行的差量
116 std::string filename = WtHelper::getExecDataDir();
117 filename += _name + ".json";
118
119 if (!StdFile::exists(filename.c_str()))
120 {
121 return;
122 }
123
124 std::string content;
125 StdFile::read_file_content(filename.c_str(), content);
126 if (content.empty())
127 return;
128
129 rj::Document root;
130 root.Parse(content.c_str());
131
132 if (root.HasParseError())
133 return;
134
135 if(root.HasMember("targets"))
136 {
137 const rj::Value& jTargets = root["targets"];
138 for (const rj::Value& jItem : jTargets.GetArray())
139 {
140 const char* stdCode = jItem["code"].GetString();
141 CodeHelper::CodeInfo cInfo = CodeHelper::extractStdCode(stdCode, NULL);
142 WTSContractInfo* ct = _bd_mgr->getContract(cInfo._code, cInfo._exchg);
143 if (ct == NULL)
144 {
145 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "[{}] Ticker {} is not valid", _name, stdCode);
146 continue;
147 }
148
149 double pos = jItem["target"].GetDouble();
150 _target_pos[stdCode] = pos;
151 }
152 }
153
154 if (root.HasMember("diffs"))
155 {
156 const rj::Value& jDiffs = root["diffs"];
157 for (const rj::Value& jItem : jDiffs.GetArray())
158 {
159 const char* stdCode = jItem["code"].GetString();
160 CodeHelper::CodeInfo cInfo = CodeHelper::extractStdCode(stdCode, NULL);
161 WTSContractInfo* ct = _bd_mgr->getContract(cInfo._code, cInfo._exchg);
162 if (ct == NULL)
163 {
164 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "[{}] Ticker {} is not valid", _name, stdCode);
165 continue;
166 }
167
168 double pos = jItem["diff"].GetDouble();
169 _diff_pos[stdCode] = pos;
170 }
171 }
172}

180void WtDiffExecuter::save_data()
181{
182 std::string filename = WtHelper::getExecDataDir();
183 filename += _name + ".json";
184
185 rj::Document root(rj::kObjectType);
186 rj::Document::AllocatorType &allocator = root.GetAllocator();
187
188 {//目标持仓数据保存
189 rj::Value jTarget(rj::kArrayType);
190
191 for (auto& v : _target_pos)
192 {
193 rj::Value jItem(rj::kObjectType);
194 jItem.AddMember("code", rj::Value(v.first.c_str(), allocator), allocator);
195 jItem.AddMember("target", v.second, allocator);
196
197 jTarget.PushBack(jItem, allocator);
198 }
199
200 root.AddMember("targets", jTarget, allocator);
201 }
202
203 {//差量持仓数据保存
204 rj::Value jDiff(rj::kArrayType);
205
206 for (auto& v : _diff_pos)
207 {
208 rj::Value jItem(rj::kObjectType);
209 jItem.AddMember("code", rj::Value(v.first.c_str(), allocator), allocator);
210 jItem.AddMember("diff", v.second, allocator);
211
212 jDiff.PushBack(jItem, allocator);
213 }
214
215 root.AddMember("diffs", jDiff, allocator);
216 }
217
218 {
219 std::string filename = WtHelper::getExecDataDir();
220 filename += _name + ".json";
221
222 BoostFile bf;
223 if (bf.create_new_file(filename.c_str()))
224 {
225 rj::StringBuffer sb;
226 rj::PrettyWriter<rj::StringBuffer> writer(sb);
227 root.Accept(writer);
228 bf.write_file(sb.GetString());
229 bf.close_file();
230 }
231 }

243ExecuteUnitPtr WtDiffExecuter::getUnit(const char* stdCode, bool bAutoCreate /* = true */)
244{
245 CodeHelper::CodeInfo codeInfo = CodeHelper::extractStdCode(stdCode, NULL);
246 std::string commID = codeInfo.stdCommID();
247
248 WTSVariant* policy = _config->get("policy");
249 std::string des = commID;
250 if (!policy->has(commID.c_str()))
251 des = "default";
252
253 //SpinLock lock(_mtx_units);
254
255 auto it = _unit_map.find(stdCode);
256 if(it != _unit_map.end())
257 {
258 return it->second;
259 }
260
261 if (bAutoCreate)
262 {
263 WTSVariant* cfg = policy->get(des.c_str());
264
265 const char* name = cfg->getCString("name");
266 ExecuteUnitPtr unit = _factory->createDiffExeUnit(name);
267 if (unit != NULL)
268 {
269 _unit_map[stdCode] = unit;
270 unit->self()->init(this, stdCode, cfg);
271
272 //如果通道已经就绪，则直接通知执行单元
273 if (_channel_ready)
274 unit->self()->on_channel_ready();
275 }
276 else
277 {
278 WTSLogger::error("Creating ExecUnit {} failed", name);
279 }
280 return unit;
281 }
282 else
283 {
284 return ExecuteUnitPtr();
285 }

501void WtDiffExecuter::on_position_changed(const char* stdCode, double diffPos)
502{
503 ExecuteUnitPtr unit = getUnit(stdCode, true);
504 if (unit == NULL)
505 return;
506
507 //如果差量为0，则直接返回
508 if (decimal::eq(diffPos, 0))
509 return;
510
511 diffPos = round(diffPos*_scale);
512
513 double oldVol = _target_pos[stdCode];
514 double& targetPos = _target_pos[stdCode];
515 targetPos += diffPos;
516
517 /*
518 * By Sunseeeeeker @ 2023.01.10
519 * 更新差量
520 */
521 double& thisDiff = _diff_pos[stdCode];
522 double prevDiff = thisDiff;
523 thisDiff += diffPos;
524
525 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "[{}] Target position of {} changed additonally: {} -> {}, diff postion changed: {} -> {}", _name, stdCode, oldVol, targetPos, prevDiff, thisDiff);
526
527 if (_trader && !_trader->checkOrderLimits(stdCode))
528 {
529 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "[{}] {} is disabled", _name, stdCode);
530 return;
531 }
532
533 //TODO 差量执行还要再看一下
534 if (_pool)
535 {
536 std::string code = stdCode;
537 _pool->schedule([unit, code, thisDiff]() {
538 unit->self()->set_position(code.c_str(), thisDiff);
539 });
540 }
541 else
542 {
543 unit->self()->set_position(stdCode, thisDiff);
544 }

554void WtDiffExecuter::set_position(const wt_hashmap<std::string, double>& targets)
555{
556 for (auto it = targets.begin(); it != targets.end(); it++)
557 {
558 const char* stdCode = it->first.c_str();
559 double newVol = it->second;
560 ExecuteUnitPtr unit = getUnit(stdCode);
561 if (unit == NULL)
562 continue;
563
564 newVol = round(newVol*_scale);
565 double oldVol = _target_pos[stdCode];
566 _target_pos[stdCode] = newVol;
567 if (decimal::eq(oldVol, newVol))
568 continue;
569
570 //差量更新
571 double& thisDiff = _diff_pos[stdCode];
572 double prevDiff = thisDiff;
573 thisDiff += (newVol - oldVol);
574
575 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "[{}] Target position of {} changed: {} -> {}, diff postion changed: {} -> {}", _name, stdCode, oldVol, newVol, prevDiff, thisDiff);
576
577 if (_trader && !_trader->checkOrderLimits(stdCode))
578 {
579 WTSLogger::log_dyn("executer", _name.c_str(), LL_WARN, "[{}] {} is disabled due to entrust limit control ", _name, stdCode);
580 continue;
581 }
582
583 //TODO 差量执行还要再看一下
584 if (_pool)
585 {
586 std::string code = stdCode;
587 _pool->schedule([unit, code, thisDiff](){
588 unit->self()->set_position(code.c_str(), thisDiff);
589 });
590 }
591 else
592 {
593 unit->self()->set_position(stdCode, thisDiff);
594 }
595 }
596
597 //在原来的目标头寸中，但是不在新的目标头寸中，则需要自动设置为0
598 for (auto it = _target_pos.begin(); it != _target_pos.end(); it++)
599 {
600 const char* stdCode = it->first.c_str();
601 double& pos = (double&)it->second;
602 auto tit = targets.find(stdCode);
603 if(tit != targets.end())
604 continue;
605
606 WTSContractInfo* cInfo = _bd_mgr->getContract(stdCode);
607 if(cInfo == NULL)
608 continue;
609
610 if(pos != 0)
611 {
612 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "[{}] {} is not in target, set to 0 automatically", _name, stdCode);
613
614 ExecuteUnitPtr unit = getUnit(stdCode);
615 if (unit == NULL)
616 continue;
617
618 //更新差量
619 double& thisDiff = _diff_pos[stdCode];
620 double prevDiff = thisDiff;
621
622 //WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "[DiffExecuter][set_position][{}] {} is not in target, thisDiff: {}, prevDiff: {}, pos: {}, new thisDiff: {}", _name, stdCode, thisDiff, prevDiff, pos, thisDiff + pos);
623
624 thisDiff -= -pos;
625 pos = 0;
626
627 if (_pool)
628 {
629 std::string code = stdCode;
630 _pool->schedule([unit, code, thisDiff]() {
631 unit->self()->set_position(code.c_str(), thisDiff);
632 });
633 }
634 else
635 {
636 unit->self()->set_position(stdCode, thisDiff);
637 }
638 }
639 }
640
641 save_data();
642}

680void WtDiffExecuter::on_trade(uint32_t localid, const char* stdCode, bool isBuy, double vol, double price)
681{
682 ExecuteUnitPtr unit = getUnit(stdCode, false);
683 if (unit == NULL)
684 return;
685
686 if (localid == 0)
687 return;
688
689 //如果localid不为0，则更新差量
690 double& curDiff = _diff_pos[stdCode];
691 double prevDiff = curDiff;
692 curDiff -= vol * (isBuy ? 1 : -1);
693
694 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "[{}] Diff of {} updated by trade: {} -> {}", _name, stdCode, prevDiff, curDiff);
695 save_data();
696
697 if (_pool)
698 {
699 std::string code = stdCode;
700 _pool->schedule([localid, unit, code, isBuy, vol, price]() {
701 unit->self()->on_trade(localid, code.c_str(), isBuy, vol, price);
702 });
703 }
704 else
705 {
706 unit->self()->on_trade(localid, stdCode, isBuy, vol, price);
