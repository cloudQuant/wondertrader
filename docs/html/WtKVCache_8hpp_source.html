183 bool resize(uint32_t newCap, CacheLogger logger = nullptr)
184 {
185 if (_cache._file == NULL)
186 return false;
187
188 //调用该函数之前,应该保证线程安全了
189 CacheBlock* cBlock = _cache._block;
190 if (cBlock->_capacity >= newCap)
191 return _cache._file->addr();
192
193 std::string filename = _cache._file->filename();
194 uint64_t uOldSize = sizeof(CacheBlock) + sizeof(CacheItem)*cBlock->_capacity;
195 uint64_t uNewSize = sizeof(CacheBlock) + sizeof(CacheItem)*newCap;
196 std::string data;
197 data.resize((std::size_t)(uNewSize - uOldSize), 0);
198 try
199 {
200 BoostFile f;
201 f.open_existing_file(filename.c_str());
202 f.seek_to_end();
203 f.write_file(data.c_str(), data.size());
204 f.close_file();
205 }
206 catch (std::exception&)
207 {
208 if (logger) logger("Got an exception while resizing cache file");
209 return false;
210 }
211
212
213 _cache._file.reset();
214 BoostMappingFile* pNewMf = new BoostMappingFile();
215 try
216 {
217 if (!pNewMf->map(filename.c_str()))
218 {
219 delete pNewMf;
220 if (logger) logger("Mapping cache file failed");
221 return false;
222 }
223 }
224 catch (std::exception&)
225 {
226 if (logger) logger("Got an exception while mapping cache file");
227 return false;
228 }
229
230 _cache._file.reset(pNewMf);
231
232 _cache._block = (CacheBlock*)_cache._file->addr();
233 _cache._block->_capacity = newCap;
234 return true;
235 }

248 bool init(const char* filename, uint32_t uDate, CacheLogger logger = nullptr)
249 {
250 bool isNew = false;
251 if (!BoostFile::exists(filename))
252 {
253 uint64_t uSize = sizeof(CacheBlock) + sizeof(CacheItem) * SIZE_STEP;
254 BoostFile bf;
255 bf.create_new_file(filename);
256 bf.truncate_file((uint32_t)uSize);
257 bf.close_file();
258
259 isNew = true;
260 }
261
262 _cache._file.reset(new BoostMappingFile);
263 if (!_cache._file->map(filename))
264 {
265 _cache._file.reset();
266 if (logger) logger("Mapping cache file failed");
267 return false;
268 }
269 _cache._block = (CacheBlock*)_cache._file->addr();
270
271 if (!isNew && _cache._block->_date != uDate)
272 {
273 _cache._block->_size = 0;
274 _cache._block->_date = uDate;
275
276 memset(& _cache._block->_items, 0, sizeof(CacheItem)* _cache._block->_capacity);
277
278 if (logger) logger("Cache file reset due to a different date");
279 }
280
281 if (isNew)
282 {
283 _cache._block->_capacity = SIZE_STEP;
284 _cache._block->_size = 0;
285 _cache._block->_date = uDate;
286 strcpy( _cache._block->_blk_flag, CACHE_FLAG);
287 }
288 else
289 {
290 //检查缓存文件是否有问题,要自动恢复
291 do
292 {
293 uint64_t uSize = sizeof(CacheBlock) + sizeof(CacheItem) * _cache._block->_capacity;
294 uint64_t realSz = _cache._file->size();
295 if (realSz != uSize)
296 {
297 uint32_t realCap = (uint32_t)((realSz - sizeof(CacheBlock)) / sizeof(CacheItem));
298 uint32_t markedCap = _cache._block->_capacity;
299 //文件大小不匹配,一般是因为capacity改了,但是实际没扩容
300 //这是做一次扩容即可
301 _cache._block->_capacity = realCap;
302 _cache._block->_size = (realCap < markedCap) ? realCap : markedCap;
303 }
304
305 } while (false);
306 }
307
308 //这里把索引加到hashmap中
309 for (uint32_t i = 0; i < _cache._block->_size; i++)
310 _indice[_cache._block->_items[i]._key] = i;
311
312 return true;
313 }
