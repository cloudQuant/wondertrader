70bool WtStraDtSel::init(WTSVariant* cfg)
71{
72 // 配置为空则返回失败
73 if (cfg == NULL)
74 return false;
75
76 // 读取双突破算法相关参数
77 _days = cfg->getUInt32("days"); // 计算区间天数
78 _k1 = cfg->getDouble("k1"); // 上边界系数
79 _k2 = cfg->getDouble("k2"); // 下边界系数
80
81 // 读取K线数据相关参数
82 _period = cfg->getCString("period"); // 周期标识
83 _count = cfg->getUInt32("count"); // K线条数
84
85 // 读取是否为股票的标记
86 _isstk = cfg->getBoolean("stock");
87
88 // 通过参数读取并初始化交易代码列表
89 std::string codes = cfg->getCString("codes");
90 // 将逗号分隔的代码字符串分割为列表
91 auto ayCodes = StrUtil::split(codes, ",");
92 // 将每个代码加入代码集合
93 for (auto& code : ayCodes)
94 _codes.insert(code);
95
96 return true;
97}

121void WtStraDtSel::on_schedule(ISelStraCtx* ctx, uint32_t uDate, uint32_t uTime)
122{
123 // 遍历所有监控的交易代码
124 for (auto& curCode : _codes)
125 {
126 // 获取交易品种的交易时间信息
127 WTSSessionInfo* sInfo = ctx->stra_get_sessinfo(curCode.c_str());
128 // 如果当前不在交易时间内，则跳过处理
129 if(!sInfo->isInTradingTime(uTime))
130 continue;
131
132 // 构建标准化代码，股票需要添加"-"
133 std::string code = curCode;
134 if (_isstk)
135 code += "-";
136
137 // 获取K线数据
138 WTSKlineSlice *kline = ctx->stra_get_bars(code.c_str(), _period.c_str(), _count);
139 if (kline == NULL)
140 {
141 //没有获取到K线数据，这里可以输出一些日志
142 return;
143 }
144
145 // 如果K线数据为空，释放资源并返回
146 if (kline->size() == 0)
147 {
148 kline->release();
149 return;
150 }
151
152 // 设置交易单位，股票为100股，其他为1
153 int32_t trdUnit = 1;
154 if (_isstk)
155 trdUnit = 100;
156
157 // 转换天数参数为int32_t类型
158 int32_t days = (int32_t)_days;
159
160 // 计算历史最高价与最低价
161 double hh = kline->maxprice(-days, -2); // 区间内最高价
162 double ll = kline->minprice(-days, -2); // 区间内最低价
163
164 // 提取收盘价数据
165 WTSValueArray* closes = kline->extractData(KFT_CLOSE);
166 double hc = closes->maxvalue(-days, -2); // 区间内最高收盘价
167 double lc = closes->minvalue(-days, -2); // 区间内最低收盘价
168 double curPx = closes->at(-1); // 当前收盘价
169 closes->release(); // 释放收盘价数据数组资源
170
171 // 获取当前交易日的开盘、最高和最低价
172 double openPx = kline->at(-1)->open;
173 double highPx = kline->at(-1)->high;
174 double lowPx = kline->at(-1)->low;
175
176 // 计算双突破上下边界
177 // 上边界 = 开盘价 + K1 * max(最高价-最低收盘价, 最高收盘价-最低价)
178 double upper_bound = openPx + _k1 * (std::max(hh - lc, hc - ll));
179 // 下边界 = 开盘价 - K2 * max(最高价-最低收盘价, 最高收盘价-最低价)
180 double lower_bound = openPx - _k2 * std::max(hh - lc, hc - ll);
181
182 // 获取品种信息
183 WTSCommodityInfo* commInfo = ctx->stra_get_comminfo(curCode.c_str());
184
185 // 获取当前持仓并转换为持仓手数
186 double curPos = ctx->stra_get_position(curCode.c_str()) / trdUnit;
187
188 // 无持仓状态下的入场逻辑
189 if (decimal::eq(curPos, 0))
190 {
191 // 价格突破上边界，进场做多
192 if (highPx >= upper_bound)
193 {
194 ctx->stra_set_position(curCode.c_str(), 1 * trdUnit, "DT_EnterLong");
195 // 记录多仓进场日志
196 ctx->stra_log_info(fmt::format("{} 向上突破{}>={},多仓进场", curCode.c_str(), highPx, upper_bound).c_str());
197 }
198 // 价格突破下边界且非股票，进场做空
199 else if (lowPx <= lower_bound && !_isstk)
200 {
201 ctx->stra_set_position(curCode.c_str(), -1 * trdUnit, "DT_EnterShort");
202 // 记录空仓进场日志
203 ctx->stra_log_info(fmt::format("{} 向下突破{}<={},空仓进场", curCode.c_str(), lowPx, lower_bound).c_str());
204 }
205 }
206 // 多头持仓状态下的出场逻辑
207 else if (decimal::gt(curPos, 0))
208 {
209 // 价格突破下边界，多头出场
210 if (lowPx <= lower_bound)
211 {
212 // 将多头持仓平仓至零
213 ctx->stra_set_position(curCode.c_str(), 0, "DT_ExitLong");
214 // 记录多仓出场日志
215 ctx->stra_log_info(fmt::format("{} 向下突破{}<={},多仓出场", curCode.c_str(), lowPx, lower_bound).c_str());
216 }
217 }
218 // 空头持仓状态下的出场逻辑
219 else if (decimal::lt(curPos, 0))
220 {
221 // 价格突破上边界且非股票，空头出场
222 if (highPx >= upper_bound && !_isstk)
223 {
224 // 将空头持仓平仓至零
225 ctx->stra_set_position(curCode.c_str(), 0, "DT_ExitShort");
226 // 记录空仓出场日志
227 ctx->stra_log_info(fmt::format("{} 向上突破{}>={},空仓出场", curCode.c_str(), highPx, upper_bound).c_str());
228 }
229 }
230
231 // 释放K线数据资源，这个释放一定要做
232 kline->release();
233 }
