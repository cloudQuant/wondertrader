41void WtExecuterMgr::set_positions(wt_hashmap<std::string, double> target_pos)
42{
43 // 如果过滤器管理器存在，则应用过滤规则
44 if(_filter_mgr != NULL)
45 {
46 wt_hashmap<std::string, double> des_port;
47 for(auto& m : target_pos)
48 {
49 const auto& stdCode = m.first;
50 double& desVol = (double&)m.second;
51 double oldVol = desVol;
52
53 // 检查合约是否被过滤，如果被过滤则desVol可能会被修改
54 bool isFltd = _filter_mgr->is_filtered_by_code(stdCode.c_str(), desVol);
55 if (!isFltd)
56 {
57 // 如果目标仓位被过滤器修改，输出日志
58 if (!decimal::eq(desVol, oldVol))
59 {
60 WTSLogger::info("[Filters] {} target position reset by code filter: {} -> {}", stdCode.c_str(), oldVol, desVol);
61 }
62
63 // 将过滤后的仓位添加到新的映射表中
64 des_port[stdCode] = desVol;
65 }
66 else
67 {
68 // 如果合约被完全过滤掉，输出日志
69 WTSLogger::info("[Filters] {} target position ignored by filter", stdCode.c_str());
70 }
71 }
72
73 // 用过滤后的映射表替换原始映射表
74 des_port.swap(target_pos);
75 }
76
77 // 遍历所有执行器，并设置目标仓位
78 for (auto& v : _executers)
79 {
80 ExecCmdPtr& executer = (ExecCmdPtr&)v.second;
81
82 // 检查执行器是否被过滤
83 if (_filter_mgr && _filter_mgr->is_filtered_by_executer(executer->name()))
84 {
85 WTSLogger::info("[Filters] Executer {} is filtered, all signals will be ignored", executer->name());
86 continue;
87 }
88 // 将目标仓位设置到执行器
89 executer->set_position(target_pos);
90 }
91}

101void WtExecuterMgr::handle_pos_change(const char* stdCode, double targetPos, double diffPos, const char* execid /* = "ALL" */)
102{
103 // 如果过滤器管理器存在，则应用过滤规则
104 if(_filter_mgr != NULL)
105 {
106 double oldVol = targetPos;
107 // 检查合约是否被过滤，如果被过滤则targetPos可能会被修改
108 bool isFltd = _filter_mgr->is_filtered_by_code(stdCode, targetPos);
109 if (!isFltd)
110 {
111 // 如果目标仓位被过滤器修改，输出日志并重新计算差量
112 if (!decimal::eq(targetPos, oldVol))
113 {
114 WTSLogger::info("[Filters] {} target position reset by filter: {} -> {}", stdCode, oldVol, targetPos);
115 // 差量也要重算，将过滤器导致的仓位变化添加到差量中
116 diffPos += (targetPos - oldVol);
117 }
118 }
119 else
120 {
121 // 如果合约被完全过滤掉，输出日志并直接返回
122 WTSLogger::info("[Filters] {} target position ignored by filter", stdCode);
123 return;
124 }
125 }
126
127 // 遍历所有执行器
128 for (auto& v : _executers)
129 {
130 ExecCmdPtr& executer = (ExecCmdPtr&)v.second;
131
132 // 检查执行器是否被过滤
133 if (_filter_mgr && _filter_mgr->is_filtered_by_executer(executer->name()))
134 {
135 WTSLogger::info("[Filters] All signals to executer {} are ignored by executer filter", executer->name());
136 continue;
137 }
138
139 // 检查执行器是否在路由表中，如果不在且execid为"ALL"，或者执行器名称与execid相同，则通知仓位变化
140 auto it = _routed_executers.find(executer->name());
141 if (it == _routed_executers.end() && strcmp(execid, "ALL") == 0)
142 executer->on_position_changed(stdCode, diffPos);
143 else if(strcmp(executer->name(), execid) == 0)
144 executer->on_position_changed(stdCode, diffPos);
145 }
146}

193void WtExecuterMgr::commit_cached_targets(double scale /* = 1.0 */)
194{
195 // 遍历所有缓存的目标仓位
196 for(auto& v : _all_cached_targets)
197 {
198 // 先对组合进行缩放
199 const char* execid = v.first.c_str();
200 TargetsMap& target_pos = (TargetsMap&)v.second;
201 for(auto& item : target_pos)
202 {
203 const auto& stdCode = item.first;
204 double& pos = (double&)item.second;
205
206 // 如果仓位为0，则跳过
207 if(decimal::eq(pos, 0))
208 continue;
209
210 // 保留仓位的正负方向，并根据缩放系数调整仓位大小
211 double symbol = pos / abs(pos);
212 pos = decimal::rnd(abs(pos)*scale)*symbol;
213 }
214
215 // 然后根据过滤器调整目标仓位
216 if (_filter_mgr != NULL)
217 {
218 TargetsMap des_port;
219 for (auto& m : target_pos)
220 {
221 const auto& stdCode = m.first;
222 double& desVol = (double&)m.second;
223 double oldVol = desVol;
224
225 // 检查合约是否被过滤，如果被过滤则desVol可能会被修改
226 bool isFltd = _filter_mgr->is_filtered_by_code(stdCode.c_str(), desVol);
227 if (!isFltd)
228 {
229 // 如果目标仓位被过滤器修改，输出日志
230 if (!decimal::eq(desVol, oldVol))
231 {
232 WTSLogger::info("[Filters] {} target position reset by code filter: {} -> {}", stdCode.c_str(), oldVol, desVol);
233 }
234
235 // 将过滤后的仓位添加到新的映射表中
236 des_port[stdCode] = desVol;
237 }
238 else
239 {
240 // 如果合约被完全过滤掉，输出日志
241 WTSLogger::info("[Filters] {} target position ignored by filter", stdCode.c_str());
242 }
243 }
244
245 // 用过滤后的映射表替换原始映射表
246 target_pos.swap(des_port);
247 }
248 }
249
250 // 遍历所有执行器
251 for (auto& e : _executers)
252 {
253 ExecCmdPtr& executer = (ExecCmdPtr&)e.second;
254 // 检查执行器是否被过滤
255 if (_filter_mgr && _filter_mgr->is_filtered_by_executer(executer->name()))
256 {
257 WTSLogger::info("[Filters] Executer {} is filtered, all signals will be ignored", executer->name());
258 continue;
259 }
260
261 // 先找自己对应的组合
262 auto it = _all_cached_targets.find(executer->name());
263
264 // 如果找不到，就找全部组合
265 if (it == _all_cached_targets.end())
266 it = _all_cached_targets.find("ALL");
267
268 // 如果仍然找不到，则跳过当前执行器
269 if (it == _all_cached_targets.end())
270 continue;
271
272 // 将目标仓位设置到执行器
273 executer->set_position(it->second);
274 }
275
276 // 提交完了以后，清理掉全部缓存的目标仓位
277 _all_cached_targets.clear();
278}

286bool WtExecuterMgr::load_router_rules(WTSVariant* config)
287{
288 // 检查配置是否有效
289 if (config == NULL || !config->isArray())
290 return false;
291
292 // 遍历配置数组中的每个路由规则
293 for(uint32_t i = 0; i < config->size(); i++)
294 {
295 WTSVariant* item = config->get(i);
296 // 获取策略名称
297 const char* straName = item->getCString("strategy");
298 // 获取执行器配置
299 WTSVariant* itemExec = item->get("executer");
300 // 如果执行器配置是数组，表示一个策略可以路由到多个执行器
301 if(itemExec->isArray())
302 {
303 uint32_t cnt = itemExec->size();
304 for(uint32_t k = 0; k < cnt; k++)
305 {
306 // 获取执行器ID
307 const char* execId = itemExec->get(k)->asCString();
308 // 将执行器ID添加到策略的路由规则中
309 _router_rules[straName].insert(execId);
310 WTSLogger::info("Signal of strategy {} will be routed to executer {}", straName, execId);
311 // 将执行器ID添加到已路由执行器集合中
312 _routed_executers.insert(execId);
313 }
314 }
315 else
316 {
317 // 如果执行器配置是单个值，表示策略只路由到一个执行器
318 const char* execId = itemExec->asCString();
319 // 将执行器ID添加到策略的路由规则中
320 _router_rules[straName].insert(execId);
321 WTSLogger::info("Signal of strategy {} will be routed to executer {}", straName, execId);
322 // 将执行器ID添加到已路由执行器集合中
323 _routed_executers.insert(execId);
324 }
325 }
326
327 // 输出加载的路由规则数量
328 WTSLogger::info("{} router rules loaded", _router_rules.size());
329
330 return true;
331}
