804void WtBtRunner::config(const char* cfgFile, bool isFile /* = true */)
805{
806 if(_inited)
807 {
808 WTSLogger::error("WtBtEngine has already been inited");
809 return;
810 }
811
812 _cfg = isFile ? WTSCfgLoader::load_from_file(cfgFile) : WTSCfgLoader::load_from_content(cfgFile, false);
813 if(_cfg == NULL)
814 {
815 WTSLogger::error("Loading config failed");
816 return;
817 }
818
819 //初始化事件推送器
820 initEvtNotifier(_cfg->get("notifier"));
821
822 _replayer.init(_cfg->get("replayer"), &_notifier, _ext_fnl_bar_loader != NULL ? this : NULL);
823
824 WTSVariant* cfgEnv = _cfg->get("env");
825 const char* mode = cfgEnv->getCString("mocker");
826 WTSVariant* cfgMode = _cfg->get(mode);
827 if (strcmp(mode, "cta") == 0 && cfgMode)
828 {
829 const char* name = cfgMode->getCString("name");
830 int32_t slippage = cfgMode->getInt32("slippage");
831 _cta_mocker = new ExpCtaMocker(&_replayer, name, slippage, &_notifier);
832 _cta_mocker->init_cta_factory(cfgMode);
833 _replayer.register_sink(_cta_mocker, name);
834 }
835 else if (strcmp(mode, "hft") == 0 && cfgMode)
836 {
837 const char* name = cfgMode->getCString("name");
838 _hft_mocker = new ExpHftMocker(&_replayer, name);
839 _hft_mocker->init_hft_factory(cfgMode);
840 _replayer.register_sink(_hft_mocker, name);
841 }
842 else if (strcmp(mode, "sel") == 0 && cfgMode)
843 {
844 const char* name = cfgMode->getCString("name");
845 int32_t slippage = cfgMode->getInt32("slippage");
846 _sel_mocker = new ExpSelMocker(&_replayer, name, slippage);
847 _sel_mocker->init_sel_factory(cfgMode);
848 _replayer.register_sink(_sel_mocker, name);
849
850 WTSVariant* cfgTask = cfgMode->get("task");
851 if(cfgTask)
852 _replayer.register_task(_sel_mocker->id(), cfgTask->getUInt32("date"), cfgTask->getUInt32("time"),
853 cfgTask->getCString("period"), cfgTask->getCString("trdtpl"), cfgTask->getCString("session"));
854 }
855 else if (strcmp(mode, "exec") == 0 && cfgMode)
856 {
857 const char* name = cfgMode->getCString("name");
858 _exec_mocker = new ExecMocker(&_replayer);
859 _exec_mocker->init(cfgMode);
860 _replayer.register_sink(_exec_mocker, name);
861 }
862}

873void WtBtRunner::run(bool bNeedDump /* = false */, bool bAsync /* = false */)
874{
875 if (_running)
876 return;
877
878 _async = bAsync;
879
880 WTSLogger::info("Backtesting will run in {} mode", _async ? "async" : "sync");
881
882 if (_cta_mocker)
883 _cta_mocker->enable_hook(_async);
884 else if (_hft_mocker)
885 _hft_mocker->enable_hook(_async);
886
887 _replayer.prepare();
888 if (!bAsync)
889 {
890 _replayer.run(bNeedDump);
891 }
892 else
893 {
894 _worker.reset(new StdThread([this, bNeedDump]() {
895 _running = true;
896 try
897 {
898 _replayer.run(bNeedDump);
899 }
900 catch (...)
901 {
902 WTSLogger::error("Exception raised while worker running");
903 //print_stack_trace([](const char* message) {
904 // WTSLogger::error(message);
905 //});
906 }
907 WTSLogger::debug("Worker thread of backtest finished");
908 _running = false;
909
910 }));
911 }

920void WtBtRunner::stop()
921{
922 if (!_running)
923 {
924 if (_worker)
925 {
926 _worker->join();
927 _worker.reset();
928 }
929 return;
930 }
931
932 _replayer.stop();
933
934 WTSLogger::debug("Notify to finish last round");
935
936 if (_cta_mocker)
937 _cta_mocker->step_calc();
938
939 if (_hft_mocker)
940 _hft_mocker->step_tick();
941
942 WTSLogger::debug("Last round ended");
943
944 if (_worker)
945 {
946 _worker->join();
947 _worker.reset();
948 }
949
950 WTSLogger::freeAllDynLoggers();
951
952 WTSLogger::debug("Backtest stopped");
953}
