152 static inline uint64_t read_file_content(const char* filename, std::string& content)
153 {
154 FILE* f = fopen(filename, "rb");
155 if (!f) {
156 // 文件打开失败，抛出异常并显示文件名
157 throw std::runtime_error("Failed to open file: " + std::string(filename));
158 }
159
160 // 获取文件大小
161 fseek(f, 0, SEEK_END);
162 long length = ftell(f);
163 if (length < 0) {
164 fclose(f);
165 // 获取文件大小失败，抛出异常并显示文件名
166 throw std::runtime_error("Failed to get file size for file: " + std::string(filename));
167 }
168 fseek(f, 0, SEEK_SET);
169
170 // 分配缓冲区
171 content.resize(length);
172
173 // 读取文件内容
174 std::size_t readSize = fread((void*)content.data(), sizeof(char), length, f);
175 fclose(f);
176
177 // 检查是否读取成功
178 if (readSize != static_cast<std::size_t>(length)) {
179 // 读取失败，抛出异常并显示文件名
180 throw std::runtime_error("Failed to read file content for file: " + std::string(filename));
181 }
182
183 return length;
184 }
