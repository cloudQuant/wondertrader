64 inline uint32_t strToTime(const char* strTime)
65 {
66 static char str[10] = { 0 };
67 const char *pos = strTime;
68 int idx = 0;
69 auto len = strlen(strTime);
70 // 移除字符串中的冒号
71 for (std::size_t i = 0; i < len; i++)
72 {
73 if (strTime[i] != ':')
74 {
75 str[idx] = strTime[i];
76 idx++;
77 }
78 }
79 str[idx] = '\0';
80
81 // 将字符串转换为整数
82 return strtoul(str, NULL, 10);

347 bool HftLatencyTool::init()
348 {
349 // 初始化日志系统
350 WTSLogger::init("logcfg.yaml");
351
352 // 加载配置文件
353 WTSVariant* _config = WTSCfgLoader::load_from_file("config.yaml");
354 if (_config == NULL)
355 {
356 WTSLogger::log_raw(LL_ERROR, "Loading config file config.yaml failed");
357 return false;
358 }
359
360 //基础数据文件
361 WTSVariant* cfgBF = _config->get("basefiles");
362 bool isUTF8 = cfgBF->getBoolean("utf-8");
363 if (cfgBF->get("session"))
364 _bd_mgr.loadSessions(cfgBF->getCString("session"));
365
366 WTSVariant* cfgItem = cfgBF->get("commodity");
367 if (cfgItem)
368 {
369 if (cfgItem->type() == WTSVariant::VT_String)
370 {
371 _bd_mgr.loadCommodities(cfgItem->asCString());
372 }
373 else if (cfgItem->type() == WTSVariant::VT_Array)
374 {
375 for (uint32_t i = 0; i < cfgItem->size(); i++)
376 {
377 _bd_mgr.loadCommodities(cfgItem->get(i)->asCString());
378 }
379 }
380 }
381
382 cfgItem = cfgBF->get("contract");
383 if (cfgItem)
384 {
385 if (cfgItem->type() == WTSVariant::VT_String)
386 {
387 _bd_mgr.loadContracts(cfgItem->asCString());
388 }
389 else if (cfgItem->type() == WTSVariant::VT_Array)
390 {
391 for (uint32_t i = 0; i < cfgItem->size(); i++)
392 {
393 _bd_mgr.loadContracts(cfgItem->get(i)->asCString());
394 }
395 }
396 }
397
398 if (cfgBF->get("hot"))
399 {
400 _hot_mgr.loadHots(cfgBF->getCString("hot"));
401 WTSLogger::log_raw(LL_INFO, "Hot rules loades");
402 }
403
404 // 初始化行动策略管理器
405 _act_mgr.init("actpolicy.yaml");
406
407 _times = _config->getUInt32("times");
408 WTSLogger::warn("{} ticks will be simulated", _times);
409
410 _core = _config->getUInt32("core");
411 WTSLogger::warn("Testing thread will be bind to core {}", _core);
412
413 initEngine(_config->get("env"));
414 initModules();
415 initStrategies();
416
417 _config->release();
418 return true;
419 }
