95void WtCtaRtTicker::on_tick(WTSTickData* curTick)
96{
97 // 如果定时器线程未启动，直接触发价格事件
98 if (_thrd == NULL)
99 {
100 trigger_price(curTick);
101 return;
102 }
103
104 // 获取行情的日期和时间
105 uint32_t uDate = curTick->actiondate();
106 uint32_t uTime = curTick->actiontime();
107
108 // 如果行情时间早于当前时间，只触发价格事件而不更新时间
109 if (_date != 0 && (uDate < _date || (uDate == _date && uTime < _time)))
110 {
111 //WTSLogger::info("行情时间{}小于本地时间{}", uTime, _time);
112 trigger_price(curTick);
113 return;
114 }
115
116 // 更新当前日期和时间
117 _date = uDate;
118 _time = uTime;
119
120 // 将时间分解为分钟和秒
121 uint32_t curMin = _time / 100000; // 当前分钟，格式为HHMM
122 uint32_t curSec = _time % 100000; // 当前秒数，格式为SS000
123
124
125 // 静态变量用于记录上一次处理的分钟和相关信息
126 static uint32_t prevMin = UINT_MAX; // 上一次处理的分钟
127 static uint32_t minutes = UINT_MAX; // 转换后的分钟数
128 static bool isSecEnd = false; // 是否是交易时段的最后一分钟
129 static uint32_t wrapMin = UINT_MAX; // 包装后的分钟时间
130
131 //By Wesley @ 2023.11.01
132 //如果新的分钟和上一次处理的分钟数不同，才进行处理
133 //否则就不用处理，减少一些开销
134 if(prevMin != curMin)
135 {
136 // 将时间转换为从交易日开始的分钟数
137 minutes = _s_info->timeToMinutes(curMin);
138 // 检查是否是交易时段的最后一分钟
139 isSecEnd = _s_info->isLastOfSection(curMin);
140 prevMin = curMin;
141
142 // 如果是交易时段的最后一分钟，需要特殊处理
143 if (isSecEnd)
144 {
145 minutes--;
146 }
147 minutes++;
148
149 // 将分钟数转换回时间格式
150 wrapMin = _s_info->minuteToTime(minutes);
151 }
152
153 // 如果当前位置是0，说明是第一次收到行情
154 if (_cur_pos == 0)
155 {
156 // 如果当前时间是0, 则直接赋值即可
157 _cur_pos = minutes;
158 }
159 // 如果当前位置小于新的分钟数，说明分钟已经前进
160 else if (_cur_pos < minutes)
161 {
162 // 如果已记录的分钟小于新的分钟, 则需要触发闭合事件
163 // 这个时候要先触发闭合, 再修改平台时间和价格
164 if (_last_emit_pos < _cur_pos)
165 {
166 // 触发数据回放模块
167 StdUniqueLock lock(_mtx);
168
169 // 优先修改时间标记
170 _last_emit_pos = _cur_pos;
171
172 // 将当前位置转换为时间格式
173 uint32_t thisMin = _s_info->minuteToTime(_cur_pos);
174
175 // 检查是否是交易日结束
176 bool bEndingTDate = false;
177 uint32_t offMin = _s_info->offsetTime(thisMin, true);
178 if (offMin == _s_info->getCloseTime(true))
179 bEndingTDate = true;
180
181 // 记录分钟线关闭信息
182 WTSLogger::info("Minute Bar {}.{:04d} Closed by data", _date, thisMin);
183 // 如果有数据存储器，通知分钟结束
184 if (_store)
185 _store->onMinuteEnd(_date, thisMin, bEndingTDate ? _engine->getTradingDate() : 0);
186
187 // 触发引擎的定时任务调度
188 _engine->on_schedule(_date, thisMin);
189
190 // 如果是交易日结束，触发交易日结束事件
191 if(bEndingTDate)
192 _engine->on_session_end();
193 }
194
195 //By Wesley @ 2022.02.09
196 //这里先修改时间，再调用trigger_price
197 //无论分钟线是否切换，先修改时间都是对的
198 if (_engine)
199 {
200 // 设置引擎的当前日期和时间
201 _engine->set_date_time(_date, wrapMin, curSec, prevMin);
202 // 设置引擎的交易日期
203 _engine->set_trading_date(curTick->tradingdate());
204 }
205 // 触发价格事件
206 trigger_price(curTick);
207
208 // 更新当前位置
209 _cur_pos = minutes;
210 }
211 else
212 {
213 // 如果分钟数还是一致的, 则直接触发行情和时间即可
214 trigger_price(curTick);
215 if (_engine)
216 _engine->set_date_time(_date, wrapMin, curSec, prevMin);
217 }
218
219 // 计算下一次检查的时间
220 uint32_t sec = curSec / 1000; // 当前秒数
221 uint32_t msec = curSec % 1000; // 当前毫秒数
222 // 计算到下一分钟的剩余毫秒数
223 uint32_t left_ticks = (60 - sec) * 1000 - msec;
224 // 设置下一次检查时间
225 _next_check_time = TimeUtils::getLocalTimeNow() + left_ticks;
226}
