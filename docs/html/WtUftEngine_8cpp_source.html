326void WtUftEngine::run()
327{
328 // 遍历所有策略上下文，调用其初始化方法
329 for (auto it = _ctx_map.begin(); it != _ctx_map.end(); it++)
330 {
331 UftContextPtr& ctx = (UftContextPtr&)it->second;
332 ctx->on_init();
333 }
334
335 // 创建并初始化实时时间计时器
336 _tm_ticker = new WtUftRtTicker(this);
337 // 如果配置中有产品信息，使用产品配置中的交易时段
338 if(_cfg && _cfg->has("product"))
339 {
340 WTSVariant* cfgProd = _cfg->get("product");
341 _tm_ticker->init(cfgProd->getCString("session"));
342 }
343 else
344 {
345 // 如果没有产品信息，使用全天交易时段
346 _tm_ticker->init("ALLDAY");
347 }
348
349 // 启动时间计时器
350 _tm_ticker->run();
351}

546void WtUftEngine::on_tick(const char* stdCode, WTSTickData* curTick)
547{
548 // 如果数据管理器存在，将Tick数据交给数据管理器处理
549 if(_data_mgr)
550 _data_mgr->handle_push_quote(stdCode, curTick);
551
552 {
553 // 查找是否有策略订阅了该合约的Tick数据
554 auto sit = _tick_sub_map.find(stdCode);
555 if (sit != _tick_sub_map.end())
556 {
557 // 遍历订阅列表中的策略ID
558 const SubList& sids = sit->second;
559 for (auto it = sids.begin(); it != sids.end(); it++)
560 {
561 uint32_t sid = *it;
562
563 // 查找对应的策略上下文
564 auto cit = _ctx_map.find(sid);
565 if (cit != _ctx_map.end())
566 {
567 // 触发策略的Tick数据回调
568 UftContextPtr& ctx = (UftContextPtr&)cit->second;
569 ctx->on_tick(stdCode, curTick);
570 }
571 }
572 }
573 }
