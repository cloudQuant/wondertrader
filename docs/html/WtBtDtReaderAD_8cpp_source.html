124bool WtBtDtReaderAD::read_raw_bars(const char* exchg, const char* code, WTSKlinePeriod period, std::string& buffer)
125{
126 //直接从LMDB读取
127 WtLMDBPtr db = get_k_db(exchg, period);
128 if (db == NULL)
129 return false;
130
131 pipe_btreader_log(_sink, LL_DEBUG, "Reading back {} bars of {}.{}...", PERIOD_NAME[period], exchg, code);
132 WtLMDBQuery query(*db);
133 LMDBBarKey rKey(exchg, code, 0xffffffff);
134 LMDBBarKey lKey(exchg, code, 0);
135 int cnt = query.get_range(std::string((const char*)&lKey, sizeof(lKey)), std::string((const char*)&rKey, sizeof(rKey)),
136 [this, &buffer, &lKey](const ValueArray& ayKeys, const ValueArray& ayVals) {
137 if (ayVals.empty())
138 return;
139
140 std::size_t cnt = ayVals.size();
141 auto szUnit = sizeof(WTSBarStruct);
142 buffer.resize(szUnit*cnt);
143 char* cursor = (char*)buffer.data();
144 for(const std::string& item : ayVals)
145 {
146 memcpy(cursor, item.data(), szUnit);
147 cursor += szUnit;
148 }
149 });
150
151 return true;
152}

206WtBtDtReaderAD::WtLMDBPtr WtBtDtReaderAD::get_k_db(const char* exchg, WTSKlinePeriod period)
207{
208 WtLMDBMap* the_map = NULL;
209 std::string subdir;
210 if (period == KP_Minute1)
211 {
212 the_map = &_exchg_m1_dbs;
213 subdir = "min1";
214 }
215 else if (period == KP_Minute5)
216 {
217 the_map = &_exchg_m5_dbs;
218 subdir = "min5";
219 }
220 else if (period == KP_DAY)
221 {
222 the_map = &_exchg_d1_dbs;
223 subdir = "day";
224 }
225 else
226 return std::move(WtLMDBPtr());
227
228 auto it = the_map->find(exchg);
229 if (it != the_map->end())
230 return std::move(it->second);
231
232 WtLMDBPtr dbPtr(new WtLMDB(true));
233 std::string path = StrUtil::printf("%s%s/%s/", _base_dir.c_str(), subdir.c_str(), exchg);
234 if (!StdFile::exists(path.c_str()))
235 return std::move(WtLMDBPtr());
236
237 if (!dbPtr->open(path.c_str()))
238 {
239 pipe_btreader_log(_sink, LL_ERROR, "Opening {} db if {} failed: {}", subdir, exchg, dbPtr->errmsg());
240 return std::move(WtLMDBPtr());
241 }
242 else
243 {
244 pipe_btreader_log(_sink, LL_DEBUG, "{} db of {} opened", subdir, exchg);
245 }
246
247 (*the_map)[exchg] = dbPtr;
248 return std::move(dbPtr);
249}
