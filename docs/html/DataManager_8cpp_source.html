67bool DataManager::init(WTSVariant* params, WTSBaseDataMgr* bdMgr, StateMonitor* stMonitor)
68{
69 _bd_mgr = bdMgr;
70 _state_mon = stMonitor;
71
72 std::string module = params->getCString("module");
73 if (module.empty())
74 module = WtHelper::get_module_dir() + DLLHelper::wrap_module("WtDataStorage");
75 else
76 module = WtHelper::get_module_dir() + DLLHelper::wrap_module(module.c_str());
77
78 DllHandle libWriter = DLLHelper::load_library(module.c_str());
79 if (libWriter)
80 {
81 FuncCreateWriter pFuncCreateWriter = (FuncCreateWriter)DLLHelper::get_symbol(libWriter, "createWriter");
82 if (pFuncCreateWriter == NULL)
83 {
84 WTSLogger::error("Initializing of data writer failed: function createWriter not found...");
85 }
86
87 FuncDeleteWriter pFuncDeleteWriter = (FuncDeleteWriter)DLLHelper::get_symbol(libWriter, "deleteWriter");
88 if (pFuncDeleteWriter == NULL)
89 {
90 WTSLogger::error("Initializing of data writer failed: function deleteWriter not found...");
91 }
92
93 if (pFuncCreateWriter && pFuncDeleteWriter)
94 {
95 _writer = pFuncCreateWriter();
96 _remover = pFuncDeleteWriter;
97 }
98 WTSLogger::info("Data storage module {} loaded", module);
99 }
100 else
101 {
102 WTSLogger::error("Initializing of data writer failed: loading module {} failed...", module.c_str());
103 return false;
104 }
105
106 return _writer->init(params, this);
107}
