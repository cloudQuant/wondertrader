59std::string WtHelper::getCWD()
60{
61 // 使用静态变量缓存工作目录，避免重复获取
62 static std::string _cwd;
63 if(_cwd.empty())
64 {
65 char buffer[256];
66#ifdef _MSC_VER
67 _getcwd(buffer, 255); // Windows系统下获取当前目录
68#else //UNIX
69 getcwd(buffer, 255); // Unix/Linux系统下获取当前目录
70#endif
71 _cwd = StrUtil::standardisePath(buffer); // 标准化路径格式
72 }
73 return _cwd;
74}
