88bool MQClient::init(const char* url, FuncMQCallback cb)
89{
90 if (_sock >= 0)
91 return true;
92
93 _cb_message = cb;
94 _sock = nn_socket(AF_SP, NN_SUB);
95 if (_sock < 0)
96 {
97 _mgr->log_client(_id, fmtutil::format("MQClient {} has an error {} while initializing", _id, _sock));
98 return false;
99 }
100
101 nn_setsockopt(_sock, NN_SUB, NN_SUB_SUBSCRIBE, "", 0);
102
103 int bufsize = RECV_BUF_SIZE;
104 nn_setsockopt(_sock, NN_SOL_SOCKET, NN_RCVBUF, &bufsize, sizeof(bufsize));
105
106 m_strURL = url;
107 if (nn_connect(_sock, url) < 0)
108 {
109 _mgr->log_client(_id, fmtutil::format("MQClient {} has an error while connecting url {}", _id, url));
110 return false;
111 }
112 else
113 {
114 _mgr->log_client(_id, fmtutil::format("MQClient {} has connected to {} ", _id, url));
115 }
116
117 m_bReady = true;
118
119 _mgr->log_client(_id, fmtutil::format("MQClient {} inited", _id));
120 return true;
121}

130void MQClient::start()
131{
132 if (m_bTerminated)
133 return;
134
135 if(_sock < 0)
136 {
137 _mgr->log_client(_id, fmtutil::format("MQClient {} has not been initialized yet", _id));
138 return;
139 }
140
141 if (m_thrdRecv == NULL)
142 {
143 m_thrdRecv.reset(new StdThread([this]() {
144
145 while (!m_bTerminated)
146 {
147 bool hasData = false;
148 for(;;)
149 {
150 int nBytes = nn_recv(_sock, _recv_buf, RECV_BUF_SIZE, NN_DONTWAIT);
151 if (nBytes > 0)
152 {
153 m_iCheckTime = TimeUtils::getLocalTimeNow();
154 m_bNeedCheck = true;
155 hasData = true;
156 _buffer.append(_recv_buf, nBytes);
157 }
158 else
159 {
160 break;
161 }
162 }
163
164 if (hasData)
165 extract_buffer();
166 else
167 {
168 if(m_iCheckTime != 0 && m_bNeedCheck)
169 {
170 int64_t now = TimeUtils::getLocalTimeNow();
171 int64_t elapse = now - m_iCheckTime;
172 if (elapse >= 60 * 1000)
173 {
174 //只通知一次，防止重复通知
175 _cb_message(_id, "TIMEOUT", "", 0);
176 m_bNeedCheck = false;
177 }
178 }
179
180 std::this_thread::sleep_for(std::chrono::milliseconds(1));
181 }
182
183 }
184 }));
185
186 _mgr->log_client(_id, fmtutil::format("MQClient {} has started successfully", _id));
187 }
188 else
189 {
190 _mgr->log_client(_id, fmtutil::format("MQClient {} has already started", _id));
191 }
192
193}

202void MQClient::extract_buffer()
203{
204 uint32_t proc_len = 0;
205 for(;;)
206 {
207 //先做长度检查
208 if (_buffer.length() - proc_len < sizeof(MQPacket))
209 break;
210
211 MQPacket* packet = (MQPacket*)(_buffer.data() + proc_len);
212
213 if (_buffer.length() - proc_len < sizeof(MQPacket) + packet->_length)
214 break;
215
216 char* data = packet->_data;
217
218 if (is_allowed(packet->_topic))
219 _cb_message(_id, packet->_topic, packet->_data, packet->_length);
220
221 proc_len += sizeof(MQPacket) + packet->_length;
222 }
223
224 if(proc_len > 0)
225 _buffer.erase(0, proc_len);
226}
