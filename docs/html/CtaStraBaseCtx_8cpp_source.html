377 if (content.empty())

399 {//读取仓位
400 double total_profit = 0;
401 double total_dynprofit = 0;
402 const rj::Value& jPos = root["positions"];

404 {
405 for (const rj::Value& pItem : jPos.GetArray())
406 {
407 const char* stdCode = pItem["code"].GetString();
408 const char* ruleTag = _engine->get_hot_mgr()->getRuleTag(stdCode);
409 bool isExpired = (strlen(ruleTag) == 0 && _engine->get_contract_info(stdCode) == NULL);
410
411 if(isExpired)
412 log_info("{} not exists or expired, position ignored", stdCode);
413
414 PosInfo& pInfo = _pos_map[stdCode];
415 pInfo._closeprofit = pItem["closeprofit"].GetDouble();
416 pInfo._last_entertime = pItem["lastentertime"].GetUint64();
417 pInfo._last_exittime = pItem["lastexittime"].GetUint64();
418 pInfo._volume = isExpired ? 0 : pItem["volume"].GetDouble();
419 if (pItem.HasMember("frozen") && !isExpired)
420 {
421 pInfo._frozen = pItem["frozen"].GetDouble();
422 pInfo._frozen_date = pItem["frozendate"].GetUint();
423 }
424
425 if (pInfo._volume == 0 || isExpired)
426 {
427 //By Wesley @ 2023.02.21
428 //加这一行的原因是，有些期权合约经常会持有到交割日
429 //所以如果合约过期了，那么需要把浮动盈亏当做平仓盈亏累加一下
430 //处理完以后，下一次加载，浮动盈亏就是0了
431 pInfo._closeprofit += pInfo._dynprofit;
432
433 pInfo._dynprofit = 0;
434 pInfo._frozen = 0;
435 }
436 else
437 pInfo._dynprofit = pItem["dynprofit"].GetDouble();
438
439 total_profit += pInfo._closeprofit;
440 total_dynprofit += pInfo._dynprofit;
441
442 const rj::Value& details = pItem["details"];
443 if (details.IsNull() || !details.IsArray() || details.Size() == 0 || isExpired)
444 continue;
445
446 for (uint32_t i = 0; i < details.Size(); i++)
447 {
448 const rj::Value& dItem = details[i];
449 double vol = dItem["volume"].GetDouble();
450 if(decimal::eq(vol, 0))
451 continue;
452
453 DetailInfo dInfo;
454 dInfo._long = dItem["long"].GetBool();
455 dInfo._price = dItem["price"].GetDouble();
456 dInfo._volume = dItem["volume"].GetDouble();
457 dInfo._opentime = dItem["opentime"].GetUint64();
458 if(dItem.HasMember("opentdate"))
459 dInfo._opentdate = dItem["opentdate"].GetUint();
460
461 if (dItem.HasMember("maxprice"))
462 dInfo._max_price = dItem["maxprice"].GetDouble();
463 else
464 dInfo._max_price = dInfo._price;
465
466 if (dItem.HasMember("minprice"))
467 dInfo._min_price = dItem["minprice"].GetDouble();
468 else
469 dInfo._min_price = dInfo._price;
470
471 dInfo._profit = dItem["profit"].GetDouble();
472 dInfo._max_profit = dItem["maxprofit"].GetDouble();
473 dInfo._max_loss = dItem["maxloss"].GetDouble();
474
475 strcpy(dInfo._opentag, dItem["opentag"].GetString());
476 if (dItem.HasMember("openbarno"))
477 dInfo._open_barno = dItem["openbarno"].GetUint();
478 else
479 dInfo._open_barno = 0;
480
481 pInfo._details.emplace_back(dInfo);
482 }
483
484 if(!isExpired)
485 {
486 log_info("Position confirmed,{} -> {}", stdCode, pInfo._volume);
487 stra_sub_ticks(stdCode);
488 }
489 }
490 }

493 _fund_info._total_dynprofit = total_dynprofit;
494 }

499 if (!jCond.IsNull() && jCond.IsObject())

501 _last_cond_min = jCond["settime"].GetUint64();
502 const rj::Value& jItems = jCond["items"];
503 for (auto& m : jItems.GetObject())

505 const char* stdCode = m.name.GetString();
506 const char* ruleTag = _engine->get_hot_mgr()->getRuleTag(stdCode);

509 log_info("{} not exists or expired, condition ignored", stdCode);

511 }
512
513 const rj::Value& cListItem = m.value;
514
515 CondList& condList = _condtions[stdCode];
516
517 for(auto& cItem : cListItem.GetArray())
518 {
519 CondEntrust condInfo;
520 strcpy(condInfo._code, stdCode);
521 strcpy(condInfo._usertag, cItem["usertag"].GetString());
522
523 condInfo._field = (WTSCompareField)cItem["field"].GetUint();
524 condInfo._alg = (WTSCompareType)cItem["alg"].GetUint();
525 condInfo._target = cItem["target"].GetDouble();
526 condInfo._qty = cItem["qty"].GetDouble();
527 condInfo._action = (char)cItem["action"].GetUint();
528
529 condList.emplace_back(condInfo);
530
531 log_info("{} condition recovered, {} {}, condition: newprice {} {}",
532 stdCode, ACTION_NAMES[condInfo._action], condInfo._qty, CMP_ALG_NAMES[condInfo._alg], condInfo._target);
533 count++;
534 }
535 }
536
537 log_info("{} conditions recovered, setup time: {}", count, _last_cond_min);
538 }
539 }

541 if (root.HasMember("signals"))
542 {
543 //读取信号
544 const rj::Value& jSignals = root["signals"];
545 if (!jSignals.IsNull() && jSignals.IsObject())
546 {
547 for (auto& m : jSignals.GetObject())

556

558
559 SigInfo& sInfo = _sig_map[stdCode];

562 sInfo._sigprice = jItem["sigprice"].GetDouble();

922void CtaStraBaseCtx::on_tick(const char* stdCode, WTSTickData* newTick, bool bEmitStrategy /* = true */)
923{
924 _price_map[stdCode] = newTick->price();
925
926 //先检查是否要信号要触发
927 {
928 auto it = _sig_map.find(stdCode);
929 if(it != _sig_map.end())
930 {
931 WTSSessionInfo* sInfo = _engine->get_session_info(stdCode, true);
932 if (sInfo->isInTradingTime(_engine->get_raw_time(), true))
933 {
934 const SigInfo sInfo = it->second;
935 //只有当信号类型不为0，即bar内信号或者条件单触发信号时，且信号没有触发过
936 do_set_position(stdCode, sInfo._volume, sInfo._usertag.c_str(), (sInfo._sigtype != 0 && !sInfo._triggered));
937 _sig_map.erase(it);
938
939 //如果是条件单触发，则回调on_condition_triggered
940 if(sInfo._sigtype == 2)
941 on_condition_triggered(stdCode, sInfo._volume, newTick->price(), sInfo._usertag.c_str());
942 }
943
944 }
945 }
946
947 //更新浮动盈亏
948 update_dyn_profit(stdCode, newTick->price());
949
951 //检查条件单
952 if(!_condtions.empty())
953 {
954 auto it = _condtions.find(stdCode);
955 if (it == _condtions.end())
956 return;
957
958 const CondList& condList = it->second;
959 for (const CondEntrust& entrust : condList)
960 {
961 double curPrice = newTick->price();
962
963 bool isMatched = false;
964 switch (entrust._alg)
965 {
966 case WCT_Equal:
967 isMatched = decimal::eq(curPrice, entrust._target);
968 break;
969 case WCT_Larger:
970 isMatched = decimal::gt(curPrice, entrust._target);
971 break;
972 case WCT_LargerOrEqual:
973 isMatched = decimal::ge(curPrice, entrust._target);
974 break;
975 case WCT_Smaller:
976 isMatched = decimal::lt(curPrice, entrust._target);
977 break;
978 case WCT_SmallerOrEqual:
979 isMatched = decimal::le(curPrice, entrust._target);
980 break;
981 default:
982 break;
983 }
984
985 if (isMatched)
986 {
987 log_info("Condition triggered[newprice {}{} targetprice {}], instrument: {}, {} {}",
988 curPrice, CMP_ALG_NAMES[entrust._alg], entrust._target, stdCode, ACTION_NAMES[entrust._action], entrust._qty);
989
990 switch (entrust._action)
991 {
992 case COND_ACTION_OL:
993 {
994 double curQty = stra_get_position(stdCode);
995 double desQty = 0;
996 if (decimal::lt(curQty, 0))
997 desQty = entrust._qty;
998 else
999 desQty = curQty + entrust._qty;
1000
1001 append_signal(stdCode, desQty, entrust._usertag, 2);
1002 }
1003 break;
1004 case COND_ACTION_CL:
1005 {
1006 double curQty = stra_get_position(stdCode);
1007 if (decimal::gt(curQty, 0))
1008 {
1009 double maxQty = min(curQty, entrust._qty);
1010 double desQty = curQty - maxQty;
1011 append_signal(stdCode, desQty, entrust._usertag, 2);
1012 }
1013 }
1014 break;
1015 case COND_ACTION_OS:
1016 {
1017 double curQty = stra_get_position(stdCode);
1018 double desQty = 0;
1019 if (decimal::gt(curQty, 0))
1020 desQty = -entrust._qty;
1021 else
1022 desQty = curQty - entrust._qty;
1023
1024 append_signal(stdCode, desQty, entrust._usertag, 2);
1025 }
1026 break;
1027 case COND_ACTION_CS:
1028 {
1029 double curQty = stra_get_position(stdCode);
1030 if (decimal::lt(curQty, 0))
1031 {
1032 double maxQty = min(abs(curQty), entrust._qty);
1033 double desQty = curQty + maxQty;
1034 append_signal(stdCode, desQty, entrust._usertag, 2);
1035 }
1036 }
1037 break;
1038 case COND_ACTION_SP:
1039 {
1040 append_signal(stdCode, entrust._qty, entrust._usertag, 2);
1041 }
1042 break;
1043 default: break;
1044 }
1045
1046 //同一个bar设置针对同一个合约的条件单, 只可能触发一条
1047 //所以这里直接清理掉即可
1048 _condtions.erase(it);
1049 break;
1050 }
1051 }
1052 }
1053
1054 if (bEmitStrategy)
1055 on_tick_updated(stdCode, newTick);
1056
1057 if(_ud_modified)
1058 {
1059 save_userdata();
1060 _ud_modified = false;
1061 }
1062}

1647void CtaStraBaseCtx::do_set_position(const char* stdCode, double qty, const char* userTag /* = "" */, bool bFireAtOnce /* = false */)
1648{
1649 PosInfo& pInfo = _pos_map[stdCode];
1650 double curPx = _price_map[stdCode];
1651 uint64_t curTm = (uint64_t)_engine->get_date() * 10000 + _engine->get_min_time();
1652 uint32_t curTDate = _engine->get_trading_date();
1653
1654 if (decimal::eq(pInfo._volume, qty))
1655 return;
1656
1657 double diff = qty - pInfo._volume;
1658
1659 WTSCommodityInfo* commInfo = _engine->get_commodity_info(stdCode);
1660 if (commInfo == NULL)
1661 return;
1662
1663 //成交价
1664 double trdPx = curPx;
1665 /* By HeJ @ 2023.03.14
1666 * 设置理论持仓时，要加个锁，避免出现组合轧差同步与信号同时触发，导致的反复发单和信号覆盖
1667 */
1668 SpinLock lock(_mutex);
1669 bool isBuy = decimal::gt(diff, 0.0);
1670 if (decimal::gt(pInfo._volume*diff, 0))
1671 {//当前持仓和仓位变化方向一致, 增加一条明细, 增加数量即可
1672 pInfo._volume = qty;
1673
1674 //如果T+1，则冻结仓位要增加
1675 if (commInfo->isT1())
1676 {
1677 //ASSERT(diff>0);
1678 pInfo._frozen += diff;
1679 pInfo._frozen_date = curTDate;
1680 log_debug("{} frozen position updated to {}", stdCode, pInfo._frozen);
1681 }
1682
1683 if (_slippage != 0)
1684 {
1685 trdPx += _slippage * commInfo->getPriceTick()*(isBuy ? 1 : -1);
1686 }
1687
1688 DetailInfo dInfo;
1689 dInfo._long = decimal::gt(qty, 0);
1690 dInfo._price = trdPx;
1691 dInfo._max_price = trdPx;
1692 dInfo._min_price = trdPx;
1693 dInfo._volume = abs(diff);
1694 dInfo._opentime = curTm;
1695 dInfo._opentdate = curTDate;
1696 dInfo._open_barno = _last_barno;
1697 wt_strcpy(dInfo._opentag, userTag);
1698 pInfo._details.emplace_back(dInfo);
1699 pInfo._last_entertime = curTm;
1700
1701 //double fee = _engine->calc_fee(stdCode, trdPx, abs(diff), 0);
1702 double fee = commInfo->calcFee(trdPx, abs(diff), 0);
1703 _fund_info._total_fees += fee;
1704 log_trade(stdCode, dInfo._long, true, curTm, trdPx, abs(diff), userTag, fee, _last_barno);
1705 }
1706 else
1707 {//持仓方向和仓位变化方向不一致, 需要平仓
1708 double left = abs(diff);
1709
1710 if (_slippage != 0)
1711 trdPx += _slippage * commInfo->getPriceTick()*(isBuy ? 1 : -1);
1712
1713 pInfo._volume = qty;
1714 if (decimal::eq(pInfo._volume, 0))
1715 pInfo._dynprofit = 0;
1716 uint32_t count = 0;
1717 for (auto it = pInfo._details.begin(); it != pInfo._details.end(); it++)
1718 {
1719 DetailInfo& dInfo = *it;
1720 if (decimal::eq(dInfo._volume, 0))
1721 {
1722 count++;
1723 continue;
1724 }
1725
1726 double maxQty = min(dInfo._volume, left);
1727 if (decimal::eq(maxQty, 0))
1728 continue;
1729
1730 dInfo._volume -= maxQty;
1731 left -= maxQty;
1732
1733 if (decimal::eq(dInfo._volume, 0))
1734 count++;
1735
1736 //计算平仓盈亏
1737 double profit = (trdPx - dInfo._price) * maxQty * commInfo->getVolScale();
1738 if (!dInfo._long)
1739 profit *= -1;
1740 pInfo._closeprofit += profit;
1741
1742 //浮盈也要做等比缩放
1743 pInfo._dynprofit = pInfo._dynprofit*dInfo._volume / (dInfo._volume + maxQty);
1744 pInfo._last_exittime = curTm;
1745 _fund_info._total_profit += profit;
1746
1747 //计算手续费
1748 //double fee = _engine->calc_fee(stdCode, trdPx, maxQty, dInfo._opentdate == curTDate ? 2 : 1);
1749 double fee = commInfo->calcFee(trdPx, maxQty, dInfo._opentdate == curTDate ? 2 : 1);
1750 _fund_info._total_fees += fee;
1751
1752 //这里写平仓记录
1753 log_close(stdCode, dInfo._long, dInfo._opentime, dInfo._price, curTm, trdPx, maxQty, profit, pInfo._closeprofit, dInfo._opentag, userTag, dInfo._open_barno, _last_barno);
1754
1755 //这里写成交记录
1756 log_trade(stdCode, dInfo._long, false, curTm, trdPx, maxQty, userTag, fee, _last_barno);
1757
1758 if (decimal::eq(left,0))
1759 break;
1760 }
1761
1762 //需要清理掉已经平仓完的明细
1763 while (count > 0)
1764 {
1765 auto it = pInfo._details.begin();
1766 pInfo._details.erase(it);
1767 count--;
1768 }
1769
1770 //最后, 如果还有剩余的, 则需要反手了
1771 if (decimal::gt(left, 0))
1772 {
1773 left = left * qty / abs(qty);
1774
1775 //如果T+1，则冻结仓位要增加
1776 if (commInfo->isT1())
1777 {
1778 pInfo._frozen += left;
1779 pInfo._frozen_date = curTDate;
1780 log_debug("{} frozen position up to {}", stdCode, pInfo._frozen);
1781 }
1782
1783 DetailInfo dInfo;
1784 dInfo._long = decimal::gt(qty, 0);
1785 dInfo._price = trdPx;
1786 dInfo._max_price = trdPx;
1787 dInfo._min_price = trdPx;
1788 dInfo._volume = abs(left);
1789 dInfo._opentime = curTm;
1790 dInfo._opentdate = curTDate;
1791 dInfo._open_barno = _last_barno;
1792 wt_strcpy(dInfo._opentag, userTag);
1793 pInfo._details.emplace_back(dInfo);
1794 pInfo._last_entertime = curTm;
1795
1796 //这里还需要写一笔成交记录
1797 double fee = commInfo->calcFee(trdPx, abs(left), 0);
1798 _fund_info._total_fees += fee;
1799 log_trade(stdCode, dInfo._long, true, curTm, trdPx, abs(left), userTag, fee, _last_barno);
1800 }
1801 }
1802
1803
1804 //存储数据
1805 save_data();
1806
1807 if (bFireAtOnce) //如果是条件单触发, 则向引擎提交变化量
1808 {
1809 _engine->handle_pos_change(_name.c_str(), stdCode, diff);
1810 }
1811}
