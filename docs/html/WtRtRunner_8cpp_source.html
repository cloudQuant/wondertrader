492{
493 TaskPeriodType ptype;
494 if (wt_stricmp(period, "d") == 0)

496 else if (wt_stricmp(period, "w") == 0)

508
509 _sel_engine.addContext(SelContextPtr(ctx), date, time, ptype, true, trdtpl, session);

780bool WtRtRunner::config(const char* cfgFile, bool isFile /* = true */)
781{
782 _config = isFile ? WTSCfgLoader::load_from_file(cfgFile) : WTSCfgLoader::load_from_content(cfgFile, false);
783
784 //基础数据文件
785 WTSVariant* cfgBF = _config->get("basefiles");
786 if (cfgBF->get("session"))
787 {
788 _bd_mgr.loadSessions(cfgBF->getCString("session"));
789 WTSLogger::info("Trading sessions loaded");
790 }
791
792 WTSVariant* cfgItem = cfgBF->get("commodity");
793 if (cfgItem)
794 {
795 if (cfgItem->type() == WTSVariant::VT_String)
796 {
797 _bd_mgr.loadCommodities(cfgItem->asCString());
798 }
799 else if (cfgItem->type() == WTSVariant::VT_Array)
800 {
801 for (uint32_t i = 0; i < cfgItem->size(); i++)
802 {
803 _bd_mgr.loadCommodities(cfgItem->get(i)->asCString());
804 }
805 }
806 }
807
808 cfgItem = cfgBF->get("contract");
809 if (cfgItem)
810 {
811 if (cfgItem->type() == WTSVariant::VT_String)
812 {
813 _bd_mgr.loadContracts(cfgItem->asCString());
814 }
815 else if (cfgItem->type() == WTSVariant::VT_Array)
816 {
817 for (uint32_t i = 0; i < cfgItem->size(); i++)
818 {
819 _bd_mgr.loadContracts(cfgItem->get(i)->asCString());
820 }
821 }
822 }
823
824 if (cfgBF->get("holiday"))
825 {
826 _bd_mgr.loadHolidays(cfgBF->getCString("holiday"));
827 WTSLogger::log_raw(LL_INFO, "Holidays loaded");
828 }
829
830 if (cfgBF->get("hot"))
831 {
832 _hot_mgr.loadHots(cfgBF->getCString("hot"));
833 WTSLogger::log_raw(LL_INFO, "Hot rules loaded");
834 }
835
836 if (cfgBF->get("second"))
837 {
838 _hot_mgr.loadSeconds(cfgBF->getCString("second"));
839 WTSLogger::log_raw(LL_INFO, "Second rules loaded");
840 }
841
842 WTSArray* ayContracts = _bd_mgr.getContracts();
843 for (auto it = ayContracts->begin(); it != ayContracts->end(); it++)
844 {
845 WTSContractInfo* cInfo = (WTSContractInfo*)(*it);
846 bool isHot = _hot_mgr.isHot(cInfo->getExchg(), cInfo->getCode());
847 bool isSecond = _hot_mgr.isSecond(cInfo->getExchg(), cInfo->getCode());
848
849 std::string hotCode = cInfo->getFullPid();
850 if (isHot)
851 hotCode += ".HOT";
852 else if (isSecond)
853 hotCode += ".2ND";
854 else
855 hotCode = "";
856
857 cInfo->setHotFlag(isHot ? 1 : (isSecond ? 2 : 0), hotCode.c_str());
858 }
859 ayContracts->release();
860
861 if(cfgBF->has("rules"))
862 {
863 auto cfgRules = cfgBF->get("rules");
864 auto tags = cfgRules->memberNames();
865 for(const std::string& ruleTag : tags)
866 {
867 _hot_mgr.loadCustomRules(ruleTag.c_str(), cfgRules->getCString(ruleTag.c_str()));
868 WTSLogger::info("{} rules loaded from {}", ruleTag, cfgRules->getCString(ruleTag.c_str()));
869 }
870 }
871
872 //初始化运行环境
873 initEngine();
874
875 //初始化数据管理
876 initDataMgr();
877
878 //初始化开平策略
879 if (!initActionPolicy())
880 return false;
881
882 //初始化行情通道
883 WTSVariant* cfgParser = _config->get("parsers");
884 if (cfgParser)
885 {
886 if (cfgParser->type() == WTSVariant::VT_String)
887 {
888 const char* filename = cfgParser->asCString();
889 if (StdFile::exists(filename))
890 {
891 WTSLogger::info("Reading parser config from {}...", filename);
892 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
893 if (var)
894 {
895 if (!initParsers(var->get("parsers")))
896 WTSLogger::error("Loading parsers failed");
897 var->release();
898 }
899 else
900 {
901 WTSLogger::error("Loading parser config {} failed", filename);
902 }
903 }
904 else
905 {
906 WTSLogger::error("Parser configuration {} not exists", filename);
907 }
908 }
909 else if (cfgParser->type() == WTSVariant::VT_Array)
910 {
911 initParsers(cfgParser);
912 }
913 }
914
915 //初始化交易通道
916 WTSVariant* cfgTraders = _config->get("traders");
917 if(cfgTraders)
918 {
919 if (cfgTraders->type() == WTSVariant::VT_String)
920 {
921 const char* filename = cfgTraders->asCString();
922 if (StdFile::exists(filename))
923 {
924 WTSLogger::info("Reading trader config from {}...", filename);
925 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
926 if (var)
927 {
928 if (!initTraders(var->get("traders")))
929 WTSLogger::error("Loading traders failed");
930 var->release();
931 }
932 else
933 {
934 WTSLogger::error("Loading trader config {} failed", filename);
935 }
936 }
937 else
938 {
939 WTSLogger::error("Trader configuration {} not exists", filename);
940 }
941 }
942 else if (cfgTraders->type() == WTSVariant::VT_Array)
943 {
944 initTraders(cfgTraders);
945 }
946 }
947
948 //初始化事件推送器
949 initEvtNotifier();
950
951 //如果不是高频引擎,则需要配置执行模块
952 if (!_is_hft)
953 {
954 WTSVariant* cfgExec = _config->get("executers");
955 if(cfgExec != NULL)
956 {
957 if(cfgExec->type() == WTSVariant::VT_String)
958 {
959 const char* filename = cfgExec->asCString();
960 if (StdFile::exists(filename))
961 {
962 WTSLogger::info("Reading executer config from {}...", filename);
963 WTSVariant* var = WTSCfgLoader::load_from_file(filename);
964 if (var)
965 {
966 if (!initExecuters(var->get("executers")))
967 WTSLogger::error("Loading executers failed");
968
969 WTSVariant* c = var->get("routers");
970 if (c != NULL)
971 _cta_engine.loadRouterRules(c);
972
973 var->release();
