75bool WtDtMgr::initStore(WTSVariant* cfg)
76{
77 // 检查配置是否为空
78 if (cfg == NULL)
79 return false;
80
81 // 获取数据存储模块名称
82 std::string module = cfg->getCString("module");
83 // 如果模块名称为空，使用默认的WtDataStorage
84 if (module.empty())
85 module = WtHelper::getInstDir() + DLLHelper::wrap_module("WtDataStorage");
86 else
87 module = WtHelper::getInstDir() + DLLHelper::wrap_module(module.c_str());
88
89 // 加载数据存储模块动态库
90 DllHandle hInst = DLLHelper::load_library(module.c_str());
91 if(hInst == NULL)
92 {
93 WTSLogger::error("Loading data reader module {} failed", module.c_str());
94 return false;
95 }
96
97 // 获取创建数据读取器的函数指针
98 FuncCreateDataReader funcCreator = (FuncCreateDataReader)DLLHelper::get_symbol(hInst, "createDataReader");
99 if(funcCreator == NULL)
100 {
101 WTSLogger::error("Loading data reader module {} failed, entrance function createDataReader not found", module.c_str());
102 DLLHelper::free_library(hInst);
103 return false;
104 }
105
106 // 创建数据读取器实例
107 _reader = funcCreator();
108 if(_reader == NULL)
109 {
110 WTSLogger::error("Creating instance of data reader module {} failed", module.c_str());
111 DLLHelper::free_library(hInst);
112 return false;
113 }
114
115 // 初始化数据读取器，传入配置、当前对象作为回调接收器和历史数据加载器
116 _reader->init(cfg, this, _loader);
117
118 return true;
119}

251void WtDtMgr::on_bar(const char* code, WTSKlinePeriod period, WTSBarStruct* newBar)
252{
253 // 生成合约和周期的键值模式
254 std::string key_pattern = fmt::format("{}-{}", code, period);
255
256 // 根据周期类型转换为字符周期和倍数
257 char speriod;
258 uint32_t times = 1;
259 switch (period)
260 {
261 case KP_Minute1:
262 speriod = 'm';
263 times = 1;
264 break;
265 case KP_Minute5:
266 speriod = 'm';
267 times = 5;
268 break;
269 default:
270 speriod = 'd';
271 times = 1;
272 break;
273 }
274
275 // 如果是订阅的基础周期，则添加到通知列表中
276 if(_subed_basic_bars.find(key_pattern) != _subed_basic_bars.end())
277 {
278 //如果是基础周期, 直接触发on_bar事件
279 //_engine->on_bar(code, speriod.c_str(), times, newBar);
280 //更新完K线以后, 统一通知交易引擎
281 _bar_notifies.emplace_back(NotifyItem(code, speriod, times, newBar));
282 }
283
284 //然后再处理非基础周期
285 if (_bars_cache == NULL || _bars_cache->size() == 0)
286 return;
287
288 WTSSessionInfo* sInfo = _engine->get_session_info(code, true);
289
290 for (auto it = _bars_cache->begin(); it != _bars_cache->end(); it++)
291 {
292 const char* key = it->first.c_str();
293 if(memcmp(key, key_pattern.c_str(), key_pattern.size()) != 0)
294 continue;
295
296 WTSKlineData* kData = (WTSKlineData*)it->second;
297 if(kData->times() != 1)
298 {
299 g_dataFact.updateKlineData(kData, newBar, sInfo, _align_by_section);
300 if (kData->isClosed())
301 {
302 //如果基础周期K线的时间和自定义周期K线的时间一致, 说明K线关闭了
303 //这里也要触发on_bar事件
304 WTSBarStruct* lastBar = kData->at(-1);
305 //_engine->on_bar(code, speriod.c_str(), times, lastBar);
306 //更新完K线以后, 统一通知交易引擎
307 _bar_notifies.emplace_back(NotifyItem(code, speriod, times*kData->times(), lastBar));
308 }
309 }
310 else
311 {
312 //如果是强制缓存的一倍周期，直接压到缓存队列里
313 kData->getDataRef().emplace_back(*newBar);
314 _bar_notifies.emplace_back(NotifyItem(code, speriod, times, newBar));
315 }
316 }

326void WtDtMgr::handle_push_quote(const char* stdCode, WTSTickData* newTick)
327{
328 // 检查新Tick数据是否有效
329 if (newTick == NULL)
330 return;
331
332 // 如果实时Tick缓存不存在，创建一个新的
333 if (_rt_tick_map == NULL)
334 _rt_tick_map = DataCacheMap::create();
335
336 // 将新的Tick数据添加到实时缓存中，并设置为自动释放
337 _rt_tick_map->add(stdCode, newTick, true);

339 // 如果复权Tick缓存存在，则处理复权数据

417WTSTickSlice* WtDtMgr::get_tick_slice(const char* stdCode, uint32_t count, uint64_t etime /* = 0 */)
418{
419 if (_reader == NULL)
420 return NULL;
421
422 /*
423 * By Wesley @ 2022.02.11
424 * 这里要重新处理一下
425 * 如果是不复权或者前复权，则直接读取底层的实时缓存即可
426 */
427 auto len = strlen(stdCode);
428 bool isHFQ = (stdCode[len - 1] == SUFFIX_HFQ);
429
430 //不是后复权，缓存直接用底层缓存
431 if(!isHFQ)
432 return _reader->readTickSlice(stdCode, count, etime);
433
434 //先转成不带+的标准代码
435 std::string pureStdCode(stdCode, len - 1);
436
437 if (_ticks_adjusted == NULL)
438 _ticks_adjusted = DataCacheMap::create();
439
440 //如果缓存没有，先重新生成一下缓存
441 auto it = _ticks_adjusted->find(pureStdCode);
442 if (it == _ticks_adjusted->end())
443 {
444 //先读取全部tick数据
445 double factor = _engine->get_exright_factor(stdCode, NULL);
446 WTSTickSlice* slice = _reader->readTickSlice(pureStdCode.c_str(), 999999, etime);
447 std::vector<WTSTickStruct> ayTicks;
448 ayTicks.resize(slice->size());
449 std::size_t offset = 0;
450 for (std::size_t bIdx = 0; bIdx < slice->get_block_counts(); bIdx++)
451 {
452 memcpy(&ayTicks[0] + offset, slice->get_block_addr(bIdx), slice->get_block_size(bIdx) * sizeof(WTSTickStruct));
453 offset += slice->get_block_size(bIdx);
454 }
455
456 //缓存的数据做一个复权处理
457 for (WTSTickStruct& tick : ayTicks)
458 {
459 tick.price *= factor;
460 tick.open *= factor;
461 tick.high *= factor;
462 tick.low *= factor;
463 }
464
465 //添加到缓存中
466 WTSHisTickData* hisTick = WTSHisTickData::create(stdCode, false, factor);
467 hisTick->getDataRef().swap(ayTicks);
468 _ticks_adjusted->add(pureStdCode, hisTick, false);
469 }
470
471 WTSHisTickData* hisTick = (WTSHisTickData*)_ticks_adjusted->get(pureStdCode);
472 uint32_t curDate, curTime, curSecs;
473 if (etime == 0)
474 {
475 curDate = get_date();
476 curTime = get_min_time();
477 curSecs = get_secs();
478
479 etime = (uint64_t)curDate * 1000000000 + curTime * 100000 + curSecs;
480 }
481 else
482 {
483 //20190807124533900
484 curDate = (uint32_t)(etime / 1000000000);
485 curTime = (uint32_t)(etime % 1000000000) / 100000;
486 curSecs = (uint32_t)(etime % 100000);
487 }
488
489 //比较时间的对象
490 WTSTickStruct eTick;
491 eTick.action_date = curDate;
492 eTick.action_time = curTime * 100000 + curSecs;
493
494 auto& ticks = hisTick->getDataRef();
495
496 WTSTickStruct* pTick = std::lower_bound(&ticks.front(), &ticks.back(), eTick, [](const WTSTickStruct& a, const WTSTickStruct& b) {
497 if (a.action_date != b.action_date)
498 return a.action_date < b.action_date;
499 else
500 return a.action_time < b.action_time;
501 });
502
503 uint32_t eIdx = pTick - &ticks.front();
504
505 //如果光标定位的tick时间比目标时间打, 则全部回退一个
506 if (pTick->action_date > eTick.action_date || pTick->action_time > eTick.action_time)
507 {
508 pTick--;
509 eIdx--;
510 }
511
512 uint32_t cnt = min(eIdx + 1, count);
513 uint32_t sIdx = eIdx + 1 - cnt;
514 WTSTickSlice* slice = WTSTickSlice::create(stdCode, &ticks.front() + sIdx, cnt);
515 return slice;
516}

577WTSKlineSlice* WtDtMgr::get_kline_slice(const char* stdCode, WTSKlinePeriod period, uint32_t times, uint32_t count, uint64_t etime /* = 0 */)
578{
579 // 如果数据读取器不存在，返回NULL
580 if (_reader == NULL)
581 return NULL;
582
583 // 使用线程本地存储创建缓存键值
584 thread_local static char key[64] = { 0 };
585 fmtutil::format_to(key, "{}-{}", stdCode, (uint32_t)period);
586
587 // 如果不强制缓存，并且重采样倍数为1，则直接读取slice返回
588 if (times == 1 && !_force_cache)
589 {
590 // 将当前合约周期添加到订阅的基础周期集合中
591 _subed_basic_bars.insert(key);
592
593 // 直接调用数据读取器获取K线切片
594 return _reader->readKlineSlice(stdCode, period, count, etime);
595 }
596
597 // 只有非基础周期的会进到下面的步骤
598 // 获取合约的交易时段信息
599 WTSSessionInfo* sInfo = _engine->get_session_info(stdCode, true);
600
601 // 如果K线缓存不存在，创建一个新的
602 if (_bars_cache == NULL)
603 _bars_cache = DataCacheMap::create();
604
605 // 生成缓存键，包含合约代码、周期和倍数
606 fmtutil::format_to(key, "{}-{}-{}", stdCode, (uint32_t)period, times);
607
608 // 从缓存中获取K线数据
609 WTSKlineData* kData = (WTSKlineData*)_bars_cache->get(key);
610 // 如果缓存中没有数据或数据不足，需要重新读取
611 if (kData == NULL || kData->size() < count)
612 {
613 // 计算需要读取的原始K线数量
614 uint32_t realCount = times==1 ? count: (count*times + times);
615 // 读取原始K线数据
616 WTSKlineSlice* rawData = _reader->readKlineSlice(stdCode, period, realCount, etime);
617 if (rawData != NULL && rawData->size() > 0)
618 {
619 if(times != 1) // 如果是多周期，需要进行重采样
620 {
621 // 使用数据工厂提取多周期K线数据
622 kData = g_dataFact.extractKlineData(rawData, period, times, sInfo, true, _align_by_section);
623 }
624 else // 如果是单周期，直接复制数据
625 {
626 // 创建K线数据对象
627 kData = WTSKlineData::create(stdCode, rawData->size());
628 kData->setPeriod(period, 1);
629 kData->setClosed(true);
630 // 复制原始K线数据
631 WTSBarStruct* pBar = kData->getDataRef().data();
632 for(uint32_t bIdx = 0; bIdx < rawData->get_block_counts(); bIdx++ )
633 {
634 memcpy(pBar, rawData->get_block_addr(bIdx), sizeof(WTSBarStruct)*rawData->get_block_size(bIdx));
635 pBar += rawData->get_block_size(bIdx);
636 }
637 }
638
639 // 释放原始K线数据
640 rawData->release();
641 }
642 else
643 {
644 return NULL;
645 }
646
647 // 如果K线数据创建成功，添加到缓存中
648 if (kData)
649 {
650 _bars_cache->add(key, kData, false);
651 // 如果是多周期，输出重采样日志
652 if(times != 1)
653 WTSLogger::debug("{} bars of {} resampled every {} bars: {} -> {}",
654 PERIOD_NAME[period], stdCode, times, realCount, kData->size());
655 }
656 }
657
658 /*
659 * By Wesley @ 2023.03.03
660 * 当多周期K线跨越小节时，如果重启了组合
661 * 这个时候就会在启动的时候拉到一条未闭合的K线
662 * 但是未闭合的K线等一下还会重新推一遍
663 * 所以这里必须要做一个修正
664 * 只处理已经闭合的K线
665 */
666 // 计算已闭合的K线数量
667 uint32_t closedSz = kData->size();
668 if (closedSz > 0 && !kData->isClosed())
669 closedSz--; // 如果最后一条未闭合，则不计入
670
671 // 计算起始索引和返回的K线数量
672 int32_t sIdx = 0;
673 uint32_t rtCnt = min(closedSz, count); // 取请求数量和可用数量的最小值
674 sIdx = closedSz - rtCnt; // 计算起始索引
675 // 获取起始位置的K线指针
676 WTSBarStruct* rtHead = kData->at(sIdx);
677 // 创建K线切片并返回
678 WTSKlineSlice* slice = WTSKlineSlice::create(stdCode, period, times, rtHead, rtCnt);
679 return slice;
680}
