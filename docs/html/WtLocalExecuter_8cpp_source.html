75bool WtLocalExecuter::init(WTSVariant* params)
76{
77 if (params == NULL)
78 return false;
79
80 _config = params;
81 _config->retain();
82
83 // 设置缩放比例
84 _scale = params->getDouble("scale");
85 // 设置是否严格同步
86 _strict_sync = params->getBoolean("strict_sync");
87
88 // 创建线程池（如果配置了池大小）
89 uint32_t poolsize = params->getUInt32("poolsize");
90 if(poolsize > 0)
91 {
92 _pool.reset(new boost::threadpool::pool(poolsize));
93 }
94
95 /*
96 * By Wesley @ 2021.12.14
97 * 从配置文件中读取自动清理的策略
98 * active: 是否启用
99 * includes: 包含列表，格式如CFFEX.IF
100 * excludes: 排除列表，格式如CFFEX.IF
101 */
102 WTSVariant* cfgClear = params->get("clear");
103 if(cfgClear)
104 {
105 _auto_clear = cfgClear->getBoolean("active");
106 WTSVariant* cfgItem = cfgClear->get("includes");
107 if(cfgItem)
108 {
109 if (cfgItem->type() == WTSVariant::VT_String)
110 _clear_includes.insert(cfgItem->asCString());
111 else if (cfgItem->type() == WTSVariant::VT_Array)
112 {
113 for(uint32_t i = 0; i < cfgItem->size(); i++)
114 _clear_includes.insert(cfgItem->get(i)->asCString());
115 }
116 }
117
118 cfgItem = cfgClear->get("excludes");
119 if (cfgItem)
120 {
121 if (cfgItem->type() == WTSVariant::VT_String)
122 _clear_excludes.insert(cfgItem->asCString());
123 else if (cfgItem->type() == WTSVariant::VT_Array)
124 {
125 for (uint32_t i = 0; i < cfgItem->size(); i++)
126 _clear_excludes.insert(cfgItem->get(i)->asCString());
127 }
128 }
129 }
130
131 WTSVariant* cfgGroups = params->get("groups");
132 if (cfgGroups)
133 {
134 auto names = cfgGroups->memberNames();
135 for(const std::string& gpname : names)
136 {
137 CodeGroupPtr& gpInfo = _groups[gpname];
138 if (gpInfo == NULL)
139 {
140 gpInfo.reset(new CodeGroup);
141 wt_strcpy(gpInfo->_name, gpname.c_str(), gpname.size());
142 }
143
144 WTSVariant* cfgGrp = cfgGroups->get(gpname.c_str());
145 auto codes = cfgGrp->memberNames();
146 for(const std::string& code : codes)
147 {
148 gpInfo->_items[code] = cfgGrp->getDouble(code.c_str());
149 _code_to_groups[code] = gpInfo;
150 }
151 }
152 }
153
154 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Local executer inited, scale: {}, auto_clear: {}, strict_sync: {}, thread poolsize: {}, code_groups: {}",
155 _scale, _auto_clear, _strict_sync, poolsize, _groups.size());
156
157 return true;
158}

170ExecuteUnitPtr WtLocalExecuter::getUnit(const char* stdCode, bool bAutoCreate /* = true */)
171{
172 // 提取合约信息
173 CodeHelper::CodeInfo codeInfo = CodeHelper::extractStdCode(stdCode, NULL);
174 // 获取商品ID
175 std::string commID = codeInfo.stdCommID();
176
177 // 获取策略配置
178 WTSVariant* policy = _config->get("policy");
179 // 默认使用商品ID作为策略名
180 std::string des = commID;
181 // 如果没有对应的策略，使用default策略
182 if (!policy->has(commID.c_str()))
183 des = "default";
184
185 // 锁定单元容器进行线程安全的访问
186 SpinLock lock(_mtx_units);
187
188 // 检查执行单元是否已存在
189 auto it = _unit_map.find(stdCode);
190 if(it != _unit_map.end())
191 {
192 return it->second;
193 }
194
195 // 如果需要自动创建
196 if (bAutoCreate)
197 {
198 // 获取策略配置
199 WTSVariant* cfg = policy->get(des.c_str());
200
201 // 获取执行单元类型名称
202 const char* name = cfg->getCString("name");
203 // 创建执行单元
204 ExecuteUnitPtr unit = _factory->createExeUnit(name);
205 if (unit != NULL)
206 {
207 // 将新建的执行单元添加到映射中
208 _unit_map[stdCode] = unit;
209 // 初始化执行单元
210 unit->self()->init(this, stdCode, cfg);
211
212 //如果通道已经就绪，则直接通知执行单元
213 if (_channel_ready)
214 unit->self()->on_channel_ready();
215 }
216 return unit;
217 }
218 else
219 {
220 return ExecuteUnitPtr();
221 }
222}

465void WtLocalExecuter::on_position_changed(const char* stdCode, double diffPos)
466{
467 // 获取合约对应的执行单元，如果不存在会自动创建
468 ExecuteUnitPtr unit = getUnit(stdCode);
469 if (unit == NULL)
470 return;
471
472 // 计算新的目标仓位
473 double oldVol = _target_pos[stdCode];
474 double newVol = oldVol + diffPos;
475 _target_pos[stdCode] = newVol;
476
477 // 应用缩放比例并四舍五入
478 double traderTarget = round(newVol * _scale);
479
480 // 如果持仓有变化，记录日志
481 if(!decimal::eq(diffPos, 0))
482 {
483 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Target position of {} changed: {} -> {} : {} with scale:{}", stdCode, oldVol, newVol, traderTarget, _scale);
484 }
485
486 // 检查合约是否有订单限制
487 if (_trader && !_trader->checkOrderLimits(stdCode))
488 {
489 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "{} is disabled", stdCode);
490 return;
491 }
492
493 // 通知执行单元设置新的目标仓位
494 unit->self()->set_position(stdCode, traderTarget);
495}

505void WtLocalExecuter::set_position(const wt_hashmap<std::string, double>& targets)
506{
507 /*
508 * 先要把目标头寒进行组合匹配
509 */
510 // 创建实际目标仓位的副本，用于处理组合合约
511 auto real_targets = targets;
512 // 遍历所有的合约组合
513 for(auto& v : _groups)
514 {
515 const CodeGroupPtr& gpInfo = v.second;
516 bool bHit = false;
517 double gpQty = DBL_MAX;
518 // 检查组合中的每个合约是否存在于目标仓位中
519 for(auto& vi : gpInfo->_items)
520 {
521 double unit = vi.second;
522 auto it = real_targets.find(vi.first);
523 if (it == real_targets.end())
524 {
525 bHit = false;
526 break;
527 }
528 else
529 {
530 bHit = true;
531 //计算最小的组合单位数量
532 gpQty = std::min(gpQty, decimal::mod(it->second, unit));
533 }
534 }
535
536 // 如果组合成功匹配且数量大于0
537 if(bHit && decimal::gt(gpQty, 0))
538 {
539 // 设置组合合约的目标仓位
540 real_targets[gpInfo->_name] = gpQty;
541 // 从原始合约中减去组合合约占用的部分
542 for (auto& vi : gpInfo->_items)
543 {
544 double unit = vi.second;
545 real_targets[vi.first] -= gpQty * unit;
546 }
547 }
548 }
549
550 // 遍历所有目标仓位，设置执行单元的目标仓位
551 for (auto it = targets.begin(); it != targets.end(); it++)
552 {
553 const char* stdCode = it->first.c_str();
554 double newVol = it->second;
555 // 获取或创建执行单元
556 ExecuteUnitPtr unit = getUnit(stdCode);
557 if (unit == NULL)
558 continue;
559
560 // 更新内部目标仓位记录
561 double oldVol = _target_pos[stdCode];
562 _target_pos[stdCode] = newVol;
563 // 账户的理论持仓要经过缩放比例修正
564 double traderTarget = round(newVol * _scale);
565
566 // 如果仓位变化，记录日志
567 if(!decimal::eq(oldVol, newVol))
568 {
569 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Target position of {} changed: {} -> {} : {} with scale{}", stdCode, oldVol, newVol, traderTarget, _scale);
570 }
571
572 // 检查合约是否有订单限制
573 if (_trader && !_trader->checkOrderLimits(stdCode))
574 {
575 WTSLogger::log_dyn("executer", _name.c_str(), LL_WARN, "{} is disabled due to entrust limit control ", stdCode);
576 continue;
577 }
578
579 // 如果配置了线程池，使用线程池并行处理
580 if (_pool)
581 {
582 std::string code = stdCode;
583 _pool->schedule([unit, code, traderTarget](){
584 unit->self()->set_position(code.c_str(), traderTarget);
585 });
586 }
587 else
588 {
589 // 直接设置执行单元的目标仓位
590 unit->self()->set_position(stdCode, traderTarget);
591 }
592 }
593
594 //在原来的目标头寒中，但是不在新的目标头寒中，则需要自动设置为0
595 for (auto it = _target_pos.begin(); it != _target_pos.end(); it++)
596 {
597 const char* code = it->first.c_str();
598 double& pos = (double&)it->second;
599 auto tit = targets.find(code);
600 // 如果新目标中存在该合约，跳过
601 if(tit != targets.end())
602 continue;
603
604 // 旧目标中有，新目标中没有的合约，自动设置为0
605 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "{} is not in target, set to 0 automatically", code);
606
607 ExecuteUnitPtr unit = getUnit(code);
608 if (unit == NULL)
609 continue;
610
611 //unit->self()->set_position(code, 0);
612 // 如果配置了线程池，使用线程池并行处理
613 if (_pool)
614 {
615 _pool->schedule([unit, code](){
616 unit->self()->set_position(code, 0);
617 });
618 }
619 else
620 {
621 // 直接设置执行单元的目标仓位为0
622 unit->self()->set_position(code, 0);
623 }
624
625 // 更新内部目标仓位记录
626 pos = 0;
627 }
628
629 //如果开启了严格同步，则需要检查通道持仓
630 //如果通道持仓不在管理中，则直接平掉
631 if(_strict_sync)
632 {
633 for(const std::string& stdCode : _channel_holds)
634 {
635 auto it = _target_pos.find(stdCode.c_str());
636 if(it != _target_pos.end())
637 continue;
638
639 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "{} is not in management, set to 0 due to strict sync mode", stdCode.c_str());
640
641 ExecuteUnitPtr unit = getUnit(stdCode.c_str());
642 if (unit == NULL)
643 continue;
644
645 if (_pool)
646 {
647 std::string code = stdCode.c_str();
648 _pool->schedule([unit, code]() {
649 unit->self()->set_position(code.c_str(), 0);
650 });
651 }
652 else
653 {
654 unit->self()->set_position(stdCode.c_str(), 0);
655 }
656 }
657 }
658}

917void WtLocalExecuter::on_position(const char* stdCode, bool isLong, double prevol, double preavail, double newvol, double newavail, uint32_t tradingday)
918{
919 // 将合约添加到通道持仓集合中
920 _channel_holds.insert(stdCode);
921
922 /*
923 * By Wesley @ 2021.12.14
924 * 先检查自动清理过期主力合约的标记是否为true
925 * 如果不为true，则直接退出该逻辑
926 */
927 if (!_auto_clear)
928 return;
929
930 // 如果不是分月期货合约，直接退出
931 if (!CodeHelper::isStdMonthlyFutCode(stdCode))
932 return;
933
934 // 获取热力合约管理器
935 IHotMgr* hotMgr = _stub->get_hot_mon();
936 // 解析标准合约代码
937 CodeHelper::CodeInfo cInfo = CodeHelper::extractStdCode(stdCode, NULL);
938 // 获取上一期的主力合约
939 std::string prevCode = hotMgr->getPrevRawCode(cInfo._exchg, cInfo._product, tradingday);
940
941 // 如果当前合约不是上一期的主力合约，则直接退出
942 if (prevCode != cInfo._code)
943 return;
944
945 // 记录日志，显示找到了上一期的主力合约
946 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Prev hot contract of {}.{} on {} is {}", cInfo._exchg, cInfo._product, tradingday, prevCode);
947
948 // 创建产品完整标识符
949 thread_local static char fullPid[64] = { 0 };
950 fmtutil::format_to(fullPid, "{}.{}", cInfo._exchg, cInfo._product);
951
952 // 先检查排除列表
953 // 如果在排除列表中，则直接退出
954 auto it = _clear_excludes.find(fullPid);
955 if(it != _clear_excludes.end())
956 {
957 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Position of {}, as prev hot contract, won't be cleared for it's in exclude list", stdCode);
958 return;
959 }
960
961 // 如果包含列表不为空，再检查是否在包含列表中
962 // 如果为空，则全部清理，不再进入该逻辑
963 if(!_clear_includes.empty())
964 {
965 it = _clear_includes.find(fullPid);
966 if (it == _clear_includes.end())
967 {
968 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Position of {}, as prev hot contract, won't be cleared for it's not in include list", stdCode);
969 return;
970 }
971 }
972
973 // 最后再进行自动清理
974 WTSLogger::log_dyn("executer", _name.c_str(), LL_INFO, "Position of {}, as prev hot contract, will be cleared", stdCode);
975 // 获取执行单元
976 ExecuteUnitPtr unit = getUnit(stdCode);
977 if (unit)
978 {
979 // 如果有线程池，则异步处理
980 if (_pool)
981 {
982 std::string code = stdCode;
983 _pool->schedule([unit, code](){
984 unit->self()->clear_all_position(code.c_str());
985 });
986 }
987 else
988 {
989 // 直接同步处理
990 unit->self()->clear_all_position(stdCode);
991 }
992 }
993}
