29void ExpHftContext::on_bar(const char* code, const char* period, uint32_t times, WTSBarStruct* newBar)
30{
31 // 如果K线数据为空，直接返回
32 if (newBar == NULL)
33 return;
34
35 // 线程本地静态变量，用于存储真实周期（周期+倍数）
36 thread_local static char realPeriod[8] = { 0 };
37 fmtutil::format_to(realPeriod, "{}{}", period, times);
38
39 // 将K线数据转发给外部接口
40 getRunner().ctx_on_bar(_context_id, code, realPeriod, newBar, ET_HFT);
41
42 // 调用基类的K线闭合回调
43 HftStraBaseCtx::on_bar(code, period, times, newBar);
44}
