130void WtHftStraDemo::do_calc(IHftStraCtx* ctx)
131{
132 const char* code = _code.c_str();
133
134 //30秒内不重复计算
135 uint64_t now = TimeUtils::makeTime(ctx->stra_get_date(), ctx->stra_get_time() * 100000 + ctx->stra_get_secs());//(uint64_t)ctx->stra_get_date()*1000000000 + (uint64_t)ctx->stra_get_time()*100000 + ctx->stra_get_secs();
136 if (now - _last_entry_time <= _freq * 1000)
137 {
138 return;
139 }
140
141 WTSTickData* curTick = ctx->stra_get_last_tick(code);
142 if (curTick == NULL)
143 return;
144
145 uint32_t curMin = curTick->actiontime() / 100000; //actiontime是带毫秒的,要取得分钟,则需要除以10w
146 if (curMin > _last_calc_time)
147 {//如果spread上次计算的时候小于当前分钟,则重算spread
148 //WTSKlineSlice* kline = ctx->stra_get_bars(code, "m5", 30);
149 //if (kline)
150 // kline->release();
151
152 //重算晚了以后,更新计算时间
153 _last_calc_time = curMin;
154 }
155
156 int32_t signal = 0;
157 double price = curTick->price();
158 //计算部分
159 double pxInThry = (curTick->bidprice(0)*curTick->askqty(0) + curTick->askprice(0)*curTick->bidqty(0)) / (curTick->bidqty(0) + curTick->askqty(0));
160
161 //理论价格大于最新价
162 if (pxInThry > price)
163 {
164 //正向信号
165 signal = 1;
166 }
167 else if (pxInThry < price)
168 {
169 //反向信号
170 signal = -1;
171 }
172
173 if (signal != 0)
174 {
175 double curPos = ctx->stra_get_position(code);
176 curPos -= _reserved;
177
178 WTSCommodityInfo* cInfo = ctx->stra_get_comminfo(code);
179
180 if (signal > 0 && curPos <= 0)
181 {//正向信号,且当前仓位小于等于0
182 //最新价+2跳下单
183 double targetPx = price + cInfo->getPriceTick() * _offset;
184 auto ids = ctx->stra_buy(code, targetPx, _unit, "enterlong");
185
186 _mtx_ords.lock();
187 for (auto localid : ids)
188 {
189 _orders.insert(localid);
190 }
191 _mtx_ords.unlock();
192 _last_entry_time = now;
193 }
194 else if (signal < 0 && (curPos > 0 || ((!_stock || !decimal::eq(_reserved, 0)) && curPos == 0)))
195 {//反向信号,且当前仓位大于0,或者仓位为0但不是股票,或者仓位为0但是基础仓位有修正
196 //最新价-2跳下单
197 double targetPx = price - cInfo->getPriceTick()*_offset;
198 auto ids = ctx->stra_sell(code, targetPx, _unit, "entershort");
199
200 _mtx_ords.lock();
201 for (auto localid : ids)
202 {
203 _orders.insert(localid);
204 }
205 _mtx_ords.unlock();
206 _last_entry_time = now;
207 }
208 }
209
210 curTick->release();
211}

241void WtHftStraDemo::check_orders()
242{
243 if (!_orders.empty() && _last_entry_time != UINT64_MAX)
244 {
245 uint64_t now = TimeUtils::makeTime(_ctx->stra_get_date(), _ctx->stra_get_time() * 100000 + _ctx->stra_get_secs());
246 if (now - _last_entry_time >= _secs * 1000) //如果超过一定时间没有成交完,则撤销
247 {
248 _mtx_ords.lock();
249 for (auto localid : _orders)
250 {
251 _ctx->stra_cancel(localid);
252 _cancel_cnt++;
253 _ctx->stra_log_info(fmt::format("Order expired, cancelcnt updated to {}", _cancel_cnt).c_str());
254 }
255 _mtx_ords.unlock();
256 }
257 }

319void WtHftStraDemo::on_order(IHftStraCtx* ctx, uint32_t localid, const char* stdCode, bool isBuy, double totalQty, double leftQty, double price, bool isCanceled, const char* userTag)
320{
321 //如果不是我发出去的订单,我就不管了
322 auto it = _orders.find(localid);
323 if (it == _orders.end())
324 return;
325
326 //如果已撤销或者剩余数量为0,则清除掉原有的id记录
327 if(isCanceled || leftQty == 0)
328 {
329 _mtx_ords.lock();
330 _orders.erase(it);
331 if (_cancel_cnt > 0)
332 {
333 _cancel_cnt--;
334 _ctx->stra_log_info(fmt::format("cancelcnt -> {}", _cancel_cnt).c_str());
335 }
336 _mtx_ords.unlock();
337
338 do_calc(ctx);
339 }
340}

347void WtHftStraDemo::on_channel_ready(IHftStraCtx* ctx)
348{
349 double undone = _ctx->stra_get_undone(_code.c_str());
350 if (!decimal::eq(undone, 0) && _orders.empty())
351 {
352 //这说明有未完成单不在监控之中,先撤掉
353 _ctx->stra_log_info(fmt::format("{}有不在管理中的未完成单 {} 手,全部撤销", _code, undone).c_str());
354
355 bool isBuy = (undone > 0);
356 OrderIDs ids = _ctx->stra_cancel(_code.c_str(), isBuy, undone);
357 for (auto localid : ids)
358 {
359 _orders.insert(localid);
360 }
361 _cancel_cnt += ids.size();
362
363 _ctx->stra_log_info(fmt::format("cancelcnt -> {}", _cancel_cnt).c_str());
364 }
365
366 _channel_ready = true;
367}
