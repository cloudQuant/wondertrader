152 StringVector ret;

154
155 // Use STL methods
156 size_t start, pos;
157 start = 0;
158 do
159 {
160 pos = str.find_first_of(delims, start);
161 if (pos == start)
162 {
163 ret.emplace_back("");
164 // Do nothing
165 start = pos + 1;
166 }
167 else if (pos == std::string::npos || (maxSplits && numSplits == maxSplits))

172 }

175 // Copy up to delimiter

177 start = pos + 1;

182
183 } while (pos != std::string::npos);

407 static inline bool match(const std::string& str, const std::string& pattern, bool caseSensitive = true)
408 {
409 std::string tmpStr = str;
410 std::string tmpPattern = pattern;
411 if (!caseSensitive)
412 {
413 toLowerCase(tmpStr);
414 toLowerCase(tmpPattern);
415 }
416
417 std::string::const_iterator strIt = tmpStr.begin();
418 std::string::const_iterator patIt = tmpPattern.begin();
419 std::string::const_iterator lastWildCardIt = tmpPattern.end();
420 while (strIt != tmpStr.end() && patIt != tmpPattern.end())
421 {
422 if (*patIt == '*')
423 {
424 lastWildCardIt = patIt;
425 // Skip over looking for next character
426 ++patIt;
427 if (patIt == tmpPattern.end())
428 {
429 // Skip right to the end since * matches the entire rest of the string
430 strIt = tmpStr.end();
431 }
432 else
433 {
434 // scan until we find next pattern character
435 while(strIt != tmpStr.end() && *strIt != *patIt)
436 ++strIt;
437 }
438 }
439 else
440 {
441 if (*patIt != *strIt)
442 {
443 if (lastWildCardIt != tmpPattern.end())
444 {
445 // The last wildcard can match this incorrect sequence
446 // rewind pattern to wildcard and keep searching
447 patIt = lastWildCardIt;
448 lastWildCardIt = tmpPattern.end();
449 }
450 else
451 {
452 // no wildwards left
453 return false;
454 }
455 }
456 else
457 {
458 ++patIt;
459 ++strIt;
460 }
461 }
462
463 }
464 // If we reached the end of both the pattern and the string, we succeeded
465 if (patIt == tmpPattern.end() && strIt == tmpStr.end())
466 {
467 return true;
468 }
469 else
470 {
471 return false;
472 }
473 }

680 {
681 std::string ret = "";

683 std::size_t lastPos = 0;

685 while(pos != std::string::npos)
686 {
687 ret += str.substr(lastPos, pos-lastPos);
